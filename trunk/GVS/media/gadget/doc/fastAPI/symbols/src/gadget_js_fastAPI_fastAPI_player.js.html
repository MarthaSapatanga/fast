 24 /**
 25  * Implementation of FastBaseAPI.Request for Player.
 26  * @constructs
 27  * @extends FastBaseAPI.Request
 28  */
 29 FastAPI.Request = Class.create(FastBaseAPI.Request,{
 30 
 31     initialize: function($super, url, options) {
 32         $super(url, options);
 33     },
 34 
 35     request: function() {
 36         var params = this.options;
 37 
 38         var _onSuccess = params.onSuccess;
 39 
 40         params.onSuccess = onSuccess;
 41 
 42         var newParams = {url:this.url, method: params["method"]};
 43         if (params["parameters"]){
 44             if (typeof(params["parameters"])=="string")
 45                 var p = parameters;
 46             else
 47                 var p = this.platform.Object.toJSON(params["parameters"]);
 48             newParams["params"] = p;
 49         }
 50         params["parameters"] = newParams;
 51         params["method"] = "POST";
 52 
 53         new Ajax.Request('/proxy', params);
 54 
 55         // This function handles a success in the asynchronous call
 56         function onSuccess(transport) {
 57             switch(params.content){
 58                 case 'xml':
 59                     (_onSuccess || Prototype.emptyFunction)(transport.responseXML);
 60                     break;
 61                 case 'text':
 62                     (_onSuccess || Prototype.emptyFunction)(transport.responseText);
 63                     break;
 64                 case 'json':
 65                     var json = transport.responseJSON;
 66                     if (!json){
 67                         json = transport.responseText.evalJSON(true);
 68                     }
 69                     (_onSuccess || Prototype.emptyFunction)(json);
 70                     break;
 71                 default:
 72                     (_onSuccess || Prototype.emptyFunction)(transport);
 73                     break;
 74             }
 75         }
 76     }
 77 });
