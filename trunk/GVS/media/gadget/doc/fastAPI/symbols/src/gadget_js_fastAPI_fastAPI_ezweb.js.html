 22 /**
 23  * Implementation of FastBaseAPI.Request for EzWeb.
 24  * @constructs
 25  * @extends FastBaseAPI.Request
 26  */
 27 FastAPI.Request = Class.create(FastBaseAPI.Request,{
 28 
 29     initialize: function($super, url, options) {
 30         $super(url, options);
 31     },
 32 
 33     request: function() {
 34         var params = this.options;
 35 
 36         var _onSuccess = params.onSuccess;
 37 
 38         params.onSuccess = onSuccess;
 39 
 40         if (params.content == 'json'){
 41             params['evalJSON'] = true;
 42             params['sanitizeJSON'] = true;
 43         }
 44 
 45         EzWebAPI.send(this.url, params.context, params);
 46 
 47         function onSuccess(transport) {
 48             switch(params.content){
 49                 case 'xml':
 50                     (_onSuccess || Prototype.emptyFunction)(transport.responseXML);
 51                     break;
 52                 case 'text':
 53                     (_onSuccess || Prototype.emptyFunction)(transport.responseText);
 54                     break;
 55                 case 'json':
 56                     var json = transport.responseJSON;
 57                     if (!json){
 58                         json = transport.responseText.evalJSON(true);
 59                     }
 60                     (_onSuccess || Prototype.emptyFunction)(json);
 61                     break;
 62                 default:
 63                     (_onSuccess || Prototype.emptyFunction)(transport);
 64                     break;
 65             }
 66         }
 67     }
 68 });

 94 FastAPI.Properties = Class.create(FastBaseAPI.Properties,{

110 var PropertiesSingleton = function() {
111     var _instance = null;
112 
113     var Properties = Class.create( /** @lends PropertiesSingleton-Properties.prototype */ {
114         initialize: function() {
115             this.equivalences = {};
116             this.equivalences[FastBaseAPI.Properties.LOGIN] = 'user_name';
117             this.equivalences[FastBaseAPI.Properties.LANGUAGE] = 'language';
118             this.equivalences[FastBaseAPI.Properties.ORIENTATION] = 'orientation';
119             this.equivalences[FastBaseAPI.Properties.GADGET_HEIGHT] = 'heightInPixels';
120             this.equivalences[FastBaseAPI.Properties.GADGET_WIDTH] = 'widthInPixels';
121             this.equivalences[FastBaseAPI.Properties.GADGET_X_POSITION] = 'xPosition';
122             this.equivalences[FastBaseAPI.Properties.GADGET_Y_POSITION] = 'yPosition';
123             this.variables = new Hash();
124         },
125 
126         get: function (variable) {
127             if(!this.equivalences[variable]){
128                 throw 'Variable not supported.' +
129                 'FastAPI.Properties :: get';
130             }
131             var obj = this.variables.get(variable);
132             if(!obj){
133                 obj = EzWebAPI.createRGadgetVariable(this.equivalences[variable], this._callBack);
134                 this.variables.set(variable, obj);
135             }
136             return obj.get();
137         },
138 
139         _callBack: function (value) {
140             return value;
141         }
142     });
143 
144     return new function() {
145         this.getInstance = function() {
146             if (_instance == null) {
147                 _instance = new Properties();
148             }
149             return _instance;
150         }
151     }
152 }();
