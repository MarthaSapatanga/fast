 41 FastBaseAPI.Request = Class.create(/** @lends FastAPI.Request.prototype */{
 42     /**
 43      * Creates an object to make AJAX requests
 44      * @constructs
 45      * @param url
 46      *      url to be requested
 47      * @param options
 48      *      - method: 'get' || 'post' || 'put' || 'delete'
 49      *      - content: 'xml' || 'text' || 'json'
 50      *      - context: context of the invoking object
 51      *      - parameters: either as a URL-encoded string or as any Hash-compatible object.
 52      *      - requestHeaders: a javascript object with properties representing headers.
 53      *      - onSuccess: Invoked when a request completes and its status code is undefined or belongs in the 2xy family.
 54      *      - onFailure: Invoked when a request completes and its status code exists but is not in the 2xy family.
 55      */
 56     initialize: function(url, options) {
 57         this.url = url;
 58 
 59         this.options = {
 60             method:         'post',
 61             asynchronous:   true,
 62             contentType:    'application/x-www-form-urlencoded',
 63             encoding:       'UTF-8',
 64             content:        '',
 65             onSuccess:      Prototype.emptyFunction,
 66             onFailure:      Prototype.emptyFunction
 67         };
 68         Object.extend(this.options, options || { });
 69 
 70         this.options.method = this.options.method.toLowerCase();
 71 
 72         if (Object.isString(this.options.parameters))
 73             this.options.parameters = this.options.parameters.toQueryParams();
 74         else if (Object.isHash(this.options.parameters))
 75             this.options.parameters = this.options.parameters.toObject();
 76         this.setRequestHeaders();
 77         this.request();
 78     },
 79 
 80     /**
 81      * Sets the default headers.
 82      * This method can be overriden and augmented by the implementations.
 83      *  - Accept defaults to 'text/javascript, text/html, application/xml, text/xml, *\/*'
 84      *  - Content-type is built based on the contentType and encoding options.
 85      * @private
 86      */
 87     setRequestHeaders: function() {
 88         var headers = {
 89             'Accept': 'text/javascript, text/html, application/xml, text/xml, application/json, */*'
 90         };
 91 
 92         if (this.options.method == 'post') {
 93             headers['Content-type'] = this.options.contentType +
 94                 (this.options.encoding ? '; charset=' + this.options.encoding : '');
 95         }
 96 
 97         Object.extend(headers, this.options.requestHeaders || { });
 98         this.options.requestHeaders = headers;
 99     },
100 
101     /**
102      * Make a general-purpose request to a remote server.
103      * @private
104      */
105     request: function() {
106         throw 'Abstract Method invocation. ' +
107               'FastAPI.Request :: request';
108     }
109 });
