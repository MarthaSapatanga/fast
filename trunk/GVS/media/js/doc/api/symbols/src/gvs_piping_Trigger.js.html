 26 var Trigger = Class.create(
 27     /** @lends Trigger.prototype */ {
 28 
 29     /**
 30      * Triggers representation
 31      * @constructs
 32      */
 33     initialize: function (/** Object */ from, /** Object */ to) {
 34 
 35         /**
 36          * Trigger source
 37          * @private
 38          * @type Object
 39          */
 40         this._from = from;
 41 
 42         /**
 43          * Trigger destination
 44          * @prvate
 45          * @type Object
 46          */
 47         this._to = to;
 48 
 49         /**
 50          * Trigger id
 51          * @private
 52          * @type String
 53          */
 54         this._id = this._createId(from, to);
 55     },
 56 
 57     /**
 58      * Returns the trigger id
 59      * @type String
 60      */
 61     getId: function() {
 62         return this._id;
 63     },
 64 
 65     getDestinationId: function() {
 66         return this._to.instance.getId();
 67     },
 68 
 69     getDestinationAction: function() {
 70         return this._to.action;
 71     },
 72 
 73     getTriggerName: function() {
 74         return this._from.name;
 75     },
 76 
 77     getSourceId: function() {
 78         return this._from.instance.getId();
 79     },
 80 
 81     getSourceInstance: function() {
 82         return this._from.instance;
 83     },
 84 
 85     /**
 86      * Returns the JSON object representing the Trigger
 87      * @type Object
 88      */
 89     toJSON: function() {
 90         return {
 91             'from': {
 92                 'buildingblock': this.getSourceId(),
 93                 'name': this.getTriggerName()
 94             },
 95             'to': {
 96                 'buildingblock': this.getDestinationId(),
 97                 'action': this.getDestinationAction()
 98             }
 99         };
100     },
101 
102     // **************** PRIVATE METHODS **************** //
103 
104     /**
105      * Creates the trigger id
106      * @private
107      * @type String
108      */
109     _createId: function (/** Object */ from, /** Object */ to) {
110         return from.instance.getId() + from.name +
111                 to.instance.getId() + to.action;
112     }
113 
114 });
