 26 var PreferencesDialog = Class.create(ConfirmDialog /** @lends PreferencesDialog.prototype */, {
 27     /**
 28      * This class handles the dialog
 29      * that shows the user preferences
 30      * @constructs
 31      * @extends ConfirmDialog
 32      */
 33     initialize: function($super) {
 34 
 35         $super("User Preferences");
 36     },
 37 
 38 
 39     // **************** PUBLIC METHODS **************** //
 40 
 41 
 42     // **************** PRIVATE METHODS **************** //
 43 
 44 
 45     /**
 46      * initDialogInterface
 47      * This function creates the dom structure and
 48      * @private
 49      * @override
 50      */
 51     _initDialogInterface: function () {
 52 
 53         this._setHeader("User preferences");
 54 
 55         var user = GVS.getUser();
 56 
 57         var formData = [
 58             {'type':'input', 'label': 'First Name:','name': 'firstName',
 59                     'value': user.getFirstName()},
 60             {'type':'input', 'label': 'Last Name:','name': 'lastName',
 61                     'value': user.getLastName()},
 62             {'type':'input', 'label': 'Email:','name': 'email', 'value': user.getEmail(),
 63                     'regExp': FormDialog.EMAIL_VALIDATION,
 64                     'message': FormDialog.INVALID_EMAIL_MESSAGE,
 65                     'required': true},
 66             {'type':'input', 'label': 'EzWeb URL:','name': 'ezWebURL',
 67                     'value': user.getEzWebURL(),
 68                     'regExp': FormDialog.URL_VALIDATION,
 69                     'message': FormDialog.INVALID_URL_MESSAGE},
 70             {'type':'checkbox', 'label': 'Magic Catalogue:',
 71                     'name': 'catalogueMagic',
 72                     'checked': user.getCatalogueMagic()},
 73             {'type':'checkbox', 'label': 'Search iServe services: <br /> (It may be slow)',
 74                     'name': 'iServe',
 75                     'checked': user.getiServe()}
 76         ];
 77 
 78         this._setContent(formData);
 79 
 80     },
 81     /**
 82      * Overriding onOk handler
 83      * @override
 84      * @private
 85      */
 86     _onOk: function($super){
 87 
 88         if (!this._getFormWidget().validate()) {
 89             return;
 90         }
 91         else {
 92             var user = GVS.getUser();
 93             user.update(this._getForm().serialize(true));
 94             $super();
 95         }
 96     },
 97 
 98     /**
 99      * Reset method to leave the form as initially
100      * @override
101      * @private
102      */
103     _reset: function ($super){
104         var user = GVS.getUser();
105         this._getForm().firstName.value = user.getFirstName();
106         this._getForm().lastName.value = user.getLastName();
107         this._getForm().email.value = user.getEmail();
108         this._getForm().ezWebURL.value = user.getEzWebURL();
109         this._getForm().catalogueMagic.checked = user.getCatalogueMagic() ? 'checked' : 'unchecked';
110         this._getForm().iServe.checked = user.getiServe() ? 'checked' : 'unchecked';
111         $super();
112     }
113 });
