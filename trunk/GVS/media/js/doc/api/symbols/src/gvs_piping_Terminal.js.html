1 /**
  2  * This class handles the creation of pipes based on WireIt.Terminal
  3  * and WireIt.Wire classes. Due to the use of these classes, which are
  4  * not Prototype compliant classes, the syntax of the class is quite
  5  * different than usual
  6  * The terminal object will be attached to each pre or post condition
  7  * of the instances (specifically, to the condition nodes of their
  8  * respective view)
  9  *
 10  * @constructs
 11  */
 12 var Terminal = function(/** DOMNode */ conditionNode, /** Object */ options,
 13                         /** ComponentInstance */ instance,
 14                         /** String */ conditionId, /** String(optional) */ action) {
 15 
 16     /**
 17      * @private
 18      * @type DOMNode
 19      */
 20     this._conditionNode = conditionNode;
 21 
 22     /**
 23      * @private
 24      * @type DOMNode
 25      */
 26     this._terminalNode = new Element('div',
 27     {
 28         'title': this._conditionNode.title
 29     });
 30 
 31     var style = {
 32         'position':'absolute',
 33         'width': '1px',
 34         'height': '1px',
 35         'z-index': '50'
 36     };
 37     this._terminalNode.setStyle(style);
 38     this._recalculatePosition();
 39     document.body.appendChild(this._terminalNode);
 40 
 41     var wireConfig = {
 42             'width': 2,
 43             'borderwidth': 2,
 44             'color': '#EAEAEA',
 45             'bordercolor': '#808080'
 46     }
 47     var extendedOptions = {};
 48     extendedOptions = Object.extend(extendedOptions, options);
 49     extendedOptions.wireConfig = Object.extend(wireConfig, options.wireConfig);
 50 
 51     WireIt.Terminal.call(this, this._terminalNode, extendedOptions);
 52 
 53     /**
 54      * Instance in where the terminal is
 55      * @type ComponentInstance
 56      * @private
 57      */
 58     this._instance = instance;
 59 
 60     /**
 61      * This is the id of the condition inside the
 62      * resource that contains the terminal
 63      * @private
 64      * @type String
 65      */
 66     this._conditionId = conditionId;
 67 
 68     /**
 69      * This is the action
 70      * @private
 71      * @type String
 72      */
 73     this._action = action ? action: "";
 74 }
 75 
 76 // Inheriting all methods
 77 Object.extend(Terminal.prototype, WireIt.Terminal.prototype);
 78 
 79 Object.extend(Terminal.prototype, /** @lends Terminal.prototype */ {
 80 
 81 
 82     // **************** PUBLIC METHODS **************** //
 83 
 84     /**
 85      * Returns the resourceUri
 86      * @type String
 87      */
 88     getBuildingBlockUri: function() {
 89         // FIXME: extrange situation for prepost instances,
 90         // when building
 91         var uri;
 92         if (this._instance.constructor == PrePostInstance) {
 93             uri = null;
 94         } else {
 95             uri = this._instance.getUri();
 96         }
 97         return uri;
 98     },
 99 
100     /**
101      * Returns the resourceId
102      * @type String
103      */
104     getBuildingBlockId: function() {
105         // FIXME: extrange situation for prepost instances,
106         // when building
107         var id;
108         if (this._instance.constructor == PrePostInstance) {
109             id = "";
110         } else {
111             id = this._instance.getId();
112         }
113         return id;
114     },
115 
116     /**
117      * Returns the conditionId
118      * @type String
119      */
120     getConditionId: function() {
121         return this._conditionId;
122     },
123 
124     /**
125      * Returns the action
126      * @type String
127      */
128     getActionId: function() {
129         return this._action;
130     },
131 
132     /**
133      * Returns the instance
134      * @type ComponentInstance
135      */
136     getInstance: function() {
137         return this._instance;
138     },
139 
140     /**
141      * Destroy the terminal
142      */
143     destroy: function() {
144         this.removeAllWires();
145         this.remove();
146         this._terminalNode.parentNode.removeChild(this._terminalNode);
147     },
148 
149     /**
150      * Updates the position when the container is moving
151      */
152     updatePosition: function() {
153         this._recalculatePosition();
154         this.redrawAllWires();
155     },
156 
157     /**
158      * Adds a handler listening for the connection or deconnection of wires
159      */
160     onWireHandler: function(/** Hash */ handlers) {
161     	var context = {};
162     	context = Object.extend(context, handlers);
163     	context['refTerminal'] = this;
164 
165         this.eventAddWire.subscribe(this._wireAddHandler.bind(context));
166         this.eventRemoveWire.subscribe(this._wireRemoveHandler.bind(context));
167     },
168 
169     /**
170      * Sets the visibility of the terminal
171      */
172     setVisible: function(/** Boolean */ visible) {
173         var style = {
174             'visibility': visible ? 'visible': 'hidden'
175         };
176         this.el.setStyle(style);
177     },
178 
179      // **************** PRIVATE METHODS **************** //
180 
181 
182      /**
183       * Recalculates the position of the terminal
184       * @private
185       */
186     _recalculatePosition: function() {
187         var position = Utils.getPosition(this._conditionNode);
188         var style = {
189             'top': position.top + 'px',
190             'left':  position.left + 'px'
191         }
192         this._terminalNode.setStyle(style);
193     },
194 
195 
196     /**
197      * Handler called whenever a new wire is added to the terminal
198      * @private
199      */
200     _wireAddHandler: function(/** Event */ event, /** Array */ params) {
201     	var wire = params[0];
202     	if (wire.terminal1.parentEl && wire.terminal2.parentEl) {
203     		if (wire.terminal1 == this.refTerminal) {
204     			this.onPipeCreation(wire);
205     		}
206     	} else if (wire.terminal1 == this.refTerminal) {
207     		this.onPipeCreationStart(wire, wire.terminal1);
208     	} else if (wire.terminal2 == this.refTerminal) {
209     		this.onPipeCreationStart(wire, wire.terminal2);
210     	}
211     },
212 
213 
214     /**
215      * Handler called whenever a wire is removed from the terminal
216      * @private
217      */
218     _wireRemoveHandler: function(/** Event */ event, /** Array */ params) {
219     	var wire = params[0];
220     	if (wire.terminal1.parentEl && wire.terminal2.parentEl) {
221     		if (wire.terminal1 == this.refTerminal) {
222     			this.onPipeDeletion(wire);
223     		}
224     	} else if (wire.terminal1 == this.refTerminal || wire.terminal2 == this.refTerminal) {
225     		this.onPipeCreationCancel(wire);
226     	}
227     }
228 });
