1 var TriggerMappingFactory = Class.create(
  2     /** @lends TriggerMappingFactory.prototype */ {
  3 
  4     /**
  5      * It handles the connection and deconnection of actions and triggers
  6      * for a screen
  7      * @constructs
  8      */
  9     initialize: function () {
 10         /**
 11          * This contains all the triggers in the screen
 12          * @type Hash
 13          * @private
 14          */
 15         this._triggers = new Hash();
 16     },
 17     // **************** PUBLIC METHODS ***************** //
 18 
 19     /**
 20      * Creates a new trigger mapping coming from a created pipe.
 21      * @type Trigger
 22      */
 23     createTrigger: function (/** Object */ element) {
 24         var trigger = this._createFromJSON(element);
 25         this._triggers.set(trigger.getId(), trigger);
 26         return trigger;
 27     },
 28 
 29     /**
 30      * Removes a trigger from the list. The trigger can be passed as
 31      * parameter
 32      * @type Trigger
 33      */
 34     removeTrigger: function(/** Object */ element) {
 35         var trigger;
 36         if (element.constructor != Trigger && element.constructor != ScreenTrigger) {
 37             trigger = this._createFromJSON(element);
 38         } else {
 39             trigger = element;
 40         }
 41         this._triggers.unset(trigger.getId());
 42         return trigger;
 43     },
 44 
 45     /**
 46      * Returns a list of triggers that the instance has connected to each of its
 47      * actions
 48      * @type Hash
 49      */
 50     getTriggerList: function(/** ComponentInstance */ instance) {
 51         var result = new Hash();
 52         this._triggers.values().each(function(trigger) {
 53            if (trigger.getDestinationId() == instance.getId()) {
 54                var array = result.get(trigger.getDestinationAction());
 55                if (!array) {
 56                    array = new Array();
 57                }
 58                array.push(trigger);
 59                result.set(trigger.getDestinationAction(), array);
 60            }
 61         });
 62         return result;
 63     },
 64 
 65     /**
 66      * Returns a list with triggers related to a given instance
 67      */
 68     getRelatedTriggers: function(/** ComponentInstance */ instance) {
 69         var result = new Array();
 70         this._triggers.values().each(function(trigger) {
 71             if (trigger.getSourceId() == instance.getId() ||
 72                 trigger.getDestinationId() == instance.getId()) {
 73                     result.push(trigger);
 74                 }
 75         });
 76         return result;
 77     },
 78 
 79     // **************** PRIVATE METHODS **************** //
 80 
 81     /**
 82      * @private
 83      */
 84     _createFromJSON: function(element) {
 85         var trigger;
 86         if (element.from.instance == ScreenTrigger.INSTANCE_NAME) {
 87             trigger = new ScreenTrigger(element.to);
 88         } else {
 89             trigger = new Trigger(element.from, element.to);
 90         }
 91         return trigger;
 92     }
 93 });
