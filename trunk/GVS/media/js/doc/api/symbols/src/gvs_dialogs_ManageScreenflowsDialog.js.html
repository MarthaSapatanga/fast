  2     /**
  3      * This class handles the dialog

  8     initialize: function($super) {

 10                                       'disableIfNotValid': true });
 11 
 12         /**
 13          * List of screens
 14          * @type Array
 15          * @private

 18     },

 24      * @override

 33 
 34     /** 
 35      * initDialogInterface
 36      * This function creates the dom structure
 37      * @private
 38      * @override
 39      */
 40     _initDialogInterface: function (){
 41 
 42         this._setHeader("Browse your screenflows",
 43                 "These are the screenflows you have created. Here you can " +
 44                 "continue editing them and share your work with the community");
 45 
 46         this._setFields([{
 47                 'title': 'Screenflow Name',
 48                 'className': 'name'
 49             }, {
 50                 'title': 'Screenflow Version',
 51                 'className': 'version'
 52             }, {
 53                 'title': 'Tags',
 54                 'className': 'tags'
 55             }, {
 56                 'title': 'Description',
 57                 'className': 'description'
 58             }, {
 59                 'title': 'Sharing',
 60                 'className': 'sharing'
 61             }
 62         ]);
 63 
 64         this._setButtons([{
 65                 'value': 'Open screenflow',
 66                 'handler': this._openScreenflow.bind(this),
 67                 'disableIfNotValid': true
 68             }, {
 69                 'value': 'Share/Unshare screenflow',
 70                 'handler': this._shareScreenflow.bind(this)

 72                 'value': 'Delete screenflow',

 74             }/*, {

 78

 82 
 83     /**
 84      * Creates the the screen list to be handled by its parent class
 85      * @private
 86      */
 87     _createScreenflowList: function() {
 88         this._emptyRows();
 89         this._screenflows.each(function(screenflow) {
 90             this._addRow({
 91                             'key': screenflow.id,
 92                             'values': [
 93                                        screenflow.name,
 94                                        '<span class="bold">Version: </span>' +
 95                                          screenflow.version,
 96                                        '<span class="bold">Tags: </span>' +
 97                                          screenflow.tags.collect(function(tag) {
 98                                            return tag.label['en-gb'];
 99                                          }).join(", "),
100                                        '<span class="bold">Description </span><br />'+
101                                 	 screenflow.description['en-gb'],
102                                        '<span class=' + (screenflow.uri ? '"shared"': '"unshared"') +
103                                          '> </span>'
104                                       ],
105                             'isValid': screenflow.definition ? true : false
106                         });
107         }.bind(this));
108     },
109     
110     /**
111      * On Success handler
112      * @private
113      */
114     _onLoadSuccess: function(/** XMLHttpRequest */ transport) {
115         this._screenflows = JSON.parse(transport.responseText);
116         this._show();
117     },
118 
119     /**
120      * On Success handler, when reload
121      * @private
122      */
123     _onReLoadSuccess: function(/** XMLHttpRequest */ transport) {
124         this._screenflows = JSON.parse(transport.responseText);
125         this._createScreenflowList();
126         this._render(false);
127     },
128 
129     /**
130      * Open a screen by its id
131      * @private
132      */
133     _openScreenflow: function(/** String */ id) {
134         var documentController = GVS.getDocumentController();
135         documentController.loadScreenflow(id);
136         this._dialog.hide();
137     },
138 
139     /**
140      * Share or unshare a screen depending of its status
141      * @private
142      */
143     _shareScreenflow: function(/** String */ id) {
144         var uri = URIs.share.replace("<id>", id);
145 
146         var screen = this._screenflows.detect(function(element) {
147             return element.id == id;
148         });
149         if (screen.uri) {
150             // Unshare screen       
151             PersistenceEngine.sendDelete(uri, this, this._reload, Utils.onAJAXError);
152         } else {
153             // Share screen
154             PersistenceEngine.sendPost(uri, null, null, this, this._reload,
155                                         Utils.onAJAXError);
156         }
157     },
158 
159     /**
160      * Starts the deletion of a screen
161      * @private
162      */
163     _deleteScreenflow: function(/** String */ id) {
164         confirm("Are you sure you want to delete the screenflow? This action cannot " +
165         "be undone", this._confirmDelete.bind({'mine': this, 'id': id}));
166     },
167 
168     /**
169      * Execute the deletion of the screen
170      * @private
171      */
172     _confirmDelete: function() {
173         var uri = URIs.buildingblock + this.id;
174         PersistenceEngine.sendDelete(uri, this.mine, this.mine._reload, Utils.onAJAXError);
175     },
176 
177     /**
178      * Starts the process of adding an external screen
179      * @private
180      */
181     _addScreenflow: function() {
182         this._dialog.hide();
183         //GVS.action("addScreenflow");
184     },
185 
186     /**
187      * Reloads the screen list
188      * @private
189      */
190     _reload: function() {
191         PersistenceEngine.sendGet(URIs.screenflow, this, this._onReLoadSuccess, Utils.onAJAXError);
192     }
193 });
