 26 var DragHandler = Class.create(
 27     /** @lends DragHandler.prototype */ {
 28 
 29     /**
 30      * Enables an object to be dragged.
 31      *
 32      * @param dragSource
 33      *      An object implementing DragSource
 34      * @constructs
 35      */
 36     initialize: function (/** DragSource */ dragSource) {
 37 
 38         /**
 39          * An object returning objects to be dragged.
 40          * @type DragSource
 41          * @private
 42          */
 43         this._dragSource = dragSource;
 44 
 45         /**
 46          * Currently dragged object
 47          * @type Object
 48          * @private
 49          */
 50         this._draggedObject = null;
 51 
 52         /**
 53          * Zone from which the dragsource is dragged.
 54          * It can be null.
 55          * @type DropZone
 56          * @private
 57          */
 58         this._initialDropZone = null;
 59 
 60         /**
 61          * Valid Zones to drop the dragged object, plus their position
 62          * @type Array
 63          * @private
 64          */
 65         this._dropZonesInfo = new Array();
 66 
 67 
 68         // Wrappers to the event handlers
 69         this._boundedStartDrag = this._startDrag.bind(this);
 70         this._boundedUpdate = this._update.bind(this);
 71         this._boundedEndDrag = this._endDrag.bind(this);
 72     },
 73 
 74 
 75     /**
 76      * Initializes the drag'n'drop handlers and sets the drop zones
 77      */
 78     enableDragNDrop: function (/** DropZone */ currentZone, /** Array */ validDropZones) {
 79 
 80         this._initialDropZone = currentZone;
 81         this._initialDropZonePosition = null;
 82 
 83         validDropZones.each(function(dropZone) {
 84             this._dropZonesInfo.push({
 85                 'dropZone': dropZone,
 86                 'position': null
 87             });
 88         }.bind(this));
 89 
 90         // add mousedown event listener
 91         Event.observe (this._dragSource.getHandlerNode(), "mousedown",
 92                 this._boundedStartDrag , true);
 93     },
 94 
 95 
 96     // **************** PRIVATE METHODS **************** //
 97 
 98 
 99     /**
100      * Drag intialization.
101      * @private
102      */
103     _startDrag: function(e) {
104         e = e || window.event; // needed for IE
105 
106         // Aux data for position calculation
107         this._mouseXStart = 0;
108         this._mouseYStart = 0;
109         this._x = 0;
110         this._y = 0;
111         this._offLimitX = 0;
112         this._offLimitY = 0;
113 
114         // Only attend to left button events
115         // (or right button for left-handed persons)
116         if (!BrowserUtils.isLeftButton(e.button)) {
117             return false;
118         }
119 
120         // An object is retrieved to be dragged
121         this._draggedObject = this._dragSource.getDraggableObject();
122         var draggableNode = this._draggedObject.getHandlerNode();
123 
124         this._initialArea = draggableNode.parentNode;
125 
126         // Set the dropZones areas in coordinates
127         this._dropZonesInfo.each (function(dropZoneInfo) {
128             var dropZoneNode = dropZoneInfo.dropZone.getNode();
129             dropZoneInfo.position = Geometry.getClientRectangle(dropZoneNode);
130         }.bind(this));
131         if (this._initialDropZone) {
132             var initialDropZoneNode = this._initialDropZone.getNode();
133             this._initialDropZonePosition = Geometry.getRectangle(initialDropZoneNode);
134         }
135 
136         this._toggleEventHandlers(true);
137 
138         this._draggedObject.onStart();
139 
140 
141         this._mouseXStart = parseInt(e.screenX);
142         this._mouseYStart = parseInt(e.screenY);
143         this._y = draggableNode.offsetTop;
144         this._x = draggableNode.offsetLeft;
145 
146         if (this._isChangingZone()) {
147             draggableNode.addClassName("dragOnChangeLayer");
148         } else {
149             draggableNode.addClassName("dragLayer");
150         }
151 
152         return false;
153     },
154 
155 
156     /**
157      * Update position event handler
158      * @private
159      */
160     _update: function (e) {
161         e = e || window.event; // needed for IE
162 
163         var mouseX = parseInt(e.screenX);
164         var mouseY = parseInt(e.screenY);
165 
166         this._updateNodePosition(mouseX, mouseY);
167         this._updateNodeStatus(this._isValidPosition());
168 
169         this._draggedObject.onUpdate(this._x, this._y);
170 
171         var draggableNode = this._draggedObject.getHandlerNode();
172     },
173 
174 
175     /**
176      * Drop event handler
177      * @private
178      */
179     _endDrag: function(e) {
180         e = e || window.event; // needed for IE
181 
182         // Only attend to left button events
183         // (or right button for left-handed persons)
184         if (!BrowserUtils.isLeftButton(e.button))
185             return false;
186 
187         this._toggleEventHandlers(false);
188 
189         var draggableNode = this._draggedObject.getHandlerNode();
190         draggableNode.removeClassName("dragLayer");
191         draggableNode.removeClassName("dragOnChangeLayer");
192 
193         //Remove element transparency
194         this._updateNodeStatus(true);
195 
196         var accepted;
197         var dropPosition;
198         // When changing zone, try to get the draggable accepted by the dropZone
199         if (this._isChangingZone()) {
200             var dropZone = this._inWhichDropZone();
201             if (dropZone) {
202                 dropPosition = Geometry.adaptDropPosition(dropZone.getNode(), draggableNode);
203                 this._initialArea.removeChild(draggableNode);
204                 accepted = dropZone.drop(this._draggedObject, dropPosition);
205             } else {
206                 this._initialArea.removeChild(draggableNode);
207                 accepted = false;
208             }
209 
210             if (!accepted) {
211                 // Destroy the element (it is an invalid copy)
212                 this._draggedObject.destroy();
213             }
214         } else {
215             accepted = true;
216             dropPosition = {
217                 'top': draggableNode.offsetTop,
218                 'left': draggableNode.offsetLeft
219             };
220         }
221 
222         if (accepted) {
223             this._draggedObject.onFinish(this._isChangingZone(), dropPosition);
224         }
225         this._draggedObject = null;
226 
227         return false;
228     },
229 
230     /**
231      * Enable and disable the different event handlers
232      * @private
233      */
234     _toggleEventHandlers: function (/** Boolean */ startDrag) {
235         if (startDrag) {
236             // disable context menu and text selection
237             document.oncontextmenu = function() { return false; };
238             document.onmousedown = function() { return false; };
239             document.onselectstart =  function() { return false; };
240 
241             Event.stopObserving (this._draggedObject.getHandlerNode(), 'mousedown',
242                     this._boundedStartDrag, true);
243             Event.observe (document, 'mouseup',   this._boundedEndDrag, true);
244             Event.observe (document, 'mousemove', this._boundedUpdate, true);
245         } else {
246             // Reenable context menu and text selection
247             document.onmousedown = null;
248             document.oncontextmenu = null;
249             document.onselectstart = null;
250             Event.observe (this._dragSource.getHandlerNode(), "mousedown",
251                     this._boundedStartDrag, true);
252             Event.stopObserving (document, "mouseup",   this._boundedEndDrag, true);
253             Event.stopObserving (document, "mousemove", this._boundedUpdate,  true);
254         }
255     },
256 
257     /**
258      * This function detects if the node is going to change from one zone to
259      * another
260      * @private
261      * @type Boolean
262      */
263     _isChangingZone: function() {
264        return (this._initialDropZone == null);
265     },
266 
267 
268      /**
269      * This function calculates whether the element
270      * is over a valid drop zone or not, and
271      * check the DropZone Restrictions
272      * @private
273      * @type Boolean
274      */
275     _isValidPosition: function(){
276         if (this._isChangingZone()) {
277             return (this._inWhichDropZone() != null);
278         } else {
279             return true;
280         }
281     },
282 
283      /**
284      * This function calculates the exact dropZone the element
285      * is
286      * @private
287      * @type DropZone
288      */
289     _inWhichDropZone: function(){
290         for (var i=0; i < this._dropZonesInfo.length; i++) {
291             if (Geometry.intersects(this._dropZonesInfo[i].position,
292                         this._getDraggableNodeRectangle())) {
293                 return this._dropZonesInfo[i].dropZone;
294             }
295         }
296         return null;
297     },
298     /**
299      * @private
300      * @type Object
301      */
302     _getDraggableNodeRectangle: function () {
303         var draggableNode = this._draggedObject.getHandlerNode();
304         return {
305             'top': draggableNode.offsetTop,
306             'left': draggableNode.offsetLeft,
307             'bottom': draggableNode.offsetTop + draggableNode.offsetHeight,
308             'right': draggableNode.offsetLeft + draggableNode.offsetWidth
309         };
310     },
311 
312     /**
313      * This function updates the position of a node
314      * regarding the absolute position of the mouse
315      * @private
316      */
317     _updateNodePosition: function(/** Number */ x, /** Number */ y) {
318         var xDelta = x - this._mouseXStart;
319         var yDelta = y - this._mouseYStart;
320         this._mouseXStart = x;
321         this._mouseYStart = y;
322 
323         var node = this._draggedObject.getHandlerNode();
324         if (!this._isChangingZone()) {
325             var ranges = Geometry.dragRanges(this._initialDropZonePosition,
326                     this._getDraggableNodeRectangle());
327 
328             var effectiveUpdateX = Geometry.updateAxis(ranges.x, xDelta, this._offLimitX);
329             xDelta = effectiveUpdateX.delta;
330             this._offLimitX = effectiveUpdateX.offLimit;
331 
332             var effectiveUpdateY = Geometry.updateAxis(ranges.y, yDelta, this._offLimitY);
333             yDelta = effectiveUpdateY.delta;
334             this._offLimitY = effectiveUpdateY.offLimit;
335         }
336 
337         this._y = this._y + yDelta;
338         this._x = this._x + xDelta;
339 
340         node.style.top = this._y + 'px';
341         node.style.left = this._x + 'px';
342     },
343 
344 
345      /**
346      * This function updates node interface,
347      * taking into account if it is in a valid
348      * position or not
349      * @private
350      */
351     _updateNodeStatus: function(/** Boolean */ isValid){
352         if (isValid){
353             this._draggedObject.getHandlerNode().removeClassName('disabled');
354         }
355         else {
356             this._draggedObject.getHandlerNode().addClassName('disabled');
357         }
358     }
359 
360 
361 
362 });
