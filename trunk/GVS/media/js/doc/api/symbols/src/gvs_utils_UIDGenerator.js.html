1 /**
  2  * Unique ID Generator
  3  * @constructs
  4  */
  5 var UIDGenerator = Class.create();
  6 
  7 // **************** STATIC ATTRIBUTES **************** //
  8 
  9 Object.extend(UIDGenerator, {
 10     /**
 11      * Next available ids
 12      * @type Object
 13      * @private
 14      */
 15     _nextIds: new Object()
 16 });
 17 
 18 // **************** STATIC METHODS ******************* //
 19 
 20 Object.extend(UIDGenerator, {
 21 
 22     /**
 23      * Returns a valid new DOM Id
 24      * @param String element  Type of element that needs the identifier
 25      * @type String
 26      */
 27     generate: function (/** String */ element) {
 28         var sanitized = element.replace(new RegExp('\\s', 'g'), "")
 29                                 .replace("_","");
 30         var nextId = this._nextIds[sanitized];
 31 
 32         if (!nextId){
 33             nextId = 1;
 34         }
 35 
 36         this._nextIds[sanitized] = nextId + 1;
 37 
 38         return sanitized + "_" + nextId;
 39     },
 40 
 41     /**
 42      * Sets the initial id for a given name
 43      */
 44     setStartId: function(/** String */ id) {
 45         var pieces = id.split("_");
 46         var name = pieces[0];
 47         var lastId = parseInt(pieces[1]);
 48         if (!this._nextIds[name] || this._nextIds[name] <= lastId) {
 49             this._nextIds[name] = lastId + 1;
 50         }
 51     }
 52 });
