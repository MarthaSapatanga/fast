 26 var StandaloneEmbeddingDialog = Class.create(ConfirmDialog /** @lends PreviewDialog.prototype */, {
 27     /**
 28      * This class handles the dialog
 29      * to preview a component
 30      * @constructs
 31      * @extends ConfirmDialog
 32      */
 33     initialize: function($super, /** Object */ publication, /** String */ url) {
 34 
 35         /**
 36          * URL to embed
 37          * @private
 38          * @type String
 39          */
 40         this._url = url;
 41 
 42         /**
 43          * Publication info
 44          * @private
 45          * @type Object
 46          */
 47         this._publication = publication;
 48 
 49         $super(this._getDialogTitle(this._publication.name), ConfirmDialog.OK);
 50         this._setContent(content);
 51     },
 52 
 53 
 54     // **************** PUBLIC METHODS **************** //
 55 
 56     /**
 57      * updateDialog
 58      * Updates dialog header and content
 59      * @public
 60      */
 61     updateDialog: function (/** Object */ publication, /** String */ url) {
 62         this._url = url;
 63         this._publication = publication;
 64         this._initDialogInterface();
 65         this._setTitle(this._getDialogTitle(this._publication.name), null);
 66     },
 67 
 68     // **************** PRIVATE METHODS **************** //
 69 
 70     /**
 71      * initDialogInterface
 72      * This function creates the dom structure and
 73      * @private
 74      * @override
 75      */
 76     _initDialogInterface: function () {
 77 
 78         var formData = [
 79             {'type':'title', 'value': 'Please, embed this HTML code into your web page:'},
 80             {'type':'pre', 'value': this._getEmbedding()}
 81         ];
 82 
 83         this._setContent(formData);
 84 
 85     },
 86 
 87     /**
 88      * getEmbedding
 89      * This function creates the dom structure and
 90      * @private
 91      * @override
 92      */
 93     _getEmbedding: function () {
 94         var height = this._publication.height;
 95         if (!height || height == ''){
 96             height = '600';
 97         }
 98         var width = this._publication.width;
 99         if (!width || width == ''){
100             width = '400';
101         }
102         return ('<object data="' + this._url + '" height="'+ height
103                 + '" width="' + width + '" class="embed"></object>').escapeHTML();
104     },
105 
106     /**
107      * getDialogTitle
108      * This function returns the dialog title
109      * @private
110      * @override
111      */
112     _getDialogTitle: function (/** String */ name) {
113         return 'Embedding of ' + name + ' Standalone Gadget';
114     }
115 
116 });
