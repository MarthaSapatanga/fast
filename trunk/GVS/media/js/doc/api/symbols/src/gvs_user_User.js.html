 26 var User = Class.create( /** @lends User.prototype */ {
 27     /**
 28      * This class handles the user account and properties
 29      * It is used directly by the GVS
 30      * @constructs
 31      */
 32     initialize: function() {
 33         /**
 34          * @type String
 35          * @private @member
 36          */
 37         this._userName = null;
 38 
 39         /**
 40          * @type String
 41          * @private @member
 42          */
 43         this._firstName = null;
 44 
 45         /**
 46          * @type String
 47          * @private @member
 48          */
 49         this._lastName = null;
 50 
 51         /**
 52          * @type String
 53          * @private @member
 54          */
 55         this._email = null;
 56 
 57         /**
 58          * @type String
 59          * @private @member
 60          */
 61         this._ezWebURL = null;
 62 
 63 
 64         /**
 65          * Defines the magic level of the catalogue.
 66          * Magic stands for:
 67          *    <li> Automatic reordering of the building blocks
 68          * @type Boolean
 69          * @private
 70          */
 71         this._catalogueMagic = false;
 72 
 73         /**
 74          * Do you want iserve integration?
 75          * @private
 76          * @type Boolean
 77          */
 78         this._iServe = false;
 79 
 80         //Bring the user from the server
 81         this._getUser();
 82     },
 83 
 84 
 85     // **************** PUBLIC METHODS **************** //
 86 
 87     /**
 88      * getUserName
 89      * @type String
 90      */
 91     getUserName: function () {
 92         return this._userName;
 93     },
 94 
 95     /**
 96      * getUserName
 97      * @type String
 98      */
 99     getFirstName: function () {
100         return this._firstName;
101     },
102 
103     /**
104      * getUserName
105      * @type String
106      */
107     getLastName: function () {
108         return this._lastName;
109     },
110 
111     /**
112      * getEmail
113      * @type String
114      */
115     getEmail: function () {
116         return this._email;
117     },
118 
119     /**
120      * getEzWebURL
121      * @type String
122      */
123     getEzWebURL: function () {
124         return this._ezWebURL;
125     },
126 
127     /**
128      * getRealName
129      * @type String
130      */
131     getRealName: function () {
132         return this._firstName + " " + this._lastName;
133     },
134 
135     /**
136      * @type Boolean
137      */
138     getCatalogueMagic: function() {
139         return this._catalogueMagic;
140     },
141 
142     /**
143      * @param Boolean
144      */
145     setCatalogueMagic: function(catalogueMagic) {
146         this._catalogueMagic = catalogueMagic;
147     },
148 
149     /**
150      * @type Boolean
151      */
152     getiServe: function() {
153         return this._iServe;
154     },
155 
156     /**
157      * @param Boolean
158      */
159     setiServe: function(iServe) {
160         this._iServe = iServe;
161     },
162 
163     /**
164      * This function updates the user object
165      * based on the data passed
166      */
167     update: function (/** Hash */ userData){
168         //TODO empty data control
169         this._firstName = userData.firstName;
170         this._lastName = userData.lastName;
171         this._email = userData.email;
172         this._catalogueMagic = (userData.catalogueMagic !== undefined);
173         this._iServe = (userData.iServe !== undefined);
174 
175         this._ezWebURL = userData.ezWebURL;
176 
177         if (this._ezWebURL != "" && this._ezWebURL[this._ezWebURL.length-1] != '/') {
178             this._ezWebURL += '/';
179         }
180 
181         URIs.ezweb = this._ezWebURL;
182 
183         //Send it to the server
184         this._updateUser();
185     },
186     // **************** PRIVATE METHODS **************** //
187 
188     /**
189      *  This function gets the user from the server
190      *  @private
191      */
192     _getUser: function (){
193         /**
194          * onSuccess handler
195          * @private
196          */
197         var onSuccess = function (/** XmlHttpRequest */ response){
198             var remoteUser = JSON.parse(response.responseText);
199 
200             this._userName = remoteUser.user.username;
201             this._firstName = remoteUser.user.first_name;
202             this._lastName = remoteUser.user.last_name;
203             this._email = remoteUser.user.email;
204             this._ezWebURL = remoteUser.profile.ezweb_url;
205 
206             if (this._ezWebURL != "" && this._ezWebURL[this._ezWebURL.length-1] != '/') {
207                 this._ezWebURL += '/';
208             }
209 
210             URIs.ezweb = this._ezWebURL;
211         }
212         /**
213          * onError handler
214          * @private
215          */
216         var onError = function (/** XmlHttpRequest */ response, /** Exception */ e){
217             Logger.serverError (response, e);
218         }
219 
220         PersistenceEngine.sendGet(URIs.userPreferences, this, onSuccess, onError);
221     },
222     /**
223      *  This function updates the user to the server
224      *  @private
225      */
226     _updateUser: function (){
227 
228         /**
229          * onSuccess handler
230          * @private
231          */
232         var onSuccess = function (/** XmlHttpRequest */ response){
233         }
234 
235         /**
236          * onError handler
237          * @private
238          */
239         var onError = function (/** XmlHttpRequest */ response){
240             Logger.serverError (response, e);
241         }
242 
243         var object = {
244             'first_name': this._firstName,
245             'last_name': this._lastName,
246             'email': this._email,
247             'ezweb_url': this._ezWebURL
248         };
249         var preferences = {preferences :Object.toJSON(object)};
250 
251         PersistenceEngine.sendUpdate(URIs.userPreferences, preferences, null,
252                                 this, onSuccess, onError);
253     }
254 
255 });
