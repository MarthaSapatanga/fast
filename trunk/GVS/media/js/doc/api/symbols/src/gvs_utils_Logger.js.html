 26 var Logger = Class.create();
 27 
 28 Object.extend(Logger, {
 29     serverError: function(/** XmlHttpRequest */ transport, /** Exception */ e){
 30         var msg;
 31         if (e) {
 32             msg = "JavaScript exception on file #{errorFile} (line: #{errorLine}): #{errorDesc}".interpolate({errorFile: e.fileName, errorLine: e.lineNumber, errorDesc: e});
 33         } else if (transport.responseXML) {
 34             msg = transport.responseXML.documentElement.textContent;
 35         } else {
 36             try {
 37                 m = JSON.parse(transport.responseText);
 38                 msg = m.message;
 39             } catch (e) {}
 40             if(!msg){
 41                 msg = "HTTP Error #{status} - #{text}".interpolate({status: transport.status, text: transport.statusText});
 42             }
 43         }
 44         console.log(msg);
 45     }
 46 
 47 });
