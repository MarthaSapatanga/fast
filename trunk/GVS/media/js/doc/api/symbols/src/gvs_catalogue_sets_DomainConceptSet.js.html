 26 var DomainConceptSet = Class.create(BuildingBlockSet, /** @lends DomainConceptSet.prototype */ {
 27     /**
 28      * This class handles a set of building blocks for a given context.
 29      * This list will be updated.
 30      * @constructs
 31      * @extends BuildingBlockSet
 32      */
 33     initialize: function($super, /** Array */ tags, /** DomainConceptFactory */ factory) {
 34         $super(tags, factory);
 35 
 36         /**
 37          * Domain concepts
 38          * @type Array
 39          * @private
 40          */
 41         this._domainConcepts = new Array();
 42 
 43     },
 44 
 45 
 46     // **************** PUBLIC METHODS **************** //
 47 
 48     /**
 49      * Starts the data retrieval of the domain concepts
 50      */
 51     startRetrievingData: function() {
 52         this._factory.getBuildingBlocks(this._tags, this._onSuccess.bind(this));
 53     },
 54 
 55 
 56     /**
 57      * Returns all the building block descriptions from the set
 58      * @type Array
 59      * @override
 60      */
 61     getBuildingBlocks: function () {
 62         return this._domainConcepts;
 63     },
 64 
 65     // **************** PRIVATE METHODS **************** //
 66 
 67     /**
 68      * @private
 69      */
 70     _onSuccess: function(/** Array */ domainConcepts) {
 71         this._domainConcepts = domainConcepts;
 72         this._listener.setChanged();
 73     }
 74 });
