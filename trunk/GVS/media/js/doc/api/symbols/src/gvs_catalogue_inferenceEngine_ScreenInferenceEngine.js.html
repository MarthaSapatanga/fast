1 var ScreenInferenceEngine = Class.create( /** @lends ScreenInferenceEngine.prototype */ InferenceEngine, {
  2     /**
  3      * This class handles the reachability and recommendation of building blocks
  4      * It communicates with the serverside catalogue to retrieve this information
  5      * @extends InferenceEngine
  6      * @constructs
  7      */
  8     initialize: function($super) {
  9         $super();
 10     },

 18      /**
 19      * Creates a body to be sent in an AJAX call to the
 20      * catalogue
 21      * @private
 22      * @overrides
 23      * @type String
 24      */
 25     _constructBody: function(/**Array*/ canvas, /** Object */ elements,
 26                     /** Array */ tags,
 27                     /** String*/ criteria, /** String(Optional) */ _method) {
 28         var method = Utils.variableOrDefault(_method, "findcheck");
 29         /*var domainContext = {
 30             'tags': tags,
 31             'user': GVS.getUser().getUserName()
 32         };*/
 33 
 34         var domainContext = {
 35             'tags': tags.collect(function(tag){ return tag.label['en-gb']; }),
 36             'user': GVS.getUser().getUserName()
 37         };
 38         var body = {
 39             'canvas': canvas,
 40             'domainContext': domainContext,
 41                 'criterion': criteria
 42         };
 43         var actions = ["check"];
 44 
 45         if (method == "findcheck") {
 46             actions.push("find");
 47             if (GVS.getUser().getiServe()) {
 48                 actions.push("iserve");
 49             }
 50 
 51         }
 52         body.actions = actions;
 53         body = Object.extend(body, elements);
 54         return Object.toJSON(body);
 55     },
 56 
 57     /**
 58      * Gets the uri for a given operation
 59      * @private
 60      * @overrides
 61      */
 62     _getUri:function (/** String */ operation) {
 63         return URIs.catalogueScreenFindCheck;
 64     },
 65 
 66     /**
 67      * onSuccess callback
 68      * @private
 69      * @overrides
 70      */
 71     _findCheckOnSuccess: function(/** XMLHttpRequest */ transport){
 72         var result = JSON.parse(transport.responseText);
 73 
 74         if (result.canvas && result.canvas.length > 0) {
 75             // There is some reachability information
 76             this.mine._updateReachability(result.canvas);
 77             if (!GVS.getUser().getCatalogueMagic()) {
 78                 var paletteComponents = result.forms.concat(result.operators,
 79                 result.backendservices);
 80                 this.mine._updateReachability(paletteComponents);
 81             }
 82         }
 83 
 84         this.callback(result);
 85     },
 86 
 87     /**
 88      * onSuccess callback
 89      * @private
 90      * @overrides
 91      */
 92     _checkOnSuccess: function(transport){
 93         try {
 94              var result = JSON.parse(transport.responseText);
 95              this.mine._updateReachability(result.canvas);
 96              this.callback(result);
 97         }
 98         catch (e) {
 99 
100         }
101     }
102 });
