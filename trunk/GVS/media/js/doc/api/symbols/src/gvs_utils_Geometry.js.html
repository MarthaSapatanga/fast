 26 /**
 27  * Notation:
 28  *    Rectangles are represented as objects:
 29  *       rectangle.left
 30  *       rectangle.top
 31  *       rectangle.botttom
 32  *       rectangle.right
 33  */
 34 var Geometry = Class.create();
 35 
 36 // **************** STATIC METHODS **************** //
 37 
 38 Object.extend(Geometry, {
 39 
 40     /**
 41      * Calculate the cumulative offLimit offset and the effective delta for
 42      * an axis given the axis range and the current offLimit offset.
 43      *
 44      * @param Object range
 45      *     range.min
 46      *     range.max
 47      * @param Object delta
 48      * @param Object offLimit
 49      * @type Object result
 50      *     result.delta
 51      *     result.offLimit
 52      */
 53     updateAxis: function(/** Object */ range, /** Number */ delta, /** Number */ offLimit) {
 54 
 55         function _positiveUpdateAxis(/** Number */ max, /** Number */ delta, /** Number */ offLimit) {
 56             var result = new Object();
 57 
 58             if (delta > max) {
 59                 // Stop at the limit
 60                 result.delta = max;
 61                 result.offLimit = offLimit + (delta - max);
 62             } else if (offLimit < 0) {
 63                 // Decreasing offLimit
 64                 if (delta + offLimit > 0) {
 65                     result.delta = delta + offLimit;
 66                     result.offLimit = 0;
 67                 } else {
 68                     result.delta = 0;
 69                     result.offLimit = delta + offLimit;
 70                 }
 71             } else {
 72                 // No fx
 73                 result.delta = delta;
 74                 result.offLimit = offLimit;
 75             }
 76 
 77             return result;
 78         }
 79 
 80         if (delta >= 0) {
 81             return _positiveUpdateAxis(range.max, delta, offLimit);
 82         } else {
 83             var result = _positiveUpdateAxis(-range.min, -delta, -offLimit);
 84             result.delta = -result.delta;
 85             result.offLimit = -result.offLimit;
 86             return result;
 87         }
 88     },
 89 
 90     contains: function(/** Object */ container, /** Object */ element) {
 91         return (element.left >= container.left) &&
 92             (element.top >= container.top) &&
 93             (element.right <= container.right) &&
 94             (element.bottom <= container.bottom);
 95     },
 96 
 97     intersects: function(/** Object */ container, /** Object */ element) {
 98         return !((element.bottom < container.top) ||
 99             (element.left > container.right) ||
100             (element.top > container.bottom) ||
101             (element.right < container.left));
102     },
103 
104     getRectangle: function(/** DOMNode */ node) {
105 
106         var position = Utils.getPosition(node);
107         return {
108             'top': position.top,
109             'left': position.left,
110             'bottom': position.top + node.offsetHeight,
111             'right': position.left + node.offsetWidth
112         };
113     },
114 
115     getClientRectangle: function(/** DOMNode */ node) {
116         var computedStyle = document.defaultView.getComputedStyle(node, null);
117         var topBorder = computedStyle.getPropertyCSSValue('border-top-width').getFloatValue(CSSPrimitiveValue.CSS_PX);
118         var leftBorder = computedStyle.getPropertyCSSValue('border-left-width').getFloatValue(CSSPrimitiveValue.CSS_PX);
119 
120         var position = Utils.getPosition(node);
121         return {
122             'top': position.top + topBorder,
123             'left': position.left + leftBorder,
124             'bottom': position.top + node.clientHeight,
125             'right': position.left + node.clientWidth
126         };
127     },
128 
129     getCenter: function(/** DOMNode */ node) {
130         return {
131             'top': (node.clientHeight - node.clientTop) / 2,
132             'left': (node.clientWidth - node.clientLeft) / 2
133         };
134     },
135 
136     dragRanges: function(/** Object */ container, /** Object */ element) {
137         return {
138             'x': {
139                 'min': Math.min(-element.left, 0),
140                 'max': Math.max((container.right - container.left) - element.right, 0)
141             },
142             'y': {
143                 'min': Math.min(-element.top, 0),
144                 'max': Math.max((container.bottom - container.top) - element.bottom, 0)
145             }
146         };
147     },
148 
149     adaptDropPosition: function(/** Element */ containerElement, /** Element */ node) {
150         var element = this.getRectangle(node);
151         var containerBounds = this.getClientRectangle(containerElement);
152 
153         var width = element.right - element.left;
154         var height = element.bottom - element.top;
155 
156         if (element.right > containerBounds.right)
157             element.left = containerBounds.right - width;
158 
159         if (element.left < containerBounds.left)
160             element.left = containerBounds.left;
161 
162         if (element.bottom > containerBounds.bottom)
163             element.top = containerBounds.bottom - height;
164 
165         if (element.top < containerBounds.top)
166             element.top = containerBounds.top;
167 
168         return {
169             'left': element.left - containerBounds.left,
170             'top': element.top - containerBounds.top
171         };
172 
173     },
174 
175     adaptInitialPosition: function(/** Element */ containerElement, /** Element */ node,
176                                     /** Object */ position) {
177         var containerBounds = {
178             'top': 0,
179             'left': 0,
180             'right': containerElement.offsetWidth,
181             'bottom': containerElement.offsetHeight
182         };
183 
184         var elementBounds = {
185             'top': position.top,
186             'left': position.left,
187             'right': position.left + node.clientWidth,
188             'bottom': position.top + node.clientHeight
189         };
190 
191         var result = {
192             'top': position.top,
193             'left': position.left
194         };
195 
196         if (elementBounds.right > containerBounds.right)
197             result.left = containerBounds.right - node.offsetWidth;
198 
199         if (elementBounds.left < containerBounds.left)
200             result.left = containerBounds.left;
201 
202         if (elementBounds.bottom > containerBounds.bottom)
203             result.top = containerBounds.bottom - node.offsetHeight;
204 
205         if (elementBounds.top < containerBounds.top)
206             result.top = containerBounds.top;
207 
208         return result;
209     }
