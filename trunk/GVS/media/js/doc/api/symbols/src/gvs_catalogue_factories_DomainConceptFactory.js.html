1 var DomainConceptFactory = Class.create(BuildingBlockFactory,
  2     /** @lends DomainConceptFactory.prototype */ {
  3 
  4     /**
  5      * Factory of domain concept building blocks.
  6      * @constructs
  7      * @extends BuildingBlockFactory
  8      */
  9     initialize: function($super) {
 10         $super();
 11         
 12     },
 13 
 14     /**
 15      * Gets the type of building block this factory mades.
 16      * @type String
 17      * @override
 18      */
 19     getBuildingBlockType: function (){
 20         return Constants.BuildingBlock.DOMAIN_CONCEPT;
 21     },
 22 
 23     // **************** PUBLIC METHODS **************** //
 24     /**
 25      * @override
 26      */
 27     getBuildingBlocks: function(/** Array */ tags, /** Function */ callback){
 28         var url = this._createUrl(tags);
 29         PersistenceEngine.sendGet(url,
 30                 {
 31                     'callback': callback
 32                 },
 33                 this._onSuccess, this._onError);
 34     },
 35    
 36     // **************** PUBLIC METHODS **************** //
 37     
 38     _createUrl: function(/** Array */ tags){
 39         if (tags.size() > 0) {
 40             var processedTags = tags.collect(function(tag) {
 41                 return tag.label['en-gb'];
 42             }).join("+");
 43             return URIs.catalogueTagConcepts.replace('<tags>', processedTags);
 44         } else {
 45             return URIs.catalogueAllConcepts;
 46         }
 47     },
 48     
 49     _onSuccess: function (/** XMLHttpRequest */ transport) {
 50         var metadata = transport.responseText.evalJSON();
 51         var result = new Array();
 52         
 53         $A(metadata).each(function(conceptProperties) {
 54             result.push(new PrePostDescription(conceptProperties));
 55         });
 56         this.callback(result);
 57     },
 58     
 59     _onError: function (/**XMLHttpRequest*/ transport, /** Exception */ e) {
 60         Logger.serverError(transport, e);
 61     }
 62     
 63 });
