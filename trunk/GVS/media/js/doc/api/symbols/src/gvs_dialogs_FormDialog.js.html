1 var FormDialog = Class.create( /** @lends FormDialog.prototype */ {

  3     /**
  4      * Dialog class
  5      * This creates a modal dialog with three zones:
  6      *     * headerNode: containing the title
  7      *     * contentNode: containing all the fields of the form
  8      *     * buttonsNode: containing the different buttons: handled by this class
  9      * @constructs
 10      * @param Hash params
 11      * @abstract
 12      */
 13     initialize: function(properties) {
 14         this._dialog = new dijit.Dialog(properties);
 15         
 16         this._headerNode = new Element ('div',{
 17             'class': 'dialogHeader'
 18         });
 19         
 20         this._contentNode = new Element ('div',{
 21             'class': 'dialogContent'
 22         });
 23         this._buttonNode = new Element ('div',{
 24             'class': 'dialogButtonZone' 
 25         });
 26         this._formWidget = null;
 27                
 28         var containerDiv = new Element ('div');      
 29         
 30         containerDiv.insert (this._headerNode);
 31         containerDiv.insert (this._contentNode);
 32         containerDiv.insert (this._buttonNode);
 33         this._dialog.attr ('content', containerDiv);
 34         
 35         this._initialized = false;
 36         dojo.connect(this._dialog,"hide", this._hide.bind(this));
 37         
 38     },
 39 
 40     
 41     // **************** PUBLIC METHODS **************** //  
 42 
 43     /**
 44      * Shows the dialog
 45      * @public
 46      */
 47     show: function() {
 48         if (!this._initialized) {
 49             this._initDialogInterface();
 50             this._initialized = true;
 51         } else {
 52             this._reset();
 53         }
 54         GVSSingleton.getInstance().setEnabled(false);
 55         this._dialog.show();
 56     },

 67     },

 69     _hide: function() {

 71     },

 73     /**
 74      * Gets the form node.
 75      * @type DOMNode

 82      * Gets the form Widget
 83      * @type dijit.form.Form
 84      * @private
 85      */
 86     _getFormWidget: function() {

 90     /**
 91      * This function adds a button with an onclick handler
 92      * @private
 93      */
 94     _addButton: function (/** String */ label, /** Function */ handler){
 95         
 96         var button = new dijit.form.Button({
 97             'label': label,
 98             onClick: handler
 99         });
100         
101         this._buttonNode.insert (button.domNode);
102     },

104     /**
105      * This function sets the header and a subtitle if passed
106      * @private
107      */
108     _setHeader: function (/** String */ title, /** String */ subtitle){
109         
110         var titleNode = new Element("h2").update(title);
111         this._headerNode.insert(titleNode);
112         
113         if (subtitle && subtitle != ""){
114             var subtitleNode = new Element("div", {
115                 "class": "line"
116             }).update(subtitle);
117             this._headerNode.insert(subtitleNode);
118         }
119     },
120 
121     /**
122      * This function set the form content based on a array-like
123      * structure, containing the different elements of the form,
124      * and, optionally, form parameters
125      * @private
126      */
127     _setContent: function (/** Array | DOMNode */ data, /** Hash */ formParams){
128         if (data instanceof Array) {
129             // Form
130             if (formParams){
131                 this._formWidget = new dijit.form.Form(formParams);
132             } else {
133                 this._formWidget = new dijit.form.Form ({
134                     'method': 'post'
135                 });
136             }
137             
138             // Instantiate form elements
139             $A(data).each (function(line){
140                 var lineNode;
141                 var inputNode;
142                 
143                 switch (line.type) {
144                     case 'title':
145                         lineNode = new Element ('h3').update(line.value);
146                         break;
147                         
148                     case 'input':
149                         if (line.regExp || line.required) {
150                             var input = new dijit.form.ValidationTextBox({
151                                             'name' : line.name,
152                                             'value': line.value,
153                                             'regExp': (line.regExp) ? line.regExp : '.*',
154                                             'required': (line.required) ? line.required : false,
155                                             'invalidMessage': (line.message) ? line.message : 'This field cannot be blank'
156                                         });
157                         } else {
158                             var input = new dijit.form.TextBox({
159                                             'name' : line.name,
160                                             'value': line.value
161                                         });
162                         }             
163                         inputNode = input.domNode;
164                         lineNode = this._createLine(line.label, inputNode);
165                         break;
166     
167                     case 'label':
168                         lineNode = new Element('div', {
169                                         'class': 'line'
170                                     }).update(line.value);
171                         break;
172                         
173                     case 'hidden':
174                         lineNode = new Element('input',{
175                                         'type': 'hidden',
176                                         'name': line.name,
177                                         'value': line.value
178                                     });
179                         break;                    
180                                      
181                     case 'comboBox':
182                         inputNode = new Element('select', {
183                             'name': line.name
184                         });
185                         
186                         $A(line.options).each(function(option) {
187                             var optionNode = new Element('option', {
188                                  'value': option.value
189                             }).update(option.label);
190                             
191                             if (option.value == line.value) {
192                                 optionNode.selected = "selected";
193                             }
194                             
195                             inputNode.appendChild(optionNode);
196                         });
197                         
198                         lineNode = this._createLine(line.label, inputNode);
199                         break;
200                         
201                     default:
202                         throw "Unimplemented form field type";
203                 }
204                       
205                 this._formWidget.domNode.appendChild(lineNode);   
206                 
207                 this._armEvents(inputNode, line.events);
208             }.bind(this));
209             
210             this._contentNode.update(this._formWidget.domNode);  
211                         
212         } else {
213             // Data is a DOMNode
214             this._contentNode.update(data);
215         }        
216 
217         //Just in case
218         dojo.parser.parse(this._contentNode);
219     },

222     /**
223      * Construct a form line provided the label text and the input node.
224      * @type DOMNode
225      * @private
226      */
227     _createLine: function(/** String */ label, /** DOMNode */ inputNode) {
228         var lineNode = new Element('div', {
229                         'class' : 'line'
230                     });
231         var labelNode = new Element ('label').update(label);
232         lineNode.appendChild(labelNode);
233         lineNode.appendChild(inputNode);
234         
235         return lineNode;
236     },

239     /**
240      * Attach the event handlers to the input DOM node
241      * @private
242      */
243     _armEvents: function(/** DOMNode */ input, /** Hash */ events) {
244         $H(events).each(function(pair) {
245             Element.observe(input, pair.key, pair.value);
246         });
247     },
248 
249     /**
250      * This method is called for reseting the dialog fields.
251      * Overload when necessary.
252      * @private
253      */
254     _reset: function() {
255         if (this._getFormWidget()) {
256             this._getFormWidget().validate();
257         }
258     }
