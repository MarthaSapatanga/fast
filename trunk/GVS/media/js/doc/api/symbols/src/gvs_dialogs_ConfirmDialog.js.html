1 var ConfirmDialog = Class.create(FormDialog, /** @lends ConfirmDialog.prototype */ {
  2     /**
  3      * This class handles dialogs
  4      * @abstract
  5      * @extends FormDialog
  6      * @constructs
  7      * @param buttons
  8      *     ['ok_cancel' (default) | 'ok']
  9      * @param _options
 10      */
 11     initialize: function($super, /** String */ title, /** String */ buttons, /** Hash */_options) {
 12         $super({
 13             'title': title,
 14             'style': 'display:none;'
 15         }, _options);
 16 
 17         //Initializing buttons
 18         switch(buttons) {
 19             case ConfirmDialog.OK:
 20                 this._okButton = this._addButton ('Ok',     this._onOk.bind(this));
 21                 break;
 22 
 23             case ConfirmDialog.SAVE_DISCARD_CANCEL:
 24                 if (_options && _options.callback) {
 25 
 26                     this._addButton('Save', this._onButtonPressed.bind({
 27                         'mine': this,
 28                         'pressedButton': ConfirmDialog.SAVE,
 29                         'callback': _options.callback
 30                     }));
 31                     this._addButton('Discard changes', this._onButtonPressed.bind({
 32                         'mine': this,
 33                         'pressedButton': ConfirmDialog.DISCARD,
 34                         'callback': _options.callback
 35                     }));
 36                     this._addButton('Cancel', this._onButtonPressed.bind({
 37                         'mine': this,
 38                         'pressedButton': ConfirmDialog.CANCEL,
 39                         'callback': _options.callback
 40                     }));
 41                 } else {
 42                     throw "Cannot create a confirm dialog with that configuration. " +
 43                           "ConfirmDialog::initialize";
 44                 }
 45                 break;
 46             case ConfirmDialog.NONE:
 47                 break;
 48             case ConfirmDialog.CUSTOM:
 49                  if (_options && _options.callback && _options.buttons) {
 50                     $H(_options.buttons).each(function(pair) {
 51                         this._addButton(pair.value, this._onButtonPressed.bind({
 52                             'mine': this,
 53                             'pressedButton': pair.key,
 54                             'callback': _options.callback
 55                         }));
 56                     }.bind(this));
 57                  } else {
 58                     throw "Cannot create a confirm dialog with that configuration. " +
 59                           "ConfirmDialog::initialize";
 60                  }
 61                  break;
 62             case ConfirmDialog.OK_CANCEL:
 63             default:
 64                 this._okButton = this._addButton ('Ok', this._onOk.bind(this));
 65                 this._addButton ('Cancel', this._onCancel.bind(this));
 66                 break;
 67         }
 68     },
 69

 77     /**
 78      * Inits the user interface
 79      * It may be overriden
 80      * @private
 81      */
 82     _initDialogInterface: function() {
 83         if (this._options.contents) {
 84             this._setContent(this._options.contents);
 85         }
 86     },
 87 
 88     /**
 89      * Enables/disables the ok button allowing/disallowing users to continue.
 90      *
 91      * @private
 92      */
 93     _setDisabled: function(disabled) {
 94         this._okButton.setDisabled(disabled);
 95     },
 96 
 97     /**
 98      * onOK
 99      * This function is called when ok button is pressed (if any)
100      * @private
101      */
102     _onOk: function(){
103         this._dialog.hide();
104     },
105 
106     /**
107      * onCancel
108      * This function is called when ok button is pressed (if any)
109      * @private
110      */
111     _onCancel: function(){
112         this._dialog.hide();
113     },
114 
115     /**
116      * On button pressed
117      * @private
118      */
119     _onButtonPressed: function() {
120         this.mine._dialog.hide();
121         this.callback(this.pressedButton);
122     },
123 
124     /**
125      * Reset method to leave the form as initially
126      * @private
127      */
128     _reset: function ($super) {
129         this._setDisabled(false);
130         $super();
131     }
132 });
