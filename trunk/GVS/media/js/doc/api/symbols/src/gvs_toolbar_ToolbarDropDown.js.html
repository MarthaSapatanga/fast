 26 var ToolbarDropDown = Class.create( /** @lends ToolbarDropDown.prototype */ {
 27     /**
 28      * DropDown as a toolbar element.
 29      * Implementing ToolbarElement interface
 30      * @constructs
 31      * @param Boolean enabled (optional)
 32      */
 33     initialize: function(/** String */ label, /** String */ iconName, /** Boolean*/ enabled) {
 34 
 35         /**
 36          * The Menu inside the DropDown
 37          * @type dijit.Menu
 38          * @private
 39          */
 40         this._menu = new dijit.Menu();
 41 
 42         /**
 43          * The DropDown Widget
 44          * @type dijit.form.DropDown
 45          * @private @member
 46          */
 47         this._widget = new dijit.form.DropDownButton ({
 48             'label': label,
 49             'showLabel': true,
 50             'iconClass': 'toolbarIcon ' + iconName + 'Icon',
 51             'dropDown': this._menu,
 52             'disabled': (enabled !== undefined)? (!enabled) : false
 53         });
 54     },
 55 
 56 
 57     // **************** PUBLIC METHODS **************** //
 58     /**
 59      * This function adds a new menu item into the DropDown
 60      */
 61     addMenuItem: function(/** String */ label, /** Function */ onClick, /** String */ iconName) {
 62         var item = new dijit.MenuItem({
 63             'label': label,
 64             'showLabel': true,
 65             'onClick': onClick
 66         });
 67         if (iconName) {
 68             item.attr('iconClass', 'dropDownIcon ' + iconName + 'Icon');
 69         }
 70         this._menu.addChild(item);
 71     },
 72 
 73     /**
 74      * Returns the DropDown widget
 75      * @type dijit.form.DropDown
 76      */
 77     getWidget: function () {
 78         return this._widget;
 79     },
 80 
 81     /**
 82      * Set the status of the widget
 83      */
 84     setEnabled: function(/** Boolean */ enabled) {
 85         this._widget.attr('disabled',!enabled);
 86     }
 87 
 88 
 89 });
