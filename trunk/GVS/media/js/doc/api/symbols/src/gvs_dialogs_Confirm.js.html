1 /**
  2  * <p>This class implements the Singleton Design Pattern to make sure there is 
  3  * only one instance of the class Preferences.
  4  *
  5  * <p> It should be accessed as follows.
  6  *
  7  * @constructor
  8  * @example
  9  * var confirm = ConfirmSingleton.getInstance();
 10  */ 
 11 var ConfirmSingleton = function() {
 12 
 13     /**
 14      * Singleton instance
 15      * @private @member
 16      */
 17     var _instance = null;
 18     
 19 
 20     var Confirm = Class.create(ConfirmDialog,
 21         /** @lends ConfirmSingleton-Confirm.prototype */ {
 22 
 23         /** 
 24          * Confirm dialog
 25          * @constructs
 26          * @extends ConfirmDialog
 27          */
 28         initialize: function($super) {
 29             $super('Warning');
 30             
 31             /**
 32              * Callback function to be called
 33              * @type Function
 34              * @private @member
 35              */
 36             this._callback = null;
 37             
 38             this._contentNode.addClassName("systemDialog"); 
 39         },
 40         
 41         /**
 42          * This function shows a message
 43          */
 44         show: function  ($super, /** String */ message, /**Function*/ callback){
 45             this._contentNode.update(message);
 46             $super();
 47             this._callback = callback;
 48         },
 49         // *********************** PRIVATE METHODS ******************//
 50         /**
 51          * @override
 52          * @private
 53          */
 54         _onOk: function ($super) {
 55             this._callback(true);
 56             this._callback = null;
 57             $super();
 58         },
 59         /**
 60          * @override
 61          * @private
 62          */        
 63         _onCancel: function ($super){
 64             this._callback(false);
 65             this._callback = null;
 66             $super();
 67         },
 68         /**
 69          * @override
 70          * @private
 71          */
 72         _initDialogInterface: function () {
 73             // Do Nothing
 74         },
 75         /**
 76          * @private
 77          * @override
 78          */
 79         _hide: function($super) {
 80             $super();
 81             if (this._callback) {
 82                 this._callback(false);
 83                 this._callback = null;
 84             }        
 85         }
 86     });
 87 
 88     return new function(){
 89         /**
 90          * Returns the singleton instance
 91          * @type Preferences
 92          */
 93         this.getInstance = function(){
 94             if (_instance == null) {
 95                 _instance = new Confirm();
 96             }
 97             return _instance;
 98         }
 99     }
100 }();
101 
102 // Browser confirm dialog override
103 if (document.getElementById) {
104     //Note that confirm is not blocking anymore
105     //so a callback function is needed
106     var browserConfirm = window.confirm;             
107     window.confirm = function(msg,callback) {
108         if (callback) {
109             ConfirmSingleton.getInstance().show(msg,callback);
110         } else{ //In case you don't use the modified version
111             browserConfirm(msg);
112         }
113     }    
114 }
