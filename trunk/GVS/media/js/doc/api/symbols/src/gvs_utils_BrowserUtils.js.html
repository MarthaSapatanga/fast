1 var BrowserUtils = Class.create();
  2 
  3 // **************** STATIC METHODS **************** //
  4 
  5 Object.extend(BrowserUtils, {
  6 
  7     /**
  8      * Gets browser window height
  9      * @type Integer
 10      */
 11     getHeight : function () {
 12         var newHeight=window.innerHeight; //Non-IE (Firefox and Opera)
 13           
 14         if (document.documentElement &&
 15                 document.documentElement.clientHeight) {
 16           //IE 6+ in 'standards compliant mode'
 17           newHeight = document.documentElement.clientHeight;
 18 
 19         } else if( document.body && document.body.clientHeight ) {
 20           //IE 4 compatible and IE 5-7 'quirk mode'
 21           newHeight = document.body.clientHeight;
 22         }
 23 
 24         return newHeight;
 25     },
 26     
 27 
 28     /**
 29      * Gets browser window width
 30      * @type Integer
 31      */
 32     getWidth : function(){  
 33         var newWidth=window.innerWidth; //Non-IE (Firefox and Opera)
 34       
 35         if (document.documentElement &&
 36                 (document.documentElement.clientWidth ||
 37                  document.documentElement.clientHeight)) {
 38             //IE 6+ in 'standards compliant mode'
 39             newWidth = document.documentElement.clientWidth;
 40         } else if (document.body && 
 41                 (document.body.clientWidth || document.body.clientHeight)) {
 42             //IE 4 compatible and IE 5-7 'quirk mode'
 43             newWidth = document.body.clientWidth;
 44         }
 45 
 46         return newWidth;
 47     },
 48         
 49 
 50     /**
 51      * Determines if a button code is the left one
 52      * (right button for left-handed people).
 53      * @type Boolean
 54      */
 55     isLeftButton : function(button) {
 56         
 57         if (button == 0 || (this.isIE() && button == 1)) {
 58             return true;
 59         } else {
 60             return false;   
 61         }
 62     },
 63 
 64     
 65     /**
 66      * Is the browser Internet Explorer?
 67      * @type Boolean
 68      */
 69     isIE : function(){ 
 70         if (/MSIE (\d+\.\d+);/.test(navigator.userAgent)) { //test for MSIE x.x;  
 71             return true;
 72         } else {
 73             return false;       
 74         }
 75     }
 76 });
