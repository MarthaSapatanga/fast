1 /**
  2  * <p>This class implements the Singleton Design Pattern to make sure there is 
  3  * only one instance of the class BrowserUtilsSingleton.
  4  *
  5  * <p> It should be accessed as follows.
  6  *
  7  * @constructor
  8  * @example
  9  * var gvs = BrowserUtilsSingleton.getInstance();
 10  */ 
 11 var BrowserUtilsSingleton = function () {
 12 
 13     /**
 14      * Singleton instance
 15      * @private @member
 16      */
 17     var _instance = null;
 18 
 19 
 20     var BrowserUtils = Class.create(
 21         /** @lends BrowserUtilsSingleton-BrowserUtils.prototype */ {
 22         
 23         /**
 24          * This class provides some utility functions in a browser independent
 25          * fashion.
 26          * @constructs
 27          */
 28         initalize: function (){},
 29         
 30 
 31         // **************** PUBLIC METHODS **************** //
 32 
 33 
 34         /**
 35          * Gets browser window height
 36          * @type Integer
 37          */
 38         getHeight : function () {
 39             var newHeight=window.innerHeight; //Non-IE (Firefox and Opera)
 40               
 41             if (document.documentElement &&
 42                     document.documentElement.clientHeight) {
 43               //IE 6+ in 'standards compliant mode'
 44               newHeight = document.documentElement.clientHeight;
 45 
 46             } else if( document.body && document.body.clientHeight ) {
 47               //IE 4 compatible and IE 5-7 'quirk mode'
 48               newHeight = document.body.clientHeight;
 49             }
 50 
 51             return newHeight;
 52         },
 53         
 54 
 55         /**
 56          * Gets browser window width
 57          * @type Integer
 58          */
 59         getWidth : function(){  
 60             var newWidth=window.innerWidth; //Non-IE (Firefox and Opera)
 61           
 62             if (document.documentElement &&
 63                     (document.documentElement.clientWidth ||
 64                      document.documentElement.clientHeight)) {
 65                 //IE 6+ in 'standards compliant mode'
 66                 newWidth = document.documentElement.clientWidth;
 67             } else if (document.body && 
 68                     (document.body.clientWidth || document.body.clientHeight)) {
 69                 //IE 4 compatible and IE 5-7 'quirk mode'
 70                 newWidth = document.body.clientWidth;
 71             }
 72 
 73             return newWidth;
 74         },
 75             
 76 
 77         /**
 78          * Determines if a button code is the left one
 79          * (right button for left-handed people).
 80          * @type Boolean
 81          */
 82         isLeftButton : function(button) {
 83             
 84             if (button == 0 || (this.isIE() && button == 1)) {
 85                 return true;
 86             } else {
 87                 return false;   
 88             }
 89         },
 90     
 91         
 92         /**
 93          * Is the browser Internet Explorer?
 94          * @type Boolean
 95          */
 96         isIE : function(){ 
 97             if (/MSIE (\d+\.\d+);/.test(navigator.userAgent)) { //test for MSIE x.x;  
 98                 return true;
 99             } else {
100                 return false;       
101             }
102         }
103     });
104 
105     return new function() {
106         /**
107          * Returns the singleton instance
108          * @type BrowserUtils
109          */
110         this.getInstance = function() {
111             if (_instance == null) {
112                 _instance = new BrowserUtils();
113             }
114             return _instance;
115         }
116     }
117     
118 }();
