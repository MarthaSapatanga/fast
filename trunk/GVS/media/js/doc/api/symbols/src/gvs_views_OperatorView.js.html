106     /**
107      * This function returns the position of a PRE/POST fact
108      *
109      */
110     _getPosition: function(/** Boolean */ isPre, /** Integer */ order, /** Integer */ size, /** Integer */ orientation) {
111     	var width = 74;
112         var height = 60;
113     	var x = width * (order+1) / (size+1);
114     	if ((order+1) != (size + 1) / 2) { // No middle
115     		x += (order < size/2)? -4 : 3;
116     	}
117 
118     	if(isPre && orientation == 0 || !isPre && orientation == 1){
119 	    	if (x > width/2) { // Choose side
120 				var y = 3*height/2 - x *(height/2)/(width/2);
121 			} else {
122 				var y = x*(height/2)/(width/2) + height/2;
123 			}
124     	} else {
125 	    	if(x > width/2) { // Choose side
126 				var y = x*(height/2)/(width/2) - height/2 - 4;
127 			} else {
128 				var y = height/2 - x *(height/2)/(width/2) - 4;
129 			}
130     	}
131         return {'x': x, 'y': y};
132     },

171     	var size = posts.length;
172     	for(var i=0; i < size; i++){
173     		var factNode = posts[i].getNode();
174     		var position = this._getPosition(false, i, size, orientation);
175     		factNode.setStyle({
176                 'top': Math.round(position.y) + 'px',
177                 'left': Math.round(position.x) + 'px'
178             });
179     	}
