 26 var ScreenflowDescription = Class.create(BuildingBlockDescription,
 27     /** @lends ScreenflowDescription.prototype */ {
 28 
 29     /**
 30      * Screenflow building block description.
 31      * @constructs
 32      * @extends BuildingBlockDescription
 33      */
 34     initialize: function($super, /** Hash */ properties) {
 35         $super(properties);
 36 
 37 
 38         this._screens = new Hash();
 39         this._preconditions = new Hash();
 40         this._postconditions = new Hash();
 41     },
 42 
 43     /**
 44      * to JSON object function
 45      * @type Object
 46      */
 47     toJSON: function() {
 48         var result = {
 49             "definition": {
 50                 "screens": this._getScreens(),
 51                 "preconditions": this.getPreconditions(),
 52                 "postconditions": this.getPostconditions()
 53             }
 54         };
 55         result = Object.extend(result,{
 56             "name": this.name,
 57             "label": this.label,
 58             "tags": this.tags,
 59             "version": this.version,
 60             "id": this.id,
 61             "creator": this.creator,
 62             "description": this.description,
 63             "rights": this.rights,
 64             "creationDate": this.creationDate,
 65             "icon": this.icon,
 66             "screenshot": this.screenshot,
 67             "homepage": this.homepage,
 68             "type": "screenflow"
 69         });
 70         return result;
 71     },
 72 
 73     /**
 74      * Adds a new screen.
 75      * @param ScreenDescription
 76      *      Screen to be added to the
 77      *      Screenflow document.
 78      */
 79     addScreen: function (/** ScreenInstance */ instance, /** Object */ position) {
 80         this._screens.set(instance.getUri(), {
 81             "buildingblock":   instance,
 82             "position": position
 83         });
 84     },
 85 
 86     /**
 87      * Updates the position of the screen
 88      */
 89     updateScreen: function (/** String */ uri, /** Object */ position) {
 90         var buildingBlock = this._screens.get(uri);
 91         var origin = buildingBlock.position;
 92 
 93         if (origin.top != position.top || origin.left != position.left) {
 94             buildingBlock.position = position;
 95             return true;
 96         } else {
 97             return false;
 98         }
 99     },
100 
101     /**
102      * Delete a screen.
103      * @param ScreenDescription
104      *      Screen to be deleted from the
105      *      Screenflow document.
106      */
107     remove: function(/** String */ uri) {
108         this._screens.unset(uri);
109         this._preconditions.unset(uri);
110         this._postconditions.unset(uri);
111     },
112     /**
113      * Adds a new *-condition to the screenflow description
114      */
115     addPrePost: function(/** PrePostInstance */ instance, /** Object */ position) {
116         switch(instance.getType()) {
117             case 'pre':
118                 this._preconditions.set(instance.getUri(), {'buildingblock': instance,
119                                                             'position': position});
120                 break;
121             case 'post':
122                 this._postconditions.set(instance.getUri(), {'buildingblock': instance,
123                                                             'position': position});
124                 break;
125             default:
126                 //Do nothing
127         }
128     },
129 
130     updatePrePost: function(/** String */ uri, /** Object */ position) {
131         var condition = this._preconditions.get(uri);
132         if (!condition) {
133             condition = this._postconditions.get(uri);
134         }
135         var origin = condition.position;
136         if (origin.top != position.top || origin.left != position.left) {
137             condition.position = position;
138             return true;
139         }
140         return false;
141     },
142 
143     getPre: function(/** String */ uri)  {
144         var pre = this._preconditions.get(uri);
145         if (pre) {
146             return pre.buildingblock;
147         } else {
148             return null;
149         }
150     },
151 
152     getPost: function(/** String */ uri)  {
153         var post = this._postconditions.get(uri);
154         if (post) {
155             return post.buildingblock;
156         } else {
157             return null;
158         }
159     },
160 
161 
162     /**
163      * Implementing the TableModel interface
164      * @type Array
165      */
166     getInfo: function() {
167         var info = new Hash();
168         info.set('Title', this.getTitle());
169         info.set('Tags', this.tags.collect(function(tag) {
170                 return tag.label['en-gb'];
171             }).join(", "));
172         info.set('Version', this.version);
173         return info;
174     },
175 
176     /**
177      * Get a list of preconditions in form of a JSON Object
178      * @type Array
179      */
180     getPreconditions: function() {
181         var list = new Array();
182         this._preconditions.values().each(function(pre) {
183             var element = Object.extend(pre.buildingblock.toJSONForScreenflow(), {'position':
184                 pre.position});
185             list.push(element);
186         }.bind(this));
187         return list;
188     },
189 
190     /**
191      * Get a list of postconditions in form of a JSON Object
192      * @type Array
193      */
194     getPostconditions: function() {
195         var list = new Array();
196         this._postconditions.values().each(function(post) {
197             var element = Object.extend(post.buildingblock.toJSONForScreenflow(), {'position':
198                 post.position});
199             list.push(element);
200         }.bind(this));
201         return list;
202     },
203 
204     /**
205      * Get a list of preconditions in form of a JSON Object
206      * @type Array
207      */
208     getPreconditionsCatalogue: function() {
209         var list = new Array();
210         this._preconditions.values().each(function(pre) {
211             var element = pre.buildingblock.toJSONForCatalogue();
212             list.push(element);
213         }.bind(this));
214         return list;
215     },
216 
217     /**
218      * Get a list of postconditions in form of a JSON Object
219      * @type Array
220      */
221     getPostconditionsCatalogue: function() {
222         var list = new Array();
223         this._postconditions.values().each(function(post) {
224             var element = post.buildingblock.toJSONForCatalogue();
225             list.push(element);
226         }.bind(this));
227         return list;
228     },
229 
230     /**
231      * Get a list of canvas instances
232      * @type Array
233      */
234     getCanvasInstances: function() {
235         var result = new Array();
236         var list = this._screens.values();
237         list.each(function(instance){
238             result.push(instance.buildingblock);
239         });
240         return result;
241     },
242 
243     /**
244      * Returns true if an element with the parameter uri is in the screen
245      * @type Boolean
246      */
247     contains: function(/** String */ uri) {
248         var list = this._screens.values().concat(this._preconditions.values());
249         list = list.concat(this._postconditions.values());
250         var element = list.detect(function(instance) {
251                     return instance.buildingblock.getOriginalUri() == uri;
252         });
253         if (element) {
254             return true;
255         } else {
256             return false;
257         }
258     },
259 
260     /**
261      * @type Array
262      */
263     getConditionInstances: function() {
264         var result = new Array();
265         this._preconditions.values().each(function(pre){
266             result.push(pre.buildingblock);
267         });
268         this._postconditions.values().each(function(post){
269             result.push(post.buildingblock);
270         });
271         return result;
272     },
273 
274     //************************ PRIVATE METHODS *******************//
275     /**
276      * @type Array
277      * @private
278      */
279     _getScreenUris: function() {
280         return this._screens.keys();
281     },
282 
283     /**
284      * Return the list of screens
285      * @type Array
286      * @private
287      */
288     _getScreens: function() {
289         var result = new Array();
290         this._screens.values().each(function(screen){
291             var data = {
292                 'uri': screen.buildingblock.getUri(),
293                 'position': screen.position,
294                 'title': screen.buildingblock.getTitle(),
295                 'originalUri': screen.buildingblock.getOriginalUri()
296             };
297             var caption = screen.buildingblock.getCaption();
298             if (caption != null && caption != "") {
299                 data["caption"] = caption;
300             }
301             result.push(data);
302         });
303         return result;
304     }
305 });
