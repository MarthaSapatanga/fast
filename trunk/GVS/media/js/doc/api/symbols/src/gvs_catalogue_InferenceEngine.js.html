1 var InferenceEngine = Class.create( /** @lends InferenceEngine.prototype */ {
  2     /**
  3      * This class handles the reachability and recommendation of building blocks
  4      * It communicates with the serverside catalogue to retrieve this information

  7     initialize: function(/** ScreenflowDocument */parent) {
  8         
  9         /**
 10          * Parent document
 11          * @type ScreenflowDocument
 12          * @private @member

 14         this._parent = parent;
 15     },
 16     
 17 
 18     // **************** PUBLIC METHODS **************** //
 19 
 20     /**
 21      * 
 22      * This function fills the initial palette based on a domain context
 23      * 
 24      */
 25     retrieveScreens: function (/** Array */ domainContext) {
 26         this.findAndCheck ([],domainContext,[],"reachability");
 27     },
 28     
 29     /**
 30      * findAndCheck
 31      * This function calls findAndCheck in the catalogue
 32      */
 33     findAndCheck: function (/**Array*/ canvas, /** Array */ domainContext, 
 34                             /** Array*/ elements,/** String*/ criteria) {
 35         
 36         /** 
 37          * onSuccess callback
 38          */
 39         var findAndCheckOnSuccess = function(/**XMLHttpRequest*/ transport){
 40             
 41             var result = JSON.parse(transport.responseText);
 42             
 43             var paletteElements = result.elements;         
 44             var canvasElements = result.canvas;          
 45             
 46             // There are elements in the canvas
 47             // Update the screenflow
 48             if (canvasElements != null && canvasElements != []){
 49                 this._parent.updateReachability(canvasElements);
 50             }
 51             
 52             var screenFactory = CatalogueSingleton.getInstance().getBuildingBlockFactory (Constants.BuildingBlock.SCREEN);
 53             screenFactory.updateElements (domainContext,paletteElements,updateElementsCallback);
 54         }
 55         /**
 56          * Error handler
 57          */
 58         var findAndCheckOnError = function(transport, e){
 59             console.log ("Error retrieving catalogue information");
 60         }
 61         /**
 62          * This function is called when the metadata is stored in the BBFactory
 63          */
 64         var mine = this;
 65         var updateElementsCallback = function (/** Array */ screens) {
 66             //Paint the palette and updateReachability
 67             var paletteController = mine._parent.getPaletteController();
 68             paletteController.paintPalettes();
 69             paletteController.updateReachability(screens);
 70         }
 71         
 72         //construct the data to be sent
 73         var domain = {
 74             'tags': domainContext,
 75             'user':null /* TODO: add user here */
 76         };
 77         var body = {
 78             'canvas': canvas,
 79             'domainContext': domain,
 80             'elements': elements,
 81             'criterion': criteria
 82         };
 83         body = Object.toJSON(body);
 84 
 85         var persistenceEngine = PersistenceEngineFactory.getInstance();
 86         persistenceEngine.sendPost(URIs.catalogueFindAndCheck, null, body, this, findAndCheckOnSuccess, findAndCheckOnError);
 87     },
 88     /**
 89      * This function calls the check operation in the catalogue
 90      */
 91     check: function (/**Array*/ canvas, /** Array */ domainContext, 
 92                     /** Array*/ elements,/** String*/ criteria) {
 93         
 94         /**
 95          * onSuccess callbcak
 96          */
 97         var checkOnSuccess = function(response){
 98                 var screenList = JSON.parse(response.responseText);
 99                 this._parent.getPaletteController().updateReachability(screenList.elements);
100                 this._parent.updateReachability(screenList.canvas);
101             }
102             
103         var checkOnError = function(transport, e){
104             console.log ("Error performing check operation");
105         }
106         //construct the data to be sent
107         var domain = {
108             'tags': domainContext,
109             'user':null /* TODO: add user here */
110         }
111         var body = {
112             'canvas': canvas,
113             'domainContext': domain,
114             'elements': elements,
115             'criterion': criteria
116         };
117         body = Object.toJSON(body);
118         var persistenceEngine = PersistenceEngineFactory.getInstance();
119         persistenceEngine.sendPost(URIs.catalogueCheck, null, body, this, checkOnSuccess, checkOnError);                   
120     }
