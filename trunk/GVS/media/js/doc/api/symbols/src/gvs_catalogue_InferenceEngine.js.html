1 var InferenceEngine = Class.create( /** @lends InferenceEngine.prototype */ {
  2     /**
  3      * This class handles the reachability and recommendation of building blocks
  4      * It communicates with the serverside catalogue to retrieve this information
  5      * @constructs
  6      */ 
  7     initialize: function() {
  8         
  9         /**
 10          * This stores the reachability data
 11          * @type Hash
 12          * @private @member
 13          */
 14         this._reachabilityData = new Hash();
 15         
 16         /**
 17          * Listeners list hashed by resource URI
 18          * @type Hash
 19          * @private @member
 20          */
 21         this._listeners = new Hash();
 22     },
 23     
 24 
 25     // **************** PUBLIC METHODS **************** //
 26 
 27     
 28     /**
 29      * This function calls findAndCheck in the catalogue and calls a
 30      * callback upon completion.
 31      */
 32     findCheck: function (/**Array*/ canvas, /** Array*/ elements,
 33                          /** Array */ domainContext, 
 34                          /** String*/ criteria,
 35                          /** Function */ callback) {
 36         
 37         var body = this._constructBody(canvas, elements, domainContext, criteria);
 38         var persistenceEngine = PersistenceEngineFactory.getInstance();
 39         var context = {
 40             'callback': callback,
 41             'mine': this
 42         };
 43         persistenceEngine.sendPost(URIs.catalogueFindAndCheck, null, body,
 44                 context, this._findCheckOnSuccess, this._onError);
 45     },
 46     
 47     /**
 48      * This function calls the check operation in the catalogue
 49      */
 50     check: function (/**Array*/ canvas, /** Array*/ elements,
 51                     /** Array */ domainContext, 
 52                     /** String*/ criteria) {
 53         var body = this._constructBody(canvas, elements, domainContext, criteria);
 54         var persistenceEngine = PersistenceEngineFactory.getInstance();
 55         persistenceEngine.sendPost(URIs.catalogueCheck, null, body, this, 
 56                                     this._checkOnSuccess, this._onError);                   
 57     },
 58     
 59     /**
 60      * Register an object for interest on the reachability of a URI-identified
 61      * resources. The listener object must implement these methods:
 62      *    
 63      *    void setReachability(Hash reachabilityData);
 64      *    
 65      * Reachability data vary for different resource types:
 66      *   * Screens: TODO
 67      *   * ...
 68      */
 69     addReachabilityListener: function(/** String */ uri, /** Object */ listener) {
 70         this._getListenerList(uri).push(listener);
 71         
 72         if (this._reachabilityData.get(uri)) {
 73             listener.setReachability(this._reachabilityData.get(uri));
 74         }
 75     },
 76     /**
 77      * De-register an object from the reachability listeners
 78      */
 79     removeReachabilityListener: function (/** String  */uri, /** Object */ listener) {
 80         var index = this._getListenerList(uri).indexOf(listener);
 81         
 82         if (index >= 0) {
 83             //Remove the element
 84             this._listeners.get(uri)[index] = null;
 85             this._listeners.set(uri,this._listeners.get(uri).compact());
 86         }
 87     },
 88     
 89     /**
 90      * Returns a boolean determining if a building block is reachable
 91      * by its uri
 92      * @type Boolean
 93      */
 94      isReachable: function (/** String */ uri) {
 95          var reachabilityData = this._reachabilityData.get(uri);
 96          if (reachabilityData) {
 97             return reachabilityData.reachability;
 98          }
 99      },
100      
101      /**
102      * Returns the reachability information about
103      * the preconditions of a given screen
104      * @type Hash
105      */
106     getPreconditionReachability: function(/** String */ uri) {
107         var reachabilityData = this._reachabilityData.get(uri);
108         if (reachabilityData.preconditions && reachabilityData.preconditions.length > 1) {
109             //More than one set of preconditions
110             console.log("OR precondition support not implemented yet");
111             return null;
112         }
113         else {
114             var preconditions = reachabilityData.preconditions[0];        
115             var result = new Hash();
116             $A(preconditions).each(function(pre) {
117                 var uri = Utils.extractURIfromPattern(pre.pattern);
118                 result.set(uri, pre.satisfied);    
119             });
120             return result;
121         }
122     },
123     
124     // **************** PRIVATE METHODS **************** //
125     /** 
126      * onSuccess callback
127      */
128     _findCheckOnSuccess: function(/**XMLHttpRequest*/ transport){
129         var result = JSON.parse(transport.responseText);
130         var paletteElements = result.elements;
131         
132         this.mine._updateReachability(paletteElements);
133         
134         // Notifying about new uris
135         var screenURIs = new Array();
136         $A(paletteElements).each(function(element) {
137            screenURIs.push(element.uri); 
138         }); 
139 
140         this.callback(screenURIs);        
141     },
142 
143     /**
144      * onSuccess callback
145      * @private
146      */
147     _checkOnSuccess: function(transport){
148         var result = JSON.parse(transport.responseText);
149         var elements = result.elements.concat(result.canvas).uniq();
150         
151         this._updateReachability(elements);
152     },
153     
154     /**
155      * Error handler
156      * @private
157      */
158     _onError: function(transport, e){
159         Logger.serverError(transport,e);
160     },
161     /**
162      * Returns a list of listeners of an uri
163      * @private
164      */
165     _getListenerList: function (/** String */ uri) {
166         var list = this._listeners.get(uri);
167         if (!list) {
168             list = new Array();
169             this._listeners.set(uri, list);
170         }
171         return list;
172     },
173     /**
174      * Updates and notifies the reachability 
175      * of a list of elements
176      * @private
177      */
178     _updateReachability: function(/** Array */ elements) {
179         elements.each(function(element) {
180             this._reachabilityData.set(element.uri, element);
181             this._notifyReachability(element.uri);
182         }.bind(this));  
183     },
184     
185     /**
186      * Notifies the reachability information to all the relevant
187      * listeners
188      * @private 
189      */
190     _notifyReachability: function(/** String */ uri) {
191         this._getListenerList(uri).each(function(listener) {
192             listener.setReachability(this._reachabilityData.get(uri));
193         }.bind(this));        
194     },
195     
196     /**
197      * Creates a body to be sent in an AJAX call to the 
198      * catalogue
199      * @private
200      * @type String
201      */
202     _constructBody: function(/**Array*/ canvas, /** Array*/ elements,
203                     /** Array */ domainContext, 
204                     /** String*/ criteria) {
205         var domain = {
206             'tags': domainContext,
207             'user': null /* TODO: add user here */
208         };
209         var body = {
210             'canvas': canvas,
211             'elements': elements,
212             'domainContext': domain,
213             'criterion': criteria
214         };
215         return Object.toJSON(body);
216     }   
217 });
