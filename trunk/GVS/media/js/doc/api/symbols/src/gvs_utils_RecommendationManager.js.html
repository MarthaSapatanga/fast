 26 var RecommendationManager = Class.create(/** @lends RecommendationManager.prototype */ {
 27 
 28 	/**
 29      * Recommendation Manager.
 30      * @constructs
 31      */
 32     initialize: function() {
 33         /**
 34          * A list of current recommendations
 35          * @type Hash
 36          * @private
 37          */
 38         this._recommendations = new Hash();
 39 
 40         /**
 41          *
 42          */
 43         this._activeRecommendations = new Hash();
 44 
 45         /**
 46          * Current recommendation id
 47          * @type Number
 48          * @private
 49          */
 50         this._currentRecommendation = 1;
 51 
 52         /**
 53          * Current animation step
 54          * @type Number
 55          * @private
 56          */
 57         this._step = 0;
 58 
 59         /**
 60          * Timestamp when the animation started
 61          *
 62          * @type Number
 63          * @private
 64          */
 65         this._startTimespam = 0;
 66 
 67         /**
 68          * setTimeout handler
 69          *
 70          * @type Number
 71          * @private
 72          */
 73         this._timeout = null;
 74     },
 75 
 76     // **************** PUBLIC METHODS **************** //
 77 
 78 	/**
 79 	 * Sets the recommendations to manage
 80 	 *
 81 	setRecommendations: function(recommendations) {
 82     	this.clear();
 83 
 84     	this._recommendations = recommendations;
 85     	this._filterRecommendations();
 86 
 87     	var timeout = this._startFact ? null : 3000;
 88     	this._startAnimation(timeout);
 89     },*/
 90 
 91     /**
 92      * Adds a single recommendation
 93      */
 94     addRecommendation: function(localNode, externalNode) {
 95 		if (this._recommendations.get(localNode.key) == undefined) {
 96 			this._recommendations.set(localNode.key, {
 97 					'className': 'recommendation' + this._currentRecommendation++,
 98 					'localNode': localNode.node,
 99 					'externalNodes': [],
100 					'nodes': [localNode.node]});
101 		}
102 
103     	var recommendation = this._recommendations.get(localNode.key);
104 		recommendation.externalNodes.push(externalNode.node);
105 		recommendation.nodes.push(externalNode.node);
106 
107 		this._dirty = true;
108     },
109 
110 	/**
111 	 * Sets the fact that will be used as source in the connection
112 	 */
113     setStartFact: function(fact) {
114     	this._clear();
115     	this._stopAnimation();
116     	this._startFact = fact;
117 
118     	this._dirty = true;
119     	if (this._startFact)
120     		this.startAnimation();
121     },
122 
123     /**
124      * Starts recommendation animation
125      */
126     startAnimation: function() {
127     	var timeout = this._startFact ? null : 3000;
128 
129     	this._clear();
130     	this._stopAnimation();
131     	if (this._dirty)
132     		this._filterRecommendations();
133 
134     	this._startAnimation(timeout);
135     },
136 
137 	/**
138 	 * Clears all recommendations
139 	 */
140 	clear: function() {
141     	this._clear();
142     	this._stopAnimation();
143     	this._recommendations = new Hash();
144     	this._activeRecommendations = new Hash();
145     	this._currentRecommendation = 1;
146     	this._dirty = false;
147     },
148 
149     // **************** PRIVATE METHODS **************** //
150 
151     /**
152      * Updates this._activeRecommedations using the current contents of
153      * this._recommendations and this._startFact.
154      *
155      * @private
156      */
157     _filterRecommendations: function() {
158     	if (this._startFact) {
159     		this._activeRecommendations = new Hash();
160     		var filteredRecommendations = this._recommendations.get(this._startFact);
161     		if (filteredRecommendations) {
162     			this._activeRecommendations.set(this._startFact, filteredRecommendations);
163     		}
164     	} else {
165     		this._activeRecommendations = this._recommendations;
166     	}
167     	this._dirty = false;
168     },
169 
170     /**
171      * Removes all node style included for recommendations.
172      *
173      * @private
174      */
175     _clear: function() {
176     	this._recommendations.each(function(recommendation) {
177     		var nodes = recommendation.value.nodes;
178     		var className = recommendation.value.className;
179 
180     		nodes.each(function(node) {
181     			node.removeClassName(className);
182     			node.getElementsByClassName('recommendationLayer')[0].style.opacity = "";
183     		}.bind(this));
184     	}.bind(this));
185     },
186 
187 	/**
188 	 * Stops recommendation animation
189 	 *
190 	 * @private
191 	 */
192     _stopAnimation: function() {
193     	try {
194     		this._startTimestamp = 0;
195     		clearTimeout(this._timeout);
196     	} catch (e) {}
197     },
198 
199 	/**
200 	 * Starts recommendation animation
201 	 *
202 	 * @private
203 	 */
204     _startAnimation: function(duration) {
205     	this._activeRecommendations.each(function(recommendation) {
206     		var nodes;
207 
208     		if (this._startFact)
209     			nodes = recommendation.value.externalNodes;
210     		else
211     			nodes = recommendation.value.nodes;
212 
213     		var className = recommendation.value.className;
214     		nodes.each(function(node) {
215     			node.getElementsByClassName('recommendationLayer')[0].style.opacity = 0;
216     			node.addClassName(className);
217     		}.bind(this));
218     	}.bind(this));
219 
220     	this._step = 0;
221     	this._duration = duration;
222     	var now = new Date();
223     	this._startTimestamp = now.getTime();
224         this._timeout = setTimeout(this._timeoutCallback.bind(this), 100);
225     },
226 
227     /**
228      * @private
229      */
230     _timeoutCallback: function() {
231         var end = false;
232     	var opacity = Math.sin(Math.PI * this._step / 10);
233     	opacity *= opacity;
234 
235     	if (opacity < 0.10) {
236     		var now = new Date();
237     		if (this._duration && ((now.getTime() - this._startTimestamp) > this._duration)) {
238     			end = true;
239     			opacity = 0;
240     		}
241     	}
242 
243     	this._activeRecommendations.each(function(recommendation) {
244     		var nodes = recommendation.value.nodes;
245     		var className = recommendation.value.className;
246     		nodes.each(function(node) {
247     			node.getElementsByClassName('recommendationLayer')[0].style.opacity = opacity;
248     		}.bind(this));
249     	}.bind(this));
250 
251     	if (!end) {
252     		this._step++;
253     		this._timeout = setTimeout(this._timeoutCallback.bind(this), 100);
254     	}
255     }
256 
257 });
