 26 var ParamsDialog = Class.create(ConfirmDialog /** @lends ParamsDialog.prototype */, {
 27     /**
 28      * This class handles the dialog
 29      * to assign triggers to actions
 30      * @constructs
 31      * @extends ConfirmDialog
 32      */
 33     initialize: function($super, /** String */ buildingblockName,
 34                             /** String */ initialParameter,
 35                             /** String */ templateParameter,
 36                             /** Function */ onChangeCallback) {
 37 
 38         $super(buildingblockName + " Parameters");
 39 
 40         /**
 41          * Building block name
 42          * @type String
 43          * @private
 44          */
 45         this._buildingblockName = buildingblockName;
 46 
 47         /**
 48          * Params initially assigned to the buildingblock
 49          * @type String
 50          * @private
 51          */
 52         this._initialParameter = initialParameter;
 53 
 54         /**
 55          * Template parameters
 56          * @type String
 57          * @private
 58          */
 59         this._templateParameter = templateParameter;
 60 
 61         /**
 62          * Textarea to edit the buildingblock params
 63          * @type Array
 64          * @private
 65          */
 66         this._textarea = new dijit.form.SimpleTextarea();
 67         this._textarea.domNode.addClassName('parameterArea');
 68         this._textarea.setValue(this._initialParameter);
 69 
 70         this._tooltipButton = new Element('div', {
 71             'class': 'initialParameterButton'
 72         }).update(" ");
 73 
 74         this._tooltipButton.observe('click', function() {
 75             var dialog = new ExternalContentDialog("Example parameter configuration");
 76             dialog.show(new Element('pre').update(this._templateParameter));
 77         }.bind(this));
 78 
 79         /**
 80          * Example parameter configuration, in a tooltip
 81          * @type dijit.Tooltip
 82          * @private
 83          */
 84         this._templateParameterTooltip = new dijit.Tooltip({
 85             'connectId': [this._tooltipButton],
 86             'label': "<h4> Example parameter configuration </h4>" +
 87                 "<pre>" + this._templateParameter + "</pre>"
 88         });
 89 
 90         /**
 91          * @type Function
 92          * @private
 93          */
 94         this._onChangeCallback = onChangeCallback;
 95     },
 96 
 97     // **************** PUBLIC METHODS **************** //
 98 
 99     /**
100      * This function returns the DOMNode of the button that shows the
101      * dialog
102      * @type DOMNode
103      */
104     getButtonNode: function() {
105 
106     	var button = new dijit.form.Button({
107             'label': '...',
108             'onClick': this.show.bind(this)
109         });
110 
111         return new Element('div', {
112             'class': 'triggerButton'
113         }).update(button.domNode);
114     },
115 
116     // **************** PRIVATE METHODS **************** //
117 
118 
119     /**
120      * initDialogInterface
121      * This function creates the dom structure
122      * @private
123      * @override
124      */
125     _initDialogInterface: function () {
126 
127         var content = new Element('div', {
128             'class': 'paramsDialog'
129         });
130         var title = new Element('h2').update("Edit " +
131         		this._buildingblockName + " parameters:");
132         content.appendChild(title);
133 
134         content.appendChild(this._textarea.domNode);
135 
136         if (this._templateParameter) {
137             content.appendChild(this._tooltipButton);
138         }
139         this._setContent(content);
140     },
141 
142     /**
143      * Overriding onOk handler
144      * @private
145      * @override
146      */
147     _onOk: function($super) {
148     	$super();
149         this._onChangeCallback(this._textarea.getValue());
150     }
151 });
