 26 var GadgetDialog = Class.create(GalleryDialog, /** @lends Builder.prototype */ {
 27     /**
 28      * On charge of building screenflows and showing the possibility to
 29      * Deploy the gadget into the mashup platrom
 30      * @constructs
 31      */
 32     initialize: function($super, /** ScreenflowDescription */ description) {
 33         $super("Available gadgets for the screenflow", {
 34             "onDblClick": this._showDeploymentInfo.bind(this)
 35         });
 36 
 37         /**
 38          * @type StoreGadgetDialog
 39          * @private @member
 40          */
 41         this._buildGadgetDialog = new BuildGadgetDialog(this._onBuildGadgetDialogCallback.bind(this));
 42 
 43         /**
 44          * @type PublishGadgetDialog
 45          * @private @member
 46          */
 47         this._publishGadgetDialog = new PublishGadgetDialog();
 48 
 49 
 50         /**
 51          * @type ScreenflowDescription
 52          * @private @member
 53          */
 54         this._description = description;
 55 
 56         /**
 57          * Array of stored gadgets
 58          * @private @member
 59          * @type Hash
 60          */
 61         this._storedGadgets = new Hash();
 62     },
 63 
 64 
 65     // **************** PUBLIC METHODS **************** //
 66 
 67     /**
 68      * Start the process of showing the Gadget dialog
 69      * @override
 70      */
 71     show: function() {
 72         var uri = URIs.getStore.replace("<screenflow_id>", this._description.id);
 73         PersistenceEngine.sendGet(uri, this, this._onLoadSuccess, Utils.onAJAXError);
 74     },
 75 
 76 
 77     // **************** PRIVATE METHODS **************** //
 78 
 79     /**
 80      * On storage list loaded
 81      * @private
 82      */
 83     _onLoadSuccess: function(/** XMLHttpRequest */ transport) {
 84         var gadgets = JSON.parse(transport.responseText);
 85         gadgets.each(function(gadget) {
 86             this._storedGadgets.set(gadget.id, gadget);
 87         }, this);
 88         if (this._storedGadgets.values().size() > 0) {
 89             this._show();
 90         } else {
 91             this._buildGadget();
 92         }
 93     },
 94 
 95     /**
 96      * initDialogInterface
 97      * This function creates the dom structure
 98      * @private
 99      * @override
100      */
101     _initDialogInterface: function (){
102 
103         this._setHeader("Stored Gadgets for this screenflow",
104                 "These are the gadgets you have created for the current screenflow." +
105                 "Here you can deploy them to several mashup platforms or create " +
106                 "new ones");
107 
108         this._setFields([{
109                 'title': 'Gadget Name',
110                 'className': 'name'
111             }, {
112                 'title': 'Gadget Vendor',
113                 'className': 'vendor'
114             }, {
115                 'title': 'Gadget Version',
116                 'className': 'version'
117             }, {
118                 'title': 'Mashup Platforms',
119                 'className': 'platforms'
120             }, {
121                 'title': 'Description',
122                 'className': 'description'
123             }
124         ]);
125 
126         this._setButtons([{
127                 'value': 'Create new gadget',
128                 'handler': this._buildGadget.bind(this),
129                 'disableIfNotValid': false
130             }, {
131                 'value': 'Show Available platforms',
132                 'handler': this._showDeploymentInfo.bind(this),
133                 'disableIfNotValid': true
134             }, {
135                 'value': 'Delete Gadget',
136                 'handler': this._confirmDeleteGadget.bind(this),
137                 'disableIfNotValid': true
138             }]);
139         this._createGadgetList();
140         this._render();
141     },
142 
143 
144     /**
145      * Creates the gadget list to be handled by its parent class
146      * @private
147      */
148     _createGadgetList: function() {
149         this._emptyRows();
150         this._storedGadgets.values().each(function(gadget) {
151             this._addRow({
152                             'key': gadget.id,
153                             'values': [
154                                        gadget.name,
155                                        '<span class="bold">Vendor: </span>' +
156                                          gadget.vendor,
157                                        '<span class="bold">Version: </span>' +
158                                          gadget.version,
159                                        '<span class="bold">Mashup platforms: </span>' +
160                                          gadget.platforms,
161                                        '<span class="bold">Description </span><br />'+
162                                 	     gadget.description
163                                       ],
164                             'isValid': true
165                         });
166         }.bind(this));
167     },
168 
169     /**
170      * Creates a gadget deployment from the ScreenflowDescription
171      * @private
172      */
173     _buildGadget: function () {
174         this._hide();
175         this._buildGadgetDialog.show({
176             'name': this._description.name,
177             'shortname': this._description.name,
178             'desc': this._description.description['en-gb']
179         });
180     },
181 
182     _onBuildGadgetDialogCallback: function(/** Object */ data) {
183 
184        var gadgetInfo = Object.extend(data, {
185                                             'description': {
186                                                 'en-gb': data.desc
187                                              },
188                                              'uri': 'buildingblock/' +
189                                                     this._description.getId(),
190                                              'label': {
191                                                  'en-gb': data.name
192                                                 },
193                                              'id': this._description.getId()
194                                             });
195        var storeParams = {
196             'gadget': Object.toJSON(gadgetInfo),
197             'screenflow': this._description.getId()
198         };
199        PersistenceEngine.sendPost(URIs.store, storeParams, null,
200                 this, this._onBuildSuccess, this._onError);
201        Utils.showMessage("Creating and storing the gadget...");
202     },
203 
204     _onBuildSuccess: function(/** XMLHttpRequest */ transport) {
205         var result = JSON.parse(transport.responseText);
206         this._storedGadgets.set(result.id, result);
207         this._createGadgetList();
208         if (this.isVisible()) {
209             this._render();
210         }
211         Utils.showMessage("Creating and storing the gadget...Done", {
212             "hide":true
213         });
214         this._showDeploymentInfo(result.id);
215     },
216 
217     _showDeploymentInfo: function(/** String */ gadget_id) {
218         var result = this._storedGadgets.get(gadget_id);
219         this._publishGadgetDialog.show(result);
220     },
221 
222     _confirmDeleteGadget: function(/** String */ gadget_id) {
223         confirm("Are you sure you want to delete the gadget? This action" +
224         " cannot be undone", function(ok) {
225             if (ok) {
226                 this._deleteGadget(gadget_id);
227             }
228         }.bind(this));
229     },
230 
231     _deleteGadget: function(/** String */ gadget_id) {
232         var uri = URIs.store + gadget_id;
233         PersistenceEngine.sendDelete(uri, this, function() {
234             this._storedGadgets.unset(gadget_id);
235             this._createGadgetList();
236             if (this.isVisible()) {
237                 this._render();
238             }
239         }, this._onError);
240     },
241 
242     _onDelete: function() {
243 
244     },
245 
246     _onError: function(/** XMLHttpRequest */ transport, /** Exception */ e) {
247         var message;
248         if (e) {
249             message = e;
250         } else {
251             message = transport.responseText;
252         }
253         Utils.showMessage("Cannot store the gadget: " + message, {
254             "error": true,
255             "hide": true
256         });
257     }
258 
259 });
