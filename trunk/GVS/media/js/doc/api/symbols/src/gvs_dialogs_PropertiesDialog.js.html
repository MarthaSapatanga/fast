 26 var PropertiesDialog = Class.create(ConfirmDialog /** @lends PropertiesDialog.prototype */, {
 27     /**
 28      * This class handles the dialog
 29      * that shows the screen/screenflow properties
 30      * @constructs
 31      * @extends ConfirmDialog
 32      */
 33     initialize: function($super, /** String */ type,
 34                         /** BuildingBlockDescription */ description,
 35                         /** Function */ onChangeHandler) {
 36 
 37         /**
 38          * Title of the dialog
 39          * @private
 40          * @type String
 41          */
 42         this._title = type + " Properties";
 43 
 44         /**
 45          * Building block description
 46          * @private
 47          * @type String
 48          */
 49         this._description = description;
 50 
 51         /**
 52          * Handler to be called when the dialog finishes
 53          * @private
 54          * @type Function
 55          */
 56         this._onFinishHandler = null;
 57 
 58         /**
 59          * Handler to be called when the dialog finishes with changes
 60          * @private
 61          * @type Function
 62          */
 63         this._onChangeHandler = onChangeHandler;
 64 
 65         $super(this._title);
 66 
 67     },
 68 
 69 
 70     // **************** PUBLIC METHODS **************** //
 71 
 72     /**
 73      * @override
 74      */
 75     show: function($super, /** Function(Optional) */ _handler) {
 76         var handler = Utils.variableOrDefault(_handler, null);
 77         this._onFinishHandler = handler;
 78         $super();
 79     },
 80 
 81     // **************** PRIVATE METHODS **************** //
 82 
 83 
 84     /**
 85      * initDialogInterface
 86      * This function creates the dom structure and
 87      * @private
 88      * @override
 89      */
 90     _initDialogInterface: function () {
 91 
 92         this._setHeader(this._title);
 93 
 94         var formData = [
 95             {'type': 'title', 'value': 'Basic information'},
 96             {'type':'input', 'label': 'Building block name:','name': 'name',
 97                     'value': this._description.name,
 98                     'required': true,
 99                     'events': {
100                         'blur': function() {
101                             this._getForm().name.value = Utils.sanitize($F(this._getForm().name));
102                         }.bind(this)
103                     }
104             },
105             {'type':'input', 'label': 'Version:','name': 'version',
106                     'value': this._description.version},
107             {'type':'input', 'label': 'Tags:','name': 'tags',
108                     'value': this._description.tags.collect(function(tag) {
109                                     return tag.label['en-gb'];
110                                }).join(", ")},
111             {'type': 'title', 'value': 'Sharing information'},
112             {'type':'textarea', 'label': 'Description:','name': 'description',
113                     'value': this._description.description['en-gb'],
114                     'required': true},
115             {'type':'input', 'label': 'Creator:','name': 'creator',
116                     'value': this._description.creator,
117                     'required': true,
118                     'disabled': true},
119             {'type':'input', 'label': 'Licence information:','name': 'rights',
120                     'value': this._description.rights,
121                     'required': true},
122             {'type':'input', 'label': 'Icon:','name': 'icon',
123                     'value': this._description.icon,
124                     'regExp': '([hH][tT][tT][pP][sS]?)://[A-Za-z0-9-_]+(\.[A-Za-z0-9-_]+)*(:\d+)?(/[a-zA-Z0-9\.\?=/#%&\+-]*)*',
125                     'message': 'Invalid URL'},
126             {'type':'input', 'label': 'Screenshot:','name': 'screenshot',
127                     'value': this._description.screenshot,
128                     'regExp': '([hH][tT][tT][pP][sS]?)://[A-Za-z0-9-_]+(\.[A-Za-z0-9-_]+)*(:\d+)?(/[a-zA-Z0-9\.\?=/#%&\+-]*)*',
129                     'message': 'Invalid URL'},
130             {'type':'input', 'label': 'Homepage:','name': 'homepage',
131                     'value': this._description.homepage,
132                     'regExp': '([hH][tT][tT][pP][sS]?)://[A-Za-z0-9-_]+(\.[A-Za-z0-9-_]+)*(:\d+)?(/[a-zA-Z0-9\.\?=/#%&\+-]*)*',
133                     'message': 'Invalid URL'}
134         ];
135 
136         this._setContent(formData);
137 
138     },
139     /**
140      * Overriding onOk handler
141      * @override
142      * @private
143      */
144     _onOk: function($super){
145         if (this._getFormWidget().validate()) {
146 
147             var tags = $F(this._getForm().tags).split(/[\s,]+/).without("");
148             var description = {
149                 'tags': Utils.getCatalogueTags(tags, null),
150                 'description': {
151                     'en-gb': $F(this._getForm().description)
152                 },
153                 'label': {
154                     'en-gb': Utils.sanitize($F(this._getForm().name))
155                 },
156                 'name': Utils.sanitize($F(this._getForm().name))
157             };
158             var form = this._getForm();
159             var elements = [form.version, form.creator, form.rights,
160                 form.icon, form.screenshot, form.homepage];
161             Object.extend(description, Form.serializeElements(elements, {'hash': true}));
162             var changed = false;
163             $H(description).each(function(pair) {
164                 if (this._description[pair.key] != pair.value) {
165                     changed = true;
166                 }
167             }.bind(this));
168 
169             this._description.addProperties(description);
170             $super();
171             if (changed) {
172                 this._onChangeHandler();
173             }
174             if (this._onFinishHandler && this._onFinishHandler instanceof Function) {
175                 this._onFinishHandler();
176                 this._onFinishHandler = null;
177             }
178         }
179     },
180 
181     /**
182      * Reset method to leave the form as initially
183      * @override
184      * @private
185      */
186     _reset: function ($super){
187         this._handler = null;
188         this._initDialogInterface();
189         $super();
190     }
191 });
