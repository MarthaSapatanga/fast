 26 var CaptionDialog = Class.create(ConfirmDialog /** @lends CaptionDialog.prototype */, {
 27     /**
 28      * This class handles the dialog
 29      * to set a caption for a screen
 30      * @constructs
 31      * @extends ConfirmDialog
 32      */
 33     initialize: function($super, /** String */ caption,
 34                             /** Function */ onChangeCallback) {
 35 
 36         $super("Add Screen caption");
 37 
 38         this._caption = caption;
 39 
 40 
 41         this._onChangeCallback = onChangeCallback;
 42 
 43     },
 44 
 45     // **************** PUBLIC METHODS **************** //
 46 
 47     /**
 48      * This function returns the DOMNode of the button that shows the
 49      * dialog
 50      * @type DOMNode
 51      */
 52     getButtonNode: function() {
 53 
 54     	var button = new dijit.form.Button({
 55             'label': '...',
 56             'onClick': this.show.bind(this)
 57         });
 58 
 59         return new Element('div', {
 60             'class': 'triggerButton'
 61         }).update(button.domNode);
 62     },
 63 
 64     // **************** PRIVATE METHODS **************** //
 65 
 66 
 67     /**
 68      * initDialogInterface
 69      * This function creates the dom structure
 70      * @private
 71      * @override
 72      */
 73     _initDialogInterface: function () {
 74         this._setHeader(
 75                 "Add Screen caption",
 76                 "The caption will be shown in screen bottom when executing the"
 77                 + " gadget.<br />You can use HTML tags to add formatting"
 78         );
 79 
 80         var formData = [
 81             {
 82                 'type':'textarea',
 83                 'label': 'Screen Caption:',
 84                 'name': 'caption',
 85                 'value': this._caption
 86             }
 87         ];
 88 
 89         this._setContent(formData);
 90     },
 91 
 92     /**
 93      * Overriding onOk handler
 94      * @private
 95      * @override
 96      */
 97     _onOk: function($super) {
 98     	$super();
 99         this._onChangeCallback(this._getForm().caption.value);
100     }
101 });
