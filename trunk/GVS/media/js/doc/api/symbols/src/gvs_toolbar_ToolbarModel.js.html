 26 var ToolbarModel = Class.create( /** @lends ToolbarModel.prototype */ {
 27     /**
 28      * Provides toolbar elements for a toolbar section.
 29      * @abstract
 30      * @constructs
 31      */
 32     initialize: function() {
 33 
 34         /**
 35          * All the toolbar elements
 36          * @type Hash
 37          * @private @member
 38          */
 39         this._toolbarElements = new Hash();
 40 
 41         /**
 42          * Order of the elements in the toolbar interface
 43          * @type Array
 44          * @private @member
 45          */
 46         this._toolbarElementOrder = new Array();
 47 
 48     },
 49 
 50 
 51     // **************** PUBLIC METHODS **************** //
 52 
 53 
 54     /**
 55      *
 56      * @type Array   Array of ToolbarElement
 57      */
 58     getToolbarElements: function() {
 59         var elements = new Array();
 60         this._toolbarElementOrder.each(function(name) {
 61             elements.push(this._toolbarElements.get(name));
 62         }.bind(this));
 63         return elements;
 64     },
 65 
 66     // **************** PRIVATE METHODS **************** //
 67 
 68 
 69     /**
 70      * Adds a new element to the list
 71      * @private
 72      */
 73     _addToolbarElement: function(/** String */ elementName, /** ToolbarElement */ element) {
 74         this._toolbarElementOrder.push(elementName);
 75         this._toolbarElements.set(elementName, element);
 76     }
 77 });
