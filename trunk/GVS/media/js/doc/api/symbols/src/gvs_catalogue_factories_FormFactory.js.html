1 var FormFactory = Class.create(BuildingBlockFactory,
  2     /** @lends FormFactory.prototype */ {
  3 
  4     /**
  5      * Factory of Form building blocks.
  6      * @constructs
  7      * @extends BuildingBlockFactory
  8      */
  9     initialize: function($super) {
 10         $super();
 11 
 12          /**
 13          * Hash table (organized by URI)
 14          * containing all the BB descriptions
 15          * @type Hash
 16          * @private @member
 17          */
 18         this._buildingBlockDescriptions = new Hash();
 19 
 20     },
 21 
 22     // **************** PUBLIC METHODS **************** //
 23     /**
 24      * Gets the type of building block this factory mades.
 25      * @type String
 26      * @override
 27      */
 28     getBuildingBlockType: function (){
 29         return Constants.BuildingBlock.FORM;
 30     },
 31 
 32 
 33     /**
 34      * Gets building block descriptions by URI
 35      * @type {BuildingBlockDescription[]}
 36      * @override
 37      */
 38     getBuildingBlocks: function (/** Array */ uris) {
 39         var result = new Array();
 40         $A(uris).each(function(uri){
 41             if(this._buildingBlockDescriptions.get(uri)) {
 42                 result.push(this._buildingBlockDescriptions.get(uri));
 43             } else {
 44                 throw "Ooops. Something went wrong. " +
 45                     "BuildingBlockFactory::getBuildingBlocks";
 46             }
 47         }.bind(this));
 48         return result;
 49     },
 50 
 51     /**
 52      * This function retrieves the pending elements from the serverside
 53      * catalogue
 54      */
 55     cacheBuildingBlocks: function (/** Array */ uris, /** Function */ callback){
 56         //URIs not already retrieved
 57         var pendingURIs = new Array();
 58         $A(uris).each (function (uri){
 59             if (!this._buildingBlockDescriptions.get(uri)){
 60                 pendingURIs.push (uri);
 61             }
 62         }.bind(this));
 63 
 64         if (pendingURIs.size() > 0) {
 65             var postData = Object.toJSON(pendingURIs);
 66             PersistenceEngine.sendPost(URIs.catalogueGetMetadata,
 67                 null, postData,
 68                 {
 69                     'mine': this,
 70                     'callback': callback
 71                 },
 72                 this._onSuccess, Utils.onAJAXError);
 73         } else {
 74             callback();
 75         }
 76     },
 77 
 78     /**
 79      * Gets a new instance of the type of the factory
 80      * @override
 81      * @type FormInstance
 82      */
 83     getInstance: function(/** BuildingBlockDescription */description, /** InferenceEngine */ engine) {
 84         return new FormInstance(description, engine);
 85     },
 86 
 87     // **************** PRIVATE METHODS **************** //
 88 
 89 
 90 
 91     /**
 92      * Callback function
 93      */
 94     _onSuccess: function(/**XMLHttpRequest*/ transport) {
 95         var metadata = transport.responseText.evalJSON();
 96         //update the Form Factory
 97         this.mine._updateBuildingBlockDescriptions(metadata.forms);
 98         //call the callback function passed as argument
 99         this.callback();
100     },
101 
102     /**
103      * This function creates the different Form Descriptions
104      * @private
105      */
106     _updateBuildingBlockDescriptions: function (/** Array */ formDescriptions) {
107 
108         for (var i=0; i< formDescriptions.length ; i++) {
109             this._buildingBlockDescriptions.set(formDescriptions[i].uri,
110                                         new FormDescription (formDescriptions[i]));
111         }
112     }
