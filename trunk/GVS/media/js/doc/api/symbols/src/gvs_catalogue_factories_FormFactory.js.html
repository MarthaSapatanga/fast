 26 var FormFactory = Class.create(BuildingBlockFactory,
 27     /** @lends FormFactory.prototype */ {
 28 
 29     /**
 30      * Factory of Form building blocks.
 31      * @constructs
 32      * @extends BuildingBlockFactory
 33      */
 34     initialize: function($super) {
 35         $super();
 36 
 37          /**
 38          * Hash table (organized by URI)
 39          * containing all the BB descriptions
 40          * @type Hash
 41          * @private @member
 42          */
 43         this._buildingBlockDescriptions = new Hash();
 44 
 45     },
 46 
 47     // **************** PUBLIC METHODS **************** //
 48     /**
 49      * Gets the type of building block this factory mades.
 50      * @type String
 51      * @override
 52      */
 53     getBuildingBlockType: function (){
 54         return Constants.BuildingBlock.FORM;
 55     },
 56 
 57 
 58     /**
 59      * Gets building block descriptions by URI
 60      * @type {BuildingBlockDescription[]}
 61      * @override
 62      */
 63     getBuildingBlocks: function (/** Array */ uris) {
 64         var result = new Array();
 65         $A(uris).each(function(uri){
 66             if(this._buildingBlockDescriptions.get(uri)) {
 67                 result.push(this._buildingBlockDescriptions.get(uri));
 68             } else {
 69                 throw "Ooops. Something went wrong. " +
 70                     "BuildingBlockFactory::getBuildingBlocks";
 71             }
 72         }.bind(this));
 73         return result;
 74     },
 75 
 76     /**
 77      * This function retrieves the pending elements from the serverside
 78      * catalogue
 79      */
 80     cacheBuildingBlocks: function (/** Array */ uris, /** Function */ callback){
 81         //URIs not already retrieved
 82         var pendingURIs = new Array();
 83         $A(uris).each (function (uri){
 84             if (!this._buildingBlockDescriptions.get(uri)){
 85                 pendingURIs.push (uri);
 86             }
 87         }.bind(this));
 88 
 89         if (pendingURIs.size() > 0) {
 90             var postData = Object.toJSON(pendingURIs);
 91             PersistenceEngine.sendPost(URIs.catalogueGetMetadata,
 92                 null, postData,
 93                 {
 94                     'mine': this,
 95                     'callback': callback
 96                 },
 97                 this._onSuccess, Utils.onAJAXError);
 98         } else {
 99             callback();
100         }
101     },
102 
103     /**
104      * Gets a new instance of the type of the factory
105      * @override
106      * @type FormInstance
107      */
108     getInstance: function(/** BuildingBlockDescription */description, /** InferenceEngine */ engine) {
109         return new FormInstance(description, engine);
110     },
111 
112     // **************** PRIVATE METHODS **************** //
113 
114 
115 
116     /**
117      * Callback function
118      */
119     _onSuccess: function(/**XMLHttpRequest*/ transport) {
120         var metadata = transport.responseText.evalJSON();
121         //update the Form Factory
122         this.mine._updateBuildingBlockDescriptions(metadata.forms);
123         //call the callback function passed as argument
124         this.callback();
125     },
126 
127     /**
128      * This function creates the different Form Descriptions
129      * @private
130      */
131     _updateBuildingBlockDescriptions: function (/** Array */ formDescriptions) {
132 
133         for (var i=0; i< formDescriptions.length ; i++) {
134             this._buildingBlockDescriptions.set(formDescriptions[i].uri,
135                                         new FormDescription (formDescriptions[i]));
136         }
137     }
138 });
