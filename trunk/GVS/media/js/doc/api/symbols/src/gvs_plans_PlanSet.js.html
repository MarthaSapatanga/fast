 26 var PlanSet = Class.create(BuildingBlockSet, /** @lends PlanSet.prototype */ {
 27     /**
 28      * This class handles a set of building blocks for a given context.
 29      * This list will be updated.
 30      * @constructs
 31      * @extends BuildingBlockSet
 32      */
 33     initialize: function($super) {
 34         $super([]);
 35 
 36         /**
 37          * List of Plans
 38          * @type Array
 39          * @private @member
 40          */
 41         this._plans = new Array();
 42 
 43 
 44         this._factory = Catalogue.
 45             getBuildingBlockFactory(Constants.BuildingBlock.SCREEN);
 46     },
 47 
 48 
 49     // **************** PUBLIC METHODS **************** //
 50 
 51 
 52     /**
 53      * Returns all the building blocks ordered by plan from the set
 54      * @type Array
 55      * @override
 56      */
 57     getBuildingBlocks: function () {
 58         var result = new Array();
 59         this._plans.each(function(plan) {
 60             result.push(this._factory.getBuildingBlocks(plan));
 61         }.bind(this));
 62         return result;
 63     },
 64 
 65     /**
 66      * Add new building blocks to the set by uri
 67      */
 68     setPlans: function (/** Array */ plans) {
 69         this._plans = plans.splice(0,10); //Just in case
 70         var uris = this._plans.flatten().uniq();
 71         this._factory.cacheBuildingBlocks(uris, this._cachedElements.bind(this));
 72     },
 73 
 74     // **************** PRIVATE METHODS **************** //
 75 
 76     /**
 77      * This is the callback called when returning from the
 78      * building block factory
 79      */
 80     _cachedElements: function () {
 81         this._listener.setChanged();
 82     }
 83 
 84 });
