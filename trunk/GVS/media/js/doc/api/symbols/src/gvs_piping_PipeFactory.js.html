1 var PipeFactory = Class.create(
  2     /** @lends PipeFactory.prototype */ {
  3 
  4     /**
  5      * Handles all pipes for a given screenflow
  6      * @constructs
  7      */ 
  8     initialize: function () {
  9         /**
 10          * Hash containing all the pipes for the screen
 11          */
 12         this._pipes = new Hash();
 13     },
 14 
 15     /**
 16      * Returns a pipe instance from the wire element
 17      */
 18     getPipe: function (/** Variable Arguments */) {
 19         var pipe = null;
 20         var wire;
 21         var source;
 22         var destination;
 23         if (arguments.length == 1) {
 24             wire = arguments[0];
 25             source = wire.terminal1;
 26             destination = wire.terminal2;
 27            
 28         } else {
 29             // Get pipe by the terminals
 30             source = arguments[0];
 31             destination = arguments[1];
 32             wire = new WireIt.Wire(source, destination, document.body, source.options.wireConfig);
 33             wire.redraw();
 34         }
 35         var id = this._getPipeId(source, destination);
 36         pipe = new Pipe(wire, id);
 37         if (pipe.isValid()) {
 38             if (!this._pipes.get(id)) {
 39                 this._pipes.set(id, pipe);
 40             }
 41         } else {
 42             pipe.destroy();
 43             pipe = null;
 44         }
 45         return pipe;
 46         
 47     },
 48 
 49     /**
 50      * Removes a pipe
 51      */
 52     removePipe: function(/** Pipe */ pipe) {
 53         this._pipes.unset(pipe.getId());
 54     },
 55 
 56     /**
 57      * Returns a pipe instance from its JSON data
 58      * @type Pipe
 59      */
 60     getPipeFromJSON: function(/** Object */ data) {
 61         return this._pipes.get(this._getPipeIdFromJSON(data));
 62     },
 63 
 64     /**
 65      * Returns the list of pipes in which an instance is involved
 66      * (as the source or the destination)
 67      * @type Array
 68      */
 69     getPipes: function(/** ComponentInstance */ instance) {
 70         var result = new Array();
 71         this._pipes.values().each(function(pipe) {
 72             var found = pipe.getSource().getBuildingBlockId() == instance.getId();
 73             found = found || pipe.getDestination().getBuildingBlockId() == instance.getId();
 74             if (found) {
 75                 result.push(pipe);
 76             }
 77         });
 78         return result;
 79     },
 80 
 81     /**
 82      * Hides all the pipes contained in the factory
 83      */
 84     hidePipes: function() {
 85         this._pipes.values().each(function(pipe) {
 86            pipe.setVisible(false);
 87         });
 88     },
 89 
 90     /**
 91      * Shows all the pipes contained in the screenflow
 92      */
 93     showPipes: function() {
 94          this._pipes.values().each(function(pipe) {
 95            pipe.setVisible(true);
 96         });
 97     },
 98     
 99     // **************** PRIVATE METHODS **************** //
100 
101     /**
102      * Gets a pipe unique id from its endpoints
103      * @private
104      * @type String
105      */
106     _getPipeId: function(/** Terminal */ source, /** Terminal */ destination) {
107         return source.getBuildingBlockUri() + source.getConditionId() +
108             destination.getBuildingBlockUri() + destination.getActionId() +
109             destination.getConditionId();
110     },
111 
112     /**
113      * Gets an id from its json data
114      * @private
115      * @type String
116      */
117     _getPipeIdFromJSON: function(/** Object */ data) {
118         return data.from.buildingblock + data.from.condition + data.to.buildingblock +
119             data.to.action + data.to.condition;
120     }
121 });
