1 var FormView = Class.create(BuildingBlockView,
  2     /** @lends FormView.prototype */ {
  3 
  4     /**
  5      * Forms graphical representation
  6      * @constructs
  7      * @extends BuildingBlockView
  8      */ 
  9     initialize: function($super, /** FormDescription */ description) {
 10 
 11         $super();
 12 
 13         /**
 14          * Actions
 15          * @type Hash
 16          * @private
 17          */
 18         this._actions = new Hash();
 19 
 20         /**
 21          * Postcondition Icons
 22          * @type Hash
 23          * @private
 24          */
 25         this._postIcons = new Hash();  
 26         
 27         /**
 28          * Trigger icons
 29          * @type Hash
 30          * @private 
 31          */         
 32         this._triggerIcons = new Hash(); 
 33 
 34         
 35         this._node = new Element("div", {
 36             "class": "view form",
 37             "title": description.name
 38         });
 39         
 40         var title = new Element("div", {
 41             "class": "title"
 42         }).update(description.label['en-gb']);
 43         
 44         this._node.appendChild(title);
 45         
 46         var actionsNode = new Element("div", {
 47             "class": "actions"
 48         });
 49         var actionsTitle = new Element("div", {
 50             "class": "title"
 51         }).update("Actions");
 52         this._node.appendChild(actionsNode);
 53         actionsNode.appendChild(actionsTitle);
 54         
 55         var actions = description.actions;      
 56         actions.each( function(action) {
 57             var actionView = new ActionView(action);
 58             this._actions.set(action.name, actionView);
 59             actionsNode.appendChild(actionView.getNode());           
 60         }.bind(this));
 61         
 62         var triggerPostContainer = new Element("div", {
 63             "class": "postTriggerContainer"    
 64         });
 65         
 66         var containerTitle = new Element("div", {
 67             "class": "title"
 68         }).update("Postconditions");
 69         triggerPostContainer.appendChild(containerTitle);
 70         
 71         var triggerArea = new Element("div", {
 72             "class": "triggerArea"
 73         });
 74         triggerPostContainer.appendChild(triggerArea);
 75         
 76         var posts;
 77         if (description.postconditions && description.postconditions[0] instanceof Array) {
 78             posts =  description.postconditions[0];
 79         } else {
 80             posts = description.postconditions;
 81         }
 82         
 83         var postArea = new Element("div", {
 84             "class": "postArea"    
 85         });
 86         
 87         
 88         
 89         posts.each (function(post) {
 90             var fact = FactFactory.getFactIcon(post, "embedded");
 91             this._postIcons.set(post.id, fact);
 92             postArea.appendChild(fact.getNode());
 93         }.bind(this));
 94         triggerPostContainer.appendChild(postArea);
 95         
 96         this._node.appendChild(triggerPostContainer);
 97         
 98         var imageUrl = description.screenshot;
 99 
100         var image = new Element ('img',{
101                 'class': 'image', 
102                 'src': imageUrl,
103                 'onerror': 'this.src = "/fast/images/gui/imageNotFound.png"; '
104         });
105         this._node.appendChild(image);      
106         
107     },
108     
109     // **************** PUBLIC METHODS **************** //
110     /**
111      * This function returns the domNode of the condition that has
112      * the id passed as parameter
113      * @type DOMNode
114      */
115     getConditionNode: function(/** String */ id, 
116                             /** String (Optional) */ _action) {
117         if (_action) {
118             return this._actions.get(_action).getConditionNode(id);
119         } else {
120             return this._postIcons.get(id).getNode();
121         }
122     },
123     
124     /**
125      * Colorize the component depending on the reachability
126      * @public @override
127      */
128     setReachability: function( /** Hash */ reachabilityData) {
129         var satisfeable = reachabilityData.reachability;
130         Utils.setSatisfeabilityClass(this._node, satisfeable);
131         reachabilityData.actions.each(function(actionData) {
132             this._actions.get(actionData.name).setReachability(actionData);
133         }.bind(this));
134 
135         this._postIcons.values().each(function(post) {
136             Utils.setSatisfeabilityClass(post.getNode(), satisfeable);
137         });
138     },
139     
140     /**
141      * Removes the DOM Elements and frees building blocks
142      * @override
143      */
144     destroy: function () {
145         // Let the garbage collector to do its job
146         
147         this._actions.each(function(pair) {
148             pair.value.destroy();
149         });
150         this._postIcons = null;
151         this._triggerIcons.each(function(pair) {
152             pair.value.destroy();    
153         });
154     }
155     
156     // **************** PRIVATE METHODS **************** //
157 });
