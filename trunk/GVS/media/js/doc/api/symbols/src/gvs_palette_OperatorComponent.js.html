1 var OperatorComponent = Class.create(PaletteComponent,
  2     /** @lends OperatorComponent.prototype */ {
  3 
  4     /**

  6      * @param BuildingBlockDescription OperatorBuildingBlockDescription
  7      * @constructs
  8      * @extends PaletteComponent
  9      */
 10     initialize: function($super, description, dropZones, inferenceEngine) {
 11         $super(description, dropZones, inferenceEngine);
 12         this._inferenceEngine.addReachabilityListener(this._buildingBlockDescription.uri,this);
 13     },
 14 
 15 
 16     // **************** PUBLIC METHODS **************** //
 17 
 18     /**
 19      * Colorize the component depending on the reachability
 20      * @public
 21      */
 22     setReachability: function( /** Boolean */ highlight) {
 23         if (highlight) {
 24             this._hover.setStyle({"opacity": "1"});
 25             dojo.fadeOut({
 26                 "duration": 2500,
 27                 "node": this._hover
 28             }).play(2500);
 29         }
 30     },
 31 
 32 
 33     // **************** PRIVATE METHODS **************** //
 34 
 35     /**
 36      * Creates a new View instance for the component
 37      * @type BuildingBlockView
 38      * @override
 39      */
 40     _createView: function () {
 41         var view = new OperatorView(this._buildingBlockDescription);
 42         this._createTooltip(view.getNode(), this._buildingBlockDescription);
 43         return view;
 44     },
 45 
 46     /**
 47      * Creates a new Operator to be dragged.
 48      * @type OperatorInstance
 49      * @override
 50      */
 51     _createInstance: function () {
 52         return new OperatorInstance(this._buildingBlockDescription, this._inferenceEngine);
 53     },
 54 
 55     /**
 56      * @type String
 57      * @override
 58      */
 59     _getTitle: function () {
 60         return this._buildingBlockDescription.label['en-gb'];
 61     },
 62 
 63     /**
 64      * This function returns a list with all the
 65      * preconditions of the component.
 66      * @type Array
 67      * @override
 68      */
 69     _getPreConditions: function() {
 70         var result = [];
 71         var actions = this._buildingBlockDescription.actions;
 72         for (var i=0; actions && i < actions.length; i++) {
 73             var preconditions = actions[i].preconditions;
 74             for (var j=0; preconditions && j < preconditions.length; j++) {
 75                 result.push(preconditions[j]);
 76             }
 77         }
 78         return result;
 79     }
 80 
 81 });
