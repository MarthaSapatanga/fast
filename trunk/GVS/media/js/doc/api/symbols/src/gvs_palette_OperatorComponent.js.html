 26 var OperatorComponent = Class.create(PaletteComponent,
 27     /** @lends OperatorComponent.prototype */ {
 28 
 29     /**
 30      * Palette component of a domain concept building block.
 31      * @param BuildingBlockDescription OperatorBuildingBlockDescription
 32      * @constructs
 33      * @extends PaletteComponent
 34      */
 35     initialize: function($super, description, dropZones, inferenceEngine) {
 36         $super(description, dropZones, inferenceEngine);
 37         this._inferenceEngine.addReachabilityListener(this._buildingBlockDescription.uri,this);
 38     },
 39 
 40 
 41     // **************** PUBLIC METHODS **************** //
 42 
 43     /**
 44      * Colorize the component depending on the reachability
 45      * @public
 46      */
 47     setReachability: function( /** Boolean */ highlight) {
 48         if (highlight) {
 49             this._hover.setStyle({"opacity": "1"});
 50             dojo.fadeOut({
 51                 "duration": 2500,
 52                 "node": this._hover
 53             }).play(2500);
 54         }
 55     },
 56 
 57 
 58     // **************** PRIVATE METHODS **************** //
 59 
 60     /**
 61      * Creates a new View instance for the component
 62      * @type BuildingBlockView
 63      * @override
 64      */
 65     _createView: function () {
 66         var view = new OperatorView(this._buildingBlockDescription);
 67         this._createTooltip(view.getNode(), this._buildingBlockDescription);
 68         return view;
 69     },
 70 
 71     /**
 72      * Creates a new Operator to be dragged.
 73      * @type OperatorInstance
 74      * @override
 75      */
 76     _createInstance: function () {
 77         return new OperatorInstance(this._buildingBlockDescription, this._inferenceEngine);
 78     },
 79 
 80     /**
 81      * @type String
 82      * @override
 83      */
 84     _getTitle: function () {
 85         return this._buildingBlockDescription.label['en-gb'];
 86     },
 87 
 88     /**
 89      * This function returns a list with all the
 90      * preconditions of the component.
 91      * @type Array
 92      * @override
 93      */
 94     _getPreConditions: function() {
 95         var result = [];
 96         var actions = this._buildingBlockDescription.actions;
 97         for (var i=0; actions && i < actions.length; i++) {
 98             var preconditions = actions[i].preconditions;
 99             for (var j=0; preconditions && j < preconditions.length; j++) {
100                 result.push(preconditions[j]);
101             }
102         }
103         return result;
104     }
105 
106 });
