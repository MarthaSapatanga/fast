 26 var ExternalDocument = Class.create(AbstractDocument, /** @lends ExternalDocument.prototype */ {
 27     /**
 28      * Represents an external content
 29      * inner content.
 30      * @abstract
 31      * @extends AbstractDocument
 32      * @constructs
 33      */
 34     initialize: function ($super,
 35             /** String */ title,
 36             /** String */ url) {
 37 
 38         /**
 39          * Url of the external content
 40          * @type String
 41          * @private
 42          */
 43         this._url = url;
 44 
 45         $super(title);
 46 
 47         var iframe = new Element('iframe', {
 48             src: url,
 49             style:'border: 0px; width: 100%; height: 100%; margin: 0px; padding:0px;'
 50         });
 51         this._tabContent.appendChild(iframe);
 52         this._addToolbarElement('reload', new ToolbarButton(
 53             'Reload current tab',
 54             'refresh',
 55             function() { iframe.contentWindow.location.reload(); },
 56             true
 57         ));
 58     },
 59 
 60     // **************** PRIVATE METHODS **************** //
 61     /**
 62      * @override
 63      */
 64     _closeDocument: function($super) {
 65         this._tabContent.update("");
 66         $super();
 67     }
 68 });
