235     /**
236      * An element has been dropped into an area inside the canvas
237      * @private
238      * @type Boolean
239      */
240     _drop: function(/** Area */ area, /** ComponentInstance */ instance, /** Object */ position, 
241         /** Boolean (Optional) */ _isLoading) {
242         var isLoading = Utils.variableOrDefault(_isLoading, false);
243         // Reject repeated elements (except domain concepts or operators)
244         if (instance.constructor != OperatorInstance  &&
245                 this._description.contains(instance.getUri())) {
246             Utils.showMessage("There is another element like this. Cannot add it", {
247                 'hide': true,
248                 'error': true
249             });
250             return false;
251         }
252         if (instance.constructor == FormInstance) {
253             if (this._formInstance) {
254                 Utils.showMessage("Only one form per screen is allowed", {
255                     'hide': true,
256                     'error': true
257                 });
258                 return false;
259             } else {
260                 this._formInstance = instance;
261                 this._description.icon = instance.getBuildingBlockDescription().icon;
262                 this._description.screenshot = instance.getBuildingBlockDescription().screenshot;
263             }
264         }
265 
266         this._addToArea(area, instance, position);
267         
268         if (!instance.getId()) {
269             instance.setId(UIDGenerator.generate(instance.getTitle()));
270         } else {
271             UIDGenerator.setStartId(instance.getId());
272         }
273           
274         if (instance.constructor != PrePostInstance) {
275             instance.createTerminals(this._onPipeHandler.bind(this));
276             this._description.addBuildingBlock(instance, position);
277         } else {
278             instance.setConfigurable(false);
279             
280             if (area.getNode().className.include("pre")) {
281                 instance.setType("pre");
282                 instance.createTerminal(this._onPipeHandler.bind(this));
283                 this._description.addPre(instance, position);
284                 instance.getView().setReachability({"satisfied": true});
285             } else if (area.getNode().className.include("post")) {
286                 instance.setType("post");
287                 instance.createTerminal();
288                 this._description.addPost(instance, position);
289             }
290         }
291         if (!isLoading) {
292             this._setSelectedElement(instance);
293         }
294         instance.setEventListener(this);
295         instance.enableDragNDrop(area,[area]);
296         instance.getView().addGhost();
297         this._setDirty(true);
298         return true;
299     },

301     /**
302      * Launched whenever a pipe is added or removed from
303      * @private
304      */
305     _onPipeHandler: function(/** Event */ event, /** Array */ params, /** Boolean */ addedPipe) {
306         var wire = params[0];
307         if (wire.terminal1.parentEl && wire.terminal2.parentEl) {
308             var pipe = this._pipeFactory.getPipe(wire);
309             if (pipe) {
310                 if (addedPipe) {
311                     this._description.addPipe(pipe);
312                 } else {
313                     this._pipeFactory.removePipe(pipe);
314                     this._description.remove(pipe);
315                 }
316                 this._refreshReachability();
317             }
318             this._setDirty(true);
319         }

461      * This function updates the reachability in all
462      * Elements: canvas and palettes
463      * @private
464      */
465     _refreshReachability: function (/** Boolean (Optional) */_isFindCheck) {
466         var isFindCheck = Utils.variableOrDefault(_isFindCheck, true);
467         var canvas = this._getCanvasUris();
468         var body = {
469             'forms': this._paletteController.getComponentUris(Constants.BuildingBlock.FORM),
470             'operators': this._paletteController.getComponentUris(Constants.BuildingBlock.OPERATOR),
471             'backendservices': this._paletteController.getComponentUris(Constants.BuildingBlock.RESOURCE),
472             'pipes': this._description.getPipes(),
473             'preconditions': this._description.getPreconditions(),
474             'postconditions': this._description.getPostconditions()
475         }
476         if (this._selectedElement && this._selectedElement.getUri()) {
477             body.selectedItem = this._selectedElement.getUri();
478         }
479         
480         if (isFindCheck) {
481             this._inferenceEngine.findCheck(canvas, body,  this._tags,
482                                     'reachability', this._onUpdateReachability.bind(this));
483         } else {
484             this._inferenceEngine.check(canvas, body, this._tags, 'reachability',
485                                         this._onUpdateReachability.bind(this));
486         }

508             }.bind(this));
509         }
510         if (reachabilityData.connections) {
511             reachabilityData.connections.each(function(connection) {
512                 // TODO
513             }.bind(this));
514         }
515         this._updatePanes();
516     },

576      * This function builds a HTML chunk with the list of triggers of an action and
577      * the button to change the trigger mappings
578      * @private
579      * @type DOMNode
580      */
581     _buildTriggerList: function(/** String */ actionName, /** Array */ triggerList) {     
582         
583         var result = new Element('div');
584         var content = "";
585         if (triggerList) {
586             triggerList.each(function(trigger) {
587                 content += trigger.getTriggerName() + ", ";
588             });
589             content = content.slice(0, -2);
590         } else {
591             content = new Element('span', {
592                 'class': 'triggerInfo'
593             }).update("No triggers for this action");
594         }
595         result.update(content);
596         var triggerDialog = new TriggerDialog(actionName, triggerList,
597                                             this._description.getCanvasInstances(),
598                                             this._onTriggerChange.bind(this));
599         result.appendChild(triggerDialog.getButtonNode());
600         return result;

603      /**
604       * Called whenever a trigger dialog finishes its job
605       * @private
606       */
607      _onTriggerChange: function(/** String*/ action, /** Array */ triggersAdded,
608                                 /** Array */ triggersRemoved) {
609         triggersAdded.each(function(triggerInfo) {
610             var triggerSplittedInfo = triggerInfo.split("#");
611             var instance;
612             if (triggerSplittedInfo[0] == ScreenTrigger.INSTANCE_NAME) {
613                 instance = triggerSplittedInfo[0];
614             } else {
615                 instance = this._description.getInstance(triggerSplittedInfo[0]);
616             }
617             var triggerData = {
618                 'from': {
619                     'instance': instance,
620                     'name': triggerSplittedInfo[1]
621                 },
622                 'to': {
623                     'instance': this._selectedElement,
624                     'action': action
625                 }
626             }
627             var trigger = this._triggerMappingFactory.createTrigger(triggerData);
628             this._description.addTrigger(trigger);
629         }.bind(this));
630 
631         triggersRemoved.each(function(triggerInfo) {
632             var triggerSplittedInfo = triggerInfo.split("#");
633             var instance;
634             if (triggerSplittedInfo[0] == ScreenTrigger.INSTANCE_NAME) {
635                 instance = triggerSplittedInfo[0];
636             } else {
637                 instance = this._description.getInstance(triggerSplittedInfo[0]);
638             }
639             var triggerData = {
640                 'from': {
641                     'instance': instance,
642                     'name': triggerSplittedInfo[1]
643                 },
644                 'to': {
645                     'instance': this._selectedElement,
646                     'action': action
647                 }
648             }
649             var trigger = this._triggerMappingFactory.removeTrigger(triggerData);
650             this._description.remove(trigger);
651         }.bind(this));
652         this._updatePanes();
653         this._setDirty(true);
654      },

743     /**
744      * Creates the pipes programmatically
745      * @private
746      */
747     _createPipes: function(/** Array */ pipes) {
748         pipes.each(function(pipeData) {
749             var fromInstance;
750             var fromTerminal;
751             if (pipeData.from.buildingblock != "") {
752                 fromInstance = this._description.getInstance(pipeData.from.buildingblock);
753                 fromTerminal = fromInstance.getTerminal(pipeData.from.condition);
754             } else {
755                 fromInstance = this._description.getPre(pipeData.from.condition);
756                 if (!fromInstance) {
757                     fromInstance = this._description.getPost(pipeData.from.condition);
758                 }
759                 fromTerminal = fromInstance.getTerminal();
760             }
761             var toInstance;
762             var toTerminal;
763             if (pipeData.to.buildingblock != "") {
764                 toInstance = this._description.getInstance(pipeData.to.buildingblock);
765                 toTerminal = toInstance.getTerminal(pipeData.to.condition, pipeData.to.action);
766             } else {
767                 toInstance = this._description.getPre(pipeData.to.condition);
768                 if (!toInstance) {
769                     toInstance = this._description.getPost(pipeData.to.condition);
770                 }
771                 toTerminal = toInstance.getTerminal();
772             }
773             var pipe = this._pipeFactory.getPipe(fromTerminal, toTerminal);
774             this._description.addPipe(pipe);
775         }.bind(this));
776     },
777 
778     /**
779      * Creates the triggers
780      * @private
781      */
782     _createTriggers: function(/** Array */ triggers) {
783         triggers.each(function(triggerJSON) {
784             var fromInstance = null;
785             if (triggerJSON.from.buildingblock != "") {
786                 fromInstance = this._description.getInstance(triggerJSON.from.buildingblock);
787             }
788             var toInstance = this._description.getInstance(triggerJSON.to.buildingblock);
789             var triggerData = {
790                 'from': {
791                     'instance': fromInstance ? fromInstance : ScreenTrigger.INSTANCE_NAME,
792                     'name': fromInstance ? triggerJSON.from.name : ScreenTrigger.ONLOAD
793                 },
794                 'to': {
795                     'instance': toInstance,
796                     'action': triggerJSON.to.action
797                 }
798             }
799             var trigger = this._triggerMappingFactory.createTrigger(triggerData);
800             this._description.addTrigger(trigger);
801         }.bind(this));
802     },
803 
804     /**
805      * Function that loads the triggers and the pipes of the loaded screens
806      * @private
807      */
808     _loadConnections: function() {
809 
810         if (this._canvasCache.areInstancesLoaded()) {
811             // Load pre and postconditions
812             this._createConditions(this._canvasCache.getPreconditions(), this._areas.get('pre'));
813             this._createConditions(this._canvasCache.getPostconditions(), this._areas.get('post'));
814             this._createPipes(this._canvasCache.getPipes());
815             this._createTriggers(this._canvasCache.getTriggers());
816             this._setDirty(false);
817             this._refreshReachability();
818         }
819     }
820 });
