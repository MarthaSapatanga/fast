 26 var PropertiesPane = Class.create( /** @lends PropertiesPane.prototype */ {
 27     /**
 28      * This class handles the properties pane
 29      * @constructs
 30      */
 31     initialize: function(/** DOMNode */ parentNode) {
 32         /**
 33          * Variable
 34          * @type Table
 35          * @private @member
 36          */
 37         this._propertiesTable = new Table(parentNode, 'Properties', 'left');
 38 
 39         this._propertiesTable.insertFieldTitles(['Property','Value']);
 40 
 41     },
 42 
 43 
 44     // **************** PUBLIC METHODS **************** //
 45 
 46 
 47     /**
 48      * This function updates the table with data coming from
 49      * the currently selected element
 50      * @param element
 51      *          Something implementing TableModel interface
 52      *              * String getTitle()
 53      *              * Array getInfo()
 54      *              * Array getTriggerMapping(): Optional method
 55      */
 56     fillTable: function (/** Object */ element) {
 57         this._clearElement();
 58 
 59         var title = element.getTitle();
 60 
 61         this._propertiesTable.insertDataValues(element.getInfo());
 62         this._propertiesTable.setTitle((title ? 'Properties of '  + title : 'Properties'));
 63     },
 64 
 65     /**
 66      * This function add a new section to the table, without removing the actual
 67      * data. A section is a title row, and values
 68      */
 69     addSection: function (/** Array */ title, /** Hash */ values) {
 70         this._propertiesTable.insertFieldTitles(title);
 71         this._propertiesTable.insertDataValues(values);
 72     },
 73 
 74     // **************** PRIVATE METHODS **************** //
 75 
 76     /**
 77      * This function empties the table
 78      * @private
 79      */
 80     _clearElement: function (){
 81         this._propertiesTable.emptyTable();
 82         this._propertiesTable.setTitle('Properties');
 83         this._propertiesTable.insertFieldTitles(['Property','Value']);
 84     }
 85 
 86 
 87 });
