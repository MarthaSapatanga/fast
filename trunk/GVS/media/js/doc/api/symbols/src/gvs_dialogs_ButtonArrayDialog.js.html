 30     /**
 31      * Inits the user interface
 32      * @private
 33      * @override
 34      */
 35     _initDialogInterface: function() {
 36         var content = new Element('div', {
 37             'class': 'buttonArray'
 38         });
 39         this._handlers.each(function(pair) {
 40             var button = new dijit.form.Button({
 41                 'label': pair.label,
 42                 'onClick': function(){
 43                     this._dialog.hide();
 44                     pair.handler();
 45                 }.bind(this)
 46             });
 47             content.appendChild(button.domNode);
 48             content.appendChild(new Element('br'));
 49         }.bind(this));
 50         this._setContent(content);
 51     }
