 37     /**
 38      * Generates the PrePost uri if doesn't have it
 39      */
 40     generateUri: function() {
 41         if (this.uri == undefined) {
 42             if (this.pattern) {
 43                 this.uri = this.pattern.split(" ")[2];
 44             }
 45         }
 46     },
 47 
 48     getUriShortcut: function() {
 49         var uri;
 50         if (this.uri) {
 51             uri = this.uri;
 52         } else {
 53             uri = Utils.extractURIfromPattern(this.pattern);
 54         }
 55         if (uri) {
 56             var pieces = uri.split("#");
 57             var identifier = "";
 58             if (pieces.length > 1){
 59                 identifier = pieces[1];
 60             }
 61             else { //The uri has not identifier, try the last part of the url
 62                 pieces = uri.split("/");
 63                 identifier = pieces[pieces.length - 1];
 64             }
 65             return identifier;
 66         } else {
 67             return "Concept";
 68         }
 69     },

 73     /**
 74      * Creates the title of the palette component
 75      * @type String
 76      * @private
 77      */
 78     _updateTitle: function() {
 79 
 80         if (this['http://www.w3.org/2000/01/rdf-schema#label']) {
 81             this.title =
 82                  this['http://www.w3.org/2000/01/rdf-schema#label'].
 83                  replace("@en","");
 84 
 85         } else if (this['label']) {
 86             var languages = $H(this['label']).keys();
 87             if (languages.size() == 1) {
 88                 this.title = this['label'][languages[0]];
 89             } else if (this['label']['en']) {
 90                 this.title = this['label']['en'];
 91             } else {
 92                 this.title = "Concept";
 93             }
 94         } else { // Extract the title from the uri
 95             this.title = this._createTitle();
 96         }
 97     },
 98 
 99 
100     /**
101      * This function creates a title from the uri identifier
102      * @private
103      * @type String
104      */
105     _createTitle: function() {
106         return this._sanitizeTitle(this.getUriShortcut());
107     },
108 
109     /**
110      * This function returns a human-readable title from an
111      * identifier
112      * @private
113      * @type String
114      */
115     _sanitizeTitle: function (/** String */ title) {
116         //FIXME: I18NString
117         var result = title.replace("@en","");
118         //separate identifier from its forming words
119         var words = result.match(/[A-Z][a-z0-9]+|\s+[a-z][a-z0-9]*/g);
120         if (words) {
121             words.map(function(e) {
122                 return e.strip();
123             });
124             result = words.join(" ");
125         }
126         return result;
127     }
