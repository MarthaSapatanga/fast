 26 var GalleryDialog = Class.create(FormDialog, /** @lends GalleryDialog.prototype */ {
 27     /**
 28      * This class handles dialogs with a gallery (a table) of elements
 29      * Any descendant must use this class as follows:
 30      *      It must call to _setFields, _setButtons and _addRows, before calling
 31      *      _render, which will be the method that builds the interface, using
 32      *      the information that has been stated in the previous methods.
 33      * TODO: Add pagination
 34      * @abstract
 35      * @extends FormDialog
 36      * @constructs
 37      */
 38     initialize: function($super, /** String */ title,
 39                         /** Object (Optional) */ _properties) {
 40         $super({
 41             'title': title,
 42             'style': 'display:none;'
 43         }, {
 44             'buttonPosition': FormDialog.POSITION_TOP
 45         });
 46 
 47         var properties = Utils.variableOrDefault(_properties, {});
 48 
 49         /**
 50          * Hash of properties of the gallery
 51          * @type Hash
 52          * @private
 53          */
 54         this._properties = new Hash();
 55 
 56 
 57         // Assigning the passed parameters, or defaults
 58         this._properties.set('showTitleRow', (properties.showTitleRow || false));
 59         this._properties.set('elementsPerPage', (properties.elementsPerPage || 10));
 60         this._properties.set('onDblClick', (properties.onDblClick || function(){}));
 61         // Disable all the events if the selected row is not valid
 62         this._properties.set('disableIfNotValid', (properties.disableIfNotValid || false));
 63 
 64         /**
 65          * Table fields
 66          * @type Array
 67          * @private
 68          */
 69         this._fields = null;
 70 
 71         /**
 72          * Buttons that will be shown when an element is clicked
 73          * @type Array
 74          * @private
 75          */
 76         this._buttons = null;
 77 
 78         /**
 79          * List of buttons that will be disabled if the selected row
 80          * is not valid
 81          * @type Array
 82          * @private
 83          */
 84         this._disabledButtons = new Array();
 85 
 86         /**
 87          * List of rows
 88          * @type Array
 89          * @private
 90          */
 91         this._rows = new Array();
 92 
 93         /**
 94          * Currently selected row
 95          * @type Object
 96          * @private
 97          */
 98         this._selectedRow = null;
 99     },
100 
101 
102     // **************** PUBLIC METHODS **************** //
103 
104     /**
105      * @override
106      */
107     show: function() {
108         // Do nothing. To be overriden
109     },
110 
111     // **************** PRIVATE METHODS **************** //
112     /**
113      * Set the field list
114      * Each of the fields is an object with the following structure:
115      *  * String title. Field human readable title
116      *  * Boolean hidden. The field won't be shown in the interface
117      *  * String className. CSS class that will be applied to the whole
118      *                  column
119      * @private
120      */
121     _setFields: function (/** Array */ fields) {
122         this._fields = fields;
123     },
124 
125     /**
126      * Set the button list.
127      * Each element is an object with the following structure:
128      *  * String value: Text of the button
129      *  * Function handler: function which will be called when the button
130      *              is clicked. The function will receive the key value of
131      *              the selected element
132      * @private
133      */
134     _setButtons: function(/** Array */ buttons) {
135         this._buttons = buttons;
136     },
137 
138     /**
139      * Adds a new row to the gallery.
140      * It has to be an Object with this structure:
141      *    * String key: key of the row
142      *    * Array values: List with all the fields. Each of them must be
143      *                    a String or a DOM node
144      * @private
145      */
146     _addRow: function(/** Object */ row) {
147         this._rows.push(row);
148     },
149 
150     /**
151      * Builds the user interface, using the gathered information
152      * @private
153      */
154     _render: function(/** Boolean(Optional) */ _loadAll) {
155         var loadAll = Utils.variableOrDefault(_loadAll, true);
156 
157         var content = new Element('div', {
158             'class': 'gallery'
159         });
160 
161         if (this._properties.get("showTitleRow")) {
162             // TODO
163         }
164 
165         var lastSelectedRowKey = this._selectedRow && this._selectedRow['key'];
166         this._selectedRow = null;
167 
168         this._rows.each(function(row) {
169             var rowNode = new Element('div', {
170                 'class': 'row'
171             });
172 
173             for (var i=0, rValues = row.values; i < rValues.size(); i++) {
174                 if (!this._fields[i].hidden) {
175                     var field = new Element('div',{
176                         'class': "field " + this._fields[i].className
177                     }).update(rValues[i]);
178                     rowNode.appendChild(field);
179                 }
180             }
181 
182             var selectRow = function() {
183                 this._unselectElements();
184                 this._selectedRow = row;
185                 rowNode.addClassName("selected");
186                 for (var i = 0; i < this._disabledButtons.length; i++) {
187                     this._disabledButtons[i].attr('disabled', ! row.isValid);
188                 }
189             }.bind(this);
190 
191             rowNode.observe('click', selectRow);
192 
193             if (this._properties.get('onDblClick') &&
194                (! this._properties.get('disableIfNotValid') || row.isValid))
195             {
196                 rowNode.observe('dblclick', function() {
197                     this._properties.get('onDblClick')(row.key);
198                 }.bind(this));
199             }
200 
201             if (lastSelectedRowKey == row.key) {
202                 selectRow();
203             }
204 
205             content.appendChild(rowNode);
206         }.bind(this));
207 
208 
209         if (loadAll && this._rows.size() > 0) {
210             this._removeButtons();
211             this._buttons.each(function(button){
212                 var buttonWidget = this._addButton(button.value, function() {
213                     button.handler(this._selectedRow.key);
214                 }.bind(this));
215                 if (button.disableIfNotValid) {
216                     this._disabledButtons.push(buttonWidget);
217                 }
218             }.bind(this));
219         }
220 
221         if (this._rows.size() == 0) {
222             var info = new Element("div", {
223                 'class': 'info'
224             }).update("Uppss....Nothing here");
225             content.appendChild(info);
226             this._removeButtons();
227             this._addButton("Close", this._dialog.hide.bind(this._dialog));
228         } else if (! this._selectedRow &&
229                    ! this._properties.get("showTitleRow") &&
230                    content.firstChild)
231         {
232             content.firstChild.addClassName("selected");
233             var row = this._selectedRow = this._rows[0];
234             for (var i = 0; i < this._disabledButtons.length; i++) {
235                 this._disabledButtons[i].attr('disabled', ! row.isValid);
236             };
237         }
238 
239         this._setContent(content);
240         this._contentNode.appendChild(this._createSearchBar());
241     },
242 
243     /**
244      * Unselect all elements
245      * @private
246      */
247     _unselectElements: function() {
248             $$('.gallery .row.selected').each(function(node) {
249                 node.removeClassName("selected");
250             });
251             this._disabledButtons.each(function(button) {
252                 button.attr('disabled', true);
253             });
254             this._selectedRow = null;
255     },
256 
257     /**
258      * Create the search bar
259      * @private
260      * @type HTMLNode
261      */
262     _createSearchBar: function() {
263         var unselectElements = this._unselectElements.bind(this);
264         var searchBar = new Element('div', {'class':'searchBar'});
265         searchBar.style.marginTop = '3px';
266         var searchBox = new PaletteSearchBox();
267         searchBox.addEventListener(function(obj, value) {
268             var searchValue = value.toLowerCase();
269             var elements = $$('.gallery .row .field.name');
270             elements.each(function(elem){
271                 var text = elem.textContent.toLowerCase();
272                 if (searchValue.blank() || text.match(searchValue)) {
273                     elem.parentNode.show();
274                 } else {
275                     var row =elem.parentNode;
276                     row.hide();
277                     if (row.hasClassName('selected')) {
278                         unselectElements();
279                     }
280                 }
281             });
282         });
283 
284         searchBar.appendChild(searchBox.getDOMNode());
285         return searchBar;
286     },
287 
288     /**
289      * Empty the list of rows
290      * @private
291      */
292     _emptyRows: function() {
293         this._rows = new Array();
294     },
295 
296     /**
297      * Function called when the content is loaded
298      * @private
299      */
300     _show: function() {
301         this._initDialogInterface();
302         GVS.setEnabled(false);
303         this._dialog.show();
304     }
305 });
