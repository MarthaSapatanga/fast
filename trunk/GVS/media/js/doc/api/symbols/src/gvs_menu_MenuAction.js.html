 26 var MenuAction = Class.create(MenuElement, /** @lends MenuAction.prototype */ {
 27     /**
 28      * Element for a menu
 29      * @constructs
 30      * @extends MenuElement
 31      * @param Hash extraParams (optional)
 32      *           Hash containing some of these extra params
 33      *           * (String)iconName Name of the icon if present
 34      *           * (Boolean)enabled state of the element
 35      *           * (String)accelKey key shortcut to access to the element
 36      */
 37     initialize: function($super, /** Object */ data, /** Boolean */ mainMenu) {
 38         $super(data.weight);
 39         if (mainMenu) {
 40             this._widget = new dijit.MenuBarItem ({
 41                 'label': data.label,
 42                 'onClick': data.handler,
 43                 'showLabel': true
 44             });
 45         } else {
 46             this._widget = new dijit.MenuItem ({
 47                 'label': data.label,
 48                 'onClick': data.handler,
 49                 'showLabel': true
 50             });
 51         }
 52 
 53         /**
 54          * Handler function
 55          * @type Function
 56          * @private
 57          */
 58         this._handler = data.handler;
 59 
 60         /**
 61          * Accel key shortcut to use the menu item,
 62          * if any
 63          * @type String
 64          * @private
 65          */
 66         this._shortcut = (data.shortcut ? data.shortcut : null);
 67 
 68         if (this._shortcut) {
 69             this._widget.attr('accelKey', this._shortcut);
 70         }
 71 
 72         /**
 73          * Item state
 74          * @type Boolean
 75          * @private
 76          */
 77         this._enabled = null;
 78         this.setEnabled(data.enabled !== undefined ? data.enabled : true);
 79 
 80         if (data.iconName) {
 81             this._widget.attr('iconClass', 'dijitMenuItemIcon menuIcon ' + data.iconName + 'MenuIcon');
 82         }
 83     },
 84 
 85 
 86     // **************** PUBLIC METHODS **************** //
 87 
 88 
 89     /**
 90      * Sets the element enabled
 91      * It must be called to set it disabled
 92      * when the menuitem is going to be left
 93      * outside of the menu
 94      */
 95     setEnabled: function(/** Boolean */ enabled) {
 96         this._enabled = enabled;
 97         this._widget.attr('disabled',!enabled);
 98     },
 99 
100     /**
101      * Register key handlers
102      * @override
103      */
104     register: function(/** KeyPressRegistry */ registry) {
105         if (this._shortcut) {
106             registry.addHandler(this._shortcut, this._keyPressHandler.bind(this));
107         }
108     },
109 
110     /**
111      * Unregister key handlers
112      * @override
113      */
114     unregister: function(/** KeyPressRegistry */ registry) {
115         if (this._shortcut) {
116             registry.removeHandler(this._shortcut);
117         }
118     },
119 
120     // ************** PRIVATE METHODS **************** //
121 
122     /**
123      * function to capture key strokes and decide if
124      * calling the handler depending on the status of the
125      * action (enabled or disabled)
126      */
127     _keyPressHandler: function() {
128         if (this._enabled) {
129            this._handler();
130         }
131     }
132 });
