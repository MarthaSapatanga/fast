 45     /**
 46      * Gets the node of the accordion pane

 53     
 54     /**
 55      * This function populates the current palette
 56      * with the building blocks associated to a given
 57      * domain context
 58      */
 59     populateBuildingBlocks: function (/** Array */ domainContext){
 60         
 61         this._domainContext = domainContext;
 62         
 63         switch(this._buildingBlockType){
 64             case Constants.BuildingBlock.SCREEN:
 65                 //find Screens and check
 66                 this._parent.getInferenceEngine().retrieveScreens(domainContext);
 67                 break;
 68             case Constants.BuildingBlock.DOMAIN_CONCEPT:
 69                 //FIXME: Handle domain concepts
 70                 break;
 71             case Constants.BuildingBlock.CONNECTOR:
 72                 this.paintComponents();
 73                 break;
 74         }
 75     },
 76     
 77     /**
 78      * TODO: describe this. maybe this method will dissapear
 79      */
 80     updateComponents: function() {
 81         switch(this._buildingBlockType){
 82             case Constants.BuildingBlock.SCREEN:
 83                 //find Screens and check
 84                 var currentDocument = this._parent;
 85                 var canvas = currentDocument.getCanvas();
 86                 var domainContext = {
 87                     "tags":currentDocument.getBuildingBlockDescription().getDomainContext(),
 88                     "user":null
 89                 };
 90                 var elements = currentDocument.getPaletteElements();
 91                 CatalogueSingleton.getInstance().findAndCheck(canvas, domainContext, elements, 'reachability');
 92                 break;
 93             case Constants.BuildingBlock.DOMAIN_CONCEPT:
 94                 CatalogueSingleton.getInstance().getDomainConcepts();
 95                 break;
 96             case Constants.BuildingBlock.CONNECTOR:
 97                 this.paintComponents();
 98                 break;
 99         }
100     },
101 
102     paintComponents: function () {
103         this._components = this._createComponents(CatalogueSingleton.getInstance().getBuildingBlockFactory(this._buildingBlockType));
104         var content = this._renderComponents();
105         this._node.setContent(content);
106     },
107 
108     getComponents: function() {
109         return this._components;
110     },

115     /**
116      * Creates the palette components from building blocks by querying a building block factory.
117      * @type {BuildingBlockDescription[]}
118      * @private
119      */
120     _createComponents: function(buildingBlockFactory) {
121         var descs = buildingBlockFactory.getBuildingBlockDescriptions(this._domainContext);
122         var components = [];
123         $A(descs).each(
124             function(desc) {
125                 components.push(desc.createPaletteComponent());
126             }
127         );
128         return components;
129     },
130 
131     /**
132      * Creates the GUI stuff that shows the content: components and separators.
133      * @type DOMNode
134      * @private
135      */
136     _renderContent: function() {
137         var uidGenerator = UIDGeneratorSingleton.getInstance();
138         var paneId = uidGenerator.generate(this._id);
139         
140         var pane = new dijit.layout.AccordionPane({
141             'id':paneId,
142             'title':CatalogueSingleton.getInstance().getBuildingBlockFactory(this._buildingBlockType).getBuildingBlockName(),
143             'class':'paletteElement'
144         });
145         return pane;
146     },
147 
148     _renderComponents: function() {
149         var uidGenerator = UIDGeneratorSingleton.getInstance();
150         var contentId = uidGenerator.generate(this._id+ "Content");
151         var content   = new Element("div", {"class": "paletteContent", "id" : contentId});
152         var separator = new Element("div", {"class": "paletteSeparator"});
153         $A(this._components).each(
154                 function(component) {
155                     content.appendChild(component.getNode());
156                     content.appendChild(separator.cloneNode(false));
157                 }
158         );
159         return content;
160     }
161 });
