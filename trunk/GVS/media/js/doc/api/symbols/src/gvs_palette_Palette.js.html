 26 var Palette = Class.create(SetListener, /** @lends Palette.prototype */ {
 27 
 28         /**
 29          * Represents a palette of droppable components of a given type.
 30          *
 31          * @constructs
 32          * @extends SetListener
 33          */
 34         initialize: function(/** BuildingBlockSet */ set, /** Array */ dropZones,
 35                 /** InferenceEngine */ inferenceEngine) {
 36             /**
 37              * @private @member
 38              * @type InferenceEngine
 39              */
 40             this._inferenceEngine = inferenceEngine;
 41 
 42             /**
 43              * Building block set
 44              * @type BuildingBlockSet
 45              * @private @member
 46              */
 47             this._set = set;
 48 
 49             /**
 50              * Zones to drop components
 51              * @type Array
 52              * @private @member
 53              */
 54             this._dropZones = dropZones;
 55 
 56             /**
 57              * Collection of components the palette offers.
 58              * @type Hash   Hash of URI to PaletteComponent
 59              * @private @member
 60              */
 61             this._components = new Hash();
 62 
 63             /**
 64              * Accordion pane node.
 65              * @type DOMNode
 66              * @private @member
 67              */
 68             this._node = null;
 69 
 70             /**
 71              * Palette content
 72              * @type DOMNode
 73              * @private @member
 74              */
 75             this._contentNode = null;
 76 
 77             this._renderUI();
 78             this._set.setListener(this);
 79         },
 80 
 81         // **************** PUBLIC METHODS **************** //
 82 
 83         /**
 84          * Gets the node of the accordion pane
 85          * @type DOMNode
 86          * @public
 87          */
 88         getNode: function() {
 89             return this._node;
 90         },
 91 
 92         /**
 93          * Gets the node of the contents
 94          * @type DOMNode
 95          * @public
 96          */
 97         getContentNode: function() {
 98             return this._contentNode;
 99         },
100 
101         /**
102          * This function will be called whenever
103          * the set of building blocks changes
104          * @overrides
105          */
106         setChanged: function () {
107             this._updateComponents();
108         },
109 
110         getBuildingBlockSet: function() {
111             return this._set;
112         },
113 
114         /**
115          * All uris of all the components
116          */
117         getComponentUris: function() {
118             if (this._set.getBuildingBlockType() ==
119                 Constants.BuildingBlock.SCREEN) {
120                 var uris = [];
121                 this._set.getBuildingBlocks().each(function(buildingBlock) {
122                     uris.push({
123                         "uri": buildingBlock.uri
124                     });
125                 });
126                 return uris;
127             } else {
128                 return this._set.getBuildingBlocks().pluck("uri");
129             }
130         },
131 
132         // **************** PRIVATE METHODS **************** //
133 
134         /**
135          * Creates the GUI stuff that shows the content: components and separators.
136          * @type DOMNode
137          * @private
138          */
139         _renderUI: function() {
140             this._node = new dijit.layout.AccordionPane({
141                 'title':this._set.getBuildingBlockName(),
142                 'class':'paletteElement'
143             });
144 
145             this._contentNode = new Element('div', {
146                 'class':'paletteContent'
147             });
148 
149             this._searchBox = new PaletteSearchBox();
150             this._searchBox.addEventListener(this._filterComponents.bind(this));
151             this._node.setContent(this._searchBox.getDOMNode());
152 
153             this._searchBox.getDOMNode().insert({after:this._contentNode});
154         },
155 
156         /**
157          * Updates the palette components from building blocks by querying its building block factory.
158          * @private
159          */
160         _updateComponents: function() {
161             var type = this._set.getBuildingBlockType();
162             var descs = $A(this._set.getBuildingBlocks());
163             if (!GVS.getUser().getCatalogueMagic() ||
164                 type == Constants.BuildingBlock.DOMAIN_CONCEPT) {
165                     descs = descs.sortBy(function(desc) {
166                         return desc.getOrder();
167                     });
168             }
169 
170             // Emptying the current components
171             if (this._components.values().size() < descs.size() &&
172                 ! GVS.getUser().getCatalogueMagic()) {
173                 Utils.showMessage("Building blocks loaded", {'hide': true});
174             }
175             this._components = new Hash();
176             this._contentNode.update("");
177 
178 
179             var component;
180             var lastComponent;
181             for (var i=0, desc; desc = descs[i]; i++) {
182                 component = this._components.get(desc.uri);
183                 if (!component) {
184                     component = this._buildComponentFor(desc);
185                     if (lastComponent) {
186                         lastComponent.insert({after: component.getNode()});
187                     } else {
188                         this._contentNode.appendChild(component.getNode())
189                     }
190                     component.getNode().insert({after: new Element("div", {"class": "paletteSeparator"})});
191                 }
192                 lastComponent = component.getNode().next();
193             }
194 
195             this._filterComponents();
196     },
197 
198     /**
199      * Hiden component if not match filter
200      */
201     _filterComponents: (function() {
202         function matchFacts(svalue, facts){
203             if (!facts) { return false }
204             return facts.any(function(fact){
205                 return fact.textContent.toLowerCase().match(svalue)
206             });
207         }
208 
209         return function(input) {
210             var svalue = this._searchBox.getValue().toLowerCase();
211             var slots = this._contentNode.select('.slot');
212             for(var i = 0; i < slots.length; i++) {
213                 var slot = slots[i];
214                 if (svalue.blank() ||
215                     slot.select('.slotTitle').first().textContent.toLowerCase().match(svalue)||
216                     matchFacts(svalue, slot.select('.contentLayer'))
217                 ) {
218                     slot.show();
219                 } else {
220                     slot.hide();
221                 }
222             }
223         }
224     })(),
225 
226     /**
227      * Build a new component
228      */
229     _buildComponentFor: function(/** BuildingBlockDescription */ desc) {
230         var component = null;
231 
232         switch(this._set.getBuildingBlockType()) {
233             case Constants.BuildingBlock.SCREEN:
234                 component = new ScreenComponent(desc, this._dropZones, this._inferenceEngine);
235                 break;
236 
237             case Constants.BuildingBlock.DOMAIN_CONCEPT:
238                 component = new DomainConceptComponent(desc, this._dropZones, this._inferenceEngine);
239                 break;
240 
241             case Constants.BuildingBlock.FORM:
242                 component = new FormComponent(desc, this._dropZones, this._inferenceEngine);
243                 break;
244 
245             case Constants.BuildingBlock.RESOURCE:
246                 component = new ResourceComponent(desc, this._dropZones, this._inferenceEngine);
247                 break;
248 
249             case Constants.BuildingBlock.OPERATOR:
250                 component = new OperatorComponent(desc, this._dropZones, this._inferenceEngine);
251                 break;
252 
253             default:
254                 throw "Unsupported building block type";
255         }
256         this._components.set(desc.uri, component);
257 
258         return component;
259     }
260 });
