 74     /**
 75      * Creates the palette components from resources by querying a resource factory.
 76      * @type {ResourceDescription[]}
 77      * @private
 78      */
 79     _createComponents: function(resourceFactory) {
 80         var descs = resourceFactory.getResourceDescriptions();
 81         var components = [];
 82         $A(descs).each(
 83                 function(desc) { components.push(desc.createPaletteComponent()); } 
 84         );
 85         return components;
 86     },

 89     /**

 91      * @type DOMNode
 92      * @private
 93      */
 94     _renderContent: function() {
 95 		var content   = new Element("div", {"class": "paletteContent", "id" : this._id + "Content"});
 96         var separator = new Element("div", {"class": "paletteSeparator"});
 97         $A(this._components).each(
 98                 function(component) {
 99                     content.appendChild(component.getNode());
100                     content.appendChild(separator.cloneNode(false));
101                 }
102         );
103         return content;
104     }
