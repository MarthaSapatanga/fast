262     /**
263      * Inits the catalogue population
264      * @private
265      */
266     _start: function() {
267         this._paletteController = new PaletteController(this._buildingBlockSets,
268                                         this._areas.values(), this._inferenceEngine);
269 
270         this._renderPaletteArea();
271         this._configureToolbar();
272         Utils.showMessage("Loading building blocks");
273         var paletteStatus = this._getEmptyPalette();
274 
275 
276         // Start retrieving data
277         this._inferenceEngine.findCheck(
278                 [],
279                 paletteStatus,
280                 this._tags,
281                 'reachability',
282                 this._findCheckCallback.bind(this)
283         );
284         var domainConceptSet = this._buildingBlockSets.detect(function(set) {
285             return set.constructor == DomainConceptSet;
286         })
287         if (domainConceptSet) {
288             domainConceptSet.startRetrievingData();
289         }
290         if (this._description.getId() == null) {
291             this._setDirty(true);
292         }
293     },

439      /**
440      * This function returns the data array containing all the
441      * facts belonging to the screenflow
442      * @type Array
443      */
444     _getAllFacts: function() {
445         var resultHash = new Hash();
446         var instanceList = this._description.getCanvasInstances().
447                                 concat(this._description.getConditionInstances());
448         instanceList.each(function(instance){
449             if (instance.constructor != PrePostInstance) {
450                 var preReachability = this._inferenceEngine.getPreconditionReachability(
451                             instance.getUri());
452                 var preconditions = instance.getPreconditionTable(preReachability);
453                 preconditions.each(function(pre) {
454                     if (!resultHash.get(pre[2]/*The uri of the pre*/)) {
455                         resultHash.set(pre[2], pre);
456                     }
457                 });
458 
459                 var postReachability = this._inferenceEngine.isReachable(
460                             instance.getUri());
461                 var postconditions = instance.getPostconditionTable(postReachability);
462                 postconditions.each(function(post) {
463                     if (!resultHash.get(post[2]/*The uri of the post*/)) {
464                         resultHash.set(post[2], post);
465                     }
466                 });
467             } else {
468                 //PrePostInstance
469                 var factInfo = instance.getConditionTable();
470                 if (!resultHash.get(factInfo[2]/*The uri of the pre/post*/)) {
471                     resultHash.set(factInfo[2], factInfo);
472                 }
473             }
474         }.bind(this));
475         return resultHash.values();
476     },

481      * @overrides
482      * @private

484     _closeDocument: function() {
485         var removeFromServer = false;
486         if (this._description.getId()) {
487             if (this._isDirty) {
488                 this._pendingOperation = this._closeDocument.bind(this);
489                 this._save(false);
490                 return false;
491             }
492         } else {
493             removeFromServer = true;
494         }
495       
496         this._description.getCanvasInstances().each(function(instance) {
497             instance.destroy(removeFromServer);
498         }.bind(this));
499 
500         this._description.getConditionInstances().each(function(instance) {
501             instance.destroy(removeFromServer);
502         }.bind(this));
503 
504         GVS.getDocumentController().closeDocument(this._tabId);
505     },

545     /**
546      * Starts the process of saving the screenflow
547      * @private
548      * @override
549      */
550     _save: function(/** Boolean (Optional) */ _showMessage) {
551         var showMessage = Utils.variableOrDefault(_showMessage, true);
552         if (showMessage) {
553             Utils.showMessage("Saving " + this._typeName);
554         }
555         if (this._description.getId() == null) {
556             // Save it for the first time
557             PersistenceEngine.sendPost(this._getSaveUri(), null, "buildingblock=" + Object.toJSON(this._description.toJSON()),
558                                        this, this._onSaveSuccess, this._onSaveError);
559         } else {
560             var uri = URIs.buildingblock + this._description.getId();
561             PersistenceEngine.sendUpdate(uri, null, "buildingblock=" + Object.toJSON(this._description.toJSON()),
562                                       this, this._onSaveSuccess, this._onSaveError);
563         }
564     },

567     /**
568      * On success handler when saving
569      * @private
570      */
571     _onSaveSuccess: function(/** XMLHttpRequest */ transport) {
572         this._setDirty(false);
573         Utils.showMessage("Saved", {
574             'hide': true
575         });
576         if (this._description.getId() == null) {
577             var data = JSON.parse(transport.responseText);
578             this._description.addProperties({'id': data.id,
579                                             'version': data.version,
580                                             'creationDate': data.creationDate});
581         }
582         if (this._pendingOperation) {
583             var operation = this._pendingOperation;
584             this._pendingOperation = null;
585             operation();
586         }
587     },
588 
589     /**
590      * On save error: the screen already exists
591      * @private
592      */
593     _onSaveError: function(/** XMLHttpRequest */ transport) {
594         // TODO: think about what to do when a screen cannot be saved
595         // (problems with wrong versions)
596         if (!this._pendingOperation) {
597             Utils.showMessage("Cannot save " + this._typeName, {
598                 'hide': true,
599                 'error': true
600             });
601         } else {
602             var operation = this._pendingOperation;
603             this._pendingOperation = null;
604             this._setDirty(false);
605             operation();
606         }

609     /**
610      * Creates the instances coming from a list of uris
611      * @private
612      */
613     _createInstances: function(/** BuildingBlockFactory */ factory, /** Array */ buildingBlocks,
614                                 /** Area */ area) {
615         buildingBlocks.each(function(buildingBlock) {
616             // More than one buildingblock for a given uri
617             var ids = this._canvasCache.getIds(buildingBlock.uri);
618             ids.each(function(id) {
619                 var instance = factory.getInstance(buildingBlock, this._inferenceEngine);
620                 instance.setId(id);
621                 var position = this._canvasCache.getPosition(id);
622                 instance.onFinish(true, position);
623                 var dropNode = area.getNode();
624                 $("main").appendChild(instance.getView().getNode());
625                 var effectivePosition = Geometry.adaptInitialPosition(dropNode,
626                                         instance.getView().getNode(), position);
627                 $("main").removeChild(instance.getView().getNode());
628                 this._drop(area, instance, effectivePosition, true);
629             }.bind(this));
630         }.bind(this));
631     },
