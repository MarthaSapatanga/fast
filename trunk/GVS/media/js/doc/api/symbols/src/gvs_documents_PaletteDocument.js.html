528      /**
529      * This function returns the data array containing all the
530      * facts belonging to the screenflow
531      * @type Array
532      */
533     _getAllFacts: function() {
534         var resultHash = new Hash();
535         var instanceList = this._description.getCanvasInstances().
536                                 concat(this._description.getConditionInstances());
537         instanceList.each(function(instance){
538             if (instance.constructor != PrePostInstance) {
539                 var preReachability = this._inferenceEngine.getPreconditionReachability(
540                             instance.getUri());
541                 var preconditions = instance.getPreconditionTable(preReachability);
542                 preconditions.each(function(pre) {
543                     if (!resultHash.get(pre[2]/*The uri of the pre*/)) {
544                         resultHash.set(pre[2], pre);
545                     }
546                 });
547 
548                 var postReachability = this._inferenceEngine.isReachable(
549                             instance.getUri());
550                 var postconditions = instance.getPostconditionTable(postReachability);
551                 postconditions.each(function(post) {
552                     if (!resultHash.get(post[2]/*The uri of the post*/)) {
553                         resultHash.set(post[2], post);
554                     }
555                 });
556             } else {
557                 //PrePostInstance
558                 var factInfo = instance.getConditionTable();
559                 if (!resultHash.get(factInfo[2]/*The uri of the pre/post*/)) {
560                     resultHash.set(factInfo[2], factInfo);
561                 }
562             }
563         }.bind(this));
564         return resultHash.values();
565     },

591     /**
592      * Effectively close the document
593      * @private
594      */
595     _effectiveCloseDocument: function(/** String */ status) {
596 
597         switch (status) {
598             case ConfirmDialog.SAVE:
599                 this._pendingOperation = this._effectiveCloseDocument.bind(this);
600                 this._save(false);
601                 break;
602 
603             case ConfirmDialog.CANCEL:
604                 break;
605 
606             case ConfirmDialog.DISCARD:
607             default:
608                 var removeFromServer = false;
609                 if (!this._description.getId()) {
610                     removeFromServer = true;
611                 }
612                 this._description.getCanvasInstances().each(function(instance) {
613                     instance.destroy(removeFromServer, true);
614                 }.bind(this));
615 
616                 this._description.getConditionInstances().each(function(instance) {
617                     instance.destroy(removeFromServer, true);
618                 }.bind(this));
619 
620                 GVS.getDocumentController().closeDocument(this._tabId);
621 
622         }
623     },

668     /**
669      * Starts the process of saving the screenflow
670      * @private
671      * @override
672      */
673     _save: function(/** Boolean (Optional) */ _showMessage) {
674         var showMessage = Utils.variableOrDefault(_showMessage, true);
675         if (showMessage) {
676             Utils.showMessage("Saving " + this._typeName);
677         }
678         var buildingblock = $H(this._description.toJSON())
679                             .merge($H(this._getExtraProperties()));
680         var params = {'buildingblock':
681             Object.toJSON(buildingblock.toObject())
682         };
683         params = Object.extend(params, this._getExtraParams());
684         var context = {
685             "mine": this,
686             "showMessage": showMessage
687         };
688         if (this._description.getId() == null) {
689             // Save it for the first time
690             PersistenceEngine.sendPost(this._getSaveUri(), params, null,
691                                        context, this._onSaveSuccess, this._onSaveError);
692         } else {
693             var uri = URIs.buildingblock + this._description.getId();
694             PersistenceEngine.sendUpdate(uri, params, null,
695                                       context, this._onSaveSuccess, this._onSaveError);
696         }
697     },

728     /**
729      * On success handler when saving
730      * @private
731      */
732     _onSaveSuccess: function(/** XMLHttpRequest */ transport) {
733         this.mine._setDirty(false);
734         if (this.showMessage) {
735                 Utils.showMessage("Saved", {
736                     'hide': true
737                 });
738         }
739         this.mine._updatePanes();
740         if (this.mine._description.getId() == null) {
741             var data = JSON.parse(transport.responseText);
742             this.mine._description.addProperties({'id': data.id,
743                                             'version': data.version,
744                                             'creationDate': data.creationDate});
745         }
746         if (this.mine._pendingOperation) {
747             var operation = this.mine._pendingOperation;
748             this.mine._pendingOperation = null;
749             operation();
750         }
751     },

774     /**
775      * Creates the instances coming from a list of uris
776      * @private
777      */
778     _createInstances: function(/** BuildingBlockFactory */ factory, /** Array */ buildingBlocks,
779                                 /** Area */ area) {
780         buildingBlocks.each(function(buildingBlock) {
781             // More than one buildingblock for a given uri
782             var ids = this._canvasCache.getIds(buildingBlock.uri);
783             ids.each(function(id) {
784                 var instance = factory.getInstance(buildingBlock, this._inferenceEngine);
785                 instance.setId(id);
786                 instance.setParams(this._canvasCache.getParams(id));
787                 var title = this._canvasCache.getTitle(id);
788                 if (title != null && title != "") {
789                     instance.setTitle(title);
790                 }
791                 var caption = this._canvasCache.getCaption(id);
792                 if (caption != null && caption != "") {
793                     instance.setCaption(caption);
794                 }
795                 var position = this._canvasCache.getPosition(id);
796                 var originalUri = this._canvasCache.getOriginalUri(id);
797                 if (originalUri) {
798                     instance.setOriginalUri(originalUri);
799                 }
800                 instance.onFinish(true, position);
801                 var dropNode = area.getNode();
802                 $("main").appendChild(instance.getView().getNode());
803                 var effectivePosition = Geometry.adaptInitialPosition(dropNode,
804                                         instance.getView().getNode(), position);
805                 $("main").removeChild(instance.getView().getNode());
806                 this._drop(area, instance, effectivePosition, this._canvasCache.getOrientation(id), true);
807             }.bind(this));
808         }.bind(this));
809     },
