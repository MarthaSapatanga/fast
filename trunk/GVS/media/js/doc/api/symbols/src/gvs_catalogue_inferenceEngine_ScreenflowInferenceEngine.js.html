 16     /**
 17      * This function calls the catalogue to create a plan for a given screen
 18      */
 19     getPlans: function(/** Array */ canvas, /** Array */ goal,
 20                         /** Function */ handler) {
 21         var body = {
 22             "goal": goal,
 23             "canvas": canvas
 24         };
 25         var bodyJSON = Object.toJSON(body);
 26         PersistenceEngine.sendPost(URIs.cataloguePlanner, null, bodyJSON, {'handler': handler},
 27                                     this._planOnSuccess, this._onError);
 28     },

 32     /**
 33      * plan onSuccess
 34      * @private
 35      */
 36     _planOnSuccess: function(transport) {
 37         var result = JSON.parse(transport.responseText);
 38         this.handler(result);
 39     },

 42     /**
 43      * Creates a body to be sent in an AJAX call to the
 44      * catalogue
 45      * @private
 46      * @overrides
 47      * @type String
 48      */
 49     _constructBody: function(/**Array*/ canvas, /** Array*/ elements,
 50                     /** Array */ tags,
 51                     /** String*/ criteria) {
 52 
 53         /*var domainContext = {
 54             'tags': tags,
 55             'user': GVS.getUser().getUserName()
 56         };*/
 57 
 58         var domainContext = {
 59             'tags': tags.collect(function(tag){ return tag.label['en-gb']}),
 60             'user': GVS.getUser().getUserName()
 61         }
 62         var body = {
 63             'domainContext': domainContext,
 64             'criterion': criteria
 65         };
 66         body = Object.extend(body, elements);
 67         body.canvas.screens = canvas;
 68         return Object.toJSON(body);
 69     },

 71     /**
 72      * Gets the uri for a given operation
 73      * @private
 74      * @overrides
 75      */
 76     _getUri:function (/** String */ operation) {
 77         switch(operation) {
 78             case "findCheck":
 79                 return URIs.catalogueScreenflowFindCheck;
 80                 break;
 81             case "check":
 82                 return URIs.catalogueScreenflowCheck;
 83                 break;
 84             default:
 85                 return "";
 86         }
 87     },
 88 
 89     /**
 90      * onSuccess callback
 91      * @private
 92      * @overrides
 93      */
 94     _findCheckOnSuccess: function(/**XMLHttpRequest*/ transport){
 95         var result = JSON.parse(transport.responseText);
 96         if (result.palette) {
 97             var paletteElements = result.palette;
 98         } else {
 99             var paletteElements = [];
100         }
101 
102         var allElements = paletteElements.clone();
103 
104         result.canvas.screens.each(function(bb) {
105             found = allElements.detect(function(element) {
106                 return (element.uri == bb.uri);
107             });
108             if (!found) {
109                 allElements.push(bb);
110             }
111         }, this);
112 
113 
114         this.mine._updateReachability(allElements);
115 
116         // Notifying about new uris
117         var screenURIs = new Array();
118         $A(paletteElements).each(function(element) {
119            screenURIs.push(element.uri);
120         });
121 
122         this.callback(screenURIs, result);
123     },
124 
125     /**
126      * onSuccess callback
127      * @private
128      * @overrides
129      */
130     _checkOnSuccess: function(transport){
131         var result = JSON.parse(transport.responseText);
132         var elements = result.palette.concat(result.canvas.screens).uniq();
133 
134         this.mine._updateReachability(elements);
135         this.callback();
136     }
