1 var PaletteController = Class.create(
  2     /** @lends PaletteController.prototype */ {
  3 
  4     /**
  5      * Manages a set of palettes.
  6      * @constructs
  7      */
  8     initialize: function(/** Array */ buildingBlockSets,
  9             /** DropZone */ dropZone, /** InferenceEngine */ inferenceEngine) {
 10         
 11         
 12         /**
 13          * List of available palettes
 14          * @type {Hash}
 15          * @private @member
 16          */
 17         this._palettes = new Hash();
 18         
 19         /**
 20          * AccordionContainer which contains the different palettes
 21          * @type AccordionContainer
 22          * @private @member
 23          */
 24         this._node = new dijit.layout.AccordionContainer({
 25                 "class":"palettePane",
 26                 "region":"left",
 27                 "minSize":"170",
 28                 "maxSize":"300",
 29                 "splitter":"true",
 30                 "livesplitters":"false",
 31                 "style":"width:220px;"
 32                 });
 33          
 34         //Create all the document necessary palettes
 35         $A(buildingBlockSets).each (function(set) {
 36            var palette = new Palette (set, dropZone, inferenceEngine); 
 37            this._palettes.set(set.getBuildingBlockType(), palette);
 38            this._node.addChild(palette.getNode());
 39         }.bind(this));
 40     },
 41 
 42     // **************** PUBLIC METHODS **************** //
 43     getPalette: function (/** String */ type) {
 44         return this._palettes.get(type);
 45     },
 46     
 47     getNode: function() {
 48         return this._node;
 49     },
 50     
 51     /**
 52      * Starts the data retrieval from the catalogue.
 53      */
 54     startRetrievingData: function() {
 55         this._palettes.each(function(pair) {
 56             pair.value.startRetrievingData(); 
 57         });
 58     },
 59     
 60     /**
 61      * All uris of all the components (of all the palettes)
 62      */
 63     getComponentUris: function() {

 65     }

 67     // **************** PRIVATE METHODS **************** //
