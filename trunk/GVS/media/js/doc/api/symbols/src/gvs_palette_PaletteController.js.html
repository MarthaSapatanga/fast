1 var PaletteController = Class.create(
  2     /** @lends PaletteController.prototype */ {
  3 
  4     /**
  5      * Manages a set of palettes.
  6      * @constructs
  7      * @param buildingBlocks
  8      *      Is an array of objects.
  9      *      Each objects contains the set for the building block
 10      *      and its associated drop zone
 11      */
 12     initialize: function(/** Array */ buildingBlockSets, /** Array */ dropZones,
 13         /** InferenceEngine */ inferenceEngine) {
 14 
 15 
 16         /**
 17          * List of available palettes
 18          * @type {Hash}
 19          * @private @member
 20          */
 21         this._palettes = new Hash();
 22 
 23         /**
 24          * AccordionContainer which contains the different palettes
 25          * @type AccordionContainer
 26          * @private @member
 27          */
 28         this._node = new dijit.layout.AccordionContainer({
 29                 "class":"palettePane",
 30                 "region":"left",
 31                 "minSize":"170",
 32                 "maxSize":"300",
 33                 "splitter":"true",
 34                 "livesplitters":"false",
 35                 "style":"width:260px;"
 36                 });
 37 
 38         //Create all the document necessary palettes
 39         $A(buildingBlockSets).each (function(set) {
 40             var validDropZones = new Array();
 41             dropZones.each(function(dropZone) {
 42                 if (dropZone.accepts().include(set.getBuildingBlockType())) {
 43                     validDropZones.push(dropZone);
 44                 }
 45             });
 46             var palette = new Palette (set, validDropZones, inferenceEngine);
 47             this._palettes.set(set.getBuildingBlockType(), palette);
 48             this._node.addChild(palette.getNode());
 49         }.bind(this));
 50     },
 51 
 52     // **************** PUBLIC METHODS **************** //
 53     getPalette: function (/** String */ type) {
 54         return this._palettes.get(type);
 55     },
 56 
 57     getNode: function() {
 58         return this._node;
 59     },
 60 
 61     /**
 62      * All uris of all the components (of all the palettes)
 63      */
 64     getComponentUris: function(/** String */ palette) {
 65         return this._palettes.get(palette).getComponentUris();
 66     }
 67 
 68     // **************** PRIVATE METHODS **************** //
 69 
 70 });
