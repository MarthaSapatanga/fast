1 var PaletteController = Class.create(
  2     /** @lends PaletteController.prototype */ {
  3 
  4     /**
  5      * Manages a set of palettes.
  6      * @constructs
  7      */
  8     initialize: function(/** AbstractDocument */ parent) {
  9         /**
 10          * List of available palettes
 11          * @type {Hash}
 12          * @private
 13          */
 14         this._palettes = new Hash();
 15         
 16          /**
 17          * Document which contains the palette
 18          * @type AbstractDocument
 19          * @private
 20          */       
 21         this._parent = parent;
 22         
 23         this._containerNode = null;
 24         
 25         var uidGenerator = UIDGeneratorSingleton.getInstance();
 26         this._paletteId = uidGenerator.generate("palette");
 27         
 28         var mine = this;
 29         
 30         //Create all the document necessary palettes
 31         this._parent.getValidBuildingBlocks().each (function(buildingBlock){
 32            var palette = new Palette (buildingBlock, mine._parent); 
 33            mine._palettes.set(buildingBlock, palette);
 34            mine.getNode().addChild(palette.getNode());
 35         });
 36     },
 37 
 38     // **************** PUBLIC METHODS **************** //
 39     /**
 40      * This function retrieve remote information 
 41      * from the catalogue based on a given domain context
 42      * 
 43      */
 44     populateBuildingBlocks: function (/** Array */ domainContext){
 45         this._palettes.each (function (pair){
 46             pair.value.populateBuildingBlocks(domainContext);
 47         });
 48     },
 49 
 50     /**
 51      * Updates each palette
 52      */
 53     updatePalettes: function () {
 54         $H(this._palettes).each (function (pair){
 55             pair.value.updateComponents();
 56         });
 57     },
 58     
 59     /**
 60      * Paints each palette from each factory
 61      */
 62     paintPalettes: function () {
 63         $H(this._palettes).each (function (pair){
 64             pair.value.paintComponents();
 65         });
 66     },
 67 
 68     getPalette: function (/** String */ type) {
 69         return this._palettes.get(type);
 70     },
 71     
 72     getNode: function() {
 73         if(this._containerNode == null){
 74             var palettePane = new dijit.layout.AccordionContainer({
 75                 "id":this._paletteId,
 76                 "class":"palettePane",
 77                 "region":"left",
 78                 "minSize":"170",
 79                 "maxSize":"300",
 80                 "splitter":"true",
 81                 "livesplitters":"false",
 82                 "style":"width:220px;"
 83                 });
 84             this._containerNode = palettePane;
 85         }
 86         return this._containerNode;
 87     },
 88     
 89     /**
 90      * Updates the reachability of each palette component
 91      */
 92     updateReachability: function(/** map id->value*/screenList){
 93         //TODO: Currently, only the screen palette is updated...
 94         var screens = this.getPalette(Constants.BuildingBlock.SCREEN).getComponents();
 95         for (var i = 0; i < screens.length; i++) {
 96             for (var j = 0; j < screenList.length; j++) {
 97                 if (screens[i].getBuildingBlockDescription().uri == screenList[j].uri) {
 98                     if (screenList[j].reachability == true) {
 99                         screens[i].getBuildingBlockDescription().satisfeable = true;
100                         break;
101                     }
102                     else {
103                         screens[i].getBuildingBlockDescription().satisfeable = false;
104                     }
105                 }
106             }
107             screens[i].colorize();
108         }
109     }
110 
111     // **************** PRIVATE METHODS **************** //
112     
113 });
