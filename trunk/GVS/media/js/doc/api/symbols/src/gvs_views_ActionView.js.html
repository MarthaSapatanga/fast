1 var ActionView = Class.create(
  2     /** @lends ActionView.prototype */ {
  3 
  4     /**
  5      * Actions graphical representation
  6      * @constructs
  7      */ 
  8     initialize: function(/** Object */ description) {
  9 
 10         /**
 11          * @type DOMNode
 12          * @private
 13          */
 14         this._node = null;
 15 
 16         /**
 17          * Condition Icons
 18          * @type Hash
 19          * @private
 20          */
 21         this._preIcons = new Hash();
 22         
 23         this._node = new Element("div", {
 24             "class": "action"
 25         });
 26         
 27         var title = new Element("div", {
 28             "class": "title"
 29         }).update(description.name);
 30         this._node.appendChild(title);
 31         
 32         var preArea = new Element("div",{
 33             "class": "preArea"
 34         });
 35         description.preconditions.each(function(pre) {
 36             var fact = FactFactory.getFactIcon(pre, "embedded");
 37             this._preIcons.set(pre.id, fact);
 38             preArea.appendChild(fact.getNode());
 39         }.bind(this));
 40         this._node.appendChild(preArea);       
 41         
 42     },
 43     
 44     // **************** PUBLIC METHODS **************** //
 45     
 46     getNode: function() {
 47         return this._node;        
 48     },
 49     
 50     /**
 51      * This function returns the domNode of the condition that has
 52      * the id passed as parameter
 53      * @type DOMNode
 54      */
 55     getConditionNode: function(/** String */ id) {
 56         return this._preIcons.get(id).getNode();
 57     },
 58 
 59     /**
 60      * Destroys the action node
 61      */
 62     destroy: function () {
 63         // Let the garbage collector to do its job
 64         this._preIcons = null;
 65         this._node = null;
 66     },
 67 
 68     setReachability: function(/** Object */ actionData) {
 69         var satisfeable = actionData.satisfied;
 70         Utils.setSatisfeabilityClass(this._node, satisfeable);
 71         actionData.preconditions.each(function(preData){
 72             Utils.setSatisfeabilityClass(this._preIcons.get(preData.id).getNode(),
 73                                         preData.satisfied);
 74         }.bind(this));
 75     }
 76     
 77     // **************** PRIVATE METHODS **************** //
 78 });
