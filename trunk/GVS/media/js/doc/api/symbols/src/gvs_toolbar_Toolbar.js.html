1 var Toolbar = Class.create( /** @lends Toolbar.prototype */ {
  2     /**
  3      * The toolbar itself
  4      * 
  5      * @constructs
  6      */ 
  7     initialize: function() {
  8         /** 
  9          * @type dijit.dialog.Toolbar
 10          * @private @member
 11          */
 12         this._toolbar = dijit.byId("toolbar");
 13         
 14         /**
 15          * List of toolbar models
 16          * @type Array
 17          * @private @member
 18          */
 19         this._models = new Array();
 20         
 21         /**
 22          * List of section elements.
 23          * Each section have another array with the Widget of the buttons plus a 
 24          * separator on the first position (index 0).
 25          * 
 26          * @type Array
 27          * @private @member
 28          */
 29         this._sections = new Array();
 30         this._sections[0] = new Array();
 31         this._sections[0][0] = dijit.byId("firstSeparator");
 32     },
 33     
 34     /**
 35      * Sets the model for the toolbar section on the given position.
 36      * ToolbarModel must provide objects implementing the following interface:
 37      * 
 38      *   interface ToolbarElement
 39      *      Widget getWidget()  // dojo widget
 40      */
 41     setModel: function(/** Number */position, /** ToolbarModel */ model) {
 42         this._removeModel(position);
 43         
 44         if (model) {
 45             var toolbarElements = model.getToolbarElements();
 46             
 47             if (toolbarElements.size() > 0) {
 48                 this._models[position] = model;
 49                 
 50                 this._initSection(position);
 51                 
 52                 toolbarElements.each(function (element) {
 53                     this._sections[position].push(element.getWidget());
 54                 }.bind(this));
 55             }
 56         }
 57         
 58         this._refreshToolbar();
 59     },
 60     
 61     // ************************ PRIVATE METHODS ************************* //
 62     
 63     _removeModel: function(/** Number */ position) {
 64         if (this._models[position]) {
 65             this._models[position] = null;
 66             this._sections[position].clear();
 67         }
 68     },
 69     
 70     _initSection: function(/** Number */ position) {
 71         if (!this._sections[position]) {
 72             this._sections[position] = new Array();
 73         }
 74         if (!this._sections[position][0]) {
 75             this._sections[position][0] = new dijit.ToolbarSeparator();
 76         }
 77     },
 78     
 79     _refreshToolbar: function() {
 80         this._toolbar.getDescendants().each(function(descendant) {
 81             this._toolbar.removeChild(descendant);
 82         }.bind(this));
 83         
 84         this._sections.each(function (section) {
 85             section.each(function (element) {
 86                 this._toolbar.addChild(element);
 87             }.bind(this));
 88         }.bind(this));
 89     }
 90     
 91 });
