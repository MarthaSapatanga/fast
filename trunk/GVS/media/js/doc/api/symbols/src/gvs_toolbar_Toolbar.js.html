 26 var Toolbar = Class.create( /** @class @lends Toolbar.prototype */ {
 27     /**
 28      * The toolbar itself
 29      *
 30      * @constructs
 31      * @constructor
 32      */
 33     initialize: function() {
 34         /**
 35          * @type dijit.dialog.Toolbar
 36          * @private @member
 37          */
 38         this._toolbar = dijit.byId("toolbar");
 39 
 40         /**
 41          * List of toolbar models
 42          * @type Array
 43          * @private @member
 44          */
 45         this._models = new Array();
 46 
 47         /**
 48          * List of section elements.
 49          * Each section have another array with the Widget of the buttons plus a
 50          * separator on the first position (index 0).
 51          *
 52          * @type Array
 53          * @private @member
 54          */
 55         this._sections = new Array();
 56         this._sections[0] = new Array();
 57     },
 58 
 59     /**
 60      * Sets the model for the toolbar section on the given position.
 61      * ToolbarModel must provide objects implementing the following interface:
 62      *
 63      *   interface ToolbarElement
 64      *      Widget getWidget()  // dojo widget
 65      * @
 66      */
 67     setModel: function(/** Number */position, /** ToolbarModel */ model) {
 68         this._removeModel(position);
 69 
 70         if (model) {
 71             var toolbarElements = model.getToolbarElements();
 72 
 73             if (toolbarElements.size() > 0) {
 74                 this._models[position] = model;
 75 
 76                 this._initSection(position);
 77 
 78                 toolbarElements.each(function (element) {
 79                     this._sections[position].push(element.getWidget());
 80                 }.bind(this));
 81             }
 82         }
 83 
 84         this._refreshToolbar();
 85     },
 86 
 87     // ************************ PRIVATE METHODS ************************* //
 88 
 89     /**
 90      * Removes the model from the given position
 91      * @private
 92      */
 93     _removeModel: function(/** Number */ position) {
 94         if (this._models[position]) {
 95             this._models[position] = null;
 96             this._sections[position].clear();
 97         }
 98     },
 99 
100     /**
101      * Initializes a section
102      * @private
103      */
104     _initSection: function(/** Number */ position) {
105         if (!this._sections[position]) {
106             this._sections[position] = new Array();
107         }
108         if (position != 0 && !this._sections[position][0]) {
109             this._sections[position][0] = new dijit.ToolbarSeparator();
110         }
111     },
112 
113     /**
114      * Refresh the toolbar interface
115      * Called when something happens (a model has been added or removed)
116      * @private
117      */
118     _refreshToolbar: function() {
119         this._toolbar.getDescendants().each(function(descendant) {
120             this._toolbar.removeChild(descendant);
121         }.bind(this));
122 
123         this._sections.each(function (section) {
124             section.each(function (element) {
125                 this._toolbar.addChild(element);
126             }.bind(this));
127         }.bind(this));
128     }
129 
130 });
