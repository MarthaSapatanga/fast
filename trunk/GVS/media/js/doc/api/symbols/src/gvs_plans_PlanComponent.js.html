 26 var PlanComponent = Class.create(DragSource,
 27     /** @lends PlanComponent.prototype */ {
 28 
 29     /**
 30      * GUI element that represents a Plan element.
 31      * @constructs
 32      * @extends DragSource
 33      */
 34     initialize: function($super,/** Array */ plan, /** DropZone */ dropZone,
 35                         /** InferenceEngine */ inferenceEngine) {
 36         $super();
 37 
 38         /**
 39          * Component and instance Drop zone
 40          * @type DropZone
 41          * @private
 42          */
 43         this._dropZone = dropZone;
 44 
 45         /**
 46          * List of buildingBlock description of the plan
 47          * @type Array
 48          * @private
 49          */
 50         this._plan = plan;
 51 
 52 
 53         /**
 54          * Inference engine
 55          * @type InferenceEngine
 56          * @private
 57          */
 58         this._inferenceEngine = inferenceEngine;
 59 
 60         /**
 61          * Plan view
 62          * @type PlanView
 63          * @private
 64          */
 65         this._view = new PlanView(this._plan);
 66 
 67         /**
 68          * Node of the component.
 69          * @type DOMNode
 70          * @private
 71          */
 72         this._node = this._view.getNode();
 73 
 74 
 75         this.enableDragNDrop(null, [this._dropZone]);
 76     },
 77 
 78     // **************** PUBLIC METHODS **************** //
 79 
 80     /**
 81      * Gets the component root node.
 82      * @type DOMNode
 83      * @public
 84      */
 85     getNode: function() {
 86         return this._node;
 87     },
 88 
 89 
 90     /**
 91      * Returns the node that can be clicked to start a drag-n-drop operation.
 92      * @type DOMNode
 93      * @override
 94      */
 95     getHandlerNode: function() {
 96         return this._node;
 97     },
 98 
 99     /**
100      * Creates a new Plan component to be dragged.
101      * Returned object must have a getNode() method.
102      * @type Object
103      * @override
104      */
105     getDraggableObject: function() {
106         var instance = new PlanInstance (this._plan, this._inferenceEngine);
107         var node = instance.getHandlerNode();
108         dijit.byId("main").domNode.appendChild(node);
109         node.setStyle({
110             'top': this._getContentOffsetTop() + 'px',
111             'left':  this._getContentOffsetLeft() + 'px',
112             'position': 'absolute'
113         });
114         return instance;
115     },
116 
117 
118     // **************** PRIVATE METHODS **************** //
119 
120     /**
121      * Calculates the distance from the window top to the palette component.
122      * @type Integer
123      * @private
124      */
125     _getContentOffsetTop: function() {
126 
127         return this._node.cumulativeOffset().top -
128                 this._node.cumulativeScrollOffset().top;
129     },
130 
131     /**
132      * Calculates the distance from the window left border to the palette
133      * component.
134      * @type Integer
135      * @private
136      */
137     _getContentOffsetLeft: function() {
138 
139        return this._node.cumulativeOffset().left -
140                 this._node.cumulativeScrollOffset().left;
141     }
142 
143 });
