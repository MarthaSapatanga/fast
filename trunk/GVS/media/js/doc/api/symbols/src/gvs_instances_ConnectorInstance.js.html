336     onPropertiesDialogChange: function(event){
337         switch (event.element().name) {
338             case 'fact':
339                 var form = this.getPropertiesDialog().getForm();
340                 var fact = $F(event.element());
341                 if (fact == 'none') {
342                     var i;
343                     for (i = form.factAttr.length - 1; i>=0; i--) {
344                         if(form.factAttr.options[i].value != ''){
345                             form.factAttr.remove(i);
346                         }
347                     }
348                 }
349                 else {
350                     var descs = CatalogueSingleton.getInstance().getBuildingBlockFactory(Constants.BuildingBlock.DOMAIN_CONCEPT).getBuildingBlockDescriptions();
351                     $A(descs).each(function(desc){
352                         if (desc.name == fact) {
353                             var i;
354                             for (i = form.factAttr.length - 1; i >= 0; i--) {
355                                 if (form.factAttr.options[i].value != '') {
356                                     form.factAttr.remove(i);
357                                 }
358                             }
359                             $A(desc.attributes).each(function(attr){
360                                 var option = new Element("option", {
361                                     value: attr
362                                 });
363                                 option.innerHTML = attr;
364                                 form.factAttr.insert(option);
365                             })
366                             form.semantics.setValue(desc.semantics);
367                             form.shortcut.setValue(desc.shortcut);
368                         }
369                     });
370                 }
371                 break;
372             case 'type':
373                 var kind = this.getPropertiesDialog().getForm().kind;
374                 var type = $F(event.element());
375                 var i;
376                 for (i = kind.length - 1; i >= 0; i--) {
377                     if (kind.options[i].value == 'event_slot') {
378                         switch (type.toLowerCase()) {
379                             case 'in':
380                                 kind.options[i].innerHTML = 'Slot';
381                                 break;
382                             case 'out':
383                                 kind.options[i].innerHTML = 'Event';
384                                 break;
385                             default:
386                                 kind.options[i].innerHTML = 'Event/slot';
387                         }
388                     }
389                 }
390                 break;
391             default:
392         }
