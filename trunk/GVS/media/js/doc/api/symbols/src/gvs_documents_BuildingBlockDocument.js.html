 30 var Observable = {
 31     subscribe: function subcribe(object) {
 32         var observers = this._observers || (this._observers = []);
 33         observers.push(object);
 34         return this;
 35     },
 36     emit: function notify() {
 37         var observers = this._observers || (this._observers = []);
 38         for (var i = observers.length; i--;) {
 39             observers[i].update.call(observers[i]);
 40         }
 41         return this;
 42     }
 43 }

 45 function Collection(objects, options) {
 46     options || (options = {});
 47     this.objects = objects || [];
 48     this.model = options.model;
 49     this.primaryKey = this.model.key || 'id';
 50 }
 51 Object.extend(Collection.prototype, Observable);
 52 Collection.prototype.find = function find(value) {
 53     var object,
 54         key = this.primaryKey,
 55         array = this.objects;
 56 
 57     for (var i = array.length; i--;) {
 58         object = array[i];
 59         if (object[key] === value) {
 60             return object;
 61         }
 62     }
 63     return null;
 64 };
 65 Collection.prototype.insert = function insert(object) {
 66     var key = object[this.primaryKey];
 67     if (this.find(key)) return;
 68     this.objects.push(object);
 69     object._collection = this;
 70     return object;
 71 };
 72 Collection.prototype.remove = function remove(/* objects */) {
 73     var index,
 74         array = this.objects,
 75         result = [];
 76 
 77     for (var i = arguments.length; i--;) {
 78         index = array.indexOf(arguments[i]);
 79         if (index >= 0) {
 80             result.push(array.splice(index, 1));
 81         }
 82     }
 83     return result;
 84 };
 85 Collection.prototype.create = function create(properties) {
 86     var object = (this.model) ?
 87         new this.model(properties):
 88         Object.clone(properties);
 89     return this.insert(object);
 90 };
 91 Collection.prototype.toJSON = function toJOSN() {
 92     return this.objects;
 93 };
 94 Collection.prototype.each = function each(iterator, context) {
 95     return this.objects.each(iterator, context);
 96 };
 97 Collection.prototype.map = function(iterator, context) {
 98     return this.objects.map(iterator, context);
 99 };
100 Collection.prototype.empty = function() {
101     return this.objects.length === 0;
102 };

273 var FormBuilder = (function() {
274     _createLine = function(/** String */ label, /** DOMNode */ inputNode) {
275         var lineNode = new Element('div', {'class': 'line'});
276         if (label) {
277             lineNode.appendChild(new Element ('label').update(label));
278         }
279         lineNode.appendChild(inputNode);
280         return lineNode;
281     };
282 
283     _buildLine = function(line, events) {
284         var lineNode;
285         var inputNode;
286 
287         switch (line.type) {
288             case 'title':
289                 lineNode = new Element ('h3').update(line.value);
290                 break;
291 
292             case 'input':
293                 var input;
294                 if (line.regExp || line.required) {
295                     input = new dijit.form.ValidationTextBox({
296                                     'name' : line.name,
297                                     'value': line.value,
298                                     'regExp': (line.regExp) ? line.regExp : '.*',
299                                     'required': (line.required) ? line.required : false,
300                                     'invalidMessage': (line.message) ? line.message : 'This field cannot be blank'
301                                 });
302                 } else {
303                     input = new dijit.form.TextBox({
304                                     'name' : line.name,
305                                     'value': line.value
306                                 });
307                 }
308                 if (line.disabled) {
309                     input.attr('disabled', line.disabled);
310                 }
311                 inputNode = input.textbox;
312                 lineNode = _createLine(line.label, input.domNode);
313                 break;
314 
315             case 'checkbox':
316                 var checkbox = new dijit.form.CheckBox({
317                             'name' : line.name,
318                             'value': line.value
319                         });
320                 if (line.disabled) {
321                     checkbox.attr('disabled', line.disabled);
322                 }
323 
324                 if (line.checked) {
325                     checkbox.attr('checked', line.checked);
326                 }
327 
328                 inputNode = checkbox.domNode;
329                 lineNode = _createLine(line.label, checkbox.domNode);
330                 break;
331 
332             case 'label':
333                 lineNode = new Element('div', {
334                                 'class': 'line',
335                                 'style': line.style
336                             }).update(line.value);
337                 break;
338 
339             case 'pre':
340                 lineNode = new Element('pre', {
341                                 'class': 'line',
342                                 'style': line.style
343                             }).update(line.value);
344                 break;
345 
346             case 'hidden':
347                 lineNode = new Element('input',{
348                                 'type': 'hidden',
349                                 'name': line.name,
350                                 'value': line.value
351                             });
352                 break;
353 
354             case 'comboBox':
355                 inputNode = new Element('select', {
356                     'name': line.name
357                 });
358 
359                 $A(line.options).each(function(option) {
360                     var optionNode = new Element('option', {
361                          'value': option.value
362                     }).update(option.label);
363 
364                     if (option.value == line.value) {
365                         optionNode.selected = "selected";
366                     }
367 
368                     inputNode.appendChild(optionNode);
369                 });
370 
371                 lineNode = _createLine(line.label, inputNode);
372                 break;
373 
374             case 'textarea':
375                 var textArea = new dijit.form.SimpleTextarea({
376                     'name': line.name
377                 });
378                 textArea.attr("value", line.value);
379 
380                 inputNode = textArea.domNode;
381                 lineNode = _createLine(line.label, inputNode);
382                 break;
383 
384             default:
385                 throw "Unimplemented form field type";
386         }
387         if (inputNode) {
388             $H(events).each(function(pair) {
389                 Element.observe(input, pair.key, pair.value);
390             });
391         }
392         return lineNode;
393     };
394 
395     return {
396         build: function(data, params) {
397             var form = new dijit.form.Form(Object.extend({method:'post'}, params));
398             $A(data).each(function(line) {
399                 var lineNode = _buildLine(line);
400                 form.domNode.appendChild(lineNode);
401             }.bind(this));
402             return form;
403         }
404     }

746     },

748         if (!this._onSave) return;
749

752

754           object = JSON.parse(text);
755         } catch (e) {
756           Utils.showErrorMessage("The properties are not well formed. It will not work");
757         }

766         var model = this._model,
767             object = formatDescription(model),
768             text = JSON.stringify(object, null, 2);
769         this._onSave = false;
770         this._codeEditor.setCode(text);
771         this._onSave = true;
772     },
773     show: function() {
774         return this._element.show();
775     },
776     hide: function() {

849 var formatDescription = (function(){
850     function formatDescription(description) {
851         var object = description.getProperties();
852         delete object.id;
853         delete object.type;
854         delete object.code;
855         delete object.codeInline;
856         delete object.creationDate;
857         object.actions = formatActions(description.actions);
858         object.postconditions = formatConditions(description.postconditions);
859         return object;
860     }
861     function formatActions(actions) {
862         return actions.map(function(action){
863             var object = action.getProperties();
864             object.preconditions = formatConditions(action.preconditions);
865             return object;
866         });
867     }
868     function formatConditions(conditions) {
869         return conditions.map(function(condition) {
870             var object = condition.getProperties();
871             delete object.position;
872             return object;
873         });
874     }
875     return formatDescription;
876 })();

879 var updateProperties = (function() {
880 
881     function updateConditions(objects, conditions) {
882         objects || (objects = []);
883 
884         conditions.each(function(condition) {
885             var props,
886                 object = objects.find(function(e){ return e.id == condition.id });
887 
888             if (!object) {
889                 conditions.remove(condition);
890             } else {
891                 props = condition.getProperties();
892                 object.position = props.position;
893                 for (var key in props) {
894                     if (!(key in object)) {
895                         delete condition[key];
896                     }
897                 }
898                 condition.setProperties(object);
899             }
900         });
901 
902         objects.each(function(properties) {
903             conditions.create(properties)
904         });
905 
906         conditions.emit();
907     }
908 
909     function updateActions(objects, actions) {
910         objects || (objects = []);
911 
912         actions.each(function(action) {
913             var props,
914                 object = objects.find(function(e){ return e.name == action.name; });
915 
916             if (!object) {
917                 actions.remove(action);
918             } else {
919                 props = action.getProperties();
920                 for (var key in props) {
921                     if (!(key in object)) {
922                         delete action[key];
923                     }
924                 }
925                 updateConditions(object.preconditions, action.preconditions);
926                 action.setProperties(object);
927             }
928         });
929 
930         objects.each(function(properties) {
931             actions.create(properties);
932         });
933 
934         actions.emit();
935     }
936 
937     function updateProperties(object, model) {
938 
939         updateActions(object.actions, model.actions);
940         updateConditions(object.postconditions, model.postconditions);
941 
942         props = model.getProperties();
943         object.id           = props.id;
944         object.type         = props.type;
945         object.code         = props.code;
946         object.codeInline   = props.codeInline;
947         object.creationDate = props.creationDate;
948 
949         object.name = (object.name && object.name.trim()) || props.name;
950 
951         for (key in props) {
952             if (!(key in object)) {
953                 delete model[key];
954             }
955         }
956 
957         model.setProperties(object);
958     }
959 
960     return updateProperties;
