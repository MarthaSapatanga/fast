 22     /**

 27         var content = this.getNode();

 29         content.addClassName("welcome");

 31         var logo = new Element('img', {

 33         });
 34         content.appendChild(logo);

 36             update("Welcome to the Gadget Visual Storyboard!");

 45         );

 47             label: "Browse Screenflows...",
 48             onClick: function() {GVS.action("browseScreenflows");}}
 49         );
 50 
 51         var newScreenButton = new dijit.form.Button({
 52             label: "Edit a new Screen",
 53             onClick: function() {GVS.action("newScreen");}}
 54         );
 55         var browseScreensButton = new dijit.form.Button({
 56             label: "Browse Screens...",
 57             onClick: function() {GVS.action("browseScreens");}}
 58         );
 59 
 60         var newBuildingBlockButton = new dijit.form.Button({
 61             label: "Add a Building Block from sources",
 62             onClick: function() {GVS.action("newBuildingBlock");}}
 63         );
 64 
 65         var browseBuildingBlockButton = new dijit.form.Button({
 66             label: "Browse Building Blocks...",
 67             onClick: function() {GVS.action("browseBuildingBlocks");}}
 68         );
 69 
 70         var openWrapperServiceButton = new dijit.form.Button({
 71             label: "Create a resource from a service",
 72             onClick: function() {GVS.action("wrapperService");}}
 73         );
 74         var mediationButton = new dijit.form.Button({
 75             label: "Create operator for concepts",
 76             onClick: function() {GVS.action("mediation");}}
 77         );

 79             label: "Manage Domain Concepts",
 80             onClick: function() {GVS.action("manageConcepts");}}
 81         );

 83         var buttonsContainer = new Element("div");

 85         buttonsContainer.appendChild(newScreenflowButton.domNode);
 86         buttonsContainer.appendChild(new Element("br"));
 87         buttonsContainer.appendChild(openScreenflowButton.domNode);
 88         buttonsContainer.appendChild(new Element("div").update("Screens"));
 89         buttonsContainer.appendChild(newScreenButton.domNode);
