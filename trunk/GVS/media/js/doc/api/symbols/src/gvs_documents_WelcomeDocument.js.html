 26 var WelcomeDocument = Class.create(AbstractDocument,
 27     /** @lends WelcomeDocument.prototype */ {
 28 
 29     /**
 30      * WelcomeDocument is the initial document and allows the user to access
 31      * to the most important GVS features as starting point.
 32      * @constructs
 33      * @extends AbstractDocument
 34      */
 35     initialize: function($super) {
 36         $super("Welcome!");
 37         this._populate ();
 38     },
 39 
 40 
 41 
 42     // **************** PUBLIC METHODS **************** //
 43 
 44     // **************** PRIVATE METHODS **************** //
 45 
 46 
 47     /**
 48      * Constructs the document content.
 49      * @private
 50      */
 51     _populate: function(){
 52         var content = this.getNode();
 53 
 54         content.addClassName("welcome");
 55 
 56         var logo = new Element('img', {
 57             'src': URIs.logoFast
 58         });
 59         content.appendChild(logo);
 60         var documentTitle = new Element ("div", {"class": "documentTitle"}).
 61             update("Welcome to the Gadget Visual Storyboard!");
 62         content.appendChild(documentTitle);
 63         var welcomeIntro = new Element ("div", {"id": "intro"}).
 64             update("Choose your desired action:");
 65         content.appendChild (welcomeIntro);
 66 
 67         var newScreenflowButton = new dijit.form.Button({
 68             label: "Edit a new Screenflow",
 69             onClick: function() {GVS.action("newScreenflow");}}
 70         );
 71         var openScreenflowButton = new dijit.form.Button({
 72             label: "Browse Screenflows...",
 73             onClick: function() {GVS.action("browseScreenflows");}}
 74         );
 75 
 76         var newScreenButton = new dijit.form.Button({
 77             label: "Edit a new Screen",
 78             onClick: function() {GVS.action("newScreen");}}
 79         );
 80         var browseScreensButton = new dijit.form.Button({
 81             label: "Browse Screens...",
 82             onClick: function() {GVS.action("browseScreens");}}
 83         );
 84 
 85         var newBuildingBlockButton = new dijit.form.Button({
 86             label: "Add a Building Block from sources",
 87             onClick: function() {GVS.action("newBuildingBlock");}}
 88         );
 89 
 90         var browseBuildingBlockButton = new dijit.form.Button({
 91             label: "Browse Building Blocks...",
 92             onClick: function() {GVS.action("browseBuildingBlocks");}}
 93         );
 94 
 95         var openWrapperServiceButton = new dijit.form.Button({
 96             label: "Create a resource from a service",
 97             onClick: function() {GVS.action("wrapperService");}}
 98         );
 99         var mediationButton = new dijit.form.Button({
100             label: "Create operator for concepts",
101             onClick: function() {GVS.action("mediation");}}
102         );
103         var manageConceptsButton = new dijit.form.Button({
104             label: "Manage Domain Concepts",
105             onClick: function() {GVS.action("manageConcepts");}}
106         );
107 
108         var buttonsContainer = new Element("div");
109         buttonsContainer.appendChild(new Element("div").update("Screenflows"));
110         buttonsContainer.appendChild(newScreenflowButton.domNode);
111         buttonsContainer.appendChild(new Element("br"));
112         buttonsContainer.appendChild(openScreenflowButton.domNode);
113         buttonsContainer.appendChild(new Element("div").update("Screens"));
114         buttonsContainer.appendChild(newScreenButton.domNode);
115         buttonsContainer.appendChild(new Element("br"));
116         buttonsContainer.appendChild(browseScreensButton.domNode);
117 
118         var wrapper = new Element("div", {
119             "style": "visibility:hidden;"
120         });
121         wrapper.appendChild(newBuildingBlockButton.domNode);
122         wrapper.appendChild(new Element("br"));
123         wrapper.appendChild(browseBuildingBlockButton.domNode);
124         wrapper.appendChild(new Element("br"));
125         wrapper.appendChild(document.createTextNode("---"));
126         if (URIs.wrapperService != "") {
127             wrapper.appendChild(new Element("br"));
128             wrapper.appendChild(openWrapperServiceButton.domNode);
129         }
130         if (URIs.dataMediation != "") {
131             wrapper.appendChild(new Element("br"));
132             wrapper.appendChild(mediationButton.domNode);
133         }
134         if (URIs.factTool != "") {
135             wrapper.appendChild(new Element("br"));
136             wrapper.appendChild(manageConceptsButton.domNode);
137         }
138 
139 
140         var link = new Element("a", {
141             "href": "javascript:"
142         }).update("Show advanced features...");
143         link.observe("click", function(e) {
144             if (wrapper.style.visibility == "hidden") {
145                 wrapper.setStyle({"visibility":"visible"});
146                 link.update("Hide advanced features...");
147             } else {
148                 wrapper.setStyle({"visibility":"hidden"});
149                 link.update("Show advanced features...");
150             }
151         });
152 
153         buttonsContainer.appendChild(new Element("br"));
154         buttonsContainer.appendChild(new Element("br"));
155         buttonsContainer.appendChild(link);
156         buttonsContainer.appendChild(new Element("br"));
157         buttonsContainer.appendChild(new Element("br"));
158         buttonsContainer.appendChild(wrapper);
159 
160         content.appendChild(buttonsContainer);
161     }
162 });
