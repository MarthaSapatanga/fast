 26 var ScreenInstance = Class.create(ComponentInstance,
 27     /** @lends ScreenInstance.prototype */ {
 28 
 29     /**
 30      * Screen instance.
 31      * @constructs
 32      * @extends ComponentInstance
 33      */
 34     initialize: function($super, /**BuildingBlockDescription*/ buildingBlockDescription,
 35              /** InferenceEngine */ inferenceEngine) {
 36         $super(buildingBlockDescription, inferenceEngine);
 37 
 38         /**
 39          * @type PreviewDialog
 40          * @private @member
 41          */
 42         this._dialog = null;
 43 
 44         this._menu.addOption('Preview', function(){
 45             this.showPreviewDialog();
 46         }.bind(this));
 47         if (this.getBuildingBlockDescription().definition != null) {
 48             this._menu.addOption('Clone', function(){
 49                 this.document.cloneElement(this);
 50             }.bind(this));
 51         }
 52         this._menu.addOption('Create a Plan', function(){
 53             this.document.getPlansElement(this);
 54         }.bind(this));
 55 
 56         /**
 57          * Screen caption (optional
 58          * @private
 59          * @type String
 60          */
 61         this._caption = null;
 62     },
 63 
 64     // **************** PUBLIC METHODS **************** //
 65 
 66     /**
 67      * @override
 68      */
 69     getInfo: function() {
 70         var info = new Hash();
 71         var titleDialog = new TitleDialog(this.getTitle(),
 72                                           this.setTitle.bind(this));
 73         var titleArea = new Element("div");
 74         var titleText = new Element("span").update(this.getTitle());
 75         titleArea.appendChild(titleText);
 76         titleArea.appendChild(titleDialog.getButtonNode());
 77         info.set('Title', titleArea);
 78 
 79         info.set('Description', this._buildingBlockDescription.description['en-gb']);
 80 
 81         var captionDialog = new CaptionDialog(this._caption,
 82                                           this.setCaption.bind(this));
 83         var captionArea = new Element("div");
 84         var captionText;
 85         if (this._caption != null) {
 86             captionText = new Element("span");
 87             var showedText = this._caption.stripTags().substring(0,10);
 88             if (showedText != this._caption.stripTags()) {
 89                 // The caption is longer than 10 chars
 90                 showedText += "...";
 91             }
 92             captionText.update(showedText);
 93         } else {
 94             captionText = new Element("span", {
 95                 "class":"triggerInfo"
 96             }).update("No caption set");
 97         }
 98         captionArea.appendChild(captionText);
 99         captionArea.appendChild(captionDialog.getButtonNode());
100         info.set('Screen caption', captionArea);
101 
102         info.set('Tags', this._buildingBlockDescription.tags.collect(function(tag) {
103                 return tag.label['en-gb'];
104             }).join(", "));
105         return info;
106     },
107 
108     /**
109      * Returns the caption of the screen if any
110      */
111     getCaption: function() {
112         return this._caption;
113     },
114 
115     /**
116      * sets the caption
117      */
118     setCaption: function(/** String  */ caption) {
119         this._caption = caption.stripScripts();
120 
121         if (this._listener && this._listener.modified) {
122             this._listener.modified(this);
123         }
124     },
125 
126 
127     /**
128      * This function shows the dialog to change
129      * the instance properties
130      */
131     showPreviewDialog: function () {
132         if (!this._dialog) {
133             this._dialog = new PreviewDialog(this.getTitle(), this._buildingBlockDescription.getPreview());
134         }
135         this._dialog.show();
136     },
137 
138     // **************** PRIVATE METHODS **************** //
139     /**
140      * Creates a new View instance for the component
141      * @type BuildingBlockView
142      * @override
143      */
144     _createView: function () {
145         return new ScreenView(this._buildingBlockDescription);
146     },
147 
148     /**
149      * This function is called when the attached view is dbl-clicked
150      * @private
151      * @override
152      */
153     _onDoubleClick: function (/** Event */ event){
154         this.showPreviewDialog();
155     }
156 });
