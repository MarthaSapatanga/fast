 87     /**
 88      * Creates the data hash to be passed to the
 89      * table
 90      * @private
 91      * @type Array
 92      */
 93     _getConditionList: function(/** String */ type, /** Hash | Boolean */ reachability) {
 94         
 95         if (this._buildingBlockDescription[type].length > 1){ //More than one set of conditions
 96             console.log("OR support not implemented yet");
 97             return null;
 98         }
 99         else {
100             var conditions = this._buildingBlockDescription[type][0];
101             
102             var factFactory = FactFactorySingleton.getInstance();
103     
104             var result = new Array();  
105             $A(conditions).each( 
106                 function(condition) {
107                     var uri = factFactory.getFactUri(condition);
108                     
109                     var fact = factFactory.getFactIcon(condition, "embedded").getNode();
110                     if (reachability.constructor == Hash) {
111                         Utils.setSatisfeabilityClass(fact, reachability.get(uri));
112                     } else {
113                         Utils.setSatisfeabilityClass(fact, reachability);
114                     }
115                     
116                     var description = condition.label['en-gb'];
117                    
118                     result.push([fact, description, uri]);
119                     
120                 }.bind(this)
121             );
122             return result;         
123         }
124     }
