1 var ScreenInstance = Class.create(ComponentInstance,
  2     /** @lends ScreenInstance.prototype */ {
  3 
  4     /**
  5      * Screen instance.
  6      * @constructs
  7      * @extends ComponentInstance
  8      */
  9     initialize: function($super, /**BuildingBlockDescription*/ buildingBlockDescription, 
 10              /** InferenceEngine */ inferenceEngine) {
 11         $super(buildingBlockDescription, inferenceEngine);
 12         
 13         /**
 14          * @type PreviewDialog
 15          * @private @member
 16          */
 17         this._dialog = null;
 18     },
 19 
 20     // **************** PUBLIC METHODS **************** //
 21 
 22     /**
 23      * Somehow something the user can comprehend
 24      * @override
 25      */
 26     getTitle: function() {
 27         return this._buildingBlockDescription.label['en-gb'];    
 28     },
 29     /**
 30      * @override
 31      */
 32     getInfo: function() {
 33         var info = new Hash();
 34         info.set('Title', this._buildingBlockDescription.label['en-gb']);
 35         info.set('Description', this._buildingBlockDescription.description['en-gb']);
 36         info.set('Tags', this._buildingBlockDescription.tags.collect(function(tag) {
 37                 return tag.label['en-gb'];
 38             }).join(", "));
 39         return info;
 40     },
 41     
 42     /**
 43      * This function shows the dialog to change
 44      * the instance properties
 45      */
 46     showPreviewDialog: function () {
 47         if (!this._dialog) {
 48             this._dialog = new PreviewDialog(this.getTitle(), this._buildingBlockDescription.getPreview());
 49         }
 50         this._dialog.show();        
 51     },
 52     
 53     /**
 54      * This function returns a list with all the 
 55      * preconditions of the instance, 
 56      * ready to be set in the FactPane
 57      * @type Array
 58      */
 59     getPreconditionTable: function(/** Hash */ reachability) {
 60         return this._getConditionList("preconditions", reachability);    
 61     },
 62 
 63     // **************** PRIVATE METHODS **************** //
 64     /**
 65      * Creates a new View instance for the component
 66      * @type BuildingBlockView
 67      * @override
 68      */
 69     _createView: function () {
 70         return new ScreenView(this._buildingBlockDescription);
 71     },
 72 
 73     /**
 74      * This function is called when the attached view is dbl-clicked
 75      * @private
 76      * @override
 77      */
 78     _onDoubleClick: function (/** Event */ event){
 79         this.showPreviewDialog(); 
 80     } 
 81 });
