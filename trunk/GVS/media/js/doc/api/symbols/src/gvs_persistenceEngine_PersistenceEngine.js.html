 27 var PersistenceEngineFactory = function () {
 28 
 29     // *********************************
 30     // SINGLETON INSTANCE
 31     // *********************************
 32     var instance = null;
 33 
 34     function PersitenceEngine () {
 35 
 36         // ****************
 37         // PUBLIC METHODS 
 38         // ****************
 39         PersitenceEngine.prototype.sendGet = function (url, context, successHandler, errorHandler, requestHeaders) {
 40             new Ajax.Request(url, {
 41                 method: 'get',
 42                 parameters: arguments[4],
 43                 onSuccess: successHandler.bind(context),
 44                 onFailure: errorHandler.bind(context),
 45                 onException: errorHandler.bind(context),
 46 				requestHeaders: requestHeaders
 47             });
 48         }
 49 
 50         PersitenceEngine.prototype.sendPost = function (url, params, body, context, successHandler, errorHandler, requestHeaders) {
 51             new Ajax.Request(url, {
 52                 method: 'post',
 53                 parameters: params,
 54                 postBody: body,
 55                 onSuccess: successHandler.bind(context),
 56                 onFailure: errorHandler.bind(context),
 57                 onException: errorHandler.bind(context),
 58 				requestHeaders: requestHeaders
 59             });
 60         }
 61 
 62         PersitenceEngine.prototype.sendDelete = function (url, context, successHandler, errorHandler, requestHeaders){
 63             new Ajax.Request(url, {
 64                 method: 'delete',
 65                 onSuccess: successHandler.bind(context),
 66                 onFailure: errorHandler.bind(context),
 67                 onException: errorHandler.bind(context),
 68 				requestHeaders: requestHeaders
 69             });
 70         }
 71 
 72         PersitenceEngine.prototype.sendUpdate = function (url, params, body, context, successHandler, errorHandler, requestHeaders){
 73         	//FIXME: If body -> it sends post
 74         	new Ajax.Request(url, {
 75                 method: 'put',
 76                 parameters: params,
 77                 postBody: body,
 78                 onSuccess: successHandler.bind(context),
 79                 onFailure: errorHandler.bind(context),
 80                 onException: errorHandler.bind(context),
 81 				requestHeaders: requestHeaders
 82             });
 83         }
 84     }
 85 
 86     // *********************************
 87     // SINGLETON GET INSTANCE
 88     // *********************************
 89     return new function() {
 90         this.getInstance = function() {
 91             if (instance == null) {
 92                 instance = new PersitenceEngine();
 93             }
 94             return instance;
 95         }
 96     }
 97 
 98 }();
