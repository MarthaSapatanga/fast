1 /**
  2  * <p>This class implements the Singleton Design Pattern to make sure there is 
  3  * only one instance of the class FactFactory.
  4  *
  5  * <p> It should be accessed as follows.
  6  *
  7  * @constructor
  8  * @example
  9  * var instance = FactFactorySingleton.getInstance();
 10  */ 
 11 var FactFactorySingleton = function() {
 12 
 13     /**
 14      * Singleton instance
 15      * @private @member
 16      */
 17     var _instance = null;
 18     
 19 
 20     var FactFactory = Class.create(
 21         /** @lends FactFactorySingleton-FactFactory.prototype */ {
 22 
 23         /**
 24          * This class is used to create facts representations.
 25          * @constructs
 26          */
 27         initialize: function() {
 28             /**
 29              * Cached facts.
 30              * @type Hash
 31              * @private @member
 32              */
 33             this._cachedFacts = new Hash();
 34             
 35             /**
 36              * This array stores the shortcuts
 37              * being used in the execution 
 38              */
 39             this._cachedShortcuts = new Array();
 40         },
 41         
 42         // **************** PUBLIC METHODS **************** //
 43 
 44         /**
 45          * Gets the root node of a icon for a give fact identified by uri.
 46          * @param String size  Icon size ("inline"|"embedded"|"standalone")
 47          * @type FactIcon
 48          */
 49         getFactIcon: function (/** Object */ factData, size) {
 50             var fact = this._getFact(factData);
 51             return new FactIcon(fact, size);
 52         },
 53         
 54         /**
 55          * Gets the fact uri
 56          * @type String
 57          */
 58         getFactUri: function (/** Object */ factData) {
 59             var uri;
 60             if (factData.uri) {
 61                 uri = factData.uri;
 62             } else if (factData.pattern) {
 63                 uri = Utils.extractURIfromPattern(factData.pattern);
 64             }
 65             else { //We don't know the uri
 66                 uri = "http://unknown.uri#?";
 67             }
 68             return uri;
 69         },
 70 
 71         // **************** PRIVATE METHODS **************** //
 72         
 73         /**
 74          * Gets a fact
 75          * @type Fact
 76          * @private
 77          */
 78         _getFact: function (/** Object */ factData) {
 79             
 80             var uri = this.getFactUri(factData);
 81             
 82             //The fact didn't exist, create a new one
 83             if(this._cachedFacts.get(uri)==null){
 84                     this._cachedFacts.set(uri, new Fact(uri, 
 85                         this._extractShortcut(uri), this._extractDescription(factData)));
 86             }
 87             return this._cachedFacts.get(uri);
 88         },
 89 
 90         
 91         /**
 92          * This function returns a shortcut coming from the URI
 93          * TODO: Add more criteria to determine the shortcut
 94          * @type String
 95          * @private
 96          */
 97         _extractShortcut: function(/** String */ uri) {
 98             var pieces = uri.split("#");
 99             var identifier = "";
100             if (pieces.length > 1){
101                 identifier = pieces[1];               
102             } else { //The uri has not identifier, try the last part of the url
103                 pieces = uri.split("/");
104                 identifier = pieces[pieces.length - 1];
105             }
106 
107             
108             //Let's try with capital letters...
109             var letters = identifier.match(/[A-Z]/g);
110             if (letters && letters.length > 1) { //More than one capital letter
111                 //try only with 2 letters
112                 //Put the second letter in lower case
113                 //letters[1]= letters[1].toLowerCase();
114                 shortcut = letters.slice(0, 2).join("");
115                 
116                 if (this._cachedShortcuts.indexOf(shortcut) == -1) {
117                     this._cachedShortcuts.push(shortcut);
118                     return shortcut;
119                 }
120             }
121     
122             //Let's try with the first two letters
123             identifier[1]= identifier[1].toLowerCase();
124             var shortcut = identifier.slice(0,2);
125             if (this._cachedShortcuts.indexOf(shortcut) == -1){
126                 this._cachedShortcuts.push(shortcut);
127                 return shortcut;
128                 
129             }
130             //If none of the above have worked out, show the first letter
131             //Despite they have been used before
132             shortcut = identifier.slice(0, 1);
133             return shortcut;
134         },
135         
136         /**
137          * This function extract the description from fact or concept metadata
138          * @private
139          * @type String
140          */
141         _extractDescription: function(/** Object */ factData) {
142             if(factData.label && factData.label['en-gb']) {
143                 return factData.label['en-gb'];
144             }
145             var comment = factData["http://www.w3.org/2000/01/rdf-schema#comment"];
146             if(comment) {
147                 return comment.replace("@en","");
148             }
149             var label = factData["http://www.w3.org/2000/01/rdf-schema#label"];
150             if(label) {
151                 return label.replace("@en","");
152             }
153         }
154     });
155     
156 
157     return new function(){
158         /**
159          * Returns the singleton instance
160          * @public
161          * @type FactFactory
162          */
163         this.getInstance = function(){
164             if (_instance == null) {
165                 _instance = new FactFactory();
166             }
167             return _instance;
168         }
169     }
170 }();
