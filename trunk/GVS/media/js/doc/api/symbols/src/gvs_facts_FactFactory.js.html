1 /**
  2  * <p>This class implements the Singleton Design Pattern to make sure there is 
  3  * only one instance of the class FactFactory.
  4  *
  5  * <p> It should be accessed as follows.
  6  *
  7  * @constructor
  8  * @example
  9  * var instance = FactFactorySingleton.getInstance();
 10  */ 
 11 var FactFactorySingleton = function() {
 12 
 13     /**
 14      * Singleton instance
 15      * @private @member
 16      */
 17     var _instance = null;
 18     
 19 
 20     var FactFactory = Class.create(
 21         /** @lends FactFactorySingleton-FactFactory.prototype */ {
 22 
 23         /**
 24          * This class is used to retrieve facts.
 25          * @constructs
 26          */
 27         initialize: function() {
 28             /**
 29              * Cached facts.
 30              * TODO: get the facts from the catalogue.
 31              * @type Hash
 32              * @private @member
 33              */
 34             this._cachedFacts = new Hash();
 35             
 36             // This is the old hardcoded structure, kept because of the temporary nature of the solution
 37             /*
 38             this._cachedFacts = {
 39                     "http://TODO/amazon#filter": new Fact("http://TODO/amazon#filter", "F", "Filter Search Criteria"),
 40                     "http://TODO/amazon#item":   new Fact("http://TODO/amazon#item",   "I", "Amazon item fact. Normally, its identifier will be an ASIN number"),
 41                     "http://TODO/amazon#shoppingCart": new Fact("http://TODO/amazon#shoppingCart", "C", "Amazon shopping cart fact. It represents a non-purchased shopping cart at Amazon"),
 42                     "http://TODO/amazon#purchase": new Fact("http://TODO/amazon#purchase", "P", "Amazon purchase fact. It represents the URL that allows to purchase the Shopping Cart"),
 43                     "http://TODO/ebay#item":     new Fact("http://TODO/ebay#item", "EI", "eBay Item. It represents an Item selled by eBay")
 44             };
 45             */
 46         },
 47         
 48         // **************** PUBLIC METHODS **************** //
 49         
 50         /**
 51          * Gets a fact identified by uri.
 52          * @type BuildingBlockDescription
 53          * @public
 54          */
 55         getFact: function (/** String */ uri) {
 56             // If the fact does not exist in the cached ones, then a default one is created.
 57             if(this._cachedFacts.get(uri)==null){
 58                 this.updateFactFromUri(uri);
 59                 if(this._cachedFacts.get(uri)==null){
 60                     this._cachedFacts.set(uri, new Fact(uri, "DF", "Default Fact"));
 61                 }
 62             }
 63             return this._cachedFacts.get(uri);
 64         },
 65 
 66         /**
 67          * Updates the facts from a JSON file.
 68          * @public
 69          */
 70         updateFacts: function () {
 71             CatalogueSingleton.getInstance().getFacts();
 72         },
 73         
 74         /**
 75          * Updates the facts from the fact uri.
 76          * @public
 77          */
 78         updateFactFromUri: function (/**String*/ stringUri) {
 79             if(stringUri.startsWith('?')){
 80                 factUri = stringUri.split(' ');
 81                 var shortcut = factUri[0].split('?')[1].truncate(2, '');
 82                 
 83                 var uri = factUri[2];
 84                 var desc = factUri[2];
 85                 this._cachedFacts.set(stringUri,new Fact(stringUri, shortcut, desc));
 86             }
 87         },
 88         
 89         /**
 90          * Sets a fact identified by uri.
 91          * @params String: Facts in JSON
 92          * @public
 93          */
 94         setFacts: function(factMetadata) {
 95             var fact_metadata = factMetadata.fact_metadata;
 96             for (var i=0; i<fact_metadata.length ; i++) {
 97                 this._cachedFacts.set(fact_metadata[i].uri,new Fact(fact_metadata[i].uri, fact_metadata[i].shortcut, fact_metadata[i].description));
 98             }
 99         },
100 
101         /**
102          * Gets the root node of a icon for a give fact identified by uri.
103          * @param String uri   Fact Building Block identifier
104          * @param String size  Icon size ("small"|"medium"|"big")
105          * @type FactIcon
106          */
107         getFactIcon: function (/** String */ uri, /** String */ size) {
108             var fact = this.getFact(uri);
109             return new FactIcon(fact, size);
110         }
111 
112         // **************** PUBLIC METHODS **************** //
113     });
114     
115 
116     return new function(){
117         /**
118          * Returns the singleton instance
119          * @public
120          * @type FactFactory
121          */
122         this.getInstance = function(){
123             if (_instance == null) {
124                 _instance = new FactFactory();
125             }
126             return _instance;
127         }
128     }
129 }();
