 83     /**
 84      * This function returns the domNode of the precondition that has
 85      * the identification passed as parameter
 86      */
 87     getConditionNode: function(/** String */ id, /** String */ action) {
 88         action = Utils.variableOrDefault(action, "postconditions");
 89         var actionIcons = this._icons.get(action);
 90         if (actionIcons) {
 91             var icon = actionIcons.get(id);
 92             if (icon) {
 93                 return icon.getNode();
 94             }
 95         }
 96 
 97         return null;

100     /**
101      * Colorize the component depending on the reachability
102      * @public @override
103      */
104     setReachability: function( /** Hash */ reachabilityData) {
105         var satisfeable = reachabilityData.reachability;
106         Utils.setSatisfeabilityClass(this._node, satisfeable);
107 
108         this._icons.get("postconditions").each(function(post) {
109             Utils.setSatisfeabilityClass(post.value.getNode(), satisfeable);
110         });
111 
112         reachabilityData.actions.each(function(actionData) {
113             var actionIcons = this._icons.get(actionData.name);
114 
115             actionData.preconditions.each(function(preData) {
116                 var factNode = this.get(preData.id).getNode();
117                 Utils.setSatisfeabilityClass(factNode, preData.satisfied);
118             }.bind(actionIcons));
119         }.bind(this));
120     },
