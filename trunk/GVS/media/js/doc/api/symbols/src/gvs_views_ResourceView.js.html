 11         $super();
 12 
 13         /**
 14          * Precondition Icons

 16          * @private

 42             }.bind(this));

 44         }.bind(this));

 48             posts =  description.postconditions[0];

 50             posts = description.postconditions;
 51         }
 52         
 53         posts.each(function(post) {
 54                 var fact = FactFactory.getFactIcon(post, "embedded");
 55                 this._postIcons.set(post.id, fact);
 56                 postArea.appendChild(fact.getNode());
 57             }.bind(this));
 58 
 59         var prePostSeparator = new Element("div",

 95     getConditionNode: function(/** String */ id) {

104     setReachability: function( /** Hash */ reachabilityData) {

106                                this._postIcons.values(), this._node);
