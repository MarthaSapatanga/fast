 26 var ResourceView = Class.create(BuildingBlockView,
 27     /** @lends ResourceView.prototype */ {
 28 
 29     /**
 30      * Resources graphical representation
 31      * @constructs
 32      * @extends BuildingBlockView
 33      */
 34     initialize: function($super,/** ResourceDescription */ description) {
 35 
 36         $super();
 37 
 38         /**
 39          * Fact Icons
 40          * @type Hash
 41          * @private
 42          */
 43         this._icons = new Hash();
 44 
 45 
 46         var preArea = new Element("div", {"class": "preArea"});
 47         var postArea = new Element("div", {"class": "postArea"});
 48 
 49         var actions = description.actions;
 50         var actionIcons;
 51         actions.each(function(action) {
 52             actionIcons = new Hash();
 53             action.preconditions.each(function(pre) {
 54                 var fact = FactFactory.getFactIcon(pre, "embedded");
 55                 actionIcons.set(pre.id, fact);
 56                 preArea.appendChild(fact.getNode());
 57             }.bind(this));
 58             this._icons.set(action.name, actionIcons)
 59         }.bind(this));
 60 
 61         var posts;
 62         if (description.postconditions && description.postconditions[0] instanceof Array) {
 63             posts = description.postconditions[0];
 64         } else {
 65             posts = description.postconditions;
 66         }
 67 
 68         actionIcons = new Hash();
 69         posts.each(function(post) {
 70                 var fact = FactFactory.getFactIcon(post, "embedded");
 71                 actionIcons.set(post.id, fact);
 72                 postArea.appendChild(fact.getNode());
 73             }.bind(this));
 74         this._icons.set("postconditions", actionIcons);
 75 
 76         var prePostSeparator = new Element("div",
 77                 {"class": "prePostSeparator"});
 78 
 79         /*if (description.icon){
 80             var imageContainer = new Element ('div',
 81                     {'class': 'screenImage' });
 82             var image = new Element ('img',{
 83                     'class': 'img',
 84                     'src': description.icon
 85             imageContainer.appendChild (image);
 86         }*/
 87 
 88         this._node = new Element("div", {
 89             "class": "view resource",
 90             "title": description.name
 91         });
 92 
 93         this._node.appendChild(preArea);
 94         this._node.appendChild(prePostSeparator);
 95 
 96         /*if (description.icon){
 97             this._node.appendChild(imageContainer);
 98         }*/
 99         this._node.appendChild(postArea);
100 
101         var titleNode = new Element("div", {"class":"title"});
102         titleNode.update(description.label['en-gb']);
103         this._node.appendChild(titleNode);
104 
105     },
106 
107     // **************** PUBLIC METHODS **************** //
108     /**
109      * This function returns the domNode of the precondition that has
110      * the identification passed as parameter
111      */
112     getConditionNode: function(/** String */ id, /** String */ action) {
113         action = Utils.variableOrDefault(action, "postconditions");
114         var actionIcons = this._icons.get(action);
115         if (actionIcons) {
116             var icon = actionIcons.get(id);
117             if (icon) {
118                 return icon.getNode();
119             }
120         }
121 
122         return null;
123     },
124 
125     /**
126      * Colorize the component depending on the reachability
127      * @public @override
128      */
129     setReachability: function( /** Hash */ reachabilityData) {
130         var satisfeable = reachabilityData.reachability;
131         Utils.setSatisfeabilityClass(this._node, satisfeable);
132 
133         this._icons.get("postconditions").each(function(post) {
134             Utils.setSatisfeabilityClass(post.value.getNode(), satisfeable);
135         });
136 
137         reachabilityData.actions.each(function(actionData) {
138             var actionIcons = this._icons.get(actionData.name);
139 
140             actionData.preconditions.each(function(preData) {
141                 var factNode = this.get(preData.id).getNode();
142                 Utils.setSatisfeabilityClass(factNode, preData.satisfied);
143             }.bind(actionIcons));
144         }.bind(this));
145     },
146 
147     /**
148      * Removes the DOM Elements and frees building blocks
149      * @override
150      */
151     destroy: function ($super) {
152         // Let the garbage collector to do its job
153         this._icons = null;
154         $super();
155     }
156 
157 
158     // **************** PRIVATE METHODS **************** //
159 });
