 26 var PlanView = Class.create(BuildingBlockView,
 27     /** @lends PlanView.prototype */ {
 28 
 29     /**
 30      * Plans graphical representation
 31      * @constructs
 32      * @extends BuildingBlockView
 33      */
 34     initialize: function($super, /** Array */ plan) {
 35 
 36         $super();
 37 
 38         this._node = new Element('div', {'class': 'plan view'});
 39 
 40         var nScreens = 0;
 41         plan.each(function(screenDescription) {
 42             var screenView = new ScreenView(screenDescription);
 43             this._node.appendChild(screenView.getNode());
 44             nScreens++;
 45         }.bind(this));
 46         // width = (size of each screen)* (number of screens)
 47         // + (margin+padding)*(number of screens)
 48         var width = 102*nScreens + 6*nScreens;
 49         var widthText = width + 'px';
 50         this._node.setStyle({'width': widthText});
 51     },
 52 
 53     // **************** PUBLIC METHODS **************** //
 54 
 55     /**
 56      * Colorize the component depending on the reachability
 57      * @public @override
 58      */
 59     setReachability: function( /** Hash */ reachabilityData) {
 60 
 61         //Do nothing
 62     },
 63 
 64     /**
 65      * Removes the DOM Elements and frees building blocks
 66      * @override
 67      */
 68     destroy: function () {
 69         // Let the garbage collector to do its job
 70 
 71     }
 72 
 73     // **************** PRIVATE METHODS **************** //
 74 });
