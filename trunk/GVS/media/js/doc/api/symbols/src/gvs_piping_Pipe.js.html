 26 var Pipe = Class.create(
 27     /** @lends Pipe.prototype */ {
 28 
 29     /**
 30      * Pipes graphical representation
 31      * @constructs
 32      */
 33     initialize: function (/** WireIt.Wire */ wire, /** String */ id) {
 34 
 35         /**
 36          * Pipe wire
 37          * @private
 38          * @type WireIt.Wire
 39          */
 40         this._wire = wire;
 41 
 42         /**
 43         * Pipe id
 44         * @private
 45         * @type String
 46         */
 47         this._id = id;
 48 
 49         /**
 50          * Valid pipe
 51          * @type Boolean
 52          * @private
 53          */
 54         this._isValid = true;
 55     },
 56 
 57     /**
 58      * Returns the WireIt wire object
 59      * @type WireIt.Wire
 60      */
 61     getWire: function() {
 62         return this._wire;
 63     },
 64 
 65     /**
 66      * Returns the pipe id
 67      * @type String
 68      */
 69     getId: function() {
 70         return this._id;
 71     },
 72 
 73     /**
 74      * Returns the source terminal of the pipe
 75      * @type Terminal
 76      */
 77     getSource: function() {
 78         return this._wire.terminal1;
 79     },
 80 
 81 
 82     /**
 83      * Returns the destination terminal of the pipe
 84      * @type Terminal
 85      */
 86     getDestination: function() {
 87         return this._wire.terminal2;
 88     },
 89 
 90     /**
 91      * Gets the JSON object representing the pipe for screen definition
 92      * @type Object
 93      */
 94     getJSONforScreen: function() {
 95         return {'from': {
 96                     'buildingblock': this.getSource().getBuildingBlockId(),
 97                     'condition': this.getSource().getConditionId()
 98                 },
 99                 'to': {
100                     'buildingblock': this.getDestination().getBuildingBlockId(),
101                     'action': this.getDestination().getActionId(),
102                     'condition': this.getDestination().getConditionId()
103                 }
104             };
105     },
106 
107     /**
108      * Gets the JSON object representing the pipe for catalogue check
109      * @type Object
110      */
111     getJSONforCheck: function() {
112         return {
113                 'from': {
114                     'buildingblock': this.getSource().getBuildingBlockUri(),
115                     'condition': this.getSource().getConditionId()
116                 },
117                 'to': {
118                     'buildingblock': this.getDestination().getBuildingBlockUri(),
119                     'action': this.getDestination().getActionId(),
120                     'condition': this.getDestination().getConditionId()
121                 }
122             };
123     },
124 
125     /**
126      * Sets the reachability coloring information of the pipe
127      */
128     setReachability: function(/** Object */ reachabilityData) {
129         var options = this._wire.options;
130         var satisfied = reachabilityData.satisfied;
131         if (!reachabilityData.correct) {
132             options = Object.extend(options, {
133                 'color': '#EAEAEA',
134                 'bordercolor': '#808080'
135             });
136         } else {
137             if (satisfied) {
138                 options = Object.extend(options, {
139                     'color': '#DDF7DD',
140                     'bordercolor': '#008000'
141                 });
142             } else {
143                 options = Object.extend(options, {
144                     'color': '#F5D9D9',
145                     'bordercolor': '#B90000'
146                 });
147             }
148         }
149         this._wire.setOptions(options);
150         this._wire.redraw();
151 
152         if (!reachabilityData.correct) {
153              this._wire.element.setStyle({'opacity': '0.5'});
154              if (this._isValid) {
155                  Utils.showMessage("The elements you have connected are not compatible." +
156                             " It may not work", {
157                             'hide': true,
158                             'error': true
159                             });
160                  this._isValid = false;
161              }
162 
163         }
164     },
165 
166     /**
167      * Is the pipe semantically correct
168      * @type Boolean
169      */
170     isValid: function() {
171         // TODO: check if there are more conditions
172         return (this.getSource().getBuildingBlockId() != this.getDestination().getBuildingBlockId())
173                 && this._isValid;
174     },
175 
176 
177     /**
178      * Destroys the wire
179      */
180     destroy: function() {
181         this.getSource().removeWire(this._wire);
182         this.getDestination().removeWire(this._wire);
183     },
184 
185 
186     /**
187      * Sets the pipe visible or hidden. To be called whenever a document
188      * is selected or deselected
189      */
190     setVisible: function(/** Boolean */ visible) {
191         if (visible) {
192             this.getSource().updatePosition();
193             this.getDestination().updatePosition();
194         }
195         var style = {
196             'visibility': visible ? 'visible': 'hidden'
197         }
198         this._wire.element.setStyle(style);
199     }
200 
201     // **************** PRIVATE METHODS **************** //
202 
203 });
