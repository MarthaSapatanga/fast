 26 var Table = Class.create( /** @lends Table.prototype */ {
 27     /**
 28      * This abstract class handles the different properties tables
 29      * belonging to different documents
 30      * @abstract
 31      * @constructs
 32      */
 33     initialize: function(/** DOMNode */ parentNode, /** String */ baseTitle,
 34             /** String */ region, /** String (optional) */ minSize) {
 35         /**
 36          * Parent node
 37          * @type DOMNode
 38          * @private @member
 39          */
 40         this._parentNode = parentNode;
 41 
 42         /**
 43          * Title of the table
 44          * @type String
 45          * @private @member
 46          */
 47         this._title = baseTitle;
 48 
 49         /**
 50          * Node of the title
 51          * @type DOMNode
 52          * @private @member
 53          */
 54         this._titleNode = new Element ('div',{
 55             'class': 'dijitAccordionTitle '
 56         }).update(this._title);
 57 
 58         /**
 59          * Node of the table
 60          * @type DOMNode
 61          * @private @member
 62          */
 63         this._tableNode = new Element ('table',{
 64             'class': 'propertiesTable'
 65         });
 66 
 67         var style = (region == "left") ? 'width: 50%': 'width: auto';
 68         var container= new dijit.layout.ContentPane({
 69             'region': region,
 70             'splitter': true,
 71             'style': style,
 72             'class': 'tableArea'
 73         });
 74         if (minSize) {
 75             container.attr('minSize', minSize);
 76         }
 77 
 78         container.domNode.insert(this._titleNode);
 79         container.domNode.insert(this._tableNode);
 80         this._parentNode.addChild(container);
 81     },
 82 
 83 
 84     // **************** PUBLIC METHODS **************** //
 85 
 86     /**
 87      * Set the table title
 88      */
 89     setTitle: function (/** String */ title) {
 90         this._title = title;
 91         this._titleNode.update(title);
 92     },
 93 
 94     /**
 95      * Returns the table title
 96      * @type String
 97      */
 98     getTitle: function () {
 99         return this._title;
100     },
101 
102     /**
103      * Returns the table Node
104      * @type DOM
105      */
106     getTableNode: function () {
107         return this._tableNode;
108     },
109 
110     /**
111      * Sets the field titles
112      * @type String
113      */
114     insertFieldTitles: function (/** Array */ fieldTitles) {
115         var tr = new Element('tr', {
116             'class': 'tableHeader'
117         });
118 
119         fieldTitles.each (function(title){
120            var td = new Element ('td').update(title);
121            tr.insert(td);
122         });
123 
124         this._tableNode.insert(tr);
125     },
126 
127     /**
128      * Updates the data in the table
129      */
130     insertDataValues: function (/** Hash */ data)  {
131 
132         data.each (function(line){
133             var tr = new Element('tr');
134 
135             //TODO: What happen with the classes, maybe another method?
136             line.each (function(field){
137                var td = new Element ('td');
138                var div = new Element ('div').update(field);
139                if (typeof field === "string") {
140                    div.setAttribute('title', field);
141                }
142                td.insert(div)
143                tr.insert(td);
144             });
145             this._tableNode.insert(tr);
146         }.bind(this));
147     },
148 
149     /**
150      * Empty the data in the table
151      */
152     emptyTable : function (){
153         this._tableNode.update("");
154     }
155 
156 
157 
158     // **************** PRIVATE METHODS **************** //
159 
160 
161 });
