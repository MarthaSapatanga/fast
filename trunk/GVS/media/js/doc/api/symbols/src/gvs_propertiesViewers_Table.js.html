1 var Table = Class.create( /** @lends Table.prototype */ {
  2     /**
  3      * This abstract class handles the different properties tables
  4      * belonging to different documents
  5      * @abstract
  6      * @constructs
  7      */
  8     initialize: function(/** DOMNode */ parentNode, /** String */ baseTitle,
  9             /** String */ region, /** String (optional) */ minSize) {
 10         /**
 11          * Parent node
 12          * @type DOMNode
 13          * @private @member
 14          */
 15         this._parentNode = parentNode;
 16 
 17         /**
 18          * Title of the table
 19          * @type String
 20          * @private @member
 21          */
 22         this._title = baseTitle;
 23 
 24         /**
 25          * Node of the title
 26          * @type DOMNode
 27          * @private @member
 28          */
 29         this._titleNode = new Element ('div',{
 30             'class': 'dijitAccordionTitle '
 31         }).update(this._title);
 32 
 33         /**
 34          * Node of the table
 35          * @type DOMNode
 36          * @private @member
 37          */
 38         this._tableNode = new Element ('table',{
 39             'class': 'propertiesTable'
 40         });
 41 
 42         var style = (region == "left") ? 'width: 50%': 'width: auto';
 43         var container= new dijit.layout.ContentPane({
 44             'region': region,
 45             'splitter': true,
 46             'style': style,
 47             'class': 'tableArea'
 48         });
 49         if (minSize) {
 50             container.attr('minSize', minSize);
 51         }
 52 
 53         container.domNode.insert(this._titleNode);
 54         container.domNode.insert(this._tableNode);
 55         this._parentNode.addChild(container);
 56     },
 57 
 58 
 59     // **************** PUBLIC METHODS **************** //
 60 
 61     /**
 62      * Set the table title
 63      */
 64     setTitle: function (/** String */ title) {
 65         this._title = title;
 66         this._titleNode.update(title);
 67     },
 68 
 69     /**
 70      * Returns the table title
 71      * @type String
 72      */
 73     getTitle: function () {
 74         return this._title;
 75     },
 76 
 77     /**
 78      * Returns the table Node
 79      * @type DOM
 80      */

 82         return this._tableNode;
 83     },
 84 
 85     /**
 86      * Sets the field titles
 87      * @type String
 88      */
 89     insertFieldTitles: function (/** Array */ fieldTitles) {
 90         var tr = new Element('tr', {
 91             'class': 'tableHeader'
 92         });
 93 
 94         fieldTitles.each (function(title){
 95            var td = new Element ('td').update(title);
 96            tr.insert(td);
 97         });
 98 
 99         this._tableNode.insert(tr);
100     },
101 
102     /**
103      * Updates the data in the table
104      */
105     insertDataValues: function (/** Hash */ data)  {
106 
107         data.each (function(line){
108             var tr = new Element('tr');
109 
110             //TODO: What happen with the classes, maybe another method?
111             line.each (function(field){
112                var td = new Element ('td');
113                var div = new Element ('div').update(field);
114                if (typeof field === "string") {
115                    div.setAttribute('title', field);
116                }
117                td.insert(div)
118                tr.insert(td);
119             });
120             this._tableNode.insert(tr);
121         }.bind(this));
122     },
123 
124     /**
125      * Empty the data in the table
126      */
127     emptyTable : function (){
128         this._tableNode.update("");
129     }
