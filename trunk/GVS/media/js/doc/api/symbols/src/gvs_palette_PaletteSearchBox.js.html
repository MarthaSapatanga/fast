1 var PaletteSearchBox = Class.create(
  2     /** @lends PaletteSearchBox.prototype */ {
  3 
  4      /**
  5      * Represents a search box.
  6      *
  7      * @constructs
  8      */
  9     initialize: function(/* String */ defaultValue) {
 10 
 11         /**
 12          * @public @member
 13          * @type String
 14          */
 15         this.defaultValue = defaultValue || "Search...";
 16 
 17         /**
 18          * @private @member
 19          * @type Array
 20          */
 21         this._listeners = [];
 22 
 23         /**
 24          * @private @member
 25          * @type String
 26          */
 27         this._textSearch = "";
 28 
 29         /**
 30          * @private @member
 31          * @type Input Html Element of prototypejs.org
 32          */
 33         this._inputElement = new Element('input', {
 34             'type':'text',
 35             'class':'defaultValue',
 36             'value':this.defaultValue
 37         });
 38         this._inputElement.observe('blur', this._lostFocus.bind(this));
 39         this._inputElement.observe('focus', this._getFocus.bind(this));
 40         new Form.Element.Observer(this._inputElement, 1, this._valueChange.bind(this));
 41 
 42         /**
 43          * @private @member
 44          * @type Div Html Element of prototypejs.org
 45          */
 46         this._rootNode = new Element('div', {'class':'searchBox'}).insert(this._inputElement);
 47     },
 48 
 49      /**
 50      * Gets the value of the textbox
 51      * @type String
 52      * @public
 53      */
 54     getValue: function() {
 55         if (this._inputElement.hasClassName('defaultValue')) {
 56             return "";
 57         }
 58         return String.interpret(this._inputElement.value);
 59     },
 60 
 61     /**
 62      * Sets the value of the textbox
 63      * @public
 64      */
 65     setValue: function(/* String */ value) {
 66         this._inputElement.value = value
 67     },
 68 
 69     /**
 70      * Gets the node of the accordion pane
 71      * @type DOMNode
 72      * @public
 73      */
 74     getDOMNode: function() {
 75         return this._rootNode;
 76     },
 77 
 78     /**
 79      * Add a new listener
 80      * @public
 81      */
 82     addEventListener: function(/* Object (Listener)*/ listener) {
 83         this._listeners.push(listener);
 84     },
 85 
 86     /**
 87      * Text box set focus. event handler
 88      * @private
 89      */
 90     _getFocus: function() {
 91         if (this._inputElement.hasClassName('defaultValue')) {
 92             this._inputElement.value = "";
 93             this._inputElement.removeClassName('defaultValue');
 94         } else {
 95             this._inputElement.select();
 96         }
 97     },
 98 
 99     /**
100      * Text box lost focus. event handler
101      * @private
102      */
103     _lostFocus: function() {
104         if (!this.getValue()) {
105             this._inputElement.value = this.defaultValue;
106             this._inputElement.addClassName('defaultValue');
107         }
108     },
109 
110     /**
111      * Value change. event handler
112      * @private
113      */
114     _valueChange: function() {
115         this._textSearch = this.getValue();
116 
117         for (var i = 0; i < this._listeners.length; i++) {
118             this._listeners[i](this, this._textSearch);
119         }
120     }
121 });
