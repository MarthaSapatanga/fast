 26 var FormDescription = Class.create(BuildingBlockDescription,
 27     /** @lends FormDescription.prototype */ {
 28 
 29     /**
 30      * Form building block description.
 31      * @constructs
 32      * @extends BuildingBlockDescription
 33      */
 34     initialize: function($super, /** Hash */ properties) {
 35         $super(properties);
 36     },
 37 
 38     /**
 39      * This method creates a DOM Node with the preview
 40      * of the Form
 41      * @type DOMNode
 42      */
 43     getPreview: function() {
 44         var node = new Element('div', {
 45             'class': 'preview'
 46         });
 47         var errorField = new Element('div', {
 48             'class': 'error'
 49         });
 50         node.appendChild(errorField);
 51 
 52         var image = new Element ('img', {
 53             'src': this.screenshot,
 54             'onerror': 'this.src = "/fast/images/gui/imageNotFound.png";'
 55         });
 56 
 57         node.appendChild(image);
 58         return node;
 59     }
 60 });
