107     /**
108      * Colorize the component depending on the reachability
109      * @public @override
110      */
111     setReachability: function( /** Hash */ reachabilityData) {
112         
113         var satisfeable = reachabilityData.reachability;
114         
115         // screen
116         Utils.setSatisfeabilityClass(this.getNode(), satisfeable);
117        
118         // posts
119         this._postIcons.each(function(postIcon) {
120             Utils.setSatisfeabilityClass(postIcon.getNode(), satisfeable);  
121         });
122         
123         // pres
124         var preconditionList = reachabilityData.preconditions;  
125         if (preconditionList.length > 1) {
126             //More than one set of preconditions is NOT YET SUPPORTED
127             console.log("OR precondition support not implemented yet");
128             return;
129         } else {
130             var preconditionData = preconditionList[0];
131         }
132         
133         //Setting precondition reachability
134         var preReachability = new Hash();
135         $A(preconditionData).each(function(precondition) {
136             preReachability.set(FactFactory.getFactUri(precondition), precondition.satisfied);
137         });
138       
139         this._preIcons.each(function(preIcon) {
140             var factUri = preIcon.getFact().getUri();
141             Utils.setSatisfeabilityClass(preIcon.getNode(), preReachability.get(factUri));
142         });
143     },
