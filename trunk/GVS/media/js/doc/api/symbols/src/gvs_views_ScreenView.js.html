1 var ScreenView = Class.create(BuildingBlockView,
  2     /** @lends ScreenView.prototype */ {

  4     /**
  5      * Screens graphical representation
  6      * @constructs
  7      * @extends BuildingBlockView
  8      */
  9     initialize: function($super,/** ScreenDescription */ description) {
 10 
 11         $super();
 12 
 13         /**
 14          * Precondition Icons
 15          * @type Array
 16          * @private
 17          */
 18         this._preIcons = new Array();
 19 
 20         /**
 21          * Postcondition Icons
 22          * @type {FactIcon[]}
 23          * @private
 24          */
 25         this._postIcons = new Array();
 26 
 27         /**
 28          * Tooltip
 29          * @type dijit.Tooltip
 30          * @private
 31          */
 32         this._tooltip = null;
 33 
 34         /**
 35          * Title node
 36          * @type DOMNode
 37          * @private
 38          */
 39         this._titleNode = new Element("div", {"class":"title"});
 40 
 41         this._titleNode.update(description.label['en-gb']);
 42 
 43         var preArea = new Element("div", {"class": "preArea"});
 44 
 45         var preconditions = description.preconditions;
 46 
 47 
 48         $A(preconditions).each(
 49                 function(pre) {
 50                     var preFact = FactFactory.getFactIcon(pre, "embedded");
 51                     this._preIcons.push(preFact);
 52                     preArea.appendChild(preFact.getNode());
 53                 }.bind(this)
 54         );
 55 
 56         var postArea = new Element("div", {"class": "postArea"});
 57 
 58         var postconditions = description.postconditions;
 59 
 60         $A(postconditions).each(
 61                 function(post) {
 62                     if (post) {
 63                         var postFact = FactFactory.getFactIcon(post, "embedded");
 64                         this._postIcons.push(postFact);
 65 
 66                         if (!post.positive)
 67                         	postFact.getNode().addClassName('negative');
 68 
 69                         postArea.appendChild(postFact.getNode());
 70                     }
 71                 }.bind(this)
 72         );
 73 
 74         var prePostSeparator = new Element("div",
 75                 {"class": "prePostSeparator"});
 76 
 77         if (description.icon){
 78             var imageContainer = new Element ('div',
 79                     {'class': 'image' });
 80             var image = new Element ('img',{
 81                     'class': 'img',
 82                     'src': description.icon
 83                     //If we want an onerror image...
 84                     /*'onError': 'if (this.src != URIs.screenImageNotFound){'+
 85                                 'this.src = URIs.screenImageNotFound;'+
 86                             '}'*/
 87                     });
 88             imageContainer.appendChild (image);
 89         }
 90 
 91         this._node = new Element("div", {
 92             "class": "view screen"
 93         });
 94         this._node.appendChild(this._titleNode);
 95         this._node.appendChild(preArea);
 96         this._node.appendChild(prePostSeparator);
 97         if (description.icon){
 98             this._node.appendChild(imageContainer);
 99         }
100         this._node.appendChild(postArea);
101     },
102

104 
105     /**
106      * Colorize the component depending on the reachability
107      * @public @override
108      */
109     setReachability: function( /** Hash */ reachabilityData) {
110 
111         var satisfeable = reachabilityData.reachability;
112 
113         // screen
114         Utils.setSatisfeabilityClass(this.getNode(), satisfeable);
115 
116         // posts
117         this._postIcons.each(function(postIcon) {
118             Utils.setSatisfeabilityClass(postIcon.getNode(), satisfeable);
119         });
120 
121         // pres
122         var preconditionList = reachabilityData.preconditions;
123         var preconditionData = preconditionList;
124 
125         //Setting precondition reachability
126         var preReachability = new Hash();
127         $A(preconditionData).each(function(precondition) {
128             preReachability.set(FactFactory.getFactUri(precondition), precondition.satisfied);
129         });
130 
131         this._preIcons.each(function(preIcon) {
132             var factUri = preIcon.getFact().getUri();
133             Utils.setSatisfeabilityClass(preIcon.getNode(), preReachability.get(factUri));
134         });
135     },
136 
137     /**
138      * Removes the DOM Elements and frees building blocks
139      * @override
140      */
141     destroy: function ($super) {
142         // Let the garbage collector to do its job
143         this._preIcons = null;
144         this._postIcons = null;
145         $super();
146     },
147 
148     /**
149      * @override
150      */
151     setTitle: function(title) {
152         this._titleNode.update(title);
153     }
154 
155     // **************** PRIVATE METHODS **************** //
156 });
