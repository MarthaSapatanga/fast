 22     /**
 23      * to JSON object function
 24      * @type Object
 25      */
 26     toJSON: function() {
 27         var result = {
 28             "preconditions": this.getPreconditions().size() > 0 ? [this.getPreconditions()] : [],
 29             "postconditions": this.getPostconditions().size() > 0 ? [this.getPostconditions()] : [],
 30             "definition": {
 31                 "buildingblocks": this._getScreenBuildingBlocks(),
 32                 "pipes": this._getScreenPipes(),
 33                 "triggers": this._getTriggers()
 34             }
 35         };
 36         result = Object.extend(result,{
 37             "name": this.name,
 38             "label": this.label,
 39             "tags": this.tags,
 40             "version": this.version,
 41             "id": this.id,
 42             "creator": this.creator,
 43             "description": this.description,
 44             "rights": this.rights,
 45             "creationDate": this.creationDate,
 46             "icon": this.icon,
 47             "screenshot": this.screenshot,
 48             "homepage": this.homepage,
 49             "type": "screen"
 50         });
 51         return result;
 52     },

 54     /**
 55      * This method creates a DOM Node with the preview
 56      * of the Screen
 57      * @type DOMNode
 58      */
 59     getPreview: function() {
 60         var node = new Element('div', {
 61             'class': 'preview'
 62         });
 63         var errorField = new Element('div', {
 64             'class': 'error'
 65         });
 66         node.appendChild(errorField);
 67         var image = new Element ('img', {
 68             'src': this.screenshot, 
 69             'onerror': "this.parentNode.childNodes[0].update('Image not available');" +
 70                 "this.src='"+ URIs.logoFast + "';"
 71         });
 72         node.appendChild(image);
 73         return node;
 74     },
 75 
 76     /**
 77      * Implementing the TableModel interface
 78      * @type Array
 79      */
 80     getInfo: function() {
 81         var info = new Hash();
 82         info.set('Title', this.label['en-gb']);
 83         info.set('Tags', this.tags.collect(function(tag) {
 84                 return tag.label['en-gb'];
 85             }).join(", "));
 86         info.set('Version', this.version);
 87         return info;
 88     },

110     /**
111      * Updates the position of a *-condition
112      */
113     updatePrePost: function(/** PrePostInstance */ prepost, /** Object */ position) {
114         var list;
115         if (prepost.getType() == "pre") {
116             list = this._preconditions;
117         } else {
118             list = this._postconditions;
119         }
120         list.get(prepost.getId()).position = position;

220      * Returns true if an element with the parameter uri is in the screen

275     /**
276      * Removes an instance from the screen description
277      */
278     remove: function(/** Object */ instance) {
279         switch (instance.constructor) {
280             case PrePostInstance:
281                 if (this._preconditions.get(instance.getId())) {
282                     this._preconditions.unset(instance.getId());
283                 } else {
284                     this._postconditions.unset(instance.getId());
285                 }
286                 break;
287             case Pipe:
288                 this._pipes.unset(instance.getId());
289                 break;
290             case Trigger:
291             case ScreenTrigger:
292                 this._triggers.unset(instance.getId());
293                 break;
294             default:
295                 this._buildingBlocks.unset(instance.getId());
296         }
297     },
