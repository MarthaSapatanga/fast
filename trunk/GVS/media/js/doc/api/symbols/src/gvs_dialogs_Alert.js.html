1 /**
  2  * <p>This class implements the Singleton Design Pattern to make sure there is 
  3  * only one instance of the class Preferences.
  4  *
  5  * <p> It should be accessed as follows.
  6  *
  7  * @constructor
  8  * @example
  9  * var alert = AlertSingleton.getInstance();
 10  */ 
 11 var AlertSingleton = function() {
 12 
 13     /**
 14      * Singleton instance
 15      * @private @member
 16      */
 17     var _instance = null;
 18     
 19 
 20     var Alert = Class.create(
 21         /** @lends AlertSingleton-Alert.prototype */ {
 22 
 23         /** 
 24          * Alert dialog
 25          * @constructs
 26          */
 27         initialize: function() {
 28             this._dialog = new FormDialog({
 29                'title': 'Warning',
 30                'style': 'display:none'
 31             });
 32             this._dialog.getContentNode().addClassName("systemDialog"); 
 33             
 34             this._dialog.addButton ('Ok', this.hide.bind(this));
 35         },
 36         /**
 37          * This function shows a message
 38          */
 39         show: function  (/** String */ message){
 40             this._dialog.getContentNode().update(message);
 41             this._dialog.show();
 42         },
 43         
 44         hide: function (){
 45             this._dialog.hide();
 46         }
 47         
 48     });
 49 
 50     return new function(){
 51         /**
 52          * Returns the singleton instance
 53          * @type Preferences
 54          */
 55         this.getInstance = function(){
 56             if (_instance == null) {
 57                 _instance = new Alert();
 58             }
 59             return _instance;
 60         }
 61     }
 62 }();
