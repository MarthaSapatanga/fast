 22          * Key press registry

 24          * @private

 26         this._registry = registry;

 29          * Array of current menuElements

 31          * @private

 37     /**
 38      * Sets the model for the menu. 
 39      * The model can contains different menus and elements
 40      */
 41     setModel: function(/** String */ keyword, /** MenuModel */ model) {
 42         this._models.unset(keyword);
 43         
 44         if (model) {
 45             this._models.set(keyword, model);
 46         }
 47         var menuConfig = this._mergeAllModels(this._models.values());
 48         this._createMenu(menuConfig);
 49     },

 54     /**
 55      * Merge all different model objects in order to create
 56      * a single menu
 57      * @private
 58      * @type Array
 59      */
 60     _mergeAllModels: function(/** Array */ models) {
 61         var resultHash = new Hash();
 62         models.each(function(model){
 63             resultHash = this._mergeMenuElements(resultHash, $H(model.getMenuElements()));
 64         }.bind(this));
 65         
 66         var result = resultHash.values().clone();
 67         
 68         result.sort(function(a,b) {
 69                         var left = a.get('weight') ? a.get('weight') : MenuElement.MAXIMUM_WEIGHT;
 70                         var right = b.get('weight') ? b.get('weight') : MenuElement.MAXIMUM_WEIGHT;
 71                         return left - right;              
 72                     });
 73         
 74         return result;
 75     },
 76     
 77     
 78     /**
 79      * Merge all menu subelements from two elements
 80      * @private
 81      * @type Hash
 82      */    
 83     _mergeMenuElements: function(/** Hash */ left, /** Hash */ right) {
 84         var result = new Hash();
 85         // All the keys without duplicates
 86         var keys = left.keys().concat(right.keys()).uniq();
 87         keys.each(function(key) {
 88             if(left.get(key) && right.get(key)) {
 89                 result.set(key, this._mergeMenuElement($H(left.get(key)),$H(right.get(key)))); 
 90                 
 91             } else if (left.get(key)) {
 92                 result.set(key,$H(left.get(key)));
 93                 
 94             } else if (right.get(key)) {
 95                 result.set(key,$H(right.get(key)));
 96             }
 97         }.bind(this));
 98         return result;
 99     },
100 
101 
102     /**
103      * Returns an element with Union of the attributes
104      * of the two passed as parameters
105      * @private
106      * @type Hash
107      */
108     _mergeMenuElement: function(/** Hash */ left, /** Hash */ right) {
109         var result = new Hash();
110                
111         // All the keys without duplicates
112         var keys = left.keys().concat(right.keys()).uniq();
113         keys.each(function(key) {
114             if(left.get(key) && right.get(key)) {
115                 if (key == 'children') {
116                     result.set(key, this._mergeMenuElements($H(left.get(key)), $H(right.get(key))));
117                                         
118                 } else {
119                     if (left.get(key) != right.get(key)) {
120                         throw "Conflicting menu element property " + key;
121                     }
122                          
123                     result.set(key, left.get(key)); 
124                 }
125                 
126             } else if (left.get(key)) {
127                 result.set(key,left.get(key));
128                 
129             } else if (right.get(key)) {
130                 result.set(key,right.get(key));
131             }
132         }.bind(this));
133         return result;
134     },

137     /**
138      * Creates the menu widget structure from a menu configuration object
139      * @private
140      */
141     _createMenu: function(/** Array */ menuConfiguration) {
142         //Unregistering the old menu elements
143         this._menuElements.each(function(element) {
144             element.unregisterKeyHandlers(this._registry);
145             element.getWidget().destroyRecursive();
146         }.bind(this));
147         this._menuElements.clear();
148         menuConfiguration.each(function(menuElement) {
149             var menuElementObject = menuElement.toObject();
150             switch(menuElementObject.type.toLowerCase()) {
151                 case "submenu":
152                     var element = new SubMenu(menuElementObject, true);
153                     break;
154                 case "action":
155                     var element = menuElementObject.action;
156                     break;
157             }
158             this._menuElements.push(element);  
159             this._menu.addChild(element.getWidget());
160             element.registerKeyHandlers(this._registry);
161         }.bind(this));
162     }
