 26 var Menu = Class.create( /** @lends Menu.prototype */ {
 27     /**
 28      * The Menu itself
 29      *
 30      * @constructs
 31      */
 32     initialize: function(/** KeyPressRegistry */ registry) {
 33         /**
 34          * @type dijit.MenuBar
 35          * @private @member
 36          */
 37         this._menu = dijit.byId("menu");
 38 
 39         /**
 40          * List of menu models
 41          * @type Hash
 42          * @private @member
 43          */
 44         this._models = new Hash();
 45 
 46         /**
 47          * Key press registry
 48          * @type KeyPressRegistry
 49          * @private
 50          */
 51         this._registry = registry;
 52 
 53         /**
 54          * Array of current menuElements
 55          * @type Array
 56          * @private
 57          */
 58         this._menuElements = new Array();
 59 
 60     },
 61 
 62     /**
 63      * Sets the model for the menu.
 64      * The model can contains different menus and elements
 65      */
 66     setModel: function(/** String */ keyword, /** MenuModel */ model) {
 67         this._models.unset(keyword);
 68 
 69         if (model) {
 70             this._models.set(keyword, model);
 71         }
 72         var menuConfig = this._mergeAllModels(this._models.values());
 73         this._createMenu(menuConfig);
 74     },
 75 
 76     // ************************ PRIVATE METHODS ************************* //
 77 
 78 
 79     /**
 80      * Merge all different model objects in order to create
 81      * a single menu
 82      * @private
 83      * @type Array
 84      */
 85     _mergeAllModels: function(/** Array */ models) {
 86         var resultHash = new Hash();
 87         models.each(function(model){
 88             resultHash = this._mergeMenuElements(resultHash, $H(model.getMenuElements()));
 89         }.bind(this));
 90 
 91         var result = resultHash.values().clone();
 92 
 93         result.sort(function(a,b) {
 94                         var left = a.get('weight') ? a.get('weight') : MenuElement.MAXIMUM_WEIGHT;
 95                         var right = b.get('weight') ? b.get('weight') : MenuElement.MAXIMUM_WEIGHT;
 96                         return left - right;
 97                     });
 98 
 99         return result;
100     },
101 
102 
103     /**
104      * Merge all menu subelements from two elements
105      * @private
106      * @type Hash
107      */
108     _mergeMenuElements: function(/** Hash */ left, /** Hash */ right) {
109         var result = new Hash();
110         // All the keys without duplicates
111         var keys = left.keys().concat(right.keys()).uniq();
112         keys.each(function(key) {
113             if(left.get(key) && right.get(key)) {
114                 result.set(key, this._mergeMenuElement($H(left.get(key)),$H(right.get(key))));
115 
116             } else if (left.get(key)) {
117                 result.set(key,$H(left.get(key)));
118 
119             } else if (right.get(key)) {
120                 result.set(key,$H(right.get(key)));
121             }
122         }.bind(this));
123         return result;
124     },
125 
126 
127     /**
128      * Returns an element with Union of the attributes
129      * of the two passed as parameters
130      * @private
131      * @type Hash
132      */
133     _mergeMenuElement: function(/** Hash */ left, /** Hash */ right) {
134         var result = new Hash();
135 
136         // All the keys without duplicates
137         var keys = left.keys().concat(right.keys()).uniq();
138         keys.each(function(key) {
139             if(left.get(key) && right.get(key)) {
140                 if (key == 'children') {
141                     result.set(key, this._mergeMenuElements($H(left.get(key)), $H(right.get(key))));
142 
143                 } else {
144                     if (left.get(key) != right.get(key)) {
145                         throw "Conflicting menu element property " + key;
146                     }
147 
148                     result.set(key, left.get(key));
149                 }
150 
151             } else if (left.get(key)) {
152                 result.set(key,left.get(key));
153 
154             } else if (right.get(key)) {
155                 result.set(key,right.get(key));
156             }
157         }.bind(this));
158         return result;
159     },
160 
161 
162     /**
163      * Creates the menu widget structure from a menu configuration object
164      * @private
165      */
166     _createMenu: function(/** Array */ menuConfiguration) {
167         //Unregistering the old menu elements
168         this._menuElements.each(function(element) {
169             element.unregister(this._registry);
170         }.bind(this));
171         this._menuElements.clear();
172         menuConfiguration.each(function(menuElement) {
173             var menuElementObject = menuElement.toObject();
174             switch(menuElementObject.type.toLowerCase()) {
175                 case "submenu":
176                     var element = new SubMenu(menuElementObject, true);
177                     break;
178                 case "action":
179                     var element = menuElementObject.action;
180                     break;
181             }
182             this._menuElements.push(element);
183             this._menu.addChild(element.getWidget());
184             element.register(this._registry);
185         }.bind(this));
186     }
187 });
