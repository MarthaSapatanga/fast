1 var ScreenComponentInstance = Class.create(ComponentInstance,
  2 /** @lends ScreenComponentInstance.prototype */ {
  3 
  4     /**
  5      * Destroy the instance
  6      * @override
  7      */
  8     destroy: function($super, /** Boolean */ removeFromServer) {
  9         $super();
 10         this._destroyTerminals();
 11     },
 12 
 13     /**
 14      * Create the terminals
 15      */
 16     createTerminals: function(handlers) {
 17         this._destroyTerminals();
 18         this._terminals = new Hash();
 19         this._buildPreConditionsTerminals(handlers, this);
 20         this._buildPostConditionsTerminals(handlers, this);
 21     },
 22 
 23     /**
 24      * Update position of the terminals
 25      * @override
 26      */
 27     updateTerminals: function() {
 28         if (this._terminals) {
 29             this._terminals.values().each(function(terminalGroup) {
 30                 terminalGroup.values().each(function(terminal) {
 31                     terminal.updatePosition();
 32                 });
 33             });
 34         }
 35     },
 36 
 37     /**
 38      * Gets a terminal from an id
 39      * @type Terminal
 40      */
 41     getTerminal: function(/** String */ id, /** String (Optional) */ _action) {
 42         var terminal = null;
 43         if (this._terminals) {
 44             if (_action) {
 45                 var actionTerminals = this._terminals.get(_action);
 46                 if (actionTerminals) {
 47                     terminal = actionTerminals.get(id);
 48                 }
 49             } else {
 50                 terminal = this._terminals.get("postconditions").get(id);
 51             }
 52         }
 53         return terminal;
 54     },
 55 
 56     /**
 57      * On position update
 58      * @override
 59      */
 60     onUpdate: function($super, /** Number */ x, /** Number */ y) {
 61         $super();
 62         this.updateTerminals();
 63     },
 64 
 65     // **************** PRIVATE METHODS **************** //
 66 
 67     /**
 68      * Destroy the terminals
 69      */
 70     _destroyTerminals: function() {
 71         if (this._terminals) {
 72             this._terminals.values().each(function(terminalGroup){
 73                 terminalGroup.values().each(function(terminal){
 74                     terminal.destroy();
 75                 });
 76             });
 77             this._terminals = null;
 78         }
 79     },
 80 
 81         /**
 82      * Creates the preconditions terminals
 83      */
 84     _buildPreConditionsTerminals: function(handlers, instance) {
 85         var options = {
 86             direction:[0,1],
 87             wireConfig: { 'drawingMethod': 'arrows' },
 88             ddConfig: {
 89                 type: 'input',
 90                 allowedTypes: ['output']
 91             }
 92         };
 93 
 94         if (this._preOffsetPosition) {
 95             options.offsetPosition = this._preOffsetPosition;
 96         }
 97 
 98         this._buildingBlockDescription.actions.each(function(action) {
 99             var preconditions = action.preconditions;
100             var actionTerminals = this._buildTerminals(
101                 preconditions,
102                 instance,
103                 options,
104                 handlers,
105                 action.name
106             );
107             this._terminals.set(action.name, actionTerminals);
108         }.bind(this));
109     },
110 
111     /**
112      * Creates the postconditions terminals
113      */
114     _buildPostConditionsTerminals: function(handlers, instance) {
115         var options = {
116             direction:[0,1],
117             wireConfig: { 'drawingMethod': 'arrows' },
118             alwaysSrc: true,
119             ddConfig: {
120                 type: 'output',
121                 allowedTypes: ['input']
122             }
123         };
124 
125         if (this._postOffsetPosition) {
126             options.offsetPosition = this._postOffsetPosition;
127         }
128 
129         var posts = this._buildingBlockDescription.postconditions;
130         if (posts && posts[0] instanceof Array) {
131             posts = this._buildingBlockDescription.postconditions[0];
132         }
133 
134         var terminals = this._buildTerminals(posts, instance, options, handlers);
135         this._terminals.set("postconditions", terminals);
136     },
137 
138     /**
139      * Build terminals Object
140      * @type Hash<ConditionId, Terminal>
141      */
142     _buildTerminals: function(conditions, instance, options, handlers, _name) {
143         var conditionTerminals = new Hash();
144         conditions.each(function(condition) {
145             var node = this._view.getConditionNode(condition.id, _name);
146             var terminal = new Terminal(node, options, instance, condition.id, _name);
147             terminal.onWireHandler(handlers);
148             conditionTerminals.set(condition.id, terminal);
149         }.bind(this));
150         return conditionTerminals;
151     }
152 });
