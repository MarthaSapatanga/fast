 26 var ScreenComponentInstance = Class.create(ComponentInstance,
 27 /** @lends ScreenComponentInstance.prototype */ {
 28 
 29     /**
 30      * Destroy the instance
 31      * @override
 32      */
 33     destroy: function($super, /** Boolean */ removeFromServer) {
 34         $super();
 35         this._destroyTerminals();
 36     },
 37 
 38     /**
 39      * Create the terminals
 40      */
 41     createTerminals: function(handlers) {
 42         this._destroyTerminals();
 43         this._terminals = new Hash();
 44         this._buildPreConditionsTerminals(handlers, this);
 45         this._buildPostConditionsTerminals(handlers, this);
 46     },
 47 
 48     /**
 49      * Update position of the terminals
 50      * @override
 51      */
 52     updateTerminals: function() {
 53         if (this._terminals) {
 54             this._terminals.values().each(function(terminalGroup) {
 55                 terminalGroup.values().each(function(terminal) {
 56                     terminal.updatePosition();
 57                 });
 58             });
 59         }
 60     },
 61 
 62     /**
 63      * Gets a terminal from an id
 64      * @type Terminal
 65      */
 66     getTerminal: function(/** String */ id, /** String (Optional) */ _action) {
 67         var terminal = null;
 68         if (this._terminals) {
 69             if (_action) {
 70                 var actionTerminals = this._terminals.get(_action);
 71                 if (actionTerminals) {
 72                     terminal = actionTerminals.get(id);
 73                 }
 74             } else {
 75                 terminal = this._terminals.get("postconditions").get(id);
 76             }
 77         }
 78         return terminal;
 79     },
 80 
 81     /**
 82      * On position update
 83      * @override
 84      */
 85     onUpdate: function($super, /** Number */ x, /** Number */ y) {
 86         $super();
 87         this.updateTerminals();
 88     },
 89 
 90     // **************** PRIVATE METHODS **************** //
 91 
 92     /**
 93      * Destroy the terminals
 94      */
 95     _destroyTerminals: function() {
 96         if (this._terminals) {
 97             this._terminals.values().each(function(terminalGroup){
 98                 terminalGroup.values().each(function(terminal){
 99                     terminal.destroy();
100                 });
101             });
102             this._terminals = null;
103         }
104     },
105 
106         /**
107      * Creates the preconditions terminals
108      */
109     _buildPreConditionsTerminals: function(handlers, instance) {
110         var options = {
111             direction:[0,1],
112             wireConfig: { 'drawingMethod': 'arrows' },
113             ddConfig: {
114                 type: 'input',
115                 allowedTypes: ['output']
116             }
117         };
118 
119         if (this._preOffsetPosition) {
120             options.offsetPosition = this._preOffsetPosition;
121         }
122 
123         this._buildingBlockDescription.actions.each(function(action) {
124             var preconditions = action.preconditions;
125             var actionTerminals = this._buildTerminals(
126                 preconditions,
127                 instance,
128                 options,
129                 handlers,
130                 action.name
131             );
132             this._terminals.set(action.name, actionTerminals);
133         }.bind(this));
134     },
135 
136     /**
137      * Creates the postconditions terminals
138      */
139     _buildPostConditionsTerminals: function(handlers, instance) {
140         var options = {
141             direction:[0,1],
142             wireConfig: { 'drawingMethod': 'arrows' },
143             alwaysSrc: true,
144             ddConfig: {
145                 type: 'output',
146                 allowedTypes: ['input']
147             }
148         };
149 
150         if (this._postOffsetPosition) {
151             options.offsetPosition = this._postOffsetPosition;
152         }
153 
154         var posts = this._buildingBlockDescription.postconditions;
155         if (posts && posts[0] instanceof Array) {
156             posts = this._buildingBlockDescription.postconditions[0];
157         }
158 
159         var terminals = this._buildTerminals(posts, instance, options, handlers);
160         this._terminals.set("postconditions", terminals);
161     },
162 
163     /**
164      * Build terminals Object
165      * @type Hash<ConditionId, Terminal>
166      */
167     _buildTerminals: function(conditions, instance, options, handlers, _name) {
168         var conditionTerminals = new Hash();
169         conditions.each(function(condition) {
170             var node = this._view.getConditionNode(condition.id, _name);
171             var terminal = new Terminal(node, options, instance, condition.id, _name);
172             terminal.onWireHandler(handlers);
173             conditionTerminals.set(condition.id, terminal);
174         }.bind(this));
175         return conditionTerminals;
176     }
177 });
