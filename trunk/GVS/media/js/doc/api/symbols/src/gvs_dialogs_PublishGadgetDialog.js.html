1 var PublishGadgetDialog = Class.create(ConfirmDialog /** @lends PublishGadgetDialog.prototype */, {
  2     /**
  3      * This class handles a dialog
  4      * able to publish a created gadget into a mashup platform
  5      * @constructs
  6      * @extends ConfirmDialog
  7      */ 
  8     initialize: function($super) {
  9         $super("Publish Gadget", "ok");   
 10         
 11         /**
 12          * Base url of the gadget being published
 13          * @private
 14          * @type String
 15          */
 16         this._gadgetBaseUrl = null;    
 17         
 18         /**
 19          * Hash containing the references to all the buttons
 20          * @type Hash
 21          * @private
 22          */
 23         this._buttons = new Hash(); 
 24     },
 25     
 26     // **************** PUBLIC METHODS **************** //
 27 
 28     
 29     /**
 30      * show
 31      * @override
 32      */
 33     show: function ($super, /** String */ gadgetBaseUrl) {
 34         this._gadgetBaseUrl = gadgetBaseUrl;
 35         this._initDialogInterface();
 36         $super();  
 37         this._buttons.get('ezweb').attr("label", "Publish it!");
 38         this._buttons.get('ezweb').attr("disabled", false);
 39         
 40     },
 41 
 42     // **************** PRIVATE METHODS **************** //
 43 
 44     /**
 45      * Publishes the created gadget
 46      * @private
 47      */
 48     _publishGadget: function(/** dijit.form.Button */ button, /** String */ mashupPlatform) {
 49         if (mashupPlatform == 'ezweb') {
 50             var gadgetUrl = this._gadgetBaseUrl + '/ezweb.xml';
 51             Utils.ezWebDeploy(button, gadgetUrl);
 52         }
 53     },
 54 
 55     /** 
 56      * initDialogInterface
 57      * This function creates the dom structure
 58      * @private
 59      * @override
 60      */
 61     _initDialogInterface: function () {
 62         var dom = new Element('div');
 63         
 64         var title = new Element('h2', {
 65             'class': 'detailsTitle'
 66         }).update("Publish Gadget");
 67         dom.appendChild(title);
 68         
 69         var contents = new Element('div', {
 70             'class': 'deployment'
 71         });
 72         
 73         var table = new Element('table');
 74         contents.appendChild(table);
 75         dom.appendChild(contents);
 76         
 77         this._buttons.set('ezweb', new dijit.form.Button({
 78             'label': 'Publish it!',
 79             'onClick': function(e) {
 80                         this._publishGadget(e.element(), 'ezweb');
 81                     }.bind(this)
 82         }));
 83 
 84         this._buttons.set('igoogle', new dijit.form.Button({
 85             'label': 'Publish it!',
 86             'onClick': function(e) {
 87                         this._publishGadget(e.element(), 'igoogle');
 88                     }.bind(this),
 89             'disabled': true
 90         }));
 91                                 
 92         
 93         var tableData = [
 94             {'className': 'tableHeader',
 95              'fields': [{
 96                 'className': 'left',
 97                 'node': 'Mashup platform'    
 98              },{
 99                 'className': 'left',
100                 'node': ''
101              }]
102             },
103             {'className': '',
104              'fields': [{
105                 'className': 'mashup',
106                 'node': 'EzWeb ' + '[<a href="' + this._gadgetBaseUrl + "/ezweb.xml" + '" target="blank">Template</a>]'
107              },{
108                 'className': '',
109                 'node': this._buttons.get('ezweb').domNode
110              }]
111             },
112              {'className': '',
113              'fields': [{
114                 'className': 'mashup',
115                 'node': 'iGoogle (Not available)'    
116              },{
117                 'className': '',
118                 'node': this._buttons.get('igoogle').domNode
119              }]
120             }
121         ];
122         
123         this._fillTable(table, tableData);
124         this._setContent(dom);
125     },
126     
127     /**
128      * This function fills a table with data
129      * @private
130      */
131     _fillTable: function(/** DOMNode */ tableNode, /** Array */ data) {
132         data.each(function(row) {
133             var tr = new Element('tr', {
134                 'class': row.className
135             });
136             row.fields.each(function(field) {
137                 var td = new Element('td', {
138                     'class': field.className    
139                 }).update(field.node);
140                 tr.appendChild(td);
141             });
142             tableNode.appendChild(tr);
143         });
144     }
145 });
