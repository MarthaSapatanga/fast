1 var PublishGadgetDialog = Class.create(ConfirmDialog /** @lends PublishGadgetDialog.prototype */, {
  2     /**
  3      * This class handles a dialog
  4      * able to publish a created gadget into a mashup platform
  5      * @constructs
  6      * @extends ConfirmDialog
  7      */
  8     initialize: function($super) {
  9         $super("Publish Gadget", ConfirmDialog.OK);
 10 
 11         /**
 12          * Publication Info
 13          * @private
 14          * @type Object
 15          */
 16         this._publication = null;
 17 
 18         /**
 19          * Hash containing the references to all the buttons
 20          * @type Hash
 21          * @private
 22          */
 23         this._buttons = new Hash();
 24 
 25         /**
 26          * Standalone Dialog
 27          * @type StandaloneEmbeddingDialog
 28          * @private
 29          */
 30         this._standaloneDialog = null;
 31     },
 32 
 33     // **************** PUBLIC METHODS **************** //
 34 
 35 
 36     /**
 37      * show
 38      * @override
 39      */
 40     show: function ($super, /** Object */ publication) {
 41         this._publication = publication;
 42         this._initDialogInterface();
 43         $super();
 44     },
 45 
 46     // **************** PRIVATE METHODS **************** //
 47 
 48     /**
 49      * Publishes the created gadget
 50      * @private
 51      */
 52     _publishGadget: function(/** dijit.form.Button */ button, /** Hash */ options) {
 53         var publicationUrl = options.url;
 54         if (GlobalOptions.isLocalStorage) {
 55             if (options.mashupPlatform == 'ezweb') {
 56                 publicationUrl = URIs.ezweb + "interfaces/gadget?template_uri=" + options.url;
 57             } else if (options.mashupPlatform == 'igoogle') {
 58                 if (options.destination=='directory') {
 59                     publicationUrl = "http://www.google.com/ig/submit?url=" + options.url;
 60                 } else if (options.destination=='personal'){
 61                     publicationUrl = "http://www.google.com/ig/adde?moduleurl=" + options.url;
 62                 }
 63             } else if (options.mashupPlatform == 'orkut') {
 64                 publicationUrl = "http://sandbox.orkut.com/Main#MyApps?appUrl=" + options.url;
 65             }
 66         }
 67         this._deploy(button, options, publicationUrl);
 68     },
 69 
 70     /**
 71      * This function deploy a gadget to EzWeb
 72      * @private
 73      */
 74     _deploy: function(/** domNode */ buttonNode, /** Hash */ options, /**String*/ url) {
 75         var button = dijit.byId(buttonNode.id);
 76         if (options.disableAfterPublishing){
 77             button.attr("label", options.doneButtonLabel);
 78             button.attr("disabled", true);
 79         }
 80         if (options.mashupPlatform == 'standalone'){
 81             if (!this._standaloneDialog){
 82                 this._standaloneDialog = new StandaloneEmbeddingDialog(this._publication, url);
 83             } else {
 84                 this._standaloneDialog.updateDialog(this._publication, url);
 85             }
 86             this._standaloneDialog.show();
 87         } else {
 88             window.open(url);
 89         }
 90         console.log(url);
 91     },
 92 
 93     /**
 94      * initDialogInterface
 95      * This function creates the dom structure
 96      * @private
 97      * @override
 98      */
 99     _initDialogInterface: function () {
100         var dom = new Element('div');
101 
102         var title = new Element('h2', {
103             'class': 'detailsTitle'
104         }).update("Publish Gadget");
105         dom.appendChild(title);
106 
107 
108         var tableData = [
109             {'className': 'tableHeader',
110              'fields': [{
111                     'className': 'left',
112                     'node': 'Mashup platform'
113                  },{
114                     'className': 'left',
115                     'node': ''
116                  }]
117             }
118         ];
119 
120         var gadgets = new Hash(this._publication.gadgets);
121 
122         var hasGadget = false;
123         gadgets.each(function(gadget) {
124                 var destinations = PublishGadgetDialog.GADGET_DESTINATIONS.get(gadget.key);
125                 destinations.each(function(destination) {
126                     if (gadget.value) {
127                         destination.url = gadget.value;
128                         tableData.push(this._createPlatformRow(destination));
129                         hasGadget = true;
130                     }
131                 }.bind(this));
132             }.bind(this));
133 
134         var contents = new Element('div', {
135             'class': 'deployment'
136         });
137         dom.appendChild(contents);
138 
139         if (hasGadget) {
140             var table = new Element('table');
141             contents.appendChild(table);
142             this._fillTable(table, tableData);
143         } else {
144             contents.update("Upps! There are no available gadgets");
145         }
146         this._setContent(dom);
147     },
148 
149     /**
150      * This function create a row for a destination platform
151      * @private
152      */
153     _createPlatformRow: function (/** Hash */ options) {
154         return {'className': '',
155             'fields': [{
156                 'className': 'mashup',
157                 'node': options.title + ' [<a href="' + options.url + '" target="blank">' + options.urlLabel + '</a>]'
158              },{
159                 'className': '',
160                 'node': this._buttons.set(options.id, new dijit.form.Button({
161                     'label': options.buttonLabel,
162                     'onClick': function(e) {
163                                 this._publishGadget(e.element(), options);
164                             }.bind(this)
165                 })).domNode
166              }]
167          };
168     },
169 
170     /**
171      * This function fills a table with data
172      * @private
173      */
174     _fillTable: function(/** DOMNode */ tableNode, /** Array */ data) {
175         data.each(function(row) {
176             var tr = new Element('tr', {
177                 'class': row.className
178             });
179             row.fields.each(function(field) {
180                 var td = new Element('td', {
181                     'class': field.className
182                 }).update(field.node);
183                 tr.appendChild(td);
184             });
185             tableNode.appendChild(tr);
186         });
187     }
188 });
