 42     /**
 43      * Creates the terminal
 44      */
 45     createTerminals: function(/** Function */ handler) {
 46         var options = {
 47             'direction':[0,1],
 48             'wireConfig': {
 49                 'drawingMethod': 'arrows'
 50             }
 51         };
 52         this._buildingBlockDescription.actions.each(function(action) {
 53             var actionTerminals = new Hash();
 54             action.preconditions.each(function(pre) { 
 55                 options.ddConfig = {
 56                     'type': 'input',
 57                     'allowedTypes': ['output']
 58                 }
 59                 options.offsetPosition =  {
 60                     'top': 6,
 61                     'left': 0
 62                 }     
 63                 var node = this._view.getConditionNode(pre.id);
 64                 var terminal = new Terminal(node, options, this, pre.id, action.name);
 65                 actionTerminals.set(pre.id, terminal);
 66             }.bind(this));
 67             this._terminals.set(action.name, actionTerminals);
 68         }.bind(this));
 69         
 70         var posts;
 71         if (this._buildingBlockDescription.postconditions && 
 72                 this._buildingBlockDescription.postconditions[0] instanceof Array) {
 73             posts = this._buildingBlockDescription.postconditions[0];
 74         } else {
 75             posts = this._buildingBlockDescription.postconditions;
 76         }
 77 
 78         var postconditionTerminals = new Hash();
 79         posts.each(function(post) {
 80                 options.alwaysSrc = true;
 81                 options.ddConfig = {
 82                      'type': 'output',
 83                      'allowedTypes': ['input']
 84                 }
 85                 options.offsetPosition = {
 86                     'top': -10,
 87                     'left': 2
 88                 }
 89                 var node = this._view.getConditionNode(post.id);
 90                 var terminal = new Terminal(node, options, this, post.id);
 91                 terminal.onWireHandler(handler);
 92                 postconditionTerminals.set(post.id, terminal);
 93             }.bind(this));
 94             this._terminals.set("postconditions", postconditionTerminals);
 95     },
 96 
 97     /**
 98      * Gets a terminal from an id
 99      * @type Terminal
100      */
101     getTerminal: function(/** String */ id, /** String (Optional) */ _action) {
102         var terminal = null;
103         if (_action) {
104             var actionTerminals = this._terminals.get(_action);
105             if (actionTerminals) {
106                 terminal = actionTerminals.get(id);
107             }
108         } else {
109             terminal = this._terminals.get("postconditions").get(id);
110         }
111         return terminal;
112     },
