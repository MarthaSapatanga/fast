 11 UIUtils.onKeyPressCanvas = function(e){
 12     
 13     if (e.keyCode == Event.KEY_DELETE) {
 14     
 15         var currentDocument = GVSSingleton.getInstance().getDocumentController().getCurrentDocument();
 16         
 17         if (currentDocument.getDocumentType() == Constants.DocumentType.SCREENFLOW) {
 18         
 19             var selectedElement = currentDocument.getSelectedElement();
 20             
 21             if (selectedElement != null) { //Delete an element from the canvas
 22                 instance = currentDocument.getBuildingBlockInstance(selectedElement.id);
 23                 if (instance.getBuildingBlockDescription().label){
 24                     var label = instance.getBuildingBlockDescription().label['en-gb'];
 25                 }
 26                 else {//FIXME: workaround for connectors
 27                     var label = instance.getBuildingBlockDescription().name;
 28                 }
 29                 title = (label) ? ('the element "' + label + '"') : "the selected element";
 30                 confirm("You are about to remove " + title + " from canvas. Are you sure?", 
 31                         UIUtils.deleteHandler);
 32             }
 33         }
 34     }
 35 }
 36 
 37 UIUtils.deleteHandler = function (/** Boolean */ remove){
 38     if (remove) {
 39         
 40         var currentDocument = GVSSingleton.getInstance().getDocumentController().getCurrentDocument();
 41         var selectedElement = currentDocument.getSelectedElement();
 42         
 43         if (selectedElement != null) {
 44             instance = currentDocument.getBuildingBlockInstance(selectedElement.id);
 45             type = instance.getBuildingBlockType();
 46 
 47             switch (type) {
 48                 case Constants.BuildingBlock.SCREEN:
 49                     currentDocument.deleteScreen(selectedElement.id);
 50                     selectedElement.parentNode.removeChild(selectedElement);                          
 51                     break;
 52                 case Constants.BuildingBlock.DOMAIN_CONCEPT:
 53                     currentDocument.deleteDomainConcept(selectedElement.id);
 54                     selectedElement.parentNode.removeChild(selectedElement);
 55                     break;
 56                 case Constants.BuildingBlock.CONNECTOR:      
 57                     currentDocument.deleteConnector(selectedElement.id);
 58                     selectedElement.parentNode.removeChild(selectedElement);
 59                     break;
 60                 default:
 61                     alert("Element cannot be deleted");
 62                     break;
 63             }
 64         }
 65     }
 66 }
