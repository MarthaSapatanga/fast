 26 var Area = Class.create( /** @lends Area.prototype */ {
 27     /**
 28      * This class represents an area to drop elements of some kind
 29      * It implements the DropZone interface
 30      * @param Function onDropHandler(DropZone zone, ComponentInstance droppedInstance)
 31      * @constructs
 32      */
 33     initialize: function(/** String */ areaClass, /** Array */ acceptedElements, /** Function */ onDropHandler, /** Object */ _options) {
 34         var options = Utils.variableOrDefault(_options, {});
 35         options = Object.extend ({
 36             style : ""
 37         }, options);
 38 
 39         /**
 40          * List of valid elements to be dropped in the area
 41          * @type Array
 42          * @private
 43          */
 44         this._acceptedElements = acceptedElements;
 45 
 46 
 47         /**
 48          * Function to be called whenever an element
 49          * is dropped into the area
 50          * The handler must return if the element is accepted
 51          * @type Function
 52          * @private
 53          */
 54         this._onDropHandler = onDropHandler;
 55 
 56         if (options.region != 'center') {
 57             if (options['minHeight'] != null) {
 58                 options.style += "height: " + options['minHeight'] + 'px;';
 59             }
 60             if (options['minWidth'] != null) {
 61                 options.style += "width: " + options['minWidth'] + 'px;';
 62             }
 63         }
 64 
 65         /**
 66          * ContentPane of the area
 67          * @type dijit.layout.ContentPane
 68          * @private
 69          */
 70         this._contentPane = new dijit.layout.ContentPane(options);
 71 
 72         /**
 73          * DOM Node of the area
 74          * @type DOMNode
 75          * @private
 76          */
 77         this._node = new Element('div', {
 78             'class': 'dropArea ' + areaClass
 79         });
 80         this._contentPane.setContent(this._node);
 81 
 82     },
 83 
 84 
 85     // **************** PUBLIC METHODS **************** //
 86 
 87     /**
 88      * Implementing DropZone Interface: getNode
 89      * @type DOMNode
 90      */
 91     getNode: function() {
 92         return this._node;
 93     },
 94 
 95     /**
 96      * Returns the  widget
 97      * @type dijit.*
 98      */
 99     getWidget: function () {
100         return this._contentPane;
101     },
102 
103     /**
104      * Implementing DropZone Interface: drop
105      * @type DOMNode
106      */
107     drop: function(/** ComponentInstance */ element, /** Object */ position) {
108 
109         var accepted =  this._onDropHandler(this, element, position);
110         return accepted;
111     },
112 
113 
114     /**
115      * Implementing DropZone Interface: accepts
116      * @type DOMNode
117      */
118     accepts: function() {
119         return this._acceptedElements;
120     },
121 
122     /**
123      * Observe node
124      */
125     observe: function(/** String */ eventName, /** Function */ observer) {
126         this.getNode().observe(eventName, observer);
127     },
128 
129     stopObserving: function(/** String */ eventName) {
130         this.getNode().stopObserving(eventName);
131     },
132 
133     setLayout: function() {
134         //TODO: Think about Layouts
135     },
136 
137     /**
138      * Insert a Element
139      */
140     insert: function(/* Element */ element, /* Object */ position) {
141         $("main").insert(element);
142         position = Geometry.adaptInitialPosition(
143             this._node,
144             element.toElement(),
145             Object.extend({top: 33, left: 33}, position)
146         );
147         element.setPosition(position);
148         this._node.insert(element);
149     }
150 
151     // **************** PRIVATE METHODS **************** //
152 
153 
154 
155 });
