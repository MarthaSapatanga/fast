1 var Area = Class.create( /** @lends Area.prototype */ {
  2     /**
  3      * This class represents an area to drop elements of some kind
  4      * It implements the DropZone interface
  5      * @param Function onDropHandler(DropZone zone, ComponentInstance droppedInstance)
  6      * @constructs
  7      */ 
  8     initialize: function(/** String */ areaClass, /** Array */ acceptedElements, /** Function */ onDropHandler, /** Object */ _options) {
  9         var options = Utils.variableOrDefault(_options, {});
 10         options = Object.extend ({
 11             style : ""
 12         }, options);
 13 
 14         /**
 15          * List of valid elements to be dropped in the area
 16          * @type Array
 17          * @private
 18          */
 19         this._acceptedElements = acceptedElements;
 20         
 21         
 22         /**
 23          * Function to be called whenever an element
 24          * is dropped into the area
 25          * The handler must return if the element is accepted
 26          * @type Function
 27          * @private
 28          */
 29         this._onDropHandler = onDropHandler;
 30 
 31         if (options.region != 'center') {
 32             if (options['minHeight'] != null) {
 33                 options.style += "height: " + options['minHeight'] + 'px;';
 34             }
 35             if (options['minWidth'] != null) {
 36                 options.style += "width: " + options['minWidth'] + 'px;';
 37             }
 38         }
 39 
 40         /**
 41          * ContentPane of the area
 42          * @type dijit.layout.ContentPane
 43          * @private
 44          */
 45         this._contentPane = new dijit.layout.ContentPane(options);
 46 
 47         /**
 48          * DOM Node of the area
 49          * @type DOMNode
 50          * @private
 51          */
 52         this._node = new Element('div', {
 53             'class': 'dropArea ' + areaClass
 54         });
 55         this._contentPane.setContent(this._node);
 56     },
 57     
 58 
 59     // **************** PUBLIC METHODS **************** //
 60 
 61     /**
 62      * Implementing DropZone Interface: getNode
 63      * @type DOMNode
 64      */
 65     getNode: function() {
 66         return this._node;
 67     },
 68     
 69     /**
 70      * Returns the  widget
 71      * @type dijit.*
 72      */
 73     getWidget: function () {
 74         return this._contentPane;
 75     },
 76 
 77     /**
 78      * Implementing DropZone Interface: drop
 79      * @type DOMNode
 80      */
 81     drop: function(/** ComponentInstance */ element, /** Object */ position) {
 82         
 83         var accepted =  this._onDropHandler(this, element, position);
 84         return accepted;
 85     },
 86 
 87    
 88     /**
 89      * Implementing DropZone Interface: accepts
 90      * @type DOMNode
 91      */
 92     accepts: function() {
 93         return this._acceptedElements;    
 94     },
 95     
 96     
 97     setLayout: function() {
 98         //TODO: Think about Layouts    
 99     }
100    
101

103 
104

106 });
