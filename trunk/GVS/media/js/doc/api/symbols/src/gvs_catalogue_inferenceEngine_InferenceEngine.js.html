 26 var InferenceEngine = Class.create( /** @lends InferenceEngine.prototype */ {
 27     /**
 28      * This abstract class represents an inference engine as a proxy for the
 29      * catalogue, handling the reachability and recommendation of building blocks
 30      * @constructs
 31      * @abstract
 32      */
 33     initialize: function() {
 34         /**
 35          * This stores the reachability data
 36          * @type Hash
 37          * @private @member
 38          */
 39         this._reachabilityData = new Hash();
 40 
 41         /**
 42          * Listeners list hashed by resource URI
 43          * @type Hash
 44          * @private @member
 45          */
 46         this._listeners = new Hash();
 47     },
 48 
 49 
 50     // **************** PUBLIC METHODS **************** //
 51 
 52     /**
 53      * This function calls findAndCheck in the catalogue and calls a
 54      * callback upon completion.
 55      */
 56     findCheck: function (/**Array*/ canvas, /** Array*/ elements,
 57                          /** Array */ tags,
 58                          /** String*/ criteria,
 59                          /** Function */ callback) {
 60 
 61         var body = this._constructBody(canvas, elements, tags, criteria);
 62         var context = {
 63             'callback': callback,
 64             'mine': this
 65         };
 66         PersistenceEngine.sendPost(this._getUri("findCheck"), null, body,
 67                 context, this._findCheckOnSuccess, this._onError);
 68     },
 69 
 70     /**
 71      * This function calls the check operation in the catalogue
 72      */
 73     check: function (/**Array*/ canvas, /** Object */ elements,
 74                     /** Array */ tags,
 75                     /** String*/ criteria,
 76                     /** Function */ callback) {
 77         var body = this._constructBody(canvas, elements, tags, criteria, "check");
 78         var context = {
 79             'callback': callback,
 80             'mine': this
 81         };
 82         PersistenceEngine.sendPost(this._getUri("check"), null, body, context,
 83                                     this._checkOnSuccess, this._onError);
 84     },
 85 
 86     /**
 87      * Register an object for interest on the reachability of a URI-identified
 88      * resources. The listener object must implement these methods:
 89      *
 90      *    void setReachability(Hash reachabilityData);
 91      *
 92      * Reachability data vary for different resource types:
 93      *   * Screens: TODO
 94      *   * ...
 95      */
 96     addReachabilityListener: function(/** String */ uri, /** Object */ listener) {
 97         this._getListenerList(uri).push(listener);
 98 
 99         if (this._reachabilityData.get(uri)) {
100             listener.setReachability(this._reachabilityData.get(uri));
101         }
102     },
103 
104     /**
105      * De-register an object from the reachability listeners
106      */
107     removeReachabilityListener: function (/** String  */uri, /** Object */ listener) {
108         var index = this._getListenerList(uri).indexOf(listener);
109 
110         if (index >= 0) {
111             //Remove the element
112             this._listeners.get(uri)[index] = null;
113             this._listeners.set(uri,this._listeners.get(uri).compact());
114         }
115     },
116 
117     /**
118      * Returns the reachability information about
119      * the preconditions of a given screen
120      * @type Hash
121      */
122     getPreconditionReachability: function(/** String */ uri) {
123         var reachabilityData = this._reachabilityData.get(uri);
124         var result = new Hash();
125         if (reachabilityData) {
126             if (reachabilityData.preconditions) {
127                 var preconditions = reachabilityData.preconditions;
128                 $A(preconditions).each(function(pre) {
129                     var uri = Utils.extractURIfromPattern(pre.pattern);
130                     result.set(uri, pre.satisfied);
131                 });
132             } else {
133                 if (reachabilityData.actions) {
134                     $A(reachabilityData.actions).each(function(action) {
135                         $A(action.preconditions).each(function(pre) {
136                             var uri = Utils.extractURIfromPattern(pre.pattern);
137                             result.set(uri,pre.satisfied);
138                         });
139                     });
140                 } else {
141                     console.log("unknown reachability data format");
142                 }
143             }
144         }
145         return result;
146     },
147 
148     /**
149      * Returns a boolean determining if a building block is reachable
150      * by its uri
151      * @type Boolean
152      */
153      isReachable: function (/** String */ uri) {
154          var reachabilityData = this._reachabilityData.get(uri);
155          if (reachabilityData) {
156             return reachabilityData.reachability;
157          } else {
158             return null;
159          }
160      },
161 
162     // **************** PRIVATE METHODS **************** //
163 
164     /**
165      * Creates a body to be sent in an AJAX call to the
166      * catalogue
167      * @private @abstract
168      * @type String
169      */
170     _constructBody: function(/**Array*/ canvas, /** Object */ elements,
171                     /** Array */ tags,
172                     /** String*/ criteria) {
173         throw "Abstract Method invocation: InferenceEngine::_constructBody";
174     },
175 
176     /**
177      * Gets the uri to be called
178      * @private @abstract
179      * @type String
180      */
181     _getUri: function (/** String */ operation) {
182         throw "Abstract Method invocation: InferenceEngine::_getUri";
183     },
184 
185 
186     /**
187      * onSuccess callback
188      * @private
189      * @abstract
190      */
191     _findCheckOnSuccess: function(/**XMLHttpRequest*/ transport){
192         throw "Abstract method invocation: InferenceEngine::_findCheckOnSuccess";
193     },
194 
195     /**
196      * onSuccess callback
197      * @private
198      * @abstract
199      */
200     _checkOnSuccess: function(transport){
201         throw "Abstract method invocation: InferenceEngine::_CheckOnSuccess";
202     },
203 
204     /**
205      * Error handler
206      * @private
207      */
208     _onError: function(transport, e) {
209         var message;
210         if (e) {
211             message = e;
212         } else {
213             message = "Catalogue cannot load";
214         }
215         Utils.showMessage("Something went wrong: " + message, {
216             'error': true,
217             'hide': true
218         });
219         Logger.serverError(transport,e);
220     },
221 
222     /**
223      * Returns a list of listeners of an uri
224      * @private
225      */
226     _getListenerList: function (/** String */ uri) {
227         var list = this._listeners.get(uri);
228         if (!list) {
229             list = new Array();
230             this._listeners.set(uri, list);
231         }
232         return list;
233     },
234     /**
235      * Updates and notifies the reachability
236      * of a list of elements
237      * @private
238      */
239     _updateReachability: function(/** Array */ elements) {
240         elements.each(function(element) {
241             if (element instanceof Object) {
242                 this._reachabilityData.set(element.uri, element);
243                 this._notifyReachability(element.uri);
244             } else {
245                 this._reachabilityData.set(element, true);
246                 this._notifyReachability(element);
247             }
248         }.bind(this));
249     },
250 
251     /**
252      * Notifies the reachability information to all the relevant
253      * listeners
254      * @private
255      */
256     _notifyReachability: function(/** String */ uri) {
257         this._getListenerList(uri).each(function(listener) {
258             listener.setReachability(this._reachabilityData.get(uri));
259         }.bind(this));
260     }
261 });
