171     /**
172      * Creates a new Tooltip for the component view
173      * @type BuildingBlockView
174      * @private
175      */
176     _createTooltip: function (/*DomNode View*/ node) {
177         var screenName = this._buildingBlockDescription.getTitle();
178         var screenVersion =  this._buildingBlockDescription.version;
179         var screenDescription =  this._buildingBlockDescription.description['en-gb'];
180         var screenCreator = this._buildingBlockDescription.creator;
181         var screenshot = this._buildingBlockDescription.screenshot;
182 
183         var pres =  this._getPreConditions();
184         var posts =  this._getPostConditions();
185 
186         var content = document.createElement('div');
187         var title = document.createElement('h3');
188         content.appendChild(title);
189         title.appendChild(document.createTextNode(screenName+' '));
190 
191         var version = document.createElement('span');
192         version.style.color = '#444';
193         version.appendChild(document.createTextNode(screenVersion + ' '));
194         title.appendChild(version);
195 
196         var by = document.createElement('span');
197         by.style.fontSize = '85%';
198         by.style.fontWeight = 'normal';
199         by.appendChild(document.createTextNode('by '));
200         title.appendChild(by);
201 
202         var creator = document.createElement('span');
203         creator.style.fontSize = '85%';
204         creator.style.fontStyle = 'italic';
205         creator.style.fontWeight = 'normal';
206         creator.style.color = '#999';
207         creator.appendChild(document.createTextNode(screenCreator));
208         title.appendChild(creator);
209 
210         if (screenshot) {
211             var screenshotDiv = document.createElement('div');
212             screenshotDiv.style.textAlign = 'center';
213             content.appendChild(screenshotDiv);
214 
215             var image = new Element('img', {
216                 'src': screenshot,
217                 'style': 'width: 250px'
218             });
219             screenshotDiv.appendChild(image);
220         }
221 
222         var description = document.createElement('p');
223         description.appendChild(document.createTextNode(screenDescription));
224         content.appendChild(description);
225 
226 
227         function buildTableConditions(conditions) {
228             var table = document.createElement('table');
229             for (var i=0;conditions && i < conditions.length; i++) {
230                 var condition = conditions[i];
231                 var label = condition.label['en-gb'];
232                 label += " (" + FactFactory.getFactUri(condition) + ")";
233                 var preFact = FactFactory.getFactIcon(condition, "embedded");
234 
235                 var td1 = document.createElement('td');
236                 td1.style.border = 'none';
237                 td1.style.width = '10%';
238                 td1.appendChild(preFact.getNode());
239 
240                 var td2 = document.createElement('td');
241                 td2.style.border = 'none';
242                 td2.appendChild(document.createTextNode(label));
243 
244                 var tr = document.createElement('tr');
245                 tr.appendChild(td1);
246                 tr.appendChild(td2);
247 
248                 table.appendChild(tr);
249             }
250             return table;
251         }
252 
253         if (pres && pres.length > 0) {
254             var inputFacts = document.createElement('h4');
255             inputFacts.appendChild(document.createTextNode('Input Facts:'));
256             content.appendChild(inputFacts);
257             content.appendChild(buildTableConditions(pres));
258         }
259         if (posts && posts.length > 0) {
260             var inputFacts = document.createElement('h4');
261             inputFacts.appendChild(document.createTextNode('Output Facts:'));
262             content.appendChild(inputFacts);
263             content.appendChild(buildTableConditions(posts));
264         }
265 
266         var tip = new dijit.Tooltip({connectId:[node],
267             label:'<div style="max-width:300px">'+content.innerHTML+'</div>'});
268         tip.startup();
269     },
