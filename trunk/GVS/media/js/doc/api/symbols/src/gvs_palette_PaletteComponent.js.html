 26 var PaletteComponent = Class.create(DragSource,
 27     /** @lends PaletteComponent.prototype */ {
 28 
 29     /**
 30      * GUI element that represents a palette element.
 31      * @constructs
 32      * @extends DragSource
 33      */
 34     initialize: function($super,/** BuildingBlockDescription */ buildingBlockDescription,
 35             /** Array */ dropZones, /** InferenceEngine */ inferenceEngine) {
 36         $super();
 37 
 38         /**
 39          * Component and instance Drop zone
 40          * @type Array
 41          * @private
 42          */
 43         this._dropZones = dropZones;
 44 
 45         /**
 46          * BuildingBlock in which this component is based.
 47          * @type BuildingBlockDescription
 48          * @private
 49          */
 50         this._buildingBlockDescription = buildingBlockDescription;
 51 
 52         /**
 53          * @type InferenceEngine
 54          * @private @member
 55          */
 56         this._inferenceEngine = inferenceEngine;
 57 
 58         /**
 59          * BuildingBlock view
 60          * @type BuildingBlockView
 61          * @private
 62          */
 63         this._view = this._createView();
 64 
 65         /**
 66          * Node of the component.
 67          * @type DOMNode
 68          * @private
 69          */
 70         this._node = this._createSlot();
 71     },
 72 
 73     // **************** PUBLIC METHODS **************** //
 74 
 75     /**
 76      * Gets the component root node.
 77      * @type DOMNode
 78      * @public
 79      */
 80     getNode: function() {
 81         return this._node;
 82     },
 83 
 84 
 85     /**
 86      * Returns the node that can be clicked to start a drag-n-drop operation.
 87      * @type DOMNode
 88      * @override
 89      */
 90     getHandlerNode: function() {
 91         return this._view.getNode();
 92     },
 93 
 94     /**
 95      * Creates a new palette component to be dragged.
 96      * Returned object must have a getNode() method.
 97      * @type Object
 98      * @override
 99      */
100     getDraggableObject: function() {
101         var instance = this._createInstance();
102         var node = instance.getHandlerNode();
103         document.body.appendChild(node);
104         node.setStyle({
105             'top': this._getContentOffsetTop() + 'px',
106             'left':  this._getContentOffsetLeft() + 'px',
107             'position': 'absolute'
108         });
109         return instance;
110     },
111 
112     getBuildingBlockDescription: function() {
113         return this._buildingBlockDescription;
114     },
115 
116     /**
117      * Returns the root node
118      * @type DOMNode
119      * @public
120      */
121     getView: function() {
122         return this._view;
123     },
124 
125 
126     // **************** PRIVATE METHODS **************** //
127     /**
128      * Creates an slot (GUI frame around a component) with a given title.
129      *
130      * @private
131      */
132     _createSlot: function () {
133         var node = new Element("div", {"class": "slot"});
134         this._hover = new Element("div", {"class": "hover"});
135         this._hover.setStyle({"opacity": "0"});
136         node.appendChild(this._hover);
137         node.appendChild(this._view.getNode());
138         var titleNode = new Element("div", {"class": "slotTitle"}).update(this._getTitle());
139         node.appendChild(titleNode);
140         this.enableDragNDrop(null, this._dropZones);
141 
142         return node;
143     },
144 
145     /**
146      * Gets the title of the palette component
147      * @type String
148      * @abstract
149      */
150     _getTitle: function () {
151         throw "Abstract Method invocation: PaletteComponent::_getTitle"
152     },
153 
154     /**
155      * Creates a new View instance for the component
156      * @type BuildingBlockView
157      * @abstract
158      */
159     _createView: function () {
160         throw "Abstract Method invocation: PaletteComponent::_createView"
161     },
162 
163 
164     /**
165      * Creates a new component to be dragged.
166      * @type ComponentInstance
167      * @abstract
168      */
169     _createInstance: function () {
170         throw "Abstract Method invocation: PaletteComponent::_createInstance"
171     },
172 
173     /**
174      * Calculates the distance from the window top to the palette component.
175      * @type Integer
176      * @private
177      */
178     _getContentOffsetTop: function() {
179 
180         return this.getView().getNode().cumulativeOffset().top -
181                 this.getView().getNode().cumulativeScrollOffset().top;
182     },
183 
184     /**
185      * Calculates the distance from the window left border to the palette
186      * component.
187      * @type Integer
188      * @private
189      */
190     _getContentOffsetLeft: function() {
191 
192        return this.getView().getNode().cumulativeOffset().left -
193                 this.getView().getNode().cumulativeScrollOffset().left;
194     },
195 
196     /**
197      * Creates a new Tooltip for the component view
198      * @type BuildingBlockView
199      * @private
200      */
201     _createTooltip: function (/*DomNode View*/ node) {
202         var screenName = this._buildingBlockDescription.getTitle();
203         var screenVersion =  this._buildingBlockDescription.version;
204         var screenDescription =  this._buildingBlockDescription.description['en-gb'];
205         var screenCreator = this._buildingBlockDescription.creator;
206         var screenshot = this._buildingBlockDescription.screenshot;
207 
208         var pres =  this._getPreConditions();
209         var posts =  this._getPostConditions();
210 
211         var content = document.createElement('div');
212         var title = document.createElement('h3');
213         content.appendChild(title);
214         title.appendChild(document.createTextNode(screenName+' '));
215 
216         var version = document.createElement('span');
217         version.style.color = '#444';
218         version.appendChild(document.createTextNode(screenVersion + ' '));
219         title.appendChild(version);
220 
221         var by = document.createElement('span');
222         by.style.fontSize = '85%';
223         by.style.fontWeight = 'normal';
224         by.appendChild(document.createTextNode('by '));
225         title.appendChild(by);
226 
227         var creator = document.createElement('span');
228         creator.style.fontSize = '85%';
229         creator.style.fontStyle = 'italic';
230         creator.style.fontWeight = 'normal';
231         creator.style.color = '#999';
232         creator.appendChild(document.createTextNode(screenCreator));
233         title.appendChild(creator);
234 
235         if (screenshot) {
236             var screenshotDiv = document.createElement('div');
237             screenshotDiv.style.textAlign = 'center';
238             content.appendChild(screenshotDiv);
239 
240             var image = new Element('img', {
241                 'src': screenshot,
242                 'style': 'width: 250px'
243             });
244             screenshotDiv.appendChild(image);
245         }
246 
247         var description = document.createElement('p');
248         description.appendChild(document.createTextNode(screenDescription));
249         content.appendChild(description);
250 
251 
252         function buildTableConditions(conditions) {
253             var table = document.createElement('table');
254             for (var i=0;conditions && i < conditions.length; i++) {
255                 var condition = conditions[i];
256                 var label = condition.label['en-gb'];
257                 label += " (" + FactFactory.getFactUri(condition) + ")";
258                 var preFact = FactFactory.getFactIcon(condition, "embedded");
259 
260                 var td1 = document.createElement('td');
261                 td1.style.border = 'none';
262                 td1.style.width = '10%';
263                 td1.appendChild(preFact.getNode());
264 
265                 var td2 = document.createElement('td');
266                 td2.style.border = 'none';
267                 td2.appendChild(document.createTextNode(label));
268 
269                 var tr = document.createElement('tr');
270                 tr.appendChild(td1);
271                 tr.appendChild(td2);
272 
273                 table.appendChild(tr);
274             }
275             return table;
276         }
277 
278         if (pres && pres.length > 0) {
279             var inputFacts = document.createElement('h4');
280             inputFacts.appendChild(document.createTextNode('Input Facts:'));
281             content.appendChild(inputFacts);
282             content.appendChild(buildTableConditions(pres));
283         }
284         if (posts && posts.length > 0) {
285             var inputFacts = document.createElement('h4');
286             inputFacts.appendChild(document.createTextNode('Output Facts:'));
287             content.appendChild(inputFacts);
288             content.appendChild(buildTableConditions(posts));
289         }
290 
291         var tip = new dijit.Tooltip({connectId:[node],
292             label:'<div style="max-width:300px">'+content.innerHTML+'</div>'});
293         tip.startup();
294     },
295 
296     /**
297      * This function returns a list with all the
298      * preconditions of the component.
299      * @private
300      * @type Array
301      */
302     _getPreConditions: function() {
303         return this._getConditions("preconditions");
304     },
305 
306     /**
307      * This function returns a list with all the
308      * postconditions of the component.
309      * @private
310      * @type Array
311      */
312     _getPostConditions: function() {
313         return this._getConditions("postconditions");
314     },
315 
316     /**
317      * This function returns a list with all the
318      * conditions of the component.
319      * @private
320      * @type Array
321      */
322     _getConditions: function(/*String*/type) {
323         return this._buildingBlockDescription[type];
324     }
325 
326 });
