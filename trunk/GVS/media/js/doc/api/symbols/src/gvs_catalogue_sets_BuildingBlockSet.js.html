1 var BuildingBlockSet = Class.create( /** @lends BuildingBlockSet.prototype */ {
  2     /**
  3      * This class handles a set of building blocks for a given context.
  4      * This list will be updated.
  5      * @constructs
  6      */
  7     initialize: function(/** Array */ tags, /** BuildingBlockFactory */ factory) {
  8         /**
  9          * Associated tags
 10          * @type Array
 11          * @private @member
 12          */
 13         this._tags = tags;
 14 
 15         /**
 16          * Building block factory
 17          * @type BuildingBlockFactory
 18          * @private @member
 19          */
 20         this._factory = factory;
 21 
 22 
 23         /**
 24          * Set listener
 25          * @type SetListener
 26          * @private @member
 27          */
 28         this._listener = null;
 29 
 30         /**
 31          * List of BuildingBlock URIs
 32          * @type Array
 33          * @private @member
 34          */
 35         this._uris = new Array();
 36     },
 37 
 38 
 39     // **************** PUBLIC METHODS **************** //
 40 
 41     /**
 42      * Returns all the building block descriptions from the set
 43      * @type Array
 44      * @override
 45      */
 46     getBuildingBlocks: function () {
 47         return this._factory.getBuildingBlocks(this._uris, false);
 48     },
 49 
 50     /**
 51      * Gets the type of building block this factory mades.
 52      * @type String
 53      */
 54     getBuildingBlockType: function (){
 55         return this._factory.getBuildingBlockType();
 56     },
 57 
 58     /**
 59      * Gets the human-readable name of the building block type.
 60      * @type String
 61      */
 62     getBuildingBlockName: function (){
 63         return this._factory.getBuildingBlockName();
 64     },
 65 
 66     setListener: function (/** SetListener */ listener) {
 67         this._listener = listener;
 68     },
 69 
 70      /**
 71      * Add new building blocks to the set by uri
 72      */
 73     setURIs: function (/** Array */ uris) {
 74         this._requestedUris = uris;
 75         this._factory.cacheBuildingBlocks(uris, this._cachedElements.bind(this));
 76     },
 77 
 78     /**
 79      * This is the callback called when returning from the
 80      * building block factory
 81      * @private
 82      */
 83     _cachedElements: function () {
 84         if (GVS.getUser().getCatalogueMagic()) {
 85             this._uris = this._requestedUris;
 86         } else {
 87             this._uris = this._uris.concat(this._requestedUris).uniq();
 88         }
 89         this._listener.setChanged();
 90     }
 91 });
