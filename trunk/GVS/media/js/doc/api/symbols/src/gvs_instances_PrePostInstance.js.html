142     /**
143      * Set the type in pre | post
144      */
145     setType: function(/** String */ type) {
146         this._buildingBlockDescription.type = type;
147         if (this._isConfigurable) {
148             var data = {
149                 'label': this.getTitle()
150             };
151             if (this._buildingBlockDescription.properties) {
152                 data = Object.extend(data, this._buildingBlockDescription.properties.ezweb)
153             } else {
154                 data = Object.extend(data, {
155                     'binding': this._buildingBlockDescription.type == 'pre' ? 'slot' : 'event',
156                     'variableName': this.getTitle().replace(" ",""),
157                     'friendcode': this.getTitle().replace(" ", "")
158                 });
159             }
160             this._onChange(data);
161         } else {
162             this._onClick();
163         }

203     getConditionTable: function(/** Boolean */ reachabilityInfo) {
204         var fact = FactFactory.getFactIcon(this._getCondition(), "embedded").getNode();
205         var reachable;
206         if (reachabilityInfo !== undefined) {

216     /**
217      * Creates the terminal
218      */
219     createTerminal: function(/** Function (Optional) */ _handler) {
220         var options = {
221             'direction':[],
222             'offsetPosition': {},
223             'wireConfig': {
224                 'drawingMethod': 'arrows'
225             }
226         };
227         if (this._buildingBlockDescription.type == 'pre') {
228             options.alwaysSrc = true;
229             options.direction = [1,0];
230             options.offsetPosition = {
231                 'top': 2, 
232                 'left': 15
233             };
234             options.ddConfig = {// A precondition in screen design is an output 
235                                 // (data to be consumed inside the screen)
236                 'type': 'output',
237                 'allowedTypes': ['input']
238             }
239         } else {
240             options.direction = [-1,0];
241             options.offsetPosition = {
242                 'top': 2, 
243                 'left': -8
244             };
245             options.ddConfig = { // Viceversa
246                 'type': 'input',
247                 'allowedTypes': ['output']
248             }   
249         }
250         
251         this._terminal = new Terminal(this._view.getNode(), options, this,
252                                         this.getId());
253         if (this._buildingBlockDescription.type == 'pre') {
254             this._terminal.onWireHandler(_handler);
255         }
256     },

270     destroy: function($super, /** Boolean */ removeFromServer) {
271         if (this._buildingBlockDescription.catalogueUri && removeFromServer) {
272             var catalogueResource = (this._buildingBlockDescription.type == 'pre') ?
273                                     URIs.pre : URIs.post;
274             this._removeFromServer(catalogueResource + this.getId());
275         }
276         if (this._terminal) {
277             this._terminal.destroy();
278         }
279         $super();
280     },

342     /**
343      * This function is called when the dialog is saved
344      * @private
345      */
346     _onChange: function (/** Object */ data) {
347         this._buildingBlockDescription.title = data.label;
348         if (data.binding) {
349             this._buildingBlockDescription.properties = {
350                 'ezweb': {
351                     'binding': data.binding,
352                     'variableName': data.variableName,
353                     'friendcode': data.friendcode
354                 }
355             }
356         }
357         
358         if (!this._buildingBlockDescription.catalogueUri) {
359             // Calling the server to add the pre/post
360             var catalogueResource = (this._buildingBlockDescription.type == 'pre') ? URIs.pre : URIs.post;
361             this._id = new Date().valueOf();
362             PersistenceEngine.sendPost(catalogueResource,
363                             null, Object.toJSON(this._toJSONForCatalogue()), this,
364                             this._onPostSuccess, Utils.onAJAXError);
365         } else {
366             this._onClick();
367         }
368     },
369     /**
370      * onSuccess callback
371      * @private
372      */
373     _onPostSuccess: function (/** XMLHttpRequest */ transport) {
374         var result = JSON.parse(transport.responseText);
375         this._buildingBlockDescription.catalogueUri = result.uri;
376         
377         
378         this._inferenceEngine.addReachabilityListener(result.uri, this._view);
379         
380         // Notify change
381         this._changeHandler(this);
382         
383     },
