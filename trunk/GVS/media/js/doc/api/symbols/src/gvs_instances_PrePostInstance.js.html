1 var PrePostInstance = Class.create(ComponentInstance,
  2     /** @lends PrePostInstance.prototype */ {
  3 
  4     /**
  5      * Pre or Post instance.
  6      * @constructs
  7      * @extends ComponentInstance
  8      */
  9     initialize: function($super, /** DomainConceptDescription */ domainConceptDescription, 
 10             /** DropZone */ dropZone, /** InferenceEngine */ inferenceEngine) {
 11         $super(domainConceptDescription, dropZone, inferenceEngine);
 12         
 13         /**
 14          * Uri of the pre or post in the catalogue
 15          * @type String
 16          * @private @member
 17          */
 18         this._uri = null;
 19         
 20         /**
 21          * Pattern of the *-condition
 22          * @type String
 23          * @private @member
 24          */
 25         this._pattern = "?x http://www.w3.org/1999/02/22-rdf-syntax-ns#type " + 
 26             this._buildingBlockDescription.uri;
 27         /**
 28          * @type String
 29          * @private @member
 30          */
 31         this._label = this._buildingBlockDescription.title;
 32         
 33         /**
 34          * @type DomainConceptDialog
 35          * @private @member
 36          */
 37         this._dialog = new PrePostDialog(this._onChange.bind(this), this.getTitle());
 38         
 39         /**
 40          * This stores the platform-dependent properties
 41          * @type Hash
 42          * @private @member
 43          */
 44         this._platformProperties = new Hash();
 45         this._platformProperties.set('ezweb', new Hash());
 46         
 47         /**
 48          * @private @member
 49          * @type Function
 50          */
 51         this._changeHandler = null;
 52         /**
 53          * Type pre/post
 54          * @private @member
 55          * @type String
 56          */
 57         this._type = null;
 58     },
 59 
 60     // **************** PUBLIC METHODS **************** //
 61 
 62     /**
 63      * Somehow something the user can comprehend
 64      * Implementing TableModel interface
 65      * @override
 66      */
 67     getTitle: function() {
 68         return this._label; 
 69     },
 70     
 71     /**
 72      * This function returns the relevant info
 73      * to the properties table
 74      * Implementing TableModel interface
 75      * @overrides
 76      */
 77     getInfo: function() {
 78         var info = new Hash();
 79         info.set('Title', this._label);
 80         info.set('Type', this._type);
 81         info.set('EzWeb Binding', this._platformProperties.get('ezweb').get('binding'));
 82         info.set('Friendcode', this._platformProperties.get('ezweb').get('friendcode'));
 83         info.set('Variable Name', this._platformProperties.get('ezweb').get('varname'));
 84         return info;
 85     },
 86     
 87     /**
 88      * Returning the type in {pre|post}
 89      */
 90     getType: function() {
 91         return this._type;
 92     },
 93     
 94     /**
 95      * Transform the instance into JSON-like
 96      * string
 97      * @type String
 98      */
 99     toJSON: function() {
100         var json = {
101             'conditions': [this._getFactData()]
102         }
103         return Object.toJSON(json);
104     },
105     
106     /**
107      * Returns an object with the relevant
108      * information to the screenflow description
109      * @type Object
110      */
111     getProperties: function() {
112         var result = {
113             'semantics': this._buildingBlockDescription.uri,
114             'label': this._label,
115             'friendcode': this._platformProperties.get('ezweb').get('friendcode'),
116             'variableName': this._platformProperties.get('ezweb').get('varname'),
117             'binding': this._platformProperties.get('ezweb').get('binding'),
118             'uri': this._uri
119         };
120         return result;
121     },
122     
123     /**
124      * @override
125      */
126     getUri: function() {
127         return this._uri;    
128     },
129     
130     /**
131      * Due to the slopy catalogue implementation, uri changes can
132      * be notified via handler.
133      * @public
134      */
135     setChangeHandler: function(/** Function */ handler) {
136         this._changeHandler = handler;        
137     },
138     
139     /**
140      * This function shows the dialog to change
141      * the instance properties
142      */
143     showPreviewDialog: function () {
144         this._dialog.show();        
145     },
146     
147     /**
148      * Returns a list the 
149      * information about the instance
150      * ready to be set in the FactPane
151      * @type Array
152      */
153     getConditionTable: function(/** Boolean */ reachable) {
154         var factFactory = FactFactorySingleton.getInstance();
155         var fact = factFactory.getFactIcon(this._getFactData(), "embedded").getNode();
156         Utils.setSatisfeabilityClass(fact, reachable);
157         return [fact, this._label, this._buildingBlockDescription.uri]; 
158     },
159 
160     // **************** PRIVATE METHODS **************** //
161     /**
162      * Creates a new View instance for the component
163      * @type BuildingBlockView
164      * @override
165      */
166     _createView: function () {
167         return new DomainConceptView(this._buildingBlockDescription);
168     },
169     
170     /**
171      * This function is called when the attached view is dbl-clicked
172      * @private
173      * @override
174      */
175     _onDoubleClick: function (/** Event */ event){
176         this._dialog.show();
177     },
178     
179     /**
180      * This function is called when the dialog is saved
181      * @private
182      */
183     _onChange: function (/** Object */ data) {
184         
185         this._type = data.type;
186         this._label = data.label;
187         this._platformProperties.get('ezweb').set('binding', data.binding);
188         this._platformProperties.get('ezweb').set('varname', data.varname);
189         this._platformProperties.get('ezweb').set('friendcode', data.friendcode);
190         
191         var catalogueResource = URIs.catalogue + 
192                 ((this._type == 'pre') ? '/slots' : '/events');
193                 
194         var persistenceEngine = PersistenceEngineFactory.getInstance();
195         persistenceEngine.sendPost(catalogueResource,
196             null, this.toJSON(), this, 
197             this._onPostSuccess, Utils.onAJAXError);
198     },
199     /**
200      * onSuccess callback
201      * @private
202      */
203     _onPostSuccess: function (/** XMLHttpRequest */ transport) {
204         var result = JSON.parse(transport.responseText);
205         var previousUri = this._uri;
206         this._uri = result.uri;
207         
208         // Update notification listeners
209         if (previousUri) {
210             this._inferenceEngine.removeReachabilityListener(previousUri, this._view);
211         }
212         this._inferenceEngine.addReachabilityListener(this._uri, this._view);
213         
214         // Notify change
215         this._changeHandler(previousUri, this);
216     },
217     /**
218      * Returns an object representing
219      * the fact
220      * @type Object
221      */
222     _getFactData: function() {
223         return {
224                 'label': {'en-gb': this._label},
225                 'pattern': this._pattern,
226                 'scope': 'design time'
227             };
228     }
229 });
