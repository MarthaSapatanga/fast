 12         $super(domainConceptDescription.clone(), inferenceEngine);

 14         if (!this._buildingBlockDescription.pattern) {

 17                 this._buildingBlockDescription.uri;

 20 
 21         if (!this._buildingBlockDescription.label) {
 22             this._buildingBlockDescription.label = {

 24             };

 63     /**
 64      * This function returns the relevant info
 65      * to the properties table
 66      * Implementing TableModel interface
 67      * @overrides
 68      */
 69     getInfo: function() {
 70         var info = new Hash();
 71         info.set('Title', this.getTitle());
 72         info.set('Uri', this._buildingBlockDescription.uri);
 73         info.set('Type', this.getType());
 74         if (this._buildingBlockDescription.properties) {
 75             info.set('EzWeb Binding', this._buildingBlockDescription.properties.ezweb.binding);
 76             info.set('Friendcode', this._buildingBlockDescription.properties.ezweb.friendcode);
 77             info.set('Variable Name',this._buildingBlockDescription.properties.ezweb.variableName);
 78         }
 79         return info;

144     /**
145      * Set the type in pre | post
146      */
147     setType: function(/** String */ type) {
148         this._buildingBlockDescription.type = type;
149         if (this._isConfigurable) {
150             var data = {
151                 'label': this.getTitle()
152             };
153             if (this._buildingBlockDescription.properties) {
154                 data = Object.extend(data, this._buildingBlockDescription.properties.ezweb);
155             } else {
156                 if (this._buildingBlockDescription.binding) {
157                     data = Object.extend (data, {
158                         'binding': this._buildingBlockDescription.binding,
159                         'variableName': this._buildingBlockDescription.variableName,
160                         'friendcode': this._buildingBlockDescription.friendcode
161                     });
162                 } else {
163                      data = Object.extend(data, {
164                         'binding': this._buildingBlockDescription.type == 'pre' ? 'slot' : 'event',
165                         'variableName':
166                         this._buildingBlockDescription.getUriShortcut().replace(/\s+/gi,""),
167                         'friendcode': this._buildingBlockDescription.getUriShortcut().replace(/\s+/gi, "")
168                     });
169                 }
170 
171             }
172             this._onChange(data);
173         } else {
174             this._onClick();
175         }
176     },

214     getConditionTable: function() {
215         var fact = FactFactory.getFactIcon(this._getCondition(), "embedded").getNode();
216         var reachable;

227     /**
228      * Creates the terminal
229      */
230     createTerminal: function(/** Function (Optional) */ _handler) {
231         var options = {
232             'direction':[],
233             'offsetPosition': {},
234             'wireConfig': {
235                 'drawingMethod': 'arrows'
236             }
237         };
238         if (this._buildingBlockDescription.type == 'pre') {
239             options.alwaysSrc = true;
240             options.direction = [1,0];
241             options.offsetPosition = {
242                 'top': 9,
243                 'left': 26
244             };
245             options.ddConfig = {// A precondition in screen design is an output
246                                 // (data to be consumed inside the screen)
247                 'type': 'output',
248                 'allowedTypes': ['input']
249             };
250         } else {
251             options.direction = [-1,0];
252             options.offsetPosition = {
253                 'top': 9,
254                 'left': -8
255             };
256             options.ddConfig = { // Viceversa
257                 'type': 'input',
258                 'allowedTypes': ['output']
259             };
260         }
261 
262         this._terminal = new Terminal(this._view.getNode(), options, this,
263                                         this.getId());
264         if (this._buildingBlockDescription.type == 'pre') {
265             this._terminal.onWireHandler(_handler);
266         }
267     },
