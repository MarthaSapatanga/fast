 26 var PrePostInstance = Class.create(ComponentInstance,
 27     /** @lends PrePostInstance.prototype */ {
 28 
 29     /**
 30      * Pre or Post instance.
 31      * @constructs
 32      * @extends ComponentInstance
 33      */
 34     initialize: function($super, /** BuildingBlockDescription */ domainConceptDescription,
 35             /** InferenceEngine */ inferenceEngine, /** Boolean (Optional) */ _isConfigurable) {
 36 
 37         $super(domainConceptDescription.clone(), inferenceEngine);
 38 
 39         if (!this._buildingBlockDescription.pattern) {
 40             this._buildingBlockDescription.pattern = "?x " +
 41                 "http://www.w3.org/1999/02/22-rdf-syntax-ns#type " +
 42                 this._buildingBlockDescription.uri;
 43         }
 44 
 45 
 46         if (!this._buildingBlockDescription.label) {
 47             this._buildingBlockDescription.label = {
 48                 'en-gb': this._buildingBlockDescription.title
 49             };
 50         }
 51 
 52         if (this._buildingBlockDescription.id) {
 53             this._id = this._buildingBlockDescription.id;
 54         }
 55 
 56         /**
 57          * @type DomainConceptDialog
 58          * @private @member
 59          */
 60         this._dialog = null;
 61 
 62 
 63         /**
 64          * @private @member
 65          * @type Function
 66          */
 67         this._changeHandler = null;
 68 
 69 
 70         /**
 71          * Terminal for screen design
 72          * @type Terminal
 73          * @private
 74          */
 75         this._terminal = null;
 76 
 77         /**
 78          * It states if the instance
 79          * can be configured by the user
 80          * @type Boolean
 81          * @private
 82          */
 83         this._isConfigurable = Utils.variableOrDefault(_isConfigurable, true);
 84     },
 85 
 86     // **************** PUBLIC METHODS **************** //
 87 
 88     /**
 89      * This function returns the relevant info
 90      * to the properties table
 91      * Implementing TableModel interface
 92      * @overrides
 93      */
 94     getInfo: function() {
 95         var info = new Hash();
 96         info.set('Title', this.getTitle());
 97         info.set('Uri', this._buildingBlockDescription.uri);
 98         info.set('Type', this.getType());
 99         if (this._buildingBlockDescription.properties) {
100             info.set('EzWeb Binding', this._buildingBlockDescription.properties.ezweb.binding);
101             info.set('Friendcode', this._buildingBlockDescription.properties.ezweb.friendcode);
102             info.set('Variable Name',this._buildingBlockDescription.properties.ezweb.variableName);
103         }
104         return info;
105     },
106 
107     /**
108      * Returning the type in {pre|post}
109      */
110     getType: function() {
111         return this._buildingBlockDescription.type;
112     },
113 
114 
115     /**
116      * Returns an object representing
117      * the fact
118      * @type Object
119      */
120     toJSONForScreen: function() {
121         return {
122                 'label': this._buildingBlockDescription.label,
123                 'pattern': this._buildingBlockDescription.pattern,
124                 'positive': true,
125                 'uri': this._buildingBlockDescription.uri,
126                 'id': this.getId(),
127                 'type': this._buildingBlockDescription.type
128             };
129     },
130 
131 
132     /**
133      * Returns an object representing
134      * the fact
135      * @type Object
136      */
137     toJSONForScreenflow: function() {
138         return {
139             'conditions': [this._getCondition()],
140             'id': this.getId(),
141             'semantics': this._buildingBlockDescription.uri,
142             'binding': this._buildingBlockDescription.properties.ezweb.binding,
143             'friendcode': this._buildingBlockDescription.properties.ezweb.friendcode,
144             'variableName': this._buildingBlockDescription.properties.ezweb.variableName,
145             'label': this.getTitle(),
146             'type': this._buildingBlockDescription.type,
147             'uri': this._buildingBlockDescription.uri
148         };
149     },
150 
151 
152     /**
153      * Returns an object representing
154      * the fact
155      * @type Object
156      * @private
157      */
158     toJSONForCatalogue: function() {
159         return this._getCondition();
160     },
161 
162     /**
163      * @override
164      */
165     getUri: function() {
166         return this._id;
167     },
168 
169     /**
170      * Set the type in pre | post
171      */
172     setType: function(/** String */ type) {
173         this._buildingBlockDescription.type = type;
174         if (this._isConfigurable) {
175             var data = {
176                 'label': this.getTitle()
177             };
178             if (this._buildingBlockDescription.properties) {
179                 data = Object.extend(data, this._buildingBlockDescription.properties.ezweb);
180             } else {
181                 if (this._buildingBlockDescription.binding) {
182                     data = Object.extend (data, {
183                         'binding': this._buildingBlockDescription.binding,
184                         'variableName': this._buildingBlockDescription.variableName,
185                         'friendcode': this._buildingBlockDescription.friendcode
186                     });
187                 } else {
188                      data = Object.extend(data, {
189                         'binding': this._buildingBlockDescription.type == 'pre' ? 'slot' : 'event',
190                         'variableName':
191                         this._buildingBlockDescription.getUriShortcut().replace(/\s+/gi,""),
192                         'friendcode': this._buildingBlockDescription.getUriShortcut().replace(/\s+/gi, "")
193                     });
194                 }
195 
196             }
197             this._onChange(data);
198         } else {
199             this._onClick();
200         }
201     },
202 
203     /**
204      * Sets the configurable status
205      */
206     setConfigurable: function(/** Boolean */ configurable) {
207         this._isConfigurable = configurable;
208     },
209 
210     /**
211      * Due to the slopy catalogue implementation, uri changes can
212      * be notified via handler.
213      * @public
214      */
215     setChangeHandler: function(/** Function */ handler) {
216         this._changeHandler = handler;
217     },
218 
219     /**
220      * This function shows the dialog to change
221      * the instance properties
222      */
223     showPreviewDialog: function () {
224         if (this._isConfigurable) {
225             if (!this._dialog) {
226                 this._dialog = new PrePostDialog(this._onChange.bind(this),
227                                                  this._buildingBlockDescription);
228             }
229             this._dialog.show();
230         }
231     },
232 
233     /**
234      * Returns a list the
235      * information about the instance
236      * ready to be set in the FactPane
237      * @type Array
238      */
239     getConditionTable: function() {
240         var fact = FactFactory.getFactIcon(this._getCondition(), "embedded").getNode();
241         var reachable;
242         if (this._reachable != null) {
243             reachable = this._reachable;
244         } else {
245             reachable = this._view.getReachability();
246         }
247         Utils.setSatisfeabilityClass(fact, reachable);
248 
249         return [fact, this.getTitle(), this._buildingBlockDescription.uri];
250     },
251 
252     /**
253      * Creates the terminal
254      */
255     createTerminal: function(/** Function (Optional) */ _handler) {
256         var options = {
257             'direction':[],
258             'offsetPosition': {},
259             'wireConfig': {
260                 'drawingMethod': 'arrows'
261             }
262         };
263         if (this._buildingBlockDescription.type == 'pre') {
264             options.alwaysSrc = true;
265             options.direction = [1,0];
266             options.offsetPosition = {
267                 'top': 9,
268                 'left': 26
269             };
270             options.ddConfig = {// A precondition in screen design is an output
271                                 // (data to be consumed inside the screen)
272                 'type': 'output',
273                 'allowedTypes': ['input']
274             };
275         } else {
276             options.direction = [-1,0];
277             options.offsetPosition = {
278                 'top': 9,
279                 'left': -8
280             };
281             options.ddConfig = { // Viceversa
282                 'type': 'input',
283                 'allowedTypes': ['output']
284             };
285         }
286 
287         this._terminal = new Terminal(this._view.getNode(), options, this,
288                                         this.getId());
289         if (this._buildingBlockDescription.type == 'pre') {
290             this._terminal.onWireHandler(_handler);
291         }
292     },
293 
294     /**
295      * Gets the terminal
296      * @type Terminal
297      */
298     getTerminal: function() {
299         return this._terminal;
300     },
301 
302     /**
303      * Destroy the instance
304      * @override
305      */
306     destroy: function($super, /** Boolean */ removeFromServer) {
307         if (this._terminal) {
308             this._terminal.destroy();
309             this._terminal = null;
310         }
311         $super();
312     },
313 
314     /**
315      * On position update
316      * @override
317      */
318     onUpdate: function(/** Number */ x, /** Number */ y) {
319         if (this._terminal) {
320             this._terminal.updatePosition();
321         }
322     },
323 
324     // **************** PRIVATE METHODS **************** //
325     /**
326      * Creates a new View instance for the component
327      * @type BuildingBlockView
328      * @override
329      */
330     _createView: function () {
331         return new DomainConceptView(this._buildingBlockDescription);
332     },
333 
334     /**
335      * This function is called when the attached view is dbl-clicked
336      * @private
337      * @override
338      */
339     _onDoubleClick: function (/** Event */ event) {
340         this.showPreviewDialog();
341     },
342 
343     /**
344      * Returns an object representing
345      * a single fact
346      * @type Object
347      * @private
348      */
349     _getCondition: function() {
350         return {
351                 'label': {
352                     'en-gb': this.getTitle()
353                 },
354                 'pattern': this._buildingBlockDescription.pattern,
355                 'positive': true,
356                 'uri': this._buildingBlockDescription.uri,
357                 'id': this._id
358             };
359     },
360 
361 
362     /**
363      * This function is called when the dialog is saved
364      * @private
365      */
366     _onChange: function (/** Object */ data) {
367         this._buildingBlockDescription.title = data.label;
368         if (data.binding) {
369             this._buildingBlockDescription.properties = {
370                 'ezweb': {
371                     'binding': data.binding,
372                     'variableName': data.variableName,
373                     'friendcode': data.friendcode
374                 }
375             };
376         }
377         this._onClick();
378     }
379 
380 });
