 26 var MenuOptions = Class.create({
 27     initialize: function(domNode) {
 28         this._dijitMenu = new dijit.Menu();
 29         if (domNode) {
 30             this.bindDomNode(domNode);
 31         }
 32     },
 33 
 34     bindDomNode: function(domNode) {
 35         this._dijitMenu.bindDomNode(domNode);
 36     },
 37 
 38     addOption: function(label, handler, options) {
 39         var menuCfg = {label:label, onClick:handler};
 40         if (options) {
 41             Object.extend(menuCfg, options)
 42         }
 43         var menuItem = new dijit.MenuItem(menuCfg);
 44         this._dijitMenu.addChild(menuItem);
 45     }
 46 });
