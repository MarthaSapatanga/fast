139         for (var i=0; i<this._screens.length; i++) {

145             if (this._connectors[i].getView().getId()==buildingBlockViewId) {
146                 return this._connectors[i];
147             }
148         }
149         for (var i=0; i<this._domainConcepts.length; i++) {
150             if (this._domainConcepts[i].getView().getId()==buildingBlockViewId) {
151                 return this._domainConcepts[i];

350     updatePropertiesPane: function( /** BuildingBlockId */ buildingBlockId, /** String */ buildingBlockType) {
351         var buildingBlockInstance = this.getBuildingBlockInstance(buildingBlockId);
352         var buildingBlockDescription = buildingBlockInstance.getBuildingBlockDescription();
353         switch(buildingBlockType){
354             case Constants.BuildingBlock.SCREEN:
355                 var propertiesHash = new Hash();
356                 propertiesHash.set('title',buildingBlockDescription.label['en-gb']);
357                 propertiesHash.set('id',buildingBlockDescription.uri);
358                 propertiesHash.set('desc',buildingBlockDescription.description['en-gb']);
359                 propertiesHash.set('tags',buildingBlockDescription.domainContext.tags);
360                 this._propertiesPane.selectElement(propertiesHash,"screen",buildingBlockDescription.label['en-gb']);
361                 break;
362 
363             case Constants.BuildingBlock.CONNECTOR:
364                 var propertiesHash = buildingBlockInstance.getProperties().clone();
365                 this._propertiesPane.selectElement(propertiesHash,"connector",propertiesHash.get('type'));
366                 break;
367 
368             case Constants.BuildingBlock.DOMAIN_CONCEPT:
369                 var propertiesHash = new Hash();
370                 propertiesHash.set('name',buildingBlockDescription.name);
371                 propertiesHash.set('description',buildingBlockDescription.description);
372                 propertiesHash.set('semantics',buildingBlockDescription.semantics);
373                 this._propertiesPane.selectElement(propertiesHash,"domain concept", buildingBlockDescription.name);
374                 break;
375 
376             default:
377                 console.debug("properties pane called without buildingBlocktype", buildingBlockType);
378                 break;
379         }

383     updateReachability: function(/** map id->value*/screenList){
384         var screens = this.getScreens();
385         for (var i = 0; i < screens.length; i++) {
386             for (var j = 0; j < screenList.length; j++) {
387                 if (screens[i].getBuildingBlockDescription().uri == screenList[j].uri) {
388                     if (screenList[j].reachability == true) {
389                         screens[i].getBuildingBlockDescription().satisfeable = true;
390                         break;
391                     }
392                     else {
393                         screens[i].getBuildingBlockDescription().satisfeable = false;
394                     }
395                 }
396             }
397             screens[i].colorize();
398         }

409     onClickCanvas: function( /**Element*/ element){
410         var clickedElement = this.getBuildingBlockElement(element);
411         var buildingBlockType = this.getBuildingBlockType(clickedElement);
412         switch (buildingBlockType) {
413             case Constants.BuildingBlock.SCREEN:
414                 console.log("screen clicked");
415                 this.setSelectedElement(clickedElement);
416                 this.updatePropertiesPane(clickedElement.id, Constants.BuildingBlock.SCREEN);
417                 break;
418             case Constants.BuildingBlock.DOMAIN_CONCEPT:
419                 console.log("domain concept clicked");
420                 this.setSelectedElement(clickedElement);
421                 this.updatePropertiesPane(clickedElement.id, Constants.BuildingBlock.DOMAIN_CONCEPT);
422                 break;
423             case Constants.BuildingBlock.CONNECTOR:
424                 console.log("connector clicked");
425                 this.setSelectedElement(clickedElement);
426                 this.updatePropertiesPane(clickedElement.id, Constants.BuildingBlock.CONNECTOR);
427                 break;
428             case 'canvas':
429                 console.log("canvas clicked");
430                 this.setSelectedElement();
431                 this._propertiesPane.clearElement();
432                 break;
433             case "unknown":
434             default:
435                 console.log("unknown clicked");
436                 this.setSelectedElement();
437                 this._propertiesPane.clearElement();
438                 break;
439         }

468     /**
469      * Gets the building block div in the canvas
470      * @param {Element} element
471      */
472     getBuildingBlockElement: function(element){
473         if (element.hasClassName('canvas') && element.hasClassName('document')) {
474             return element;
475         } else {
476             while (element.up(0) != undefined) {
477                 if (element.up(0).hasClassName('canvas') && element.up(0).hasClassName('document')) {
478                     return element;
479                 } else {
480                     element = element.up(0);
481                 }
482             }
483         }
484         return null;
485     },
486 
487     getBuildingBlockType: function(element){
488         if (element.hasClassName('canvas') && element.hasClassName('document')) {
489             return 'canvas';
490         } else {
491             var elementClass = $w(element.className);
492             elementClass = elementClass.without("unknown").without("satisfeable").
493                             without("unsatisfeable").without("selected").without("view");
494             return ((elementClass.size() >= 1) ? elementClass[0] : 'unknown');
495         }
496     },

523         var mainBorderContainer = new dijit.layout.BorderContainer({
524             id:borderContainerId,
525             design:"sidebar",
526             liveSplitters:"false",
527             splitter:"true"
528         });
529         
530         var rightBorderContainer = new dijit.layout.BorderContainer({
531             id:borderContainerId + "Child",
532             design:"headline",
533             liveSplitters:"false",
534             region:"center"
535         });
536 
537         var documentContent = new Element("div", {
538             "id": this._tabContentId,
539             "class": "document screenflow canvas"
540         });
541         documentContent.observe('click', function(event){
542             this.onClickCanvas(event.element());
543         }.bind(this));
544         documentContent.observe('dblclick', function(event){
545             this.onDblClickCanvas(event.element());
546         }.bind(this));
547         var documentPaneId = uidGenerator.generate("documentPane");
548         var documentPane = new dijit.layout.ContentPane({
549             id:documentPaneId,
550             region:"center"
551         });
552         documentPane.setContent(documentContent);
553         
554         var inspectorArea = this._createInspectorArea();   
555 
556         /* Properties pane*/
557         this._propertiesPane = new PropertiesPane (inspectorArea);
558 
559         /* Pre/Post pane*/
560         this._prePostPane = new PrePostPane(inspectorArea);
561 
562         /* Facts pane*/
563         this._factsPane = new InspectorPane(inspectorArea);
564 
565 
566         rightBorderContainer.addChild(documentPane);
567         rightBorderContainer.addChild(inspectorArea);
568         mainBorderContainer.addChild(this.getPaletteController().getNode());
569         mainBorderContainer.addChild(rightBorderContainer);
570 
571         this._tab.setContent(mainBorderContainer.domNode);
572     },

616     /**
617      * This function updates the reachability in all
618      * Elements: canvas and palettes
619      * @private
620      */
621     _updateReachability: function () {
622         var canvas = this.getCanvas();
623         var domainContext = this.getBuildingBlockDescription().getDomainContext();
624         
625         var palette = this.getPaletteElements();
626         
627         if (URIs.catalogueFlow =='check'){
628             this._inferenceEngine.check(canvas, domainContext, palette, 'reachability');
629         } else {
630             this._inferenceEngine.findAndCheck(canvas, domainContext, palette, 'reachability');
631         }
