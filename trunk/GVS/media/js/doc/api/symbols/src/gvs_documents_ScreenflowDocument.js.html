 96     /**
 97      * Implementing DropZone interface.
 98      * @override
 99      */
100     drop: function(/** ComponentInstance */ droppedElement) {
101         // Reject repeated elements (except domain concepts)
102         if (this._canvasInstances.get(droppedElement.getUri()) &&
103             (droppedElement.constructor != DomainConceptInstance)) {
104             return false;
105         }
106         
107         this._canvasInstances.set(droppedElement.getUri(), droppedElement);
108         
109         switch (droppedElement.constructor) {
110             case ScreenInstance:
111                 this._description.addScreen(droppedElement.getUri(), droppedElement.getPosition());
112                 this._refreshReachability();
113                 break;
114                 
115             case DomainConceptInstance:
116                 // TODO
117                 break;
118                 
119             default:
120                 alert("Don't know how to accept that kind of element. ScreenflowDocument::drop");    
121         }            
122 
123         this.setSelectedElement(droppedElement);
124         return true; // Accept element
125     },

143     /**
144      * Delete a screen.
145      * @param ComponentInstance
146      *      Instance to be deleted from the
147      *      Screenflow document.
148      * @override
149      */
150     _deleteInstance: function(instance) {
151         if (this._canvasInstances.get(instance.getUri())) {
152             
153             var node = instance.getView().getNode();
154             node.parentNode.removeChild(node);
155             
156             this._canvasInstances.unset(instance.getUri());
157             
158             switch(instance.constructor) {
159                 case ScreenInstance:
160                     this._description.removeScreen(instance.getUri());
161                     break;
162                     
163                 case DomainConceptInstance:
164                     alert("Not yet implemented. ScreenflowDocument::deleteInstance");
165                     break;
166                     
167                 default:
168                     throw "Illegal state. ScreenflowDocument::deleteInstance";
169             }
170             
171             this._refreshReachability();
172             this.setSelectedElement();
173             instance.destroy();
174         }
175     },

177     /**
178      * Gets the elements of the canvas
179      * @type String[]
180      * @private
181      */
182     _getCanvas: function () {
183         var canvas = [];
184         
185         this._canvasInstances.keys().each(function(uri) {
186             canvas.push({
187                 'uri': uri
188             });
189         });
190         return canvas;

284     /**
285      * This function updates the reachability in all
286      * Elements: canvas and palettes
287      * @private
288      */
289     _refreshReachability: function () {
290         var canvas = this._getCanvas();
291         var palette = this._paletteController.getComponentUris();
292         
293         if (URIs.catalogueFlow =='check') {
294             this._inferenceEngine.check(canvas, palette, this._domainContext, 'reachability');
295         } else {
296             this._inferenceEngine.findAndCheck(canvas, palette,  this._domainContext, 'reachability');
297         }
298         
299         // FIXME: we must learn about document reachability from the inference 
300         //        engine. By the moment, one screen == deployable screenflow ;)
301         this._toolbarElements.get('deploy').setEnabled(canvas.size() > 0);
302     },
303     /**

307      */
308     _updatePropertiesPane: function() {
309         if (!this._selectedElement) {
310             this._propertiesPane.clearElement();
311         } else {
312             this._propertiesPane.selectElement(this._selectedElement);
313         }
314     },
