300     /**
301      * Implementing DropZone interface
302      * @type Object
303      *      It must contain two variables
304      *      * Boolean accepted: the element is accepted
305      *      * Boolean handledByDropZone: The position of the element
306      *                                   will be defined by the dropzone and
307      *                                   not by the draghandler
308      */
309     _drop: function(/** Area */ area, /** ComponentInstance */ instance,
310                     /**Object */ position, /** Number */ orientation,
311                     /** Boolean (Optional) */ _isLoading) {
312         var isLoading = Utils.variableOrDefault(_isLoading, false);
313 
314         // Reject repeated elements (except domain concepts)
315         if (instance.constructor != PrePostInstance &&
316             instance.constructor != PlanInstance) {
317             var uri;
318             if (isLoading) {
319                 uri = instance.getOriginalUri();
320             } else {
321                 // If the instance is new, _uri attribute is the original one
322                 uri = instance.getUri();
323             }
324             if (this._description.contains(uri)) {
325                 Utils.showMessage("There is another element like this. Cannot add it", {
326                     'hide': true,
327                     'error': true
328                 });
329                 return false;
330             }
331         }
332 
333         if (instance.constructor != PlanInstance) {
334             this._addToArea(area, instance, position);
335             instance.setEventListener(this);
336             instance.enableDragNDrop(area,[area]);
337         }
338 
339         switch (instance.constructor) {
340             case ScreenInstance:
341                 if (isLoading) {
342                     this._description.addScreen(instance, position);
343                     this._inferenceEngine.addReachabilityListener(instance.getUri(),
344                     instance.getView());
345                 } else {
346                     instance.createCatalogueCopy(function() {
347                         this._description.addScreen(instance, position);
348                         this._refreshReachability();
349                         this._setSelectedElement(instance);
350                         this._save(false);
351                     }.bind(this));
352                 }
353 
354                 break;
355 
356             case PrePostInstance:
357                 instance.setChangeHandler(this._onPrePostChange.bind({
358                         'mine':this,
359                         'position': position,
360                         'isLoading': isLoading
361                     }));
362                 if (area.getNode().className.include("pre")) {
363                     instance.setType("pre");
364                     instance.getView().setReachability({"satisfied": true});
365                 } else if (area.getNode().className.include("post")) {
366                     instance.setType("post");
367                 }
368                 if (!instance.getId()) {
369                     instance.setId(UIDGenerator.generate(instance.getTitle()));
370                     this._save(false);
371                 } else {
372                     UIDGenerator.setStartId(instance.getId());
373                 }
374                 this._description.addPrePost(instance, position);
375                 this._refreshReachability();
376                 break;
377 
378             case PlanInstance:
379                 this._planPanel.hide();
380                 this._addPlan(instance, position);
381                 break;
382         }
383 
384         this._setDirty(true);
385         return true;
386 
387     },

409     /**
410      * Callback to be called when the findCheck operation
411      * finishes
412      * @private
413      */
414     _findCheckCallback: function(/** Array */ uris, /** Object */ reachabilityData) {
415         // Update screen palette
416         var screenPalette = this._paletteController.getPalette(Constants.BuildingBlock.SCREEN);
417         var screenSet = screenPalette.getBuildingBlockSet();
418         screenSet.setURIs(uris);
419 
420         if (reachabilityData.canvas.postconditions) {
421             reachabilityData.canvas.postconditions.each(function(post) {
422                 var postInstance = this._description.getPost(post.id);
423                 if (postInstance) {
424                     var view = postInstance.getView();
425                     if (view) {
426                         view.setReachability(post);
427                     }
428                 }
429             }.bind(this));
430         }
431         this._refreshVisibility();
432         this._updatePanes();
433     },

629     /**
630      * This function updates the properties table and
631      * the pre/post pane depending on the selected element
632      * @private
633      */
634     _updatePanes: function() {
635         if (!this._selectedElement) {
636             var facts = this._getAllFacts();
637             this._propertiesPane.fillTable(this._description);
638             this._factPane.fillTable([], [], facts);
639         } else {
640             this._propertiesPane.fillTable(this._selectedElement);
641 
642             if (this._selectedElement.constructor == ScreenInstance) {
643                 var preReachability = this._inferenceEngine.getPreconditionReachability(
644                             this._selectedElement.getUri());
645                 var preconditions = this._selectedElement.getPreconditionTable(preReachability);
646 
647                 var postReachability = this._inferenceEngine.isReachable(
648                             this._selectedElement.getUri());
649                 var postconditions = this._selectedElement.getPostconditionTable(postReachability);
650 
651                 this._factPane.fillTable(preconditions,postconditions,[]);
652             } else {
653                 //PrePostInstance
654                 if (this._selectedElement.getType()) {
655                     //Pre or post condition
656                     var factInfo = [this._selectedElement.getConditionTable()];
657                     this._factPane.fillTable([], [], factInfo);
658                 }
659             }
660         }
661     },

741     /**
742      * This function adds all the (new) screens of the plan
743      * to the screenflow
744      * @private
745      */
746     _addPlan: function(/** PlanInstance */ plan, /** Object */ position){
747         var screenPosition = {
748             'left': position.left + 3, //with margin
749             'top': position.top + 3
750         };
751 
752         var pendingScreens = new Hash();
753 
754         plan.getPlanElements().each(function(screenDescription) {
755             if (!this._description.contains(screenDescription.uri)) {
756                 var screen = new ScreenInstance(screenDescription,
757                         this._inferenceEngine);
758                 pendingScreens.set(screenDescription.uri, screen);
759             }
760         }.bind(this));
761 
762         var request = pendingScreens.keys().collect(function(uri) {
763             return {"uri": uri};
764         });
765 
766         PersistenceEngine.sendPost(URIs.catalogueCopy, null, Object.toJSON(request), this,
767             function(transport) {
768                 var screens = JSON.parse(transport.responseText);
769                 screens.each(function(screenData) {
770                     var screenInstance = pendingScreens.get(screenData.uri);
771                     screenInstance.setCopyUri(screenData.clone);
772 
773                     this._addToArea(this._areas.get('screen'), screenInstance, screenPosition);
774                     this._description.addScreen(screenInstance, screenPosition);
775 
776                     //Incrementing the screen position for the next screen
777                     screenPosition.left += 108; // Screen size=100 + margin=6 + border=2
778 
779                     screenInstance.setEventListener(this);
780                     screenInstance.enableDragNDrop(this._areas.get('screen'),[this._areas.get('screen')]);
781                     screenInstance.onFinish(true);
782                 }.bind(this));
783                 this._setSelectedElement();
784                 this._refreshReachability();
785 
786             }.bind(this), Utils.onAJAXError);
787 
788 
789     },

806      */
807      _createConditions: function(/** Array */ conditionList, /** DropZone */ area) {
808         conditionList.each(function(condition) {
809             var description = new PrePostDescription(condition);
810             var instance = new PrePostInstance(description, this._inferenceEngine);
811             instance.onFinish(true, condition.position);
812             var zonePosition = area.getNode();
813             $("main").appendChild(instance.getView().getNode());
814             var effectivePosition = Geometry.adaptInitialPosition(zonePosition,
815                                     instance.getView().getNode(), condition.position);
816             $("main").removeChild(instance.getView().getNode());
817             this._drop(area, instance, effectivePosition, true);
818             this._description.addPrePost(instance, effectivePosition);
819         }.bind(this));
820     },
821 
822     /**
823      * Function that loads the pre and post conditions
824      * @private
825      */
826     _loadConditions: function() {
827         // Load pre and postconditions
828         this._createConditions(this._canvasCache.getPreconditions(), this._areas.get('pre'));
829         this._createConditions(this._canvasCache.getPostconditions(), this._areas.get('post'));
830         this._setDirty(false);
831         this._refreshReachability();
832         if (this._description.cloned) {
833             this._saveAs(true);
834         }
835     },
836

838      * Creates a clone of the selected screen
839      * @private

841     _cloneSelectedElement: function() {
842         this.cloneElement(this._selectedElement);
843     },

847      * (hidden/shown)
848      * @private
849      */
850     _toggleUnreachableScreens: function() {
851         this._unreachableShown = !this._unreachableShown;
852         this._refreshVisibility();
853     },

855     /**
856      * Refreshes the visibility of unreachable screens
857      * @private
858      */
859     _refreshVisibility: function() {
860 
861         var screenPalette = this._paletteController.getPalette(Constants.BuildingBlock.SCREEN);
862         var paletteNode = screenPalette.getContentNode();
863 
864         var visibility = this._unreachableShown ? "block": "none";
865         $A(paletteNode.childNodes).each(function(element) {
866             if (element.match(".slot")) {
867 
868                 if (element.firstChild.match(".unsatisfeable")) {
869                     element.setStyle({
870                         "display": visibility
871                     });
872                 }
873                 if (element.firstChild.match(".satisfeable")) {
874                     element.setStyle({
875                         "display": "block"
876                     });
877                 }
878             }
879         }, this);
880     }
