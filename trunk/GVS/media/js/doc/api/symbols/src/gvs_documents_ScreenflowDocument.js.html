179     /**
180      * Implementing DropZone interface
181      * @type Object
182      *      It must contain two variables
183      *      * Boolean accepted: the element is accepted
184      *      * Boolean handledByDropZone: The position of the element
185      *                                   will be defined by the dropzone and
186      *                                   not by the draghandler
187      */
188     _drop: function(/** Area */ area, /** ComponentInstance */ instance, /** Object */ position,
189         /** Boolean (Optional) */ _isLoading) {
190         var isLoading = Utils.variableOrDefault(_isLoading, false);
191 
192         // Reject repeated elements (except domain concepts)
193         if (this._description.contains(instance.getUri())) {
194             Utils.showMessage("There is another element like this. Cannot add it", {
195                 'hide': true,
196                 'error': true
197             });
198             return false;
199         }
200 
201         if (!instance.getId()) {
202             instance.setId(UIDGenerator.generate(instance.getTitle()));
203         } else {
204             if (instance.constructor != PrePostInstance) {
205                 UIDGenerator.setStartId(instance.getId());
206             }
207         }
208 
209         if (instance.constructor != PlanInstance) {
210             this._addToArea(area, instance, position);
211             instance.setEventListener(this);
212             instance.enableDragNDrop(area,[area]);
213         }
214 
215         switch (instance.constructor) {
216             case ScreenInstance:
217                 this._description.addScreen(instance, position);
218                 if (!isLoading) {
219                     this._setSelectedElement(instance);
220                 }
221                 
222                 break;
223 
224             case PrePostInstance:
225                 instance.setChangeHandler(this._onPrePostChange.bind({
226                         'mine':this,
227                         'position': position,
228                         'isLoading': isLoading
229                     }));
230                 if (area.getNode().className.include("pre")) {
231                     instance.setType("pre");                
232                 } else if (area.getNode().className.include("post")) {
233                     instance.setType("post");
234                 }
235                 break;
236 
237             case PlanInstance:
238                 this._planPanel.hide();
239                 this._addPlan(instance, position);
240                 this._refreshReachability();
241                 this._setSelectedElement();
242                 break;
243         }
244         this._setDirty(true);
245         // Only for piping
246         //instance.getView().addGhost();
247         return true;
248     },

388     /**
389      * This function updates the reachability in all
390      * Elements: canvas and palettes
391      * @private
392      */
393     _refreshReachability: function (/** Boolean (Optional) */_isFindCheck) {
394         var isFindCheck = Utils.variableOrDefault(_isFindCheck, false);
395         var canvas = this._getCanvasUris();
396         var palette = this._paletteController.getComponentUris(Constants.BuildingBlock.SCREEN);
397         
398         if (isFindCheck) {
399             this._inferenceEngine.findCheck(canvas, palette, this._tags,
400                                     'reachability', this._updatePanes.bind(this));
401         } else {
402             this._inferenceEngine.check(canvas, palette, this._tags,
403                                     'reachability', this._updatePanes.bind(this));
404         }      
405         
406         // FIXME: we must learn about document reachability from the inference 
407         //        engine. By the moment, one screen == deployable screenflow ;)
408         this._toolbarElements.get('build').setEnabled(canvas.size() > 0);
409     },

412     /**
413      * This function updates the properties table and 
414      * the pre/post pane depending on the selected element
415      * @private
416      */
417     _updatePanes: function() {
418         var facts = this._getAllFacts();
419         if (!this._selectedElement) {
420             this._propertiesPane.fillTable(this._description);          
421             this._factPane.fillTable([], [], facts);
422         } else {
423             this._propertiesPane.fillTable(this._selectedElement);
424            
425             if (this._selectedElement.constructor == ScreenInstance) {
426                 var preReachability = this._inferenceEngine.getPreconditionReachability(
427                             this._selectedElement.getUri());
428                 var preconditions = this._selectedElement.getPreconditionTable(preReachability);
429                 
430                 var postReachability = this._inferenceEngine.isReachable(
431                             this._selectedElement.getUri());
432                 var postconditions = this._selectedElement.getPostconditionTable(postReachability);
433                 
434                 this._factPane.fillTable(preconditions,postconditions,[]);           
435             } else {
436                 //PrePostInstance
437                 if (this._selectedElement.getType()) {
438                     //Pre or post condition
439                     var factInfo = [this._selectedElement.getConditionTable(
440                         this._inferenceEngine.isReachable(this._selectedElement.getUri())
441                     )];
442                     this._factPane.fillTable([], [], factInfo);
443                 }
444             }    
445         }
446     },

448     /**
449      * This function returns the data array containing all the
450      * facts belonging to the screenflow
451      * @type Array
452      */
453     _getAllFacts: function() {
454         var resultHash = new Hash();
455         this._description.getCanvasInstances().each(function(instance) {
456             if (instance.constructor == ScreenInstance) {
457                 var preReachability = this._inferenceEngine.getPreconditionReachability(
458                             instance.getUri());
459                 var preconditions = instance.getPreconditionTable(preReachability);               
460                 preconditions.each(function(pre) {
461                     if (!resultHash.get(pre[2]/*The uri of the pre*/)) {
462                         resultHash.set(pre[2], pre);
463                     }
464                 });
465                 
466                 var postReachability = this._inferenceEngine.isReachable(
467                             instance.getUri());
468                 var postconditions = instance.getPostconditionTable(postReachability);               
469                 postconditions.each(function(post) {
470                     if (!resultHash.get(post[2]/*The uri of the post*/)) {
471                         resultHash.set(post[2], post);
472                     }
473                 });
474             } else {
475                 //PrePostInstance
476                 var factInfo = instance.getConditionTable(
477                     this._inferenceEngine.isReachable(instance.getUri())
478                 );
479                 if (!resultHash.get(factInfo[2]/*The uri of the pre/post*/)) {
480                     resultHash.set(factInfo[2], factInfo);
481                 }
482             }
483         }.bind(this));
484         return resultHash.values();
485     },

542     /**
543      * This function adds all the (new) screens of the plan
544      * to the screenflow
545      * @private
546      */
547     _addPlan: function(/** PlanInstance */ plan, /** Object */ position){
548         var screenPosition = {
549             'left': position.left + 3, //with margin
550             'top': position.top + 3
551         };
552         
553         plan.getPlanElements().each(function(screenDescription) {
554             if (!this._description.contains(screenDescription.uri)) {
555                 var screen = new ScreenInstance(screenDescription,
556                         this._inferenceEngine);      
557                 this._description.addScreen(screen, screenPosition);
558                 screen.onFinish(true);
559                 
560                 this._addToArea(this._areas.get('screen'), screen, screenPosition);
561                 //Incrementing the screen position for the next screen
562                 screenPosition.left += 108; // Screen size=100 + margin=6 + border=2
563                 screen.setEventListener(this);
564                 screen.enableDragNDrop(this._areas.get('screen'),[this._areas.get('screen')]);
565             }  
566         }.bind(this));
567     },

581     /**
582      * Create the pre and post conditions
583      * @private
584      */
585      _createConditions: function(/** Array */ conditionList, /** DropZone */ area) {
586         conditionList.each(function(condition) {
587             var description = new PrePostDescription(condition);
588             var instance = new PrePostInstance(description, this._inferenceEngine);
589             instance.onFinish(true, condition.position);
590             var zonePosition = area.getNode();
591             $("main").appendChild(instance.getView().getNode());
592             var effectivePosition = Geometry.adaptInitialPosition(zonePosition,
593                                     instance.getView().getNode(), condition.position);
594             $("main").removeChild(instance.getView().getNode());
595             this._drop(area, instance, effectivePosition, true);
596             this._description.addPrePost(instance, effectivePosition);
597         }.bind(this));
598     },
