 26 var SubMenu = Class.create(MenuElement, /** @lends SubMenu.prototype */ {
 27     /**
 28      * Element for a menu
 29      * @constructs
 30      * @extends MenuElement
 31      */
 32     initialize: function($super, /** Object */ data, /** Boolean */ mainMenu) {
 33         $super(data.weight);
 34 
 35         var menu = new dijit.Menu({});
 36 
 37         if (mainMenu) {
 38             this._widget = new dijit.PopupMenuBarItem({
 39                     'label': data.label,
 40                     'popup': menu
 41                 });
 42         } else {
 43             this._widget = new dijit.PopupMenuItem({
 44                     'label': data.label,
 45                     'popup': menu
 46                 });
 47         }
 48 
 49 
 50         /**
 51          * Array for storing the different groups
 52          * that contains the Submenu children
 53          * @type Array
 54          * @private
 55          */
 56         this._groups = new Array();
 57 
 58         var child;
 59         $H(data.children).each(function(pair) {
 60             var childData = $H(pair.value).toObject();
 61             switch (childData.type.toLowerCase()) {
 62                 case 'action':
 63                     child = childData.action;
 64                     break;
 65                 case 'submenu':
 66                     child = new SubMenu(childData);
 67                     break;
 68             }
 69             this._addChild(childData.group, child);
 70         }.bind(this));
 71         this._createSubMenu(menu);
 72     },
 73 
 74 
 75     // **************** PUBLIC METHODS **************** //
 76 
 77     /**
 78      * Register key handlers
 79      * @override
 80      */
 81     register: function(/** KeyPressRegistry */ registry) {
 82         this._groups.each(function(group) {
 83             group.each(function(child){
 84                 child.register(registry);
 85             });
 86         });
 87     },
 88 
 89     /**
 90      * Unregister key handlers and destroy the widget
 91      * @override
 92      */
 93     unregister: function(/** KeyPressRegistry */ registry) {
 94         this._groups.each(function(group) {
 95             group.each(function(child){
 96                 child.unregister(registry);
 97             });
 98         });
 99         this._widget.destroy(false);
100     },
101 
102     // ***************** PRIVATE METHODS *************** //
103 
104      /**
105      * Adds a new child to the submenu
106      * @private
107      */
108     _addChild: function(/** Number */ group, /** MenuElement */ child) {
109         if (!this._groups[group]) {
110             this._groups[group] = new Array();
111         }
112         this._groups[group].push(child);
113         this._groups[group].sort(function(a, b){
114             return a.getWeight() - b.getWeight();
115         });
116     },
117 
118 
119     /**
120      * creates the submenu widget with the information
121      * stored in the children's list
122      * @private
123      */
124     _createSubMenu: function(/** Menu */ parent) {
125 
126         for(var i=0; i < this._groups.size(); i++) {
127             if (this._groups[i]) {
128                 this._groups[i].each(function(element){
129                     parent.addChild(element.getWidget());
130                 });
131                 if (i != this._groups.size() - 1) {
132                     // Not last group
133                     parent.addChild(new dijit.MenuSeparator());
134                 }
135             }
136         }
137         parent.startup();
138     }
139 });
