1 var SubMenu = Class.create(MenuElement, /** @lends SubMenu.prototype */ {
  2     /**
  3      * Element for a menu
  4      * @constructs
  5      * @extends MenuElement
  6      */
  7     initialize: function($super, /** Object */ data, /** Boolean */ mainMenu) {
  8         $super(data.weight);
  9 
 10         var menu = new dijit.Menu({});
 11 
 12         if (mainMenu) {
 13             this._widget = new dijit.PopupMenuBarItem({
 14                     'label': data.label,
 15                     'popup': menu
 16                 });
 17         } else {
 18             this._widget = new dijit.PopupMenuItem({
 19                     'label': data.label,
 20                     'popup': menu
 21                 });
 22         }
 23 
 24 
 25         /**
 26          * Array for storing the different groups
 27          * that contains the Submenu children
 28          * @type Array
 29          * @private
 30          */
 31         this._groups = new Array();
 32 
 33         var child;
 34         $H(data.children).each(function(pair) {
 35             var childData = $H(pair.value).toObject();
 36             switch (childData.type.toLowerCase()) {
 37                 case 'action':
 38                     child = childData.action;
 39                     break;
 40                 case 'submenu':
 41                     child = new SubMenu(childData);
 42                     break;
 43             }
 44             this._addChild(childData.group, child);
 45         }.bind(this));
 46         this._createSubMenu(menu);
 47     },
 48 
 49 
 50     // **************** PUBLIC METHODS **************** //
 51 
 52     /**
 53      * Register key handlers
 54      * @override
 55      */
 56     register: function(/** KeyPressRegistry */ registry) {
 57         this._groups.each(function(group) {
 58             group.each(function(child){
 59                 child.register(registry);
 60             });
 61         });
 62     },
 63 
 64     /**
 65      * Unregister key handlers and destroy the widget
 66      * @override
 67      */
 68     unregister: function(/** KeyPressRegistry */ registry) {
 69         this._groups.each(function(group) {
 70             group.each(function(child){
 71                 child.unregister(registry);
 72             });
 73         });
 74         this._widget.destroy(false);
 75     },
 76 
 77     // ***************** PRIVATE METHODS *************** //
 78 
 79      /**
 80      * Adds a new child to the submenu
 81      * @private
 82      */
 83     _addChild: function(/** Number */ group, /** MenuElement */ child) {
 84         if (!this._groups[group]) {
 85             this._groups[group] = new Array();
 86         }
 87         this._groups[group].push(child);
 88         this._groups[group].sort(function(a, b){
 89             return a.getWeight() - b.getWeight();
 90         });
 91     },
 92 
 93 
 94     /**
 95      * creates the submenu widget with the information
 96      * stored in the children's list
 97      * @private
 98      */
 99     _createSubMenu: function(/** Menu */ parent) {
100 
101         for(var i=0; i < this._groups.size(); i++) {
102             if (this._groups[i]) {
103                 this._groups[i].each(function(element){
104                     parent.addChild(element.getWidget());
105                 });
106                 if (i != this._groups.size() - 1) {
107                     // Not last group
108                     parent.addChild(new dijit.MenuSeparator());
109                 }
110             }
111         }
112         parent.startup();
113     }
114 });
