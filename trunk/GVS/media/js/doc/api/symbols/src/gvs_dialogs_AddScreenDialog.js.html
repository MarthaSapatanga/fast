 40     /**
 41      * initDialogInterface
 42      * This function creates the dom structure and
 43      * @private
 44      * @overrides
 45      */
 46     _initDialogInterface: function (){
 47 
 48         this._setHeader("Fulfill Screen Information",
 49                              "Please fulfill the required information in order to" +
 50                              " add a new screen to the catalogue.");
 51 
 52         var formData = [
 53             {'type':'title', 'value': 'Screen information'},
 54             {'type':'input', 'label': 'Label:','name': 'label', 'value': 'Label of the screen...'},
 55             {'type':'input', 'label': 'Description:','name': 'description', 'value': 'Short description of the screen...'},
 56             {'type':'input', 'label': 'Creator URL (*):','name': 'creator', 'value': 'Creator URL...'}, //TODO: validation
 57             {'type':'input', 'label': 'Rights URL (*):','name': 'rights', 'value': 'Rights URL...'},
 58             {'type':'input', 'label': 'Version:','name': 'version', 'value': '1.0'},
 59             {'type':'input', 'label': 'Icon URL (*):','name': 'icon', 'value': 'icon URL...'},
 60             {'type':'input', 'label': 'Screenshot URL (*):','name': 'screenshot', 'value': 'Screenshot URL...'},
 61             {'type':'input', 'label': 'Domain Context:','name': 'tags', 'value': 'Write domain context as tags separated by ","...'},
 62             {'type':'input', 'label': 'Homepage (*):','name': 'homepage', 'value': 'Homepage URL...'},
 63             {'type':'input', 'label': 'Preconditions:','name': 'preconditions', 'value': 'If any, write preconditions separated by ","...'},
 64             {'type':'input', 'label': 'Postconditions:','name': 'postconditions', 'value': 'If any, write postconditions separated by ","...'},
 65             {'type':'input', 'label': 'Screen code URL (*):','name': 'code', 'value': 'Screencode URL...'},
 66             {'type':'hidden', 'name': 'creationDate', 'value': ''},
 67             {'type':'label', 'value': '(*): Required Field'}
 68         ];
 69 
 70         this._setContent(formData);
 71 
 72     },
 73 
 74     /**
 75      * Overriding onOk handler
 76      * @overrides
 77      * @private
 78      */
 79     _onOk: function($super){
 80         $super();
 81         var creationDate = new Date();
 82         var form = this._getForm();
 83 
 84 
 85         form.creationDate.setValue(Utils.getIsoDateNow(creationDate));
 86 
 87         var formToSend = form.serialize(true);
 88         formToSend.label = {
 89             "en-GB": form.label.getValue()
 90         };
 91         formToSend.description = {
 92             "en-GB": form.description.getValue()
 93         };
 94 
 95         // TODO: Review this
 96         var tagsArray = form.tags.getValue().split(',');
 97         for (var i = 0; i < tagsArray.length; i++) {
 98             var aux = tagsArray[i].strip();
 99             if (aux && aux != "") {
100                 tagsArray[i] = aux;
101             }
102             else {
103                 tagsArray[i] = null;
104             }
105         }
106         tagsArray = tagsArray.compact();
107         formToSend.tags = Utils.getCatalogueTags(tagsArray, null);
108 
109         var preconditionsArray = form.preconditions.getValue().split(',');
110         for (var i = 0; i < preconditionsArray.length; i++) {
111             var aux = preconditionsArray[i].strip();
112             if (aux && aux != "") {
113                 preconditionsArray[i] = aux;
114             }
115             else {
116                 preconditionsArray[i] = null;
117             }
118         }
119         preconditionsArray = preconditionsArray.compact();
120         formToSend.preconditions = preconditionsArray;
121 
122         var postconditionsArray = form.postconditions.getValue().split(',');
123         for (var i = 0; i < postconditionsArray.length; i++) {
124             var aux = postconditionsArray[i].strip();
125             if (aux && aux != "") {
126                 postconditionsArray[i] = aux;
127             }
128             else {
129                 postconditionsArray[i] = null;
130             }
131         }
132         postconditionsArray = postconditionsArray.compact();
133         formToSend.postconditions = postconditionsArray;
134 
135         console.log("Before json");
136         console.log(formToSend);
137         console.log("After json");
138         console.log(Object.toJSON(formToSend));
139 
140         Catalogue.createScreen(Object.toJSON(formToSend));
141 
142     }
143 });
