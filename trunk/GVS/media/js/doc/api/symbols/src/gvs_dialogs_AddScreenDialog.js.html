218         var formToSend = form.serialize(true);
219         formToSend.label = {

221         };
222         formToSend.description = {
223             "en-GB": form.description.getValue()
224         };
225         
226         var domainContextArray = form.domainContext.getValue().split(',');
227         for (var i = 0; i < domainContextArray.length; i++) {
228             var aux = domainContextArray[i].strip();
229             if (aux && aux != "") {
230                 domainContextArray[i] = aux;
231             }
232             else {
233                 domainContextArray[i] = null;
234             }
235         }
236         domainContextArray = domainContextArray.compact();
237         formToSend.domainContext = {
238             "tags": domainContextArray,

242         var preconditionsArray = form.preconditions.getValue().split(',');
243         for (var i = 0; i < preconditionsArray.length; i++) {
