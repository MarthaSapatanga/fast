 26 var Utils = Class.create();
 27 
 28 // **************** STATIC METHODS **************** //
 29 
 30 Object.extend(Utils, {
 31 
 32     getIsoDateNow: function(/**Date*/ dateObject) {
 33         var dateString = dateObject.toJSON();
 34         dateString = dateString.replace(/"/,'');
 35         if(dateString.endsWith('Z"')){
 36             dateString=dateString.truncate(-2,'');
 37             dateString=dateString+'+0000';
 38         }
 39         return dateString;
 40     },
 41 
 42     /**
 43      * Copy the properties of an object into another.
 44      */
 45     addProperties: function(/** Object */ to, /** Object */ properties) {
 46         if (properties) {
 47             $H(properties).clone().each(function(pair) {
 48                 to[pair.key] = pair.value;
 49             });
 50         }
 51     },
 52 
 53     setSatisfeabilityClass: function (/** DOMNode */ node, /** Boolean */ satisfeable) {
 54         if (satisfeable === null || satisfeable === undefined) {
 55             //Unknown satisfeability
 56             node.removeClassName('satisfeable');
 57             node.removeClassName('unsatisfeable');
 58         } else {
 59             node.removeClassName(satisfeable ? 'unsatisfeable' : 'satisfeable');
 60             node.addClassName(satisfeable ? 'satisfeable' : 'unsatisfeable');
 61         }
 62     },
 63 
 64     /**
 65      * This function extracts an uri from a rdf pattern
 66      * @type String
 67      */
 68     extractURIfromPattern: function(/** String */ pattern) {
 69         if (pattern) {
 70             var pieces = pattern.split(" ");
 71             return pieces[2];
 72         } else {
 73             return "";
 74         }
 75     },
 76 
 77     /**
 78      * Logs an AJAX error
 79      */
 80     onAJAXError: function(transport, e){
 81         Logger.serverError(transport, e);
 82 
 83         var message = "Try Reloading";
 84         if (e) {
 85             message = e;
 86         }
 87         Utils.showMessage("Ooops. Something unexpected happened: " + message,{
 88             "error": true,
 89             "hide": true
 90         });
 91     },
 92 
 93     /**
 94     * This function returns the position of a node
 95     * @type Object
 96     */
 97     getPosition: function (/** DOMNode */ node){
 98         var left = 0;
 99         var top  = 0;
100 
101         while (node.offsetParent){
102             left += node.offsetLeft - node.scrollLeft;
103             top  += node.offsetTop - node.scrollTop;
104             node  = node.offsetParent;
105         }
106 
107         left += node.offsetLeft;
108         top  += node.offsetTop;
109 
110         return {'left':left, 'top':top};
111     },
112 
113     /**
114      * Shows a message on top of the GVS
115      */
116     showMessage: function(/** String */ text, /** Object(optional) */ options) {
117         $("messages").removeClassName("error");
118         $("messages").removeClassName("hidden");
119         $("messages").update(text);
120 
121         var position = Math.floor(($("header").clientWidth / 2) -
122                                     ($("messages").clientWidth / 2));
123         $("messages").setStyle({'opacity': '1', 'left': position + 'px'});
124         if (options) {
125             if (options.error) {
126                 $("messages").addClassName("error");
127             }
128             if (options.hide) {
129                  dojo.fadeOut({
130                     'duration': 2500,
131                     'node': $("messages")
132                 }).play(options.error ? 2500 : 1500);
133             }
134         }
135     },
136 
137     hideMessage: function() {
138         $("messages").setStyle({"opacity": "0"});
139     },
140 
141     showErrorMessage: function(/** String */ text, /** Object(optional) */ options) {
142         this.showMessage(text, Object.extend(options || {}, {'error':true}));
143     },
144     /**
145      * This function returns the value of a variable, or a default one if it
146      * is undefined
147      * @type Object
148      */
149     variableOrDefault: function(/** Object */ variable, /** Object */ defaultValue){
150         var result = variable;
151         if (result === undefined) {
152             result = defaultValue;
153         }
154         return result;
155     },
156 
157     /**
158      * Converts an Array of strings into an array
159      * of objects representing the tag structure of
160      * the catalogue
161      */
162     getCatalogueTags: function(/** Array */ tags, /** String */ user) {
163         var result = new Array();
164         tags.each(function(tag) {
165             result.push({
166                 'label': {
167                     'en-gb': tag
168                 }
169             });
170         });
171         return result;
172     },
173 
174     /**
175      * This function sanitizes a string, removing:
176      *    - Heading and trailing white spaces, and extra white spaces
177      *    - Tabs
178      *    - Carrier return
179      * @type String
180      */
181     sanitize: function(/** String */ text) {
182         var result = text.replace(/^\s+|\n|\s+$/g,"");
183         result = result.replace(/\s+/g, " ");
184         return result;
185     }
186 });
