 68     /**
 69     * This function returns the position of a node
 70     * @type Object
 71     */
 72     getPosition: function (/** DOMNode */ node){
 73         var left = 0;
 74         var top  = 0;
 75 
 76         while (node.offsetParent){
 77             left += node.offsetLeft - node.scrollLeft;
 78             top  += node.offsetTop - node.scrollTop;
 79             node  = node.offsetParent;
 80         }
 81 
 82         left += node.offsetLeft;
 83         top  += node.offsetTop;
 84 
 85         return {'left':left, 'top':top};
 86     },
