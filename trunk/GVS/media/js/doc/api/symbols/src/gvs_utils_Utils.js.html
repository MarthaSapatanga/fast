 10 Utils.getIsoDateNow = function(/**Date*/ dateObject) {
 11     var dateString = dateObject.toJSON();
 12     dateString = dateString.replace(/"/,'');
 13     if(dateString.endsWith('Z"')){
 14         dateString=dateString.truncate(-2,'');
 15         dateString=dateString+'+0000';
 16     }
 17     return dateString;
