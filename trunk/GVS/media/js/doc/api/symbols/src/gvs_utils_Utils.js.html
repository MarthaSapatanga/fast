 71     /**
 72     * This function returns the position of a node
 73     * @type Object
 74     */
 75     getPosition: function (/** DOMNode */ node){
 76         var left = 0;
 77         var top  = 0;
 78     
 79         while (node.offsetParent){
 80             left += node.offsetLeft - node.scrollLeft;
 81             top  += node.offsetTop - node.scrollTop;
 82             node  = node.offsetParent;
 83         }
 84     
 85         left += node.offsetLeft;
 86         top  += node.offsetTop;
 87     
 88         return {'left':left, 'top':top};
 89     },
