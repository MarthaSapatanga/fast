1 var AbstractDocument = Class.create(ToolbarModel, /** @lends AbstractDocument.prototype */ {
  2     /**
  3      * Represents a document and its tab. Subclasses must provide the 
  4      * inner content.
  5      * @extends ToolbarModel
  6      * @abstract
  7      * @constructs
  8      */ 
  9     initialize: function($super, /** String */ title) {
 10         $super();
 11         
 12         var uidGenerator = UIDGeneratorSingleton.getInstance();
 13         
 14         /**
 15          * Tab title
 16          * @type String
 17          * @private
 18          */
 19         this._title = title;
 20         
 21         /**
 22          * Actual Tab Id
 23          * @type String
 24          * @private
 25          */
 26         this._tabId = uidGenerator.generate("tab");
 27        
 28         
 29         /**
 30          * Initial tab content (empty by default)
 31          * @type DOMNode
 32          * @private
 33          */
 34         this._tabContent = new Element("div", {
 35             "class": "document"
 36         });        
 37         
 38         /**
 39          * Actual tab
 40          * @type DOMNode
 41          * @private
 42          */
 43         this._tab = new dijit.layout.ContentPane({
 44             title: this._title,
 45             id: this._tabId,
 46             closable: true,
 47             onClose: this._closeDocument.bind(this)
 48         }, null);
 49         
 50         
 51         this._tab.setContent(this._tabContent);
 52     },
 53     
 54 
 55     // **************** PUBLIC METHODS **************** //
 56 
 57     /**
 58      * Gets the tab id.
 59      * @type String
 60      */
 61     getTabId: function() {
 62         return this._tabId;
 63     },
 64     
 65     
 66     /**
 67      * Gets the tab node.
 68      * @type DOMNode
 69      */
 70     getTab: function() {
 71         return this._tab;
 72     },
 73     
 74     
 75     /**
 76      * Gets the title.
 77      * @type String
 78      */
 79     getTitle: function() {
 80         return this._title;
 81     },
 82 
 83     /**
 84      * Gets the content node.
 85      * @type DOMNode
 86      */
 87     getNode: function() {
 88         return this._tabContent;
 89     },
 90     
 91     /**
 92      * Method called on keypress. Might be overloaded
 93      */
 94     onKeyPressed: function(/** Event */ e) {
 95         // Do nothing
 96     },
 97     
 98 
 99     
100     
101     // **************** PRIVATE METHODS **************** //
102 
103     /**
104      * Close document event handler.
105      * @private
106      */
107     _closeDocument: function() {
108         var gvs = GVSSingleton.getInstance();
109         gvs.getDocumentController().closeDocument(this._tabId);
110         return true;
111     }
112 });
