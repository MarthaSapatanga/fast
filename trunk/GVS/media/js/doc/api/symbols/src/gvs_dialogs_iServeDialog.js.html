 26 var iServeDialog = Class.create(ConfirmDialog /** @lends iServeDialog.prototype */, {
 27     /**
 28      * Shows iServe services
 29      * @constructs
 30      * @extends ConfirmDialog
 31      */
 32     initialize: function($super, iServeList) {
 33         $super("iServe External Services", ConfirmDialog.OK);
 34 
 35         /**
 36          * List of iServe services
 37          * @private
 38          * @type Array
 39          */
 40         this._iServeList = iServeList;
 41     },
 42 
 43     // **************** PUBLIC METHODS **************** //
 44 
 45 
 46     // **************** PRIVATE METHODS **************** //
 47 
 48 
 49     /**
 50      * initDialogInterface
 51      * This function creates the dom structure
 52      * @private
 53      * @override
 54      */
 55     _initDialogInterface: function () {
 56         var content = new Element("div");
 57 
 58         var header = new Element("h2").update("About this services");
 59         content.appendChild(header);
 60 
 61         var p1 = new Element("p").update(
 62             'This services are extracted from ' +
 63             '<a href="http://iserve.kmi.open.ac.uk/">iServe</a>, ' +
 64             'which is a platform to discover services through their semantic ' +
 65             'information.<br /> All of them are related to the building block you ' +
 66             'are creating, so hopefully some of them will be useful for you.'
 67         );
 68         content.appendChild(p1);
 69 
 70         var p2 = new Element("p").update(
 71             'The proposed services are not GVS-compliant, so you ' +
 72             'will have to wrap them. You can use the GVS to do so. '
 73         );
 74         content.appendChild(p2);
 75 
 76         var table = new Element("table", {"class": "propertiesTable"});
 77         table.setStyle();
 78         content.appendChild(table);
 79 
 80         var tHead = new Element("tr", {"class": "tableHeader"});
 81         table.appendChild(tHead);
 82 
 83 
 84         var serviceTitle = new Element("td").update("Service Name");
 85         tHead.appendChild(serviceTitle);
 86 
 87         var serviceUri = new Element("td").update("Service URL");
 88         tHead.appendChild(serviceUri);
 89 
 90         var info = new Element("td").update("Get more info");
 91         tHead.appendChild(info);
 92 
 93         this._iServeList.each(function(service) {
 94             var row = new Element("tr");
 95             table.appendChild(row);
 96 
 97             var name = service.label || "-";
 98             var serviceName = new Element("td").update(name);
 99             row.appendChild(serviceName);
100 
101             var url = new Element("td").update(service.address);
102             row.appendChild(url);
103 
104             var moreinfo = new Element("td").update(
105                 '<a target="_blank" style="text-decoration:none" href="' + service.service + '">+</a>'
106             );
107             row.appendChild(moreinfo);
108         });
109 
110         this._setContent(content);
111     }
112 });
