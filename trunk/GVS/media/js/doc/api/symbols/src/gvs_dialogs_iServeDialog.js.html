1 var iServeDialog = Class.create(ConfirmDialog /** @lends iServeDialog.prototype */, {
  2     /**
  3      * Shows iServe services
  4      * @constructs
  5      * @extends ConfirmDialog
  6      */
  7     initialize: function($super, iServeList) {
  8         $super("iServe External Services", ConfirmDialog.OK);
  9 
 10         /**
 11          * List of iServe services
 12          * @private
 13          * @type Array
 14          */
 15         this._iServeList = iServeList;
 16     },
 17 
 18     // **************** PUBLIC METHODS **************** //
 19 
 20 
 21     // **************** PRIVATE METHODS **************** //
 22 
 23 
 24     /**
 25      * initDialogInterface
 26      * This function creates the dom structure
 27      * @private
 28      * @override
 29      */
 30     _initDialogInterface: function () {
 31         var content = new Element("div");
 32 
 33         var header = new Element("h2").update("About this services");
 34         content.appendChild(header);
 35 
 36         var p1 = new Element("p").update(
 37             'This services are extracted from ' +
 38             '<a href="http://iserve.kmi.open.ac.uk/">iServe</a>, ' +
 39             'which is a platform to discover services through their semantic ' +
 40             'information.<br /> All of them are related to the building block you ' +
 41             'are creating, so hopefully some of them will be useful for you.'
 42         );
 43         content.appendChild(p1);
 44 
 45         var p2 = new Element("p").update(
 46             'The proposed services are not GVS-compliant, so you ' +
 47             'will have to wrap them. You can use the GVS to do so. '
 48         );
 49         content.appendChild(p2);
 50 
 51         var table = new Element("table", {"class": "propertiesTable"});
 52         table.setStyle();
 53         content.appendChild(table);
 54 
 55         var tHead = new Element("tr", {"class": "tableHeader"});
 56         table.appendChild(tHead);
 57 
 58 
 59         var serviceTitle = new Element("td").update("Service Name");
 60         tHead.appendChild(serviceTitle);
 61 
 62         var serviceUri = new Element("td").update("Service URL");
 63         tHead.appendChild(serviceUri);
 64 
 65         var info = new Element("td").update("Get more info");
 66         tHead.appendChild(info);
 67 
 68         this._iServeList.each(function(service) {
 69             var row = new Element("tr");
 70             table.appendChild(row);
 71 
 72             var name = service.label || "-";
 73             var serviceName = new Element("td").update(name);
 74             row.appendChild(serviceName);
 75 
 76             var url = new Element("td").update(service.address);
 77             row.appendChild(url);
 78 
 79             var moreinfo = new Element("td").update(
 80                 '<a target="_blank" style="text-decoration:none" href="' + service.service + '">+</a>'
 81             );
 82             row.appendChild(moreinfo);
 83         });
 84 
 85         this._setContent(content);
 86     }
 87 });
