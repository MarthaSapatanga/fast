147     /**

157                             'values': [

159                                     screen.name,
160                                 '<span class="bold">Version: </span>' +
161                                      screen.version,
162                                 '<span class="bold">Tags: </span>' +
163                                     screen.tags.collect(function(tag) {
164                                         return tag.label['en-gb'];
165                                     }).join(", "),
166                                 '<span class="bold">Description </span><br />'+
167                                     screen.description['en-gb'],

186     /**
187      * Share or unshare a screen depending of its status
188      * @private
189      */
190     _shareBuildingBlock: function(/** String */ id) {
191         var uri = URIs.share.replace("<id>", id);
192 
193         var screen = this._buildingBlocks.detect(function(element) {
194             return element.id == id;
195         });
196         if (screen.uri) {
197             // Unshare screen
198             PersistenceEngine.sendDelete(uri, this, this._reload, Utils.onAJAXError);
199         } else {
200             // Share screen
201             PersistenceEngine.sendPost(uri, null, null, this, this._reload,
202                                         Utils.onAJAXError);
203         }
204     },

238     /**
239      * Load selected building blocks
240      * @private
241      */
242     _loadBuildinBlocks: function(onSucess) {
243         var types = [];
244         this._buildingBlocks = [];
245 
246         if (this._showForms) {
247             types.push('form');
248         }
249         if (this._showOperators) {
250             types.push('operator');
251         }
252         if (this._showResources) {
253             types.push('resource');
254         }
255         this._getBuildingBlocks(types, this, onSucess);
256     },
257 
258     /**
259      * Get selected building blocks
260      * @private
261      */
262     _getBuildingBlocks: function(/*Array<String>*/types, /*Object*/context, /*Function*/onSucess) {
263         var type = types.pop();
264         if (!type && onSucess instanceof Function) {
265             onSucess.call(context);
266         } else {
267             PersistenceEngine.sendGet(URIs[type], context, function(transport) {
268                 this._buildingBlocks = this._buildingBlocks.concat(JSON.parse(transport.responseText));
269                 this._getBuildingBlocks(types, context, onSucess);
270             }, Utils.onAJAXError);
271         }
272     }
