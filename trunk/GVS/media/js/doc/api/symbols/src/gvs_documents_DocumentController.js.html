 70     /**
 71      * Adds a new document.
 72      */
 73     addDocument: function(document){
 74         this._documents[document.getTabId()] = document;
 75         dijit.byId("documentContainer").addChild(document.getTab());
 76         dijit.byId("documentContainer").selectChild(document.getTabId());
 77         $$(".tabLabel").each(function(canvas){canvas.observe("focus",function(e){
 78            var element = Event.element(e);
 79            element.blur();
 80         })});
 81     },

 91

100     /**

102      */
103     showDeployCurrentDocDialog: function(){
104         switch(this._currentDocument.getDocumentType()) {
105             case Constants.DocumentType.WELCOME:
106                 break;
107             case Constants.DocumentType.SCREENFLOW:
108                 this.getCurrentDocument().showDeployGadgetDialog();

111             default:

115     /**
116      * this function closes a document by its Id
117      * @param String id
118      */
119     closeDocument: function (id){
120 
121         //Go to the previous tab and not to the
122         //initial tab
123         dijit.byId("documentContainer").back();
124         
125         //Remove the tab
126         dijit.byId("documentContainer").removeChild(dijit.byId(id));
127         dijit.byId(id).destroyRecursive(true);
128          
129         delete this._documents[id];
130         
131         if ($H(this._documents).keys().length == 0){
132             //Show the welcome Document
133             var welcome = new WelcomeDocument();
134             this.addDocument(welcome);
135         }
136     },

141     /**
142      * Select Document event handler.
143      * @private
144      */
145     _selectDocument: function( /** {String|DOMNode} */ tab) {
146         var id;
147 
148         if (tab.id) {  // it is tab widget
149             id = tab.id;
150         } else { //it is a string id
151             id = tab;
152         }
153         this._currentDocument = this._documents[id];
154         //Update the set of buttons
155         this._currentDocument.updateToolbar();
156     }
