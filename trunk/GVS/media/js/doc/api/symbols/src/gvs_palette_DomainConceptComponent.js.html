 39      * @type String
 40      * @override
 41      */
 42     _getTitle: function () {
 43         if (!this._buildingBlockDescription.title) {
 44             this._updateTitle();
 45         }
 46         return this._buildingBlockDescription.title;

 49     /**
 50      * Creates the title of the palette component
 51      * @type String
 52      * @private
 53      */
 54     _updateTitle: function() {
 55         
 56         if (this._buildingBlockDescription['http://www.w3.org/2000/01/rdf-schema#label']) {
 57             this._buildingBlockDescription.title = 
 58                         this._buildingBlockDescription['http://www.w3.org/2000/01/rdf-schema#label'].
 59                         replace("@en","");      
 60 
 61         } else { //Extract the title from the uri
 62             var uri;
 63             if (this._buildingBlockDescription['uri']) {
 64                 uri = this._buildingBlockDescription['uri'];
 65             } else {
 66                 uri = this._extractURI(this._buildingBlockDescription['pattern']);
 67             }
 68             this._buildingBlockDescription.title = this._createTitle(uri);
 69         }
 70     },
 71     /**
 72      * This function extracts an uri from a rdf pattern
 73      * TODO: this is the same function as FactFactory one. Extract to a common library
 74      * @private
 75      * @type String
 76      */
 77     _extractURI: function(/** String */ triplePattern) {
 78         if (triplePattern) {
 79             var pieces = triplePattern.split(" ");
 80             return pieces[2];
 81         }
 82         else {
 83             return "";
 84         }      
 85     },
 86     
 87     /**
 88      * This function creates a title from the uri identifier
 89      * @private
 90      * @type String
 91      */
 92     _createTitle: function(/** String */ uri) {
 93         if (uri) {
 94             var pieces = uri.split("#");
 95             var identifier = "";
 96             if (pieces.length > 1){
 97                 identifier = pieces[1];               
 98             }
 99             else { //The uri has not identifier, try the last part of the url
100                 pieces = uri.split("/");
101                 identifier = pieces[pieces.length - 1];
102             }
103             return this._sanitizeTitle(identifier);
104         } else {
105             return "Unknown Domain Concept";
106         }
107     },
108     
109     /**
110      * This function returns a human-readable title from an
111      * identifier
112      * @private
113      * @type String
114      */
115     _sanitizeTitle: function (/** String */ title) {
116         //FIXME
117         var result = title.replace("@en","");
118         //separate identifier from its forming words
119         var words = result.match(/[A-Z][a-z0-9]+|\s+[a-z][a-z0-9]*/g);
120         if (words) {
121             //TODO: compatibility with IE 
122             //https://developer.mozilla.org/en/Core_JavaScript_1.5_Reference/Objects/Array/map#Compatibility
123             words.map(function(e) {
124                 return e.strip();
125             });
126             result = words.join(" ");
127         }
128         return result;       
129     }
