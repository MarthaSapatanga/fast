 20     /**
 21      * Creates a new View instance for the component
 22      * @type BuildingBlockView
 23      * @override
 24      */
 25     _createView: function () {
 26         var view = new DomainConceptView(this._buildingBlockDescription);
 27         this._createTooltip(view.getNode(), this._buildingBlockDescription);
 28         return view;
 29     },
 30 
 31     /**
 32      * Creates a new Tooltip for the component view
 33      * @type BuildingBlockView
 34      * @override
 35      */
 36     _createTooltip: function (/*DomNode View*/ node) {
 37         var conceptName = this._buildingBlockDescription.getTitle();
 38 
 39         var content = document.createElement('div');
 40         var title = document.createElement('h3');
 41         title.appendChild(document.createTextNode(conceptName+' '));
 42         content.appendChild(title);
 43 
 44         if (this._buildingBlockDescription['description']) {
 45             var languages = $H(this._buildingBlockDescription['description']).keys();
 46             if (languages.size() > 0) {
 47                 var description = this._buildingBlockDescription['description'][languages[0]];
 48             } else {
 49                 var description = this._buildingBlockDescription['description'];
 50             }
 51             var descriptionElement = document.createElement('p');
 52             descriptionElement.appendChild(document.createTextNode(description));
 53             content.appendChild(descriptionElement);
 54         }
 55 
 56         if (this._buildingBlockDescription['subClassOf']) {
 57             var subClassOf = document.createElement('h4');
 58             subClassOf.appendChild(document.createTextNode('Subclass of:'));
 59             content.appendChild(subClassOf);
 60             content.appendChild(document.createTextNode(this._buildingBlockDescription['subClassOf']));
 61         }
 62 
 63         if (this._buildingBlockDescription['uri']) {
 64             var subClassOf = document.createElement('h4');
 65             subClassOf.appendChild(document.createTextNode('URI:'));
 66             content.appendChild(subClassOf);
 67             content.appendChild(document.createTextNode(this._buildingBlockDescription['uri']));
 68         }
 69 
 70         var tip = new dijit.Tooltip({connectId:[node],
 71             label:'<div style="max-width:300px">'+content.innerHTML+'</div>'});
 72         tip.startup();
 73     },
 74 
 75     /**
 76      * Creates a new domain concept to be dragged.
 77      * @type DomainConceptInstance
 78      * @override
 79      */
 80     _createInstance: function () {
 81         return new PrePostInstance(this._buildingBlockDescription, this._inferenceEngine);
 82     },
 83 
 84     /**
 85      * @type String
 86      * @override
 87      */
 88     _getTitle: function () {
 89         return this._buildingBlockDescription.getTitle();
 90     }
 91 });
