 26 var DomainConceptComponent = Class.create(PaletteComponent,
 27     /** @lends DomainConceptComponent.prototype */ {
 28 
 29     /**
 30      * Palette component of a domain concept building block.
 31      * @param BuildingBlockDescription domainConceptBuildingBlockDescription
 32      * @constructs
 33      * @extends PaletteComponent
 34      */
 35     initialize: function($super, description, dropZones, inferenceEngine) {
 36         $super(description, dropZones, inferenceEngine);
 37     },
 38 
 39 
 40     // **************** PUBLIC METHODS **************** //
 41 
 42 
 43     // **************** PRIVATE METHODS **************** //
 44 
 45     /**
 46      * Creates a new View instance for the component
 47      * @type BuildingBlockView
 48      * @override
 49      */
 50     _createView: function () {
 51         var view = new DomainConceptView(this._buildingBlockDescription);
 52         this._createTooltip(view.getNode(), this._buildingBlockDescription);
 53         return view;
 54     },
 55 
 56     /**
 57      * Creates a new Tooltip for the component view
 58      * @type BuildingBlockView
 59      * @override
 60      */
 61     _createTooltip: function (/*DomNode View*/ node) {
 62         var conceptName = this._buildingBlockDescription.getTitle();
 63 
 64         var content = document.createElement('div');
 65         var title = document.createElement('h3');
 66         title.appendChild(document.createTextNode(conceptName+' '));
 67         content.appendChild(title);
 68 
 69         if (this._buildingBlockDescription['description']) {
 70             var languages = $H(this._buildingBlockDescription['description']).keys();
 71             if (languages.size() > 0) {
 72                 var description = this._buildingBlockDescription['description'][languages[0]];
 73             } else {
 74                 var description = this._buildingBlockDescription['description'];
 75             }
 76             var descriptionElement = document.createElement('p');
 77             descriptionElement.appendChild(document.createTextNode(description));
 78             content.appendChild(descriptionElement);
 79         }
 80 
 81         if (this._buildingBlockDescription['subClassOf']) {
 82             var subClassOf = document.createElement('h4');
 83             subClassOf.appendChild(document.createTextNode('Subclass of:'));
 84             content.appendChild(subClassOf);
 85             content.appendChild(document.createTextNode(this._buildingBlockDescription['subClassOf']));
 86         }
 87 
 88         if (this._buildingBlockDescription['uri']) {
 89             var subClassOf = document.createElement('h4');
 90             subClassOf.appendChild(document.createTextNode('URI:'));
 91             content.appendChild(subClassOf);
 92             content.appendChild(document.createTextNode(this._buildingBlockDescription['uri']));
 93         }
 94 
 95         var tip = new dijit.Tooltip({connectId:[node],
 96             label:'<div style="max-width:300px">'+content.innerHTML+'</div>'});
 97         tip.startup();
 98     },
 99 
100     /**
101      * Creates a new domain concept to be dragged.
102      * @type DomainConceptInstance
103      * @override
104      */
105     _createInstance: function () {
106         return new PrePostInstance(this._buildingBlockDescription, this._inferenceEngine);
107     },
108 
109     /**
110      * @type String
111      * @override
112      */
113     _getTitle: function () {
114         return this._buildingBlockDescription.getTitle();
115     }
116 });
