1 var BrowserUtilsSingleton = function () {
  2 
  3 	// *********************************
  4 	// SINGLETON INSTANCE
  5 	// *********************************
  6 	var instance = null;
  7 
  8 	var BrowserUtils = Class.create({
  9 		initalize: function (){},
 10 		// ****************
 11 		// PUBLIC METHODS 
 12 		// ****************
 13 		
 14 		getHeight : function () {
 15 			var newHeight=window.innerHeight; //Non-IE (Firefox and Opera)
 16 			  
 17 			if( document.documentElement && document.documentElement.clientHeight ) {
 18 			  //IE 6+ in 'standards compliant mode'
 19 			  newHeight = document.documentElement.clientHeight;
 20 			} else if( document.body && document.body.clientHeight ) {
 21 			  //IE 4 compatible and IE 5-7 'quirk mode'
 22 			  newHeight = document.body.clientHeight;
 23 			}
 24 			return newHeight;
 25 		},
 26 		
 27 		//gets total width
 28 		getWidth : function(){	
 29 			var newWidth=window.innerWidth; //Non-IE (Firefox and Opera)
 30 		  
 31 			if( document.documentElement && ( document.documentElement.clientWidth || document.documentElement.clientHeight ) ) {
 32 			    //IE 6+ in 'standards compliant mode'
 33 		  		newWidth = document.documentElement.clientWidth;
 34 			} else if( document.body && ( document.body.clientWidth || document.body.clientHeight ) ) {
 35 		  		//IE 4 compatible and IE 5-7 'quirk mode'
 36 			  newWidth = document.body.clientWidth;
 37 			}
 38 			return newWidth;
 39 		},
 40 			
 41 		isLeftButton : function(button){
 42 			
 43 			if (button == 0 || (this.isIE() && button == 1))
 44 				return true;
 45 			else
 46 				return false;	
 47 		},
 48 	
 49 		getBrowser : function(){ 
 50 			if (/MSIE (\d+\.\d+);/.test(navigator.userAgent)){ //test for MSIE x.x;
 51 				var ieversion=new Number(RegExp.$1) // capture x.x portion and store as a number
 52 				if (ieversion>=8)
 53 					return "IE8";
 54 				else if (ieversion>=7)
 55 					return "IE7";
 56 				else if (ieversion>=6)
 57 					return "IE6";
 58 				else if (ieversion>=5)
 59 					return "IE5";
 60 			} else if (/Firefox[\/\s](\d+\.\d+)/.test(navigator.userAgent)){ //test for Firefox/x.x or Firefox x.x (ignoring remaining digits);
 61 				var ffversion=new Number(RegExp.$1) // capture x.x portion and store as a number
 62 				if (ffversion>=3)
 63 			 		return "FF3"
 64 				else if (ffversion>=2)
 65 			  		return "FF2"
 66 			 	else if (ffversion>=1)
 67 			  		return "FF1"
 68 			} else{
 69 				return "OTHER";
 70 			}			
 71 		},
 72 		
 73 		isIE : function(){ 
 74 			if (/MSIE (\d+\.\d+);/.test(navigator.userAgent)) //test for MSIE x.x;	
 75 				return true;
 76 			else
 77 				return false;		
 78 		}
 79 	});
 80 
 81 	// *********************************
 82 	// SINGLETON GET INSTANCE
 83 	// *********************************
 84 	return new function() {
 85     	this.getInstance = function() {
 86     		if (instance == null) {
 87         		instance = new BrowserUtils();
 88          	}
 89          	return instance;
 90        	}
 91 	}
 92 	
 93 }();
