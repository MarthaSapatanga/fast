  4 var ScreenComponent = Class.create(PaletteComponent,{
  5 	initialize: function($super, resourceDescription) {
  6 		$super(resourceDescription);
  7 		
  8    		/**********
  9 		 * Public properties
 10 		 */
 11 		
 12 		/**********
 13 		 * Private properties
 14 		 */
 15 		var uidGenerator = UIDGeneratorSingleton.getInstance();
 16 		this._id = uidGenerator.generate("screenComponent");
 17 		
 18 		var title = new Element("div", {"class":"screenTitle unknown"});
 19 		title.update(this._resourceDescription.name);
 20 		
 21 		var factFactory = FactFactorySingleton.getInstance();
 22 		var preArea = new Element("div", {"class": "preArea"});
 23 		var preIcons = [];
 24 		$A(this._resourceDescription.pre).each(
 25 				function(pre) {
 26 					var preFact = factFactory.getFactIcon(pre, "medium");
 27 					preIcons.push(preFact);
 28 					preArea.appendChild(preFact.getNode());
 29 				}
 30 		);
 31 		this._preIcons = preIcons;
 32 		
 33 		var postArea = new Element("div", {"class": "postArea"});
 34 		postIcons = [];
 35 		$A(this._resourceDescription.post).each(
 36 				function(post) {
 37 					var postFact = factFactory.getFactIcon(post, "medium");
 38 					postIcons.push(postFact);
 39 					postArea.appendChild(postFact.getNode());
 40 				}
 41 		);
 42 		this._postIcons = postIcons;
 43 		
 44 		var prePostSeparator = new Element("div", {"class":"prePostSeparator"});
 45 		
 46 		this._content = new Element("div",{"id":this._id,"class":"component screen unknown"});
 47 		this._content.appendChild(title);
 48 		this._content.appendChild(preArea);
 49 		this._content.appendChild(prePostSeparator);
 50 		this._content.appendChild(postArea);
 51 		
 52 		this._node = this._createSlot(this._resourceDescription.name, this._content);
 53 	}
 54 });
