1 var AbstractDocument = Class.create( /** @lends AbstractDocument.prototype */ {
  2     /**
  3      * Represents a document and its tab. Subclasses must provide the 
  4      * inner content.
  5      * @abstract
  6      * @constructs
  7      */ 
  8     initialize: function(/** String */ title) {
  9         var uidGenerator = UIDGeneratorSingleton.getInstance();
 10 
 11         /**
 12          * Accepted resources
 13          * @type {String[]}
 14          * @private
 15          */
 16         this._validResources = [];
 17         
 18         /**
 19          * Tab title
 20          * @type String
 21          * @private
 22          */
 23         this._title = title;
 24         
 25         /**
 26          * Actual Tab Id
 27          * @type String
 28          * @private
 29          */
 30         this._tabId = uidGenerator.generate("tab");
 31         
 32         /**
 33          * Tab content id
 34          * @type String
 35          * @private
 36          */
 37         this._tabContentId = uidGenerator.generate("tabContent");
 38         
 39         /**
 40          * Initial tab content (empty by default)
 41          * @type DOMNode
 42          * @private
 43          */
 44         this._tabContent = new Element("div", {
 45             "id":    this._tabContentId,
 46             "class": "document"
 47         });        
 48         
 49         /**
 50          * Actual tab
 51          * @type DOMNode
 52          * @private
 53          */
 54         this._tab = new dijit.layout.ContentPane({
 55             title: this._title,
 56             id: this._tabId,
 57             closable: true
 58         }, null);
 59         
 60         this._tab.setContent(this._tabContent);
 61     },
 62     
 63 
 64     // **************** PUBLIC METHODS **************** //
 65     
 66 
 67     /**
 68      * Returns the list of valid resources for the document type
 69      * @type {String[]}

 71     getValidResources: function () {

 73     },
 74     
 75     
 76     /**
 77      * Gets the tab id.
 78      * @type String
 79      */
 80     getTabId: function () {
 81         return this._tabId;
 82     },
 83     
 84     
 85     /**
 86      * Gets the tab node.
 87      * @type DOMNode
 88      */
 89     getTab: function() {
 90         return this._tab;
 91     },
 92     
 93     
 94     /**
 95      * Gets the title.
 96      * @type String
 97      */
 98     getTitle: function () {
 99         return this._title;
100     },
101 
102     
103     /**
104      * Gets the content node.
105      * @type DOMNode
106      */
107     getContent: function () {

109     },
110     
111     
112     /**
113      * Gets the content id.
114      * @type String
115      */
116     getContentId: function () {

118     },
119 
120
